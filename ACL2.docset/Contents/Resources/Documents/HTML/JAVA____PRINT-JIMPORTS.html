Function: print-jimports

(defun print-jimports (imports indent-level)
       (declare (xargs :guard (and (jimport-listp imports)
                                   (natp indent-level))))
       (let ((__function__ 'print-jimports))
            (declare (ignorable __function__))
            (cond ((endp imports) nil)
                  (t (cons (print-jimport (car imports)
                                          indent-level)
                           (print-jimports (cdr imports)
                                           indent-level))))))
