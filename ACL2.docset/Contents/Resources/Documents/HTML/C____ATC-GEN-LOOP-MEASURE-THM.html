This is like the applicability condition, except that it uses the generated measure function (to treat the measure as a black box, as discussed in atc-gen-loop-measure-fn), and that it is a type prescription rule (which seems needed, as opposed a rewrite rule, based on proof experiments).

(defun
 atc-gen-loop-measure-thm
 (fn fn-appconds appcond-thms measure-of-fn
     measure-formals names-to-avoid wrld)
 (declare (xargs :guard (and (symbolp fn)
                             (symbol-symbol-alistp fn-appconds)
                             (keyword-symbol-alistp appcond-thms)
                             (symbolp measure-of-fn)
                             (symbol-listp measure-formals)
                             (symbol-listp names-to-avoid)
                             (plist-worldp wrld))))
 (declare (xargs :guard (irecursivep+ fn wrld)))
 (let
  ((__function__ 'atc-gen-loop-measure-thm))
  (declare (ignorable __function__))
  (b*
   ((appcond-thm (cdr (assoc-eq (cdr (assoc-eq fn fn-appconds))

         (packn-pos (list 'natp-of-measure-of- fn)
