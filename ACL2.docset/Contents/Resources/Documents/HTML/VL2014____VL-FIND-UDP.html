Definitions and Theorems

Function: vl-find-udp

(defun
     vl-find-udp (name x)
     (declare (xargs :guard (and (stringp name) (vl-udplist-p x))))
     (let ((__function__ 'vl-find-udp))
          (declare (ignorable __function__))
          (cond ((atom x) nil)
                ((equal (string-fix name)
                        (vl-udp->name (car x)))
                 (vl-udp-fix (car x)))
                (t (vl-find-udp name (cdr x))))))

(defthm vl-find-udp-under-iff
