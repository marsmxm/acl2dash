Function: identifier-part-p

(defun identifier-part-p (char)
       (declare (xargs :guard (unicodep char)))
       (let ((__function__ 'identifier-part-p))
            (declare (ignorable __function__))
            (b* ((char (mbe :logic (unicode-fix char)
                            :exec char)))
                (cond ((asciip char)
                       (ascii-identifier-part-p char))
                      (t (nonascii-identifier-part-p char))))))
