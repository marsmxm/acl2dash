Function: vl-jp-modinst-fn

(defun
     vl-jp-modinst-fn (x ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard (vl-modinst-p x)))
     (let ((__function__ 'vl-jp-modinst))
          (declare (ignorable __function__))
          (vl-ps-seq (vl-print "{\"tag\": ")
                     (jp-sym ':vl-modinst)
                     (vl-print ", ")
                     (vl-print-str "\"instname\": ")
                     (jp-maybe-string (vl-modinst->instname x))
                     (vl-println? ", ")
                     (vl-print-str "\"modname\": ")
                     (jp-str (vl-modinst->modname x))
                     (vl-println? ", ")
                     (vl-print-str "\"range\": ")
                     (vl-jp-maybe-range (vl-modinst->range x))
                     (vl-println? ", ")
                     (vl-print-str "\"paramargs\": ")
                     (vl-jp-paramargs (vl-modinst->paramargs x))
                     (vl-println? ", ")
                     (vl-print-str "\"portargs\": ")
                     (vl-jp-arguments (vl-modinst->portargs x))
                     (vl-println? ", ")
                     (vl-print-str "\"str\": ")
                     (vl-jp-maybe-gatestrength (vl-modinst->str x))
                     (vl-println? ", ")
                     (vl-print-str "\"delay\": ")
                     (vl-jp-maybe-gatedelay (vl-modinst->delay x))
                     (vl-println? ", ")
                     (vl-print-str "\"atts\": ")
                     (vl-jp-atts (vl-modinst->atts x))
                     (vl-println? ", ")
                     (vl-print-str "\"loc\": ")
                     (vl-jp-location (vl-modinst->loc x))
                     (vl-println? "}"))))
