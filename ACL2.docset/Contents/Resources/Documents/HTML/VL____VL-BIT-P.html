Function: vl-bit-p

(defun vl-bit-p (x)
       (declare (xargs :guard t))
       (or (eq x ':vl-0val)
           (eq x ':vl-1val)
           (eq x ':vl-xval)
           (eq x ':vl-zval)))

(defthm type-when-vl-bit-p
        (implies (vl-bit-p x)
                 (if (symbolp x)
                     (if (not (equal x 't))
                         (not (equal x 'nil))
                         'nil)
                     'nil))
