Function: vl-10-integer-token-p

(defun
 vl-10-integer-token-p (x)
 (declare (xargs :guard (vl-token-p x)))
 (let
  ((__function__ 'vl-10-integer-token-p))
  (declare (ignorable __function__))
  (case
    (vl-token->type x)
    (:vl-inttoken (let ((etext (vl-inttoken->etext x)))
                       (and (eql (len etext) 2)
                            (eql (vl-echar->char (first etext)) #\1)
                            (eql (vl-echar->char (second etext))
                                 #\0))))
    (otherwise nil))))
