Function: vl-port->loc

(defun
   vl-port->loc (x)
   (declare (xargs :guard (vl-port-p x)))
   (let ((__function__ 'vl-port->loc))
        (declare (ignorable __function__))
        (b* ((x (vl-port-fix x)))
            (case (tag x)
                  (:vl-regularport (vl-regularport->loc x))
                  (:vl-interfaceport (vl-interfaceport->loc x))
