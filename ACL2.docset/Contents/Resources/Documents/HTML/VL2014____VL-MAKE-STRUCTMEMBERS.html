Function: vl-make-structmembers

(defun
 vl-make-structmembers
 (atts rand type decls)
 (declare (xargs :guard (and (vl-atts-p atts)
                             (vl-randomqualifier-p rand)
                             (vl-datatype-p type)
                             (vl-vardeclassignlist-p decls))))
 (let
    ((__function__ 'vl-make-structmembers))
    (declare (ignorable __function__))
    (b* (((when (atom decls)) nil)
         ((vl-vardeclassign decl) (car decls)))
        (cons (make-vl-structmember
                   :atts atts
                   :rand rand
                   :type (vl-datatype-update-udims decl.dims type)
                   :name decl.id
                   :rhs decl.expr)
              (vl-make-structmembers atts rand type (cdr decls))))))
