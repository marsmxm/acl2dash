Function: integer-setp

(defun integer-setp (x)
       (declare (xargs :guard t))
       (let ((acl2::__function__ 'integer-setp))
            (declare (ignorable acl2::__function__))
            (and (setp x) (set-all-integerp x))))

Theorem: booleanp-of-integer-setp

(defthm booleanp-of-integer-setp
        (acl2::b* ((yes/no (integer-setp x)))

Theorem: setp-when-integer-setp

(defthm setp-when-integer-setp
        (implies (integer-setp ints)
                 (setp ints)))

Theorem: integer-setp-of-insert

(defthm integer-setp-of-insert
        (equal (integer-setp (insert int ints))
               (and (integerp int)
                    (integer-setp (sfix ints)))))

Theorem: integer-setp-of-union

(defthm integer-setp-of-union
        (equal (integer-setp (union ints1 ints2))
               (and (integer-setp (sfix ints1))
                    (integer-setp (sfix ints2)))))
