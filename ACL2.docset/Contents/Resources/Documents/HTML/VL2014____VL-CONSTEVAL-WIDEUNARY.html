(defun
 vl-consteval-wideunary (op aval)
 (declare
  (xargs
    :guard
    (and (member op
                 '(:vl-unary-plus :vl-unary-minus :vl-unary-bitnot))
         (natp aval))))
 (let ((__function__ 'vl-consteval-wideunary))
      (declare (ignorable __function__))
      (b* ((aval (lnfix aval)))
          (case op (:vl-unary-plus aval)
                (:vl-unary-minus (- aval))
