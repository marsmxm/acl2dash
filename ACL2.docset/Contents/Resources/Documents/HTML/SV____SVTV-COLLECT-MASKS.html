Function: svtv-collect-masks

(defun
  svtv-collect-masks (x)
  (declare (xargs :guard (svtv-lines-p x)))
  (let ((__function__ 'svtv-collect-masks))
       (declare (ignorable __function__))
       (b* (((when (atom x)) nil)
            ((svtv-line xf) (car x))
            (vars (svtv-entries->vars xf.entries))
            (overrideconds (svtv-entries->overrideconds xf.entries))
            (mask (lhs->mask xf.lhs)))
           (append (pairlis$ vars (replicate (len vars) mask))
                   (pairlis$ overrideconds
                             (replicate (len overrideconds) mask))
                   (svtv-collect-masks (cdr x))))))
