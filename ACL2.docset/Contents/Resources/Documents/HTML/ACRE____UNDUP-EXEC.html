Function: undup-exec

(defun
     undup-exec (x acc)
     (declare (xargs :guard (and (true-listp x) (alistp acc))))
     (let ((__function__ 'undup-exec))
          (declare (ignorable __function__))
          (if (atom x)
              (rev-keys (fast-alist-free acc) nil)
              (undup-exec (cdr x)
                          (b* ((x1 (car x)))
                              (if (hons-get x1 acc)
                                  acc (hons-acons x1 nil acc)))))))
