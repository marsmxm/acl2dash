(defun
  idiv-spec$inline (size dst src)
  (declare (type (member 1 2 4 8) size))
  (declare
       (xargs :guard (and (not (equal src 0))
                          (case size (1 (and (n08p src) (n16p dst)))
                                (2 (and (n16p src) (n32p dst)))
                                (4 (and (n32p src) (n64p dst)))
                                (8 (and (n64p src) (n128p dst)))
                                (otherwise nil)))))

        (2 (idiv-spec-16 dst src))
        (4 (idiv-spec-32 dst src))
        (8 (idiv-spec-64 dst src))
