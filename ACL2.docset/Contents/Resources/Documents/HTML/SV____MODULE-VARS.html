Function: module-vars

(defun module-vars (x)
       (declare (xargs :guard (module-p x)))
       (let ((__function__ 'module-vars))

                (append (assigns-vars x.assigns)

Theorem: vars-of-module

(defthm
 vars-of-module
 (implies
  (and (not (member v (lhspairs-vars aliases)))
       (not (member v (assigns-vars assigns)))
       (not (member v (constraintlist-vars constraints)))
       (not (member v (assigns-vars fixups))))
  (not
   (member
       v
       (module-vars (module wires insts
                            assigns fixups constraints aliases))))))
