Function: int-valuep

(defun int-valuep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'int-valuep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :int)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((int (std::da-nth 0 (cdr x))))
                     (sbyte32p int)))))
