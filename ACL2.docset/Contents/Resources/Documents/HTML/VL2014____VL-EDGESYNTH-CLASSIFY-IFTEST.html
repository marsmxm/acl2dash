Function: vl-edgesynth-classify-iftest

(defun
 vl-edgesynth-classify-iftest
 (condition edgetable)
 (declare (xargs :guard (and (vl-expr-p condition)
                             (vl-edgetable-p edgetable))))
 (let
  ((__function__ 'vl-edgesynth-classify-iftest))
  (declare (ignorable __function__))
  (b*
     ((used-names (vl-expr-names condition))
      ((unless (intersectp-equal used-names (alist-keys edgetable)))
       (mv :data condition))
      ((unless (equal (len used-names) 1))
       (mv :error condition)))
     (vl-edgesynth-classify-iftest-aux condition))))
