Function: parse-dec-val

(defun
 parse-dec-val (input)
 (declare (xargs :guard (nat-listp input)))
 (seq
  input (tree := (parse-ichar #\d input))
  (trees := (parse-1*digit input))
  (tree-rest := (parse-dec-val-rest input))
  (return
      (make-tree-nonleaf :rulename? *dec-val*
                         :branches (list (list tree)
                                         trees (list tree-rest))))))
