Function: vl-parse-op-alist-p

(defun vl-parse-op-alist-p (arity x)
       (declare (xargs :guard (maybe-natp arity)))
       (let ((__function__ 'vl-parse-op-alist-p))
            (declare (ignorable __function__))
            (if (atom x)
                (not x)
                (and (consp (car x))
                     (symbolp (caar x))
                     (vl-op-p (cdar x))
                     (eql arity (vl-op-arity (cdar x)))
