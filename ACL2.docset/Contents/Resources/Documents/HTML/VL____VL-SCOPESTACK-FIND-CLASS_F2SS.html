Function: vl-scopestack-find-class/ss

(defun
 vl-scopestack-find-class/ss (name ss)
 (declare (xargs :guard (and (stringp name)
                             (vl-scopestack-p ss))))
 (let
  ((__function__ 'vl-scopestack-find-class/ss))
  (declare (ignorable __function__))
  (b*
   ((ss (vl-scopestack-fix ss)))
   (vl-scopestack-case
     ss
     :null (mv nil nil)
     :global (b* ((class (vl-scope-find-class-fast name ss.design)))
                 (mv class (vl-scopestack-fix ss)))
     :local (b* ((class (vl-scope-find-class-fast name ss.top))
                 ((when class) (mv class ss)))
                (vl-scopestack-find-class/ss name ss.super))))))
