Function: vl-modulelist-portcheck

(defun
 vl-modulelist-portcheck (x)
 (declare (xargs :guard (vl-modulelist-p x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-modulelist-portcheck))
  (declare (ignorable __function__))
  (mbe
     :logic (if (consp x)
                (cons (vl-module-portcheck (car x))
                      (vl-modulelist-portcheck (cdr x)))
                nil)
     :exec
     (if (atom x)
         nil
         (with-local-nrev (vl-modulelist-portcheck-nrev x nrev))))))
