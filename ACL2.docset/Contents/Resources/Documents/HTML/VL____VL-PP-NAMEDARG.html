Function: vl-pp-namedarg-fn

(defun
 vl-pp-namedarg-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (vl-namedarg-p x)))
 (let
  ((__function__ 'vl-pp-namedarg))
  (declare (ignorable __function__))
  (b*
    (((vl-namedarg x)))
    (vl-ps-seq
         (if x.atts (vl-pp-atts x.atts) ps)
         (vl-print ".")
         (vl-ps-span "vl_id"
                     (vl-print (vl-maybe-escape-identifier x.name)))
         (if (and x.nameonly-p x.expr (vl-idexpr-p x.expr)
                  (equal (vl-idexpr->name x.expr) x.name))
             ps
             (vl-ps-seq (vl-print "(")
                        (if x.expr (vl-pp-expr x.expr) ps)
                        (vl-print ")")))))))
