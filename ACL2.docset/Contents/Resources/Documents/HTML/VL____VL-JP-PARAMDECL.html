(defun vl-jp-paramdecl-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-paramdecl-p x)))
       (let ((__function__ 'vl-jp-paramdecl))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-paramdecl)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-paramdecl->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"type\": ")
                       (vl-jp-paramtype (vl-paramdecl->type x))
                       (vl-println? ", ")
                       (vl-print-str "\"loc\": ")
                       (vl-jp-location (vl-paramdecl->loc x))
                       (vl-println? ", ")
                       (vl-print-str "\"localp\": ")
                       (jp-bool (vl-paramdecl->localp x))
                       (vl-println? ", ")
                       (vl-print-str "\"overriddenp\": ")
                       (jp-bool (vl-paramdecl->overriddenp x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-paramdecl->atts x))
                       (vl-println? "}"))))
