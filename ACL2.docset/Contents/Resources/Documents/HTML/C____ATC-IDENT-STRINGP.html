Function: atc-ident-stringp

(defun atc-ident-stringp (str)
       (declare (xargs :guard (stringp str)))
       (let ((__function__ 'atc-ident-stringp))
            (declare (ignorable __function__))
            (and (atc-ident-char-listp (acl2::explode str))
                 (not (member-equal str *ckeywords*)))))
