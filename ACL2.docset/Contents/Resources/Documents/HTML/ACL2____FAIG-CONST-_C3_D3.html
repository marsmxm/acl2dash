Function: faig-const-<=

(defun faig-const-<= (x y)
       (declare (xargs :guard t))
       (let ((__function__ 'faig-const-<=))
            (declare (ignorable __function__))
            (let ((x (faig-const-fix x))
                  (y (faig-const-fix y)))
                 (or (equal x y) (equal x (faig-x))))))
