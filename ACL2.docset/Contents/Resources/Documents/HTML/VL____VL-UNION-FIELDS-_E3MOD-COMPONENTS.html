Iterates over the fields of a union, creating svex module components for each field using vl-datatype-elem->mod-components

Signature
(vl-union-fields->mod-components members subwires submodnames) 
  → 
(mv wires insts aliases width)
Arguments
members — Guard (vl-structmemberlist-p members).
subwires — Guard (sv::wirelist-p subwires).
submodnames — Guard (sv::maybe-modnamelist-p submodnames).
Returns
wires — Type (sv::wirelist-p wires).
insts — Type (sv::modinstlist-p insts).
aliases — Type (sv::lhspairs-p aliases).

Definitions and Theorems

Function: vl-union-fields->mod-components

(defun
 vl-union-fields->mod-components
 (members subwires submodnames)
 (declare
      (xargs :guard (and (vl-structmemberlist-p members)
                         (sv::wirelist-p subwires)
                         (sv::maybe-modnamelist-p submodnames))))
 (declare (xargs :guard (and (equal (len subwires) (len members))
                             (equal (len submodnames)
                                    (len members)))))
 (let ((__function__ 'vl-union-fields->mod-components))
      (declare (ignorable __function__))
      (b* (((when (atom members))
            (mv nil nil nil 0))
           ((vl-structmember x) (car members))
           ((mv wires insts aliases width)
            (vl-union-fields->mod-components (cdr members)
                                             (cdr subwires)
                                             (cdr submodnames)))
           ((mv wire1 insts1 aliases1)
            (vl-datatype-elem->mod-components x.name (car subwires)
                                              0 (car submodnames))))
          (mv (cons wire1 wires)
              (append-without-guard insts1 insts)
              (append-without-guard aliases1 aliases)
              (max (sv::wire->width wire1) width)))))
