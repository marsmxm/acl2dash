Function: typo-partition

(defun typo-partition (x)
       (declare (xargs :guard (character-listp x)))
       (let ((__function__ 'typo-partition))
            (declare (ignorable __function__))
            (b* (((when (atom x)) nil)
                 ((mv prefix remainder)
                  (typo-read-part x))
                 ((unless prefix) nil))
                (cons prefix (typo-partition remainder)))))
