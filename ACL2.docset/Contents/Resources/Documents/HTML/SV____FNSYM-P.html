Function: fnsym-p

(defun fnsym-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'fnsym-p))
            (declare (ignorable __function__))
            (and (symbolp x)
                 (not (eq x 'quote))
                 (not (keywordp x)))))
