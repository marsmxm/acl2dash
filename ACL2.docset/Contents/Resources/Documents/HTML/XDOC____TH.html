Macro: th

(defmacro
   th (&rest args)
   (mv-let
        (trees attributes)
        (partition-macro-args args 'th)
        (let ((attributes (keyword-macro-args-to-terms attributes)))
             (list 'th-fn
                   (cons 'list attributes)
                   (cons 'list trees)))))

(defun th-fn (attributes trees)

                                   (tree-listp trees))))

Theorem: stringp-of-th

(defthm stringp-of-th
        (equal (treep (th-fn attributes trees))
               (and (keyword-tree-alistp attributes)
                    (tree-listp trees))))
