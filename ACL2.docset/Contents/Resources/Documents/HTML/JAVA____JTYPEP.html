Function: jtypep

(defun jtypep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'jtypep))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x) (eq (car x) :prim))
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((type (std::da-nth 0 (cdr x))))
                                 (primitive-typep type))))
                       ((eq (car x) :class)
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((name (std::da-nth 0 (cdr x))))
                                 (stringp name))))
                       (t (and (eq (car x) :array)
                               (and (true-listp (cdr x))
                                    (eql (len (cdr x)) 1))
                               (b* ((comp (std::da-nth 0 (cdr x))))
                                   (jtypep comp))))))))

Theorem: consp-when-jtypep

(defthm consp-when-jtypep
        (implies (jtypep x) (consp x))
        :rule-classes :compound-recognizer)
