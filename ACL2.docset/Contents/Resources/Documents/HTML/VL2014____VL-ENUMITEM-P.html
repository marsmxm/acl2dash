Function: vl-enumitem-p

(defun
     vl-enumitem-p (x)
     (declare (xargs :guard t))
     (let ((__function__ 'vl-enumitem-p))
          (declare (ignorable __function__))
          (and (consp x)
               (eq (car x) :vl-enumitem)
               (std::prod-consp (cdr x))
               (std::prod-consp (std::prod-cdr (cdr x)))
               (b* ((name (std::prod-car (cdr x)))
                    (range (std::prod-car (std::prod-cdr (cdr x))))
                    (value (std::prod-cdr (std::prod-cdr (cdr x)))))
                   (and (stringp name)
                        (vl-maybe-range-p range)
                        (vl-maybe-expr-p value))))))
