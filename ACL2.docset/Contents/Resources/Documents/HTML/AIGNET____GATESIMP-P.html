(defun
  gatesimp-p (x)
  (declare (xargs :guard t))
  (let ((__function__ 'gatesimp-p))
       (declare (ignorable __function__))
       (and (mbe :logic (unsigned-byte-p 6 x)
                 :exec (and (natp x) (< x 64)))
