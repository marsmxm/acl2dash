Function: rangemap-p

(defun rangemap-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'rangemap-p))
            (declare (ignorable __function__))
            (if (atom x)
                (eq x nil)
                (and (consp (car x))
                     (address-p (caar x))
                     (rangelist-p (cdar x))
                     (rangemap-p (cdr x))))))
