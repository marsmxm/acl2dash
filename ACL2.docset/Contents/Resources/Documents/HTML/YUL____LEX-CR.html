Function: lex-cr

(defun
 lex-cr (abnf::input)
 (declare (xargs :guard (nat-listp abnf::input)))
 (let
  ((__function__ 'lex-cr))
  (declare (ignorable __function__))
  (b*
   (((mv abnf::treess abnf::input)
     (b*
      (((mv abnf::treess1 abnf::input1)
        (b* (((mv abnf::tree abnf::input)
              (abnf::parse-direct (list 13)
                                  abnf::input))
             ((when (resulterrp abnf::tree))
              (mv (fty::err-push abnf::tree)
                  abnf::input))
             (abnf::trees1 (list abnf::tree))
             (abnf::treess (list abnf::trees1)))
            (mv abnf::treess abnf::input)))
       ((when (not (resulterrp abnf::treess1)))
        (mv abnf::treess1 abnf::input1)))
      (mv
       (err
        (list
            :found (list abnf::treess1)
            :required '(((:repetition (:repeat 1 (:finite 1))
                                      (:num-val (:direct (13))))))))
       abnf::input)))
    ((when (resulterrp abnf::treess))
     (mv (fty::err-push abnf::treess)
         (acl2::nat-list-fix abnf::input))))
   (mv (abnf::make-tree-nonleaf :rulename? (abnf::rulename "cr")
                                :branches abnf::treess)
       abnf::input))))
