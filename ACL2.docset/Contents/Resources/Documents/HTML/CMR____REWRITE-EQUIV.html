Theorem: rewrite-equiv-is-an-equivalence

(defthm rewrite-equiv-is-an-equivalence
        (and (booleanp (rewrite-equiv x y))
             (rewrite-equiv x x)
             (implies (rewrite-equiv x y)
                      (rewrite-equiv y x))
             (implies (and (rewrite-equiv x y)
                           (rewrite-equiv y z))
                      (rewrite-equiv x z)))
        :rule-classes (:equivalence))
