Function: vl-lintconfig-loadconfig

(defun
     vl-lintconfig-loadconfig
     (config defines)
     (declare (xargs :guard (and (vl-lintconfig-p config)
                                 (vl-defines-p defines))))
     (let ((__function__ 'vl-lintconfig-loadconfig))
          (declare (ignorable __function__))
          (b* (((vl-lintconfig config)))
              (make-vl-loadconfig :edition config.edition
                                  :strictp config.strict
                                  :start-files config.start-files
                                  :search-path config.search-path
                                  :search-exts config.search-exts
                                  :include-dirs config.include-dirs
                                  :plusargs config.plusargs
                                  :defines defines
                                  :mintime 1/100
                                  :debugp config.debug))))
