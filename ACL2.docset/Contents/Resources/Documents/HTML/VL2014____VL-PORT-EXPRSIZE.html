Function: vl-port-exprsize

(defun vl-port-exprsize (x ss warnings)
       (declare (xargs :guard (and (vl-port-p x)
                                   (vl-scopestack-p ss)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-port-exprsize))
            (declare (ignorable __function__))
            (b* ((x (vl-port-fix x))
                 (warnings (vl-warninglist-fix warnings)))
                (b* ((x (vl-port-fix x)))
                    (if (eq (tag x) :vl-interfaceport)
                        (vl-interfaceport-exprsize x ss warnings)
                        (vl-regularport-exprsize x ss warnings))))))
