(defun
 sd-pp-problem-brief-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (sd-problem-p x)))
 (let ((__function__ 'sd-pp-problem-brief))
      (declare (ignorable __function__))
      (b* (((sd-problem x) x)
           (htmlp (vl-ps->htmlp)))
          (vl-ps-seq (sd-pp-problem-header x)
                     (if htmlp

                         (vl-println ""))))))
