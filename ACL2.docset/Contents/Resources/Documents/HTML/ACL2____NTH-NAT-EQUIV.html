Function: nth-nat-equiv

(defun nth-nat-equiv (x y)
       (declare (xargs :non-executable t))
       (declare (xargs :guard t))
       (prog2$ (throw-nonexec-error 'nth-nat-equiv
                                    (list x y))
               (let ((n (nth-nat-equiv-witness x y)))
                    (and (nat-equiv (nth n x) (nth n y))))))

(defthm nth-nat-equiv-necc

Theorem: nth-nat-equiv-is-an-equivalence

(defthm nth-nat-equiv-is-an-equivalence
        (and (booleanp (nth-nat-equiv x y))
             (nth-nat-equiv x x)
             (implies (nth-nat-equiv x y)
                      (nth-nat-equiv y x))
             (implies (and (nth-nat-equiv x y)
                           (nth-nat-equiv y z))
                      (nth-nat-equiv x z)))
        :rule-classes (:equivalence))
