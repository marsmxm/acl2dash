Function: pprint-declor

(defun pprint-declor (declor)
       (declare (xargs :guard (declorp declor)))
       (let ((__function__ 'pprint-declor))
            (declare (ignorable __function__))
            (msg "~s0~@1"
                 (if (declor->pointerp declor) "*" "")
                 (pprint-ident (declor->ident declor)))))

Theorem: msgp-of-pprint-declor

(defthm msgp-of-pprint-declor
        (b* ((part (pprint-declor declor)))
            (msgp part))
        :rule-classes :rewrite)

(defthm pprint-declor-of-declor-fix-declor
        (equal (pprint-declor (declor-fix declor))
