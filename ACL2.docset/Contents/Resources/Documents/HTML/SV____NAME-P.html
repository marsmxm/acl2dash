Function: name-p

(defun name-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'name-p))
            (declare (ignorable __function__))
            (or (stringp x)
                (integerp x)
                (eq x :self)
                (and (consp x)
                     (eq (car x) :anonymous)))))
