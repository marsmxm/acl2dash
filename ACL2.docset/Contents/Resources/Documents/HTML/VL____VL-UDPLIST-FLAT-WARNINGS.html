Gather a flat list of all warnings from a vl-udplist-p.

This is an ordinary defmapappend.

Function: vl-udplist-flat-warnings

(defun vl-udplist-flat-warnings (x)
       (declare (xargs :guard (vl-udplist-p x)))
       (mbe :logic (if (consp x)
                       (append (vl-udp->warnings (car x))
                               (vl-udplist-flat-warnings (cdr x)))
                       nil)
            :exec (reverse (vl-udplist-flat-warnings-exec x nil))))
