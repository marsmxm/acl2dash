Function: positive-cofactor

(defun positive-cofactor (n truth numvars)
       (declare (xargs :guard (and (natp n)
                                   (integerp truth)
                                   (natp numvars))))
       (declare (xargs :guard (< n numvars)))
       (let ((__function__ 'positive-cofactor))
            (declare (ignorable __function__))
            (b* ((mask (logand (var n numvars)

                (logior mask
                        (ash mask (- (ash 1 (lnfix n))))))))
