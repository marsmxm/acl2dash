Function: pseudo-lambdap

(defun pseudo-lambdap (x)
       (declare (xargs :guard t))
       (let ((__function__ 'pseudo-lambdap))
            (declare (ignorable __function__))
            (and (true-listp x)
                 (= (len x) 3)
                 (eq (first x) 'lambda)
                 (symbol-listp (second x))
                 (pseudo-termp (third x)))))

(defthm pseudo-lambdap-of-car-when-pseudo-termp
        (implies (and (pseudo-termp term)
                      (consp term)

Theorem: pseudo-termp-of-cons-when-pseudo-lambdap

(defthm pseudo-termp-of-cons-when-pseudo-lambdap
        (implies (and (pseudo-lambdap lambd)
                      (pseudo-term-listp terms)
                      (equal (len terms)
                             (len (lambda-formals lambd))))
                 (pseudo-termp (cons lambd terms))))
