                             (vl-scopestack-p ss))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-modinstlist-scopesubst))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-modinst-scopesubst (car x) ss)
                    (vl-modinstlist-scopesubst (cdr x) ss))

    nil
