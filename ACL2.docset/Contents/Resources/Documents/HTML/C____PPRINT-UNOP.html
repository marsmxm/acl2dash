Function: pprint-unop

(defun pprint-unop (op)
       (declare (xargs :guard (unopp op)))
       (let ((__function__ 'pprint-unop))
            (declare (ignorable __function__))
            (unop-case op
                       :address "&"
                       :indir "*"
                       :plus "+"
                       :minus "-"
                       :bitnot "~~"
                       :lognot "!")))
