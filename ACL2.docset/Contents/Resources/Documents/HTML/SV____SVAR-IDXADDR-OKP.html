(defun svar-idxaddr-okp (x bound)
       (declare (xargs :guard (and (svar-p x) (natp bound))))
       (let ((__function__ 'svar-idxaddr-okp))
            (declare (ignorable __function__))
            (and (svar-addr-p x)
                 (b* ((addr (svar->address x))
