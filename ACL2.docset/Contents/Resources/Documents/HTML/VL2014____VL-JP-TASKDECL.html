   vl-jp-taskdecl-fn (x ps)
   (declare (xargs :stobjs (ps)))
   (declare (xargs :guard (vl-taskdecl-p x)))
   (let ((__function__ 'vl-jp-taskdecl))
        (declare (ignorable __function__))
        (vl-ps-seq (vl-print "{\"tag\": ")
                   (jp-sym ':vl-taskdecl)

                   (vl-jp-importlist (vl-taskdecl->imports x))
                   (vl-println? ", ")
                   (vl-print-str "\"vardecls\": ")
                   (vl-jp-vardecllist (vl-taskdecl->vardecls x))
                   (vl-println? ", ")
                   (vl-print-str "\"paramdecls\": ")
                   (vl-jp-paramdecllist (vl-taskdecl->paramdecls x))
                   (vl-println? ", ")
                   (vl-print-str "\"body\": ")
                   (vl-jp-stmt (vl-taskdecl->body x))
                   (vl-println? ", ")
                   (vl-print-str "\"atts\": ")
                   (vl-jp-atts (vl-taskdecl->atts x))
