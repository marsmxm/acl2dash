Macro: desc-apt-input-wrapper-name

       'xdoc::ul

          'nil))

            'nil))

       (cons
        'xdoc::p
        (cons
         '"This input may be present only if the @(':wrapper') input is @('t')."
         'nil))

        (cons

          '"In the rest of this documentation page,
      let @('wrapper') be this function."
