Function: vl-make-udp-portdecls

(defun
 vl-make-udp-portdecls
 (ids dir type atts)
 (declare (xargs :guard (and (vl-idtoken-list-p ids)
                             (vl-direction-p dir)
                             (vl-datatype-p type)
                             (vl-atts-p atts))))
 (let ((__function__ 'vl-make-udp-portdecls))
      (declare (ignorable __function__))
      (if (atom ids)
          nil
          (cons (make-vl-portdecl :loc (vl-token->loc (car ids))
                                  :name (vl-idtoken->name (car ids))
                                  :dir dir
                                  :type type
                                  :atts atts)
                (vl-make-udp-portdecls (cdr ids)
                                       dir type atts)))))
