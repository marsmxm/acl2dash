Function: min-upper

(defun min-upper (i k)
       (min (sel-upper-div k (+ 1/2 (/ i 128)))
            (sel-upper-div k (+ 1/2 (/ (1+ i) 128)))))

(defthm sel-limits-check-8

(defun select-digit-d8 (a i)
