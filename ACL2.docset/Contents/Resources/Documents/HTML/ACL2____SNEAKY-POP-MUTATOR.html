(defun sneaky-pop-mutator (stored-vals name)
       (declare (xargs :guard (consp stored-vals)))
       (let ((__function__ 'sneaky-pop-mutator))

            (b* ((old-val (car stored-vals))

                  (cw "; Sneaky-pop of empty ~x0: ~x1~%"
