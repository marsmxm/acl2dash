Function: svstack-lookup

(defun svstack-lookup (k stack)
       (declare (xargs :guard (and (svar-p k) (svstack-p stack))))
       (let ((__function__ 'svstack-lookup))
            (declare (ignorable __function__))
            (if (atom stack)
                nil
                (or (svex-fastlookup k (car stack))
                    (svstack-lookup k (cdr stack))))))
