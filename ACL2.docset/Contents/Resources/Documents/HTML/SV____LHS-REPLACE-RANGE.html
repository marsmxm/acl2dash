Function: lhs-replace-range

(defun
   lhs-replace-range (start w repl x)
   (declare (xargs :guard (and (natp start)
                               (natp w)
                               (lhs-p repl)
                               (lhs-p x))))
   (let ((__function__ 'lhs-replace-range))
        (declare (ignorable __function__))
        (lhs-concat start x
                    (lhs-concat w repl
                                (lhs-rsh (+ (lnfix start) (lnfix w))
                                         x)))))
