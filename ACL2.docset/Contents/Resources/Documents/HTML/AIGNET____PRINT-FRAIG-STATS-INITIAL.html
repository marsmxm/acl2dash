(defun
 print-fraig-stats-initial (fraig-stats)
 (declare (xargs :stobjs (fraig-stats)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'print-fraig-stats-initial))
  (declare (ignorable __function__))
  (cw
   "Fraig initial equiv classes: ~x0     const lits: ~x1    class lits: ~x2~%Lits remaining: ~x3 (~s4)~%"
   (fraig-initial-nclasses fraig-stats)
   (fraig-initial-nconst-lits fraig-stats)
   (fraig-initial-nclass-lits fraig-stats)
   (+ (fraig-initial-nconst-lits fraig-stats)
      (fraig-initial-nclass-lits fraig-stats))
   "100.00%")))
