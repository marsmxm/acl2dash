Function: vl-parse-error-fn

(defun
 vl-parse-error-fn
 (description function tokstream)
 (declare (xargs :stobjs (tokstream)))
 (declare (xargs :guard (and (stringp description)
                             (symbolp function))))
 (let
  ((__function__ 'vl-parse-error))
  (declare (ignorable __function__))
  (b*
   ((tokens (vl-tokstream->tokens))
    (warning
       (make-vl-warning :type :vl-parse-error
                        :msg "Parse error at ~a0: ~s1"
                        :args (list (if (consp tokens)
                                        (vl-token->loc (car tokens))
                                        "EOF")
                                    description)
                        :fn function
                        :fatalp t)))
   (mv warning nil tokstream))))
