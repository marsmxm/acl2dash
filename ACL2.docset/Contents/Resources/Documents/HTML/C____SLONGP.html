Function: slongp

(defun slongp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'slongp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :slong)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (slong-integerp get)))))
