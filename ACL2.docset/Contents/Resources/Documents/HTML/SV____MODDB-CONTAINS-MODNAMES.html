Function: moddb-contains-modnames

(defun moddb-contains-modnames (x moddb)
       (declare (xargs :stobjs (moddb)))
       (declare (xargs :guard (and (modnamelist-p x)
                                   (moddb-basics-ok moddb))))
       (let ((__function__ 'moddb-contains-modnames))
            (declare (ignorable __function__))
            (if (atom x)
                t
                (and (moddb-modname-get-index (car x) moddb)
                     (moddb-contains-modnames (cdr x)
                                              moddb)))))
