Function: expdata-gen-forth-of-terms

(defun
 expdata-gen-forth-of-terms
 (terms arg/res-surjmaps)
 (declare
  (xargs
     :guard (and (pseudo-term-listp terms)
                 (expdata-symbol-surjmap-alistp arg/res-surjmaps))))
 (declare (xargs :guard (= (len terms) (len arg/res-surjmaps))))
 (let
  ((__function__ 'expdata-gen-forth-of-terms))
  (declare (ignorable __function__))
  (b*
   (((when (endp terms)) nil)
    (term (car terms))
    (surjmap (cdar arg/res-surjmaps))
    (forth (expdata-surjmap->forth surjmap))
    (new-term (apply-term* forth term))
    (new-terms (expdata-gen-forth-of-terms (cdr terms)
                                           (cdr arg/res-surjmaps))))
   (cons new-term new-terms))))
