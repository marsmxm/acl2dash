Function: ipasir-add-ternary

(defun
 ipasir-add-ternary (ipasir a b c)
 (declare (xargs :stobjs (ipasir)))
 (declare (xargs :guard (and (litp a) (litp b) (litp c))))
 (declare
     (xargs :guard (and (not (eq (ipasir-get-status ipasir) :undef))
                        (ipasir-empty-new-clause ipasir))))
 (let ((__function__ 'ipasir-add-ternary))
      (declare (ignorable __function__))
      (b* ((ipasir (ipasir-cancel-new-clause ipasir))
           (ipasir (ipasir-add-lit ipasir a))
           (ipasir (ipasir-add-lit ipasir b))
           (ipasir (ipasir-add-lit ipasir c)))
          (ipasir-finalize-clause ipasir))))
