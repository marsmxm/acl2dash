Theorem: logxor*

(defthm logxor*
        (implies (and (force (integerp i))
                      (force (integerp j)))
                 (equal (logxor i j)
                        (logcons (b-xor (logcar i) (logcar j))
                                 (logxor (logcdr i) (logcdr j)))))
        :rule-classes
        ((:definition :clique (binary-logxor)
                      :controller-alist ((binary-logxor t t)))))
