Function: explode-list

(defun explode-list (x)
       (declare (xargs :guard (string-listp x)))
       (let ((__function__ 'explode-list))
            (declare (ignorable __function__))
            (if (atom x)
                nil
                (cons (explode (car x))
                      (explode-list (cdr x))))))

(defthm explode-list-of-cons
