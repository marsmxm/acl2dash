Function: bfr-eval-list

(defun bfr-eval-list (x env)
       (declare (xargs :guard t))
       (let ((__function__ 'bfr-eval-list))
            (declare (ignorable __function__))
            (if (atom x)
                nil
                (cons (bfr-eval (car x) env)
                      (bfr-eval-list (cdr x) env)))))
