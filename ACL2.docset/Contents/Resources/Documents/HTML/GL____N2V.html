Function: n2v

(defun n2v (n)
       (declare (xargs :guard t))
       (let ((__function__ 'n2v))
            (declare (ignorable __function__))
            (if (eql (nfix n) 0)
                nil
                (bfr-ucons (equal 1 (logcar n))
                           (n2v (logcdr n))))))
