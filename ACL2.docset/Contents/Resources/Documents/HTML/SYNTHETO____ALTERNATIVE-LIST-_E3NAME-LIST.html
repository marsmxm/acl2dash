Function: alternative-list->name-list

(defun
 alternative-list->name-list (x)
 (declare (xargs :guard (alternative-listp x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'alternative-list->name-list))
  (declare (ignorable __function__))
  (mbe :logic (if (consp x)
                  (cons (alternative->name (car x))
                        (alternative-list->name-list (cdr x)))
                  nil)
       :exec
       (if (atom x)
           nil
           (acl2::with-local-nrev
                (alternative-list->name-list-nrev x acl2::nrev))))))
