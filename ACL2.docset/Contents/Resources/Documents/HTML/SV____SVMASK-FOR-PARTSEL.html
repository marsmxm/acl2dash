Function: svmask-for-partsel

(defun
 svmask-for-partsel (mask args)
 (declare (xargs :guard (and (4vmask-p mask)
                             (svexlist-p args))))
 (let
  ((__function__ 'svmask-for-partsel))
  (declare (ignorable __function__))
  (b*
   (((svex-nths lsb width in) args)
    (mask (4vmask-fix mask)))
   (b*
    (((when (4vmask-empty mask))
      (list 0 0 0))
     (widthval (svex-s4xeval width))
     (lsbval (svex-s4xeval lsb))
     ((unless (s4vec-2vec-p widthval))
      (if
        (s4vec-2vec-p lsbval)
        (list -1 -1
              (sparseint-ash mask
                             (sparseint-val (s4vec->upper lsbval))))
        (list -1 -1 -1)))
     (widthval (s4vec->upper widthval))
     ((when (sparseint-< widthval 0))
      (list 0 -1 0))
     (widthval (sparseint-val widthval))
     ((unless (s4vec-2vec-p lsbval))
      (if (sparseint-equal 0
                           (sparseint-concatenate widthval mask 0))
          (list -1 -1 0)
          (list -1 -1 -1)))
     (lsbval (s4vec->upper lsbval)))
    (list -1 -1
          (sparseint-ash (sparseint-concatenate widthval mask 0)
                         (sparseint-val lsbval)))))))
