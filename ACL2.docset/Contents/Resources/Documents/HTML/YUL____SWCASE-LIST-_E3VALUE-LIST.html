Function: swcase-list->value-list

(defun
 swcase-list->value-list (x)
 (declare (xargs :guard (swcase-listp x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'swcase-list->value-list))
  (declare (ignorable __function__))
  (mbe
     :logic (if (consp x)
                (cons (swcase->value (car x))
                      (swcase-list->value-list (cdr x)))
                nil)
     :exec (if (atom x)
               nil
               (acl2::with-local-nrev
                    (swcase-list->value-list-nrev x acl2::nrev))))))
