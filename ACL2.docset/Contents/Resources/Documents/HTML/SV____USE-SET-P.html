(defun use-set-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'use-set-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (b* ((uses (std::prod-car x))
                      (sets (std::prod-cdr x)))
                     (and (rangemap-p uses)
                          (rangemap-p sets))))))
