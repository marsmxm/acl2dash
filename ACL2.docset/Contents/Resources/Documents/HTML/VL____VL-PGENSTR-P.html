Function: vl-pgenstr-p

(defun vl-pgenstr-p (prefix str)
       (declare (xargs :guard (and (stringp prefix) (stringp str))))
       (let ((__function__ 'vl-pgenstr-p))
            (declare (ignorable __function__))
            (b* ((plen (length prefix))
                 (slen (length str)))
                (and (< (+ 1 plen) slen)
                     (str::strprefixp prefix str)
                     (eql (char str plen) #\_)
                     (str::dec-digit-string-p-aux str (+ 1 plen)
                                                  slen)))))

(defthm vl-pgenstr-p-of-vl-pgenstr
        (implies (and (force (stringp prefix))
                      (force (natp n)))
