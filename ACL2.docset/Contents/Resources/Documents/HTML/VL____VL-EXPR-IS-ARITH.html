Function: vl-expr-is-arith

(defun
 vl-expr-is-arith (x)
 (declare (xargs :guard (vl-expr-p x)))
 (let
  ((__function__ 'vl-expr-is-arith))
  (declare (ignorable __function__))
  (b*
   ((x (vl-expr-fix x)))
   (vl-expr-case
      x
      :vl-unary (and (member x.op '(:vl-unary-plus :vl-unary-minus))
                     (vl-expr-is-arith x.arg))
      :vl-binary
      (member x.op
              '(:vl-binary-plus :vl-binary-minus
                                :vl-binary-times :vl-binary-div))
      :otherwise nil))))
