(defun
   check-alternative-list
   (alternatives ctxt)
   (declare (xargs :guard (and (alternative-listp alternatives)
                               (contextp ctxt))))

        (xargs :guard (and (null (context->functions ctxt))
                           (omap::empty (context->variables ctxt))

   (let ((__function__ 'check-alternative-list))
