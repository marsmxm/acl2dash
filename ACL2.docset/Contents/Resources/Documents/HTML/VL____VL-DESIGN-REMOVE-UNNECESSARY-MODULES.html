(defun
  vl-design-remove-unnecessary-modules

                              (vl-design-p design))))
  (let ((__function__ 'vl-design-remove-unnecessary-modules))
       (declare (ignorable __function__))
       (b* ((design (vl-design-fix design))
            ((unless keep) design)
