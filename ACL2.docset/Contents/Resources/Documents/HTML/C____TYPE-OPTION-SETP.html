Function: type-option-setp

(defun type-option-setp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (type-optionp (car x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (acl2::fast-<< (car x) (cadr x))
                         (type-option-setp (cdr x)))))))
