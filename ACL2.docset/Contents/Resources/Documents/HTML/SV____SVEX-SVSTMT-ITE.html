Function: svex-svstmt-ite

(defun
 svex-svstmt-ite (test then else)
 (declare (xargs :guard (and (svex-p test)
                             (svex-p then)
                             (svex-p else))))
 (let
  ((__function__ 'svex-svstmt-ite))
  (declare (ignorable __function__))
  (b*
   ((test (svex-fix test))
    (then (svex-fix then))
    (else (svex-fix else)))
   (or
      (svex-case test
                 :quote (if (eql (4vec-reduction-or test.val) -1)
                            then
                            (if (eql (4vec-reduction-or test.val) 0)
                                else nil))
                 :otherwise nil)
      (and (hons-equal then else) then)
      (svex-call '?*
                 (list test then else))))))
