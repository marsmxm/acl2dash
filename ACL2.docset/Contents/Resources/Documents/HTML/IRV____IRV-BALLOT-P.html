Function: irv-ballot-p

(defun irv-ballot-p (xs)
       (declare (xargs :guard t))
       (let ((__function__ 'irv-ballot-p))
            (declare (ignorable __function__))
            (if (consp xs)
                (b* ((one (car xs)) (rest (cdr xs)))
                    (and (nat-listp one)
                         one (no-duplicatesp-equal one)
                         (irv-ballot-p rest)))
                (equal xs nil))))
