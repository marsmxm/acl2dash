 (declare (xargs :guard t))

   ((__function__ 'vl-bitlist->charlist-exec))

Function: vl-bitlist->charlist

(defun
 vl-bitlist->charlist (x)
 (declare (xargs :guard (vl-bitlist-p x)))
 (declare (xargs :guard t))
 (let
   ((__function__ 'vl-bitlist->charlist))
   (declare (ignorable __function__))
   (mbe :logic (if (consp x)
                   (cons (vl-bit->char (car x))
                         (vl-bitlist->charlist (cdr x)))
                   nil)
        :exec
        (if (atom x)
            nil
            (with-local-nrev (vl-bitlist->charlist-nrev x nrev))))))
