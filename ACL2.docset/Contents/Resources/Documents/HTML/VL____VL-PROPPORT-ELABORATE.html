 vl-propport-elaborate-fn
 (x elabindex reclimit config)
 (declare (xargs :stobjs (elabindex)))
 (declare (xargs :guard (and (vl-propport-p x)
                             (natp reclimit)
                             (vl-simpconfig-p config))))
 (let
  ((__function__ 'vl-propport-elaborate))
  (declare (ignorable __function__))
  (b*
   (((vl-propport x) (vl-propport-fix x)))
   (b*
     (((mv ok warnings type elabindex)
       (vl-datatype-elaborate-fn x.type elabindex reclimit config))
      ((mv ok1 warnings1 arg elabindex)
       (vl-propactual-elaborate-fn x.arg elabindex reclimit config))
      (ok (and ok1 ok))
      (warnings (append-without-guard warnings1 warnings)))
     (mv ok warnings
         (change-vl-propport x
                             :type type
                             :arg arg)
         elabindex)))))
