Function: a4vec-lsh

(defun
 a4vec-lsh (amt x mask)
 (declare (xargs :guard (and (a4vec-p amt)
                             (a4vec-p x)
                             (4vmask-p mask))))
 (let
  ((__function__ 'a4vec-lsh))
  (declare (ignorable __function__))
  (b*
   (((a4vec amt))
    ((a4vec x))
    (mask (4vmask-fix mask)))
   (a4vec-ite
    (a2vec-p amt)
    (b*
     ((shamt amt.upper)
      (sign (aig-sign-s shamt))
      ((mv upper-left lower-left)
       (b*
         (((when (eq sign t)) (mv nil nil))
          (lsh-amt (if (sparseint-< mask 0)
                       shamt
                       (aig-logcollapse-ns
                            (integer-length (sparseint-length mask))
                            shamt))))
         (mv (aig-ash-ss 1 x.upper lsh-amt)
             (aig-ash-ss 1 x.lower lsh-amt))))
      ((mv upper-right lower-right)
       (if (not sign)
           (mv nil nil)
           (mv (aig-right-shift-ss 1 x.upper shamt)
               (aig-right-shift-ss 1 x.lower shamt))))
      (upper (aig-ite-bss-fn sign upper-right upper-left))
      (lower (aig-ite-bss-fn sign lower-right lower-left)))
     (a4vec upper lower))
    (a4vec-x)))))
