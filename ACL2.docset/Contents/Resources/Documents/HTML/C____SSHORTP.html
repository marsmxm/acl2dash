Function: sshortp

(defun sshortp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'sshortp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :sshort)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (sshort-integerp get)))))
