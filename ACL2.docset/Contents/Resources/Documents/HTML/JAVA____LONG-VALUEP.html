Function: long-valuep

(defun long-valuep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'long-valuep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :long)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((int (std::da-nth 0 (cdr x))))
                     (sbyte64p int)))))
