Indicates whether this is a begin/end, fork/join, fork/join_any, or fork/join_none statement.

This is an ordinary defenum.

Function: vl-blocktype-p

(defun vl-blocktype-p (x)
       (declare (xargs :guard t))
       (or (eq x ':vl-beginend)
           (eq x ':vl-forkjoin)
           (eq x ':vl-forkjoinany)
           (eq x ':vl-forkjoinnone)))

(defthm type-when-vl-blocktype-p
        (implies (vl-blocktype-p x)
                 (if (symbolp x)
                     (if (not (equal x 't))
                         (not (equal x 'nil))
                         'nil)
                     'nil))
