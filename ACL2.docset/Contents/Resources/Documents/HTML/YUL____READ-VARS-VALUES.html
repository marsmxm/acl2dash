Function: read-vars-values

(defun read-vars-values (vars cstate)
       (declare (xargs :guard (and (identifier-listp vars)
                                   (cstatep cstate))))
       (let ((__function__ 'read-vars-values))
            (declare (ignorable __function__))
            (b* (((when (endp vars)) nil)
                 ((ok val)
                  (read-var-value (car vars) cstate))
                 ((ok vals)
                  (read-vars-values (cdr vars) cstate)))
                (cons val vals))))
