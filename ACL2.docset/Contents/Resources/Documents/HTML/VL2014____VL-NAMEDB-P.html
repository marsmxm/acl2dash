Function: vl-namedb-p

(defun
     vl-namedb-p (x)
     (declare (xargs :guard t))
     (let ((__function__ 'vl-namedb-p))
          (declare (ignorable __function__))
          (and (consp x)
               (eq (car x) :vl-namedb)
               (std::prod-consp (cdr x))
               (std::prod-consp (std::prod-cdr (cdr x)))
               (b* ((names (std::prod-car (cdr x)))
                    (pmap (std::prod-car (std::prod-cdr (cdr x))))
                    (pset (std::prod-cdr (std::prod-cdr (cdr x)))))
                   (and (vl-namedb-nameset-p names)
                        (vl-namedb-prefixmap-p pmap)
                        (vl-namedb-nameset-p pset)
                        (vl-namedb-pmap-okp names pmap)
                        (vl-namedb-pset-okp pmap pset))))))
