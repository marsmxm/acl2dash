Function: vl-scope-find-portdecl

(defun
 vl-scope-find-portdecl (name scope)
 (declare (xargs :guard (and (stringp name)
                             (vl-scope-p scope))))
 (let
  ((__function__ 'vl-scope-find-portdecl))
  (declare (ignorable __function__))
  (b*
   ((scope (vl-scope-fix scope)))
   (case
       (tag scope)
       (:vl-interface (vl-interface-scope-find-portdecl name scope))
       (:vl-module (vl-module-scope-find-portdecl name scope))
       (:vl-genblob (vl-genblob-scope-find-portdecl name scope))
       (otherwise nil)))))
