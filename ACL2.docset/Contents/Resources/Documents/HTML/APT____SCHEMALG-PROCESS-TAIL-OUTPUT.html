 (let
  ((__function__ 'schemalg-process-tail-output))
  (declare (ignorable __function__))
  (b*
   ((schemas-allowed (list :divconq-oset-0-1))
    ((when (and tail-output?
                (not (member-eq schema schemas-allowed))))
     (er-soft+

           tail-output))

          y "The :TAIL-OUTPUT input" t nil))

     (ensure-value-is-not-in-list$

               x-x1...xn old)
          "The :TAIL-OUTPUT input" t nil)))
