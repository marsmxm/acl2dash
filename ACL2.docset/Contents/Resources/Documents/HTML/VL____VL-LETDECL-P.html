Function: vl-letdecl-p

(defun
 vl-letdecl-p (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-letdecl-p))
  (declare (ignorable __function__))
  (and
   (consp x)
   (eq (car x) :vl-letdecl)
   (mbe
      :logic (and (alistp (cdr x))
                  (equal (strip-cars (cdr x))
                         '(name portdecls expr atts loc)))
      :exec (fty::alist-with-carsp (cdr x)
                                   '(name portdecls expr atts loc)))
   (b* ((name (cdr (std::da-nth 0 (cdr x))))
        (portdecls (cdr (std::da-nth 1 (cdr x))))
        (expr (cdr (std::da-nth 2 (cdr x))))
        (atts (cdr (std::da-nth 3 (cdr x))))
        (loc (cdr (std::da-nth 4 (cdr x)))))
       (and (stringp name)
            (vl-portdecllist-p portdecls)
            (vl-expr-p expr)
            (vl-atts-p atts)
            (vl-location-p loc))))))
