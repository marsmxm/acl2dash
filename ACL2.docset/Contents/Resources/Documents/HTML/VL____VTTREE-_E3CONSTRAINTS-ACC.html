Function: vttree->constraints-acc

(defun
 vttree->constraints-acc (x context acc)
 (declare (xargs :guard (and (vttree-p x)
                             (sv::constraintlist-p acc))))
 (let
  ((__function__ 'vttree->constraints-acc))
  (declare (ignorable __function__))
  (vttree-case
      x
      :none (sv::constraintlist-fix acc)
      :constraints
      (revappend-without-guard
           (if context
               (constraintlist-add-ctx x.constraints context)
               x.constraints)
           (sv::constraintlist-fix acc))
      :warnings (sv::constraintlist-fix acc)
      :context (vttree->constraints-acc x.subtree (or x.ctx context)
                                        acc)
      :branch (vttree->constraints-acc
                   x.right context
                   (vttree->constraints-acc x.left context acc)))))
