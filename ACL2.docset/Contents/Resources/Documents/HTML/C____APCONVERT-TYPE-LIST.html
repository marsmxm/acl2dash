Function: apconvert-type-list

(defun
 apconvert-type-list (x)
 (declare (xargs :guard (type-listp x)))
 (let
    ((__function__ 'apconvert-type-list))
    (declare (ignorable __function__))
    (mbe :logic (if (consp x)
                    (cons (apconvert-type (car x))
                          (apconvert-type-list (cdr x)))
                    nil)
         :exec (if (atom x)
                   nil
                   (acl2::with-local-nrev
                        (apconvert-type-list-nrev x acl2::nrev))))))
