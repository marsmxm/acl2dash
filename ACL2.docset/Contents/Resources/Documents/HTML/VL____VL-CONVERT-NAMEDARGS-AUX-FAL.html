(defun
 vl-convert-namedargs-aux-fal
 (args alist ports)
 (declare
   (xargs :guard (and (vl-namedarglist-p args)
                      (vl-portlist-p ports)
                      (equal alist (vl-make-namedarg-alist args)))))
 (declare
      (xargs :guard (not (member nil (vl-portlist->names ports)))))
 (let
  ((__function__ 'vl-convert-namedargs-aux-fal))
  (declare (ignorable __function__))
  (mbe
   :logic (vl-convert-namedargs-aux args ports)
   :exec
   (b*
    (((when (atom ports)) nil)
     (namedarg (vl-fast-find-namedarg (vl-port->name (car ports))
                                      args alist))
     (plainarg
        (if namedarg
            (make-vl-plainarg :expr (vl-namedarg->expr namedarg)
                              :atts (vl-namedarg->atts namedarg))
            (make-vl-plainarg :expr nil
                              :atts '(("VL_MISSING_CONNECTION"))))))
    (cons plainarg
          (vl-convert-namedargs-aux-fal args alist (cdr ports)))))))
