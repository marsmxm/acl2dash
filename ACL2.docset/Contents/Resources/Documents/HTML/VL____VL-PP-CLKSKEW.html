Function: vl-pp-clkskew-fn

(defun
 vl-pp-clkskew-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (vl-clkskew-p x)))
 (let
  ((__function__ 'vl-pp-clkskew))
  (declare (ignorable __function__))
  (b*
   (((vl-clkskew x)))
   (vl-ps-seq
       (if (eq x.edge :vl-noedge)
           ps
           (vl-ps-span "vl_key"
                       (case x.edge
                             (:vl-posedge (vl-print "posedge"))
                             (:vl-negedge (vl-print "negedge"))
                             (:vl-edge (vl-print "edge"))
                             (otherwise (progn$ (impossible) ps)))))
       (if (and (not (eq x.edge :vl-noedge))
                x.delay)
           (vl-print " ")
           ps)
       (if x.delay (vl-pp-expr x.delay) ps)))))
