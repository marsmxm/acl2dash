Theorem: input-decomposition-of-parse-any

(defthm input-decomposition-of-parse-any
        (b* (((mv error? nat? rest-input)
              (parse-any input)))
            (implies (not error?)
                     (equal (cons nat? rest-input)
                            (nat-list-fix input)))))
