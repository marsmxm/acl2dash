Function: pinst-kind$inline

(defun
     pinst-kind$inline (x)
     (declare (xargs :guard (pinst-p x)))
     (let ((acl2::__function__ 'pinst-kind))
          (declare (ignorable acl2::__function__))
          (mbe :logic (cond ((or (atom x) (eq (car x) :flat)) :flat)
                            ((eq (car x) :keyline) :keyline)
                            ((eq (car x) :dot) :dot)
                            ((eq (car x) :quote) :quote)
                            ((eq (car x) :wide) :wide)

               :exec (car x))))
