Function: nth-set-bit-pos

(defun nth-set-bit-pos (n x)
       (declare (xargs :guard (and (natp n) (integerp x))))
       (let ((__function__ 'nth-set-bit-pos))
            (declare (ignorable __function__))
            (if (zip x)
                nil
                (b* ((bit (logcar x))
                     ((when (and (zp n) (eql bit 1))) 0)
                     (rest (nth-set-bit-pos (- (lnfix n) bit)
                                            (logcdr x))))
                    (and rest (+ 1 rest))))))
