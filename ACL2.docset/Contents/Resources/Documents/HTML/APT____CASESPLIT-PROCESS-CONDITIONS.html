(defun casesplit-process-conditions-aux
       (conditions pos old$ verify-guards$ ctx state)
       (declare (xargs :stobjs (state)))

       (let ((__function__ 'casesplit-process-conditions-aux))
            (declare (ignorable __function__))
            (cond ((endp conditions) (value nil))
                  (t (b* (((er cond$)
                           (casesplit-process-condition
                                (car conditions)
                                pos old$ verify-guards$ ctx state))

                                old$ verify-guards$ ctx state)))
