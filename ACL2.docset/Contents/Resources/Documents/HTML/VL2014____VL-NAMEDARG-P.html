Function: vl-namedarg-p

(defun
 vl-namedarg-p (x)
 (declare (xargs :guard t))
 (let ((__function__ 'vl-namedarg-p))
      (declare (ignorable __function__))
      (and (consp x)
           (eq (car x) :vl-namedarg)
           (std::prod-consp (cdr x))
           (std::prod-consp (std::prod-car (cdr x)))
           (std::prod-consp (std::prod-cdr (cdr x)))
           (b* ((name (std::prod-car (std::prod-car (cdr x))))
                (expr (std::prod-cdr (std::prod-car (cdr x))))
                (nameonly-p (std::prod-car (std::prod-cdr (cdr x))))
                (atts (std::prod-cdr (std::prod-cdr (cdr x)))))
               (and (stringp name)
                    (vl-maybe-expr-p expr)
                    (booleanp nameonly-p)
                    (vl-atts-p atts))))))
