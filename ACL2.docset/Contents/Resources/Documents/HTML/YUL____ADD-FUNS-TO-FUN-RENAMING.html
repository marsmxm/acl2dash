Function: add-funs-to-fun-renaming

(defun
 add-funs-to-fun-renaming (old new ren)
 (declare (xargs :guard (and (identifier-listp old)
                             (identifier-listp new)
                             (renamingp ren))))
 (let
  ((__function__ 'add-funs-to-fun-renaming))
  (declare (ignorable __function__))
  (b*
   (((when (endp old))
     (if
        (endp new)
        (renaming-fix ren)
        (err (list :mismatch-extra-new (identifier-list-fix new)))))
    ((when (endp new))
     (err (list :mismatch-extra-old (identifier-list-fix old))))
    ((ok ren)
     (add-fun-to-fun-renaming (car old)
                              (car new)
                              ren)))
   (add-funs-to-fun-renaming (cdr old)
                             (cdr new)
                             ren))))
