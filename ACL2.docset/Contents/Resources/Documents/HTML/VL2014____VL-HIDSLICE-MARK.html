Function: vl-hidslice-mark

(defun
 vl-hidslice-mark
 (mtype hid left right ss st ctx)
 (declare (xargs :guard (and (member mtype '(:used :set))
                             (vl-expr-p hid)
                             (vl-expr-p left)
                             (vl-expr-p right)
                             (vl-scopestack-p ss)
                             (vl-lucidstate-p st)
                             (vl-context1-p ctx))))
 (declare (xargs :guard (vl-hidexpr-p hid)))
 (let ((__function__ 'vl-hidslice-mark))
      (declare (ignorable __function__))
      (b* (((mv err trace tail)
            (vl-follow-hidexpr hid ss ctx))
           ((when err)
            (b* (((vl-lucidstate st)))
                (change-vl-lucidstate st
                                      :warnings st.warnings)))
           ((cons (vl-hidstep step) rest) trace)
           (key (make-vl-lucidkey
                     :item step.item
                     :scopestack (vl-normalize-scopestack step.ss)))
           (occ (if (vl-hidexpr->endp tail)
                    (make-vl-lucidocc-slice :left left
                                            :right right
                                            :ctx ctx
                                            :ss ss)
                    (make-vl-lucidocc-tail :ctx ctx
                                           :ss ss)))
           (st (vl-hidtrace-mark-interfaces mtype rest ss st ctx))
           (st (vl-lucidstate-mark mtype key occ st ctx)))
          st)))
