Function: vl-streaming-concat-trunc/extend

(defun
  vl-streaming-concat-trunc/extend
  (bitstream target-size concat-size)
  (declare (xargs :guard (and (sv::svex-p bitstream)
                              (natp target-size)
                              (natp concat-size))))
  (let ((__function__ 'vl-streaming-concat-trunc/extend))
       (declare (ignorable __function__))
       (b* ((target-size (lnfix target-size))
            (concat-size (lnfix concat-size)))
           (cond ((> target-size concat-size)
                  (sv::svcall sv::concat
                              (svex-int (- target-size concat-size))
                              (svex-int 0)
                              bitstream))
                 ((< target-size concat-size)
                  (sv::svex-rsh (- concat-size target-size)
                                bitstream))
                 (t (sv::svex-fix bitstream))))))
