Function: funenv-dead

(defun funenv-dead (funenv)
       (declare (xargs :guard (funenvp funenv)))
       (let ((__function__ 'funenv-dead))
            (declare (ignorable __function__))
            (cond ((endp funenv) nil)
                  (t (cons (funscope-dead (car funenv))
                           (funenv-dead (cdr funenv)))))))
