Function: type-setp

(defun type-setp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (typep (car x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (acl2::fast-<< (car x) (cadr x))
                         (type-setp (cdr x)))))))
