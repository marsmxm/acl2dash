Function: vl-scope-find-class

(defun
 vl-scope-find-class (name scope)
 (declare (xargs :guard (and (stringp name)
                             (vl-scope-p scope))))
 (let ((__function__ 'vl-scope-find-class))
      (declare (ignorable __function__))
      (b* ((scope (vl-scope-fix scope)))
          (case (tag scope)
                (:vl-design (vl-design-scope-find-class name scope))
                (otherwise nil)))))
