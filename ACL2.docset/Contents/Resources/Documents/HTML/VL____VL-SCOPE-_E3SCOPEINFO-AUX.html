(defun
 vl-scope->scopeinfo-aux (scope design)

                             (vl-maybe-design-p design))))
 (let
  ((__function__ 'vl-scope->scopeinfo-aux))
  (declare (ignorable __function__))
  (b*
   ((scope (vl-scope-fix scope)))
   (case
    (tag scope)
