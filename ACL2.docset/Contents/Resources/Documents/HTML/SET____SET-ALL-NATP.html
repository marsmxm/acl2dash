Function: set-all-natp

(defun set-all-natp (set)
       (declare (xargs :guard (setp set)))
       (let ((acl2::__function__ 'set-all-natp))
            (declare (ignorable acl2::__function__))
            (or (empty set)
                (and (natp (head set))
                     (set-all-natp (tail set))))))
