(defun
 atc-gen-print-result-aux (events)
 (declare (xargs :guard (pseudo-event-form-listp events)))
 (let
  ((__function__ 'atc-gen-print-result-aux))
  (declare (ignorable __function__))
  (cond
   ((endp events) nil)
   (t (cons (cons 'cw-event
                  (cons '"~%~x0~|"
                        (cons (cons 'quote (cons (car events) 'nil))
                              'nil)))
            (atc-gen-print-result-aux (cdr events)))))))

Function: atc-gen-print-result

(defun atc-gen-print-result
       (events output-file)
       (declare (xargs :guard (and (pseudo-event-form-listp events)
                                   (stringp output-file))))
       (let ((__function__ 'atc-gen-print-result))
            (declare (ignorable __function__))
            (append (atc-gen-print-result-aux events)
                    (list (cons 'cw-event
                                (cons '"~%File ~s0.~%"
                                      (cons output-file 'nil)))))))
