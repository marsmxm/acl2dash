Function: package-namep

(defun package-namep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'package-namep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :package-name)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((identifiers (std::da-nth 0 (cdr x))))
                     (identifier-listp identifiers)))))
