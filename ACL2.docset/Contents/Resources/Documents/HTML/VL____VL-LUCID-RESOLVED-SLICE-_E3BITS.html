 (declare (xargs :guard (and (equal (vl-lucidocc-kind x) :slice)
                             (vl-lucid-resolved-slice-p x))))
 (let
  ((__function__ 'vl-lucid-resolved-slice->bits))
  (declare (ignorable __function__))
  (b*
   (((vl-lucidocc-slice x))
