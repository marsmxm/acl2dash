Function: 4vs-iff-lists

(defun 4vs-iff-lists (x y)
       (declare (xargs :guard (equal (len x) (len y))))
       (if (atom x)
           nil
           (cons (4vs-iff (car x) (car y))
                 (4vs-iff-lists (cdr x) (cdr y)))))
