Theorem: tree-fix-when-nonleaf-norulename-nobranches

(defthm tree-fix-when-nonleaf-norulename-nobranches
        (implies (and (equal (tree-kind tree) :nonleaf)
                      (not (tree-nonleaf->rulename? tree))
                      (not (consp (tree-nonleaf->branches tree))))
                 (equal (tree-fix tree)
                        '(:nonleaf nil nil))))
