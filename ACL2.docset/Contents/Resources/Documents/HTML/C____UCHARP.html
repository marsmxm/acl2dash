Function: ucharp

(defun ucharp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'ucharp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :uchar)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (uchar-integerp get)))))
