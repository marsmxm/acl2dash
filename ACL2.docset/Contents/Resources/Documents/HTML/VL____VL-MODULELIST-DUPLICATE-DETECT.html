Function: vl-modulelist-duplicate-detect

(defun
 vl-modulelist-duplicate-detect (x ss)
 (declare (xargs :guard (and (vl-modulelist-p x)
                             (vl-scopestack-p ss))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-modulelist-duplicate-detect))
  (declare (ignorable __function__))
  (mbe :logic (if (consp x)
                  (cons (vl-module-duplicate-detect (car x) ss)
                        (vl-modulelist-duplicate-detect (cdr x)
                                                        ss))
                  nil)
       :exec
       (if (atom x)
           nil
           (with-local-nrev
                (vl-modulelist-duplicate-detect-nrev x ss nrev))))))
