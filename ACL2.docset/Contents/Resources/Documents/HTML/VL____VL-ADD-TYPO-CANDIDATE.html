Function: vl-add-typo-candidate

(defun vl-add-typo-candidate (ss varname typos)
       (declare (xargs :guard (and (vl-scopestack-p ss)
                                   (stringp varname)
                                   (vl-typocandidates-p typos))))
       (let ((__function__ 'vl-add-typo-candidate))
            (declare (ignorable __function__))
            (b* ((ss (vl-scopestack-fix ss))
                 (varname (string-fix varname))
                 (typos (vl-typocandidates-fix typos))
                 (old (cdr (hons-get ss typos))))
                (hons-acons ss (cons varname old)
                            typos))))
