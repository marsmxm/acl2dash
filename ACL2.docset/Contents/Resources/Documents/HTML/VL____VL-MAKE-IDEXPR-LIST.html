(vl-make-idexpr-list x) maps vl-idexpr across a list.

Signature
(vl-make-idexpr-list x) → exprs
Arguments
x — Guard (string-listp x).
Returns
exprs — Type (and (vl-exprlist-p exprs) (vl-idexprlist-p exprs)).

Each expression is given the specified finalwidth and finaltype.

(defun
  vl-make-idexpr-list-exec (x acc)
  (declare (xargs :guard (string-listp x)))
  (declare (xargs :guard t))
  (let ((__function__ 'vl-make-idexpr-list-exec))
       (declare (ignorable __function__))
       (if (consp x)

Function: vl-make-idexpr-list

(defun
 vl-make-idexpr-list (x)
 (declare (xargs :guard (string-listp x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-make-idexpr-list))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-idexpr (car x))
                    (vl-make-idexpr-list (cdr x)))
              nil)
   :exec (if (atom x)
             nil
             (with-local-nrev (vl-make-idexpr-list-nrev x nrev))))))
