Function: branch-list->condition-list

(defun
 branch-list->condition-list (x)
 (declare (xargs :guard (branch-listp x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'branch-list->condition-list))
  (declare (ignorable __function__))
  (mbe :logic (if (consp x)
                  (cons (branch->condition (car x))
                        (branch-list->condition-list (cdr x)))
                  nil)
       :exec
       (if (atom x)
           nil
           (acl2::with-local-nrev
                (branch-list->condition-list-nrev x acl2::nrev))))))
