Function: all-valid-key-paths-p

(defun all-valid-key-paths-p (paths addresses)
       (declare (xargs :guard (and (bip32-path-setp paths)
                                   (natp addresses))))
       (or (empty paths)
           (and (valid-key-path-p (head paths)
                                  addresses)
                (all-valid-key-paths-p (tail paths)
                                       addresses))))

Theorem: booleanp-of-all-valid-key-paths-p

(defthm booleanp-of-all-valid-key-paths-p
        (b* ((yes/no (all-valid-key-paths-p paths addresses)))
            (booleanp yes/no))
        :rule-classes :rewrite)

Theorem: all-valid-key-paths-p-of-insert

(defthm all-valid-key-paths-p-of-insert
        (equal (all-valid-key-paths-p (insert path paths)
                                      addresses)
               (and (valid-key-path-p path addresses)
                    (all-valid-key-paths-p paths addresses))))
