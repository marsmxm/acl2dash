(defun vl-jp-basictype-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-basictype-p x)))
       (let ((__function__ 'vl-jp-basictype))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-basictype)
                       (vl-print ", ")
                       (vl-print-str "\"kind\": ")
                       (jp-basictypekind (vl-basictype->kind x))
                       (vl-println? "}"))))
