(defun vl-range-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-range-p))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :vl-range)
                 (std::prod-consp (cdr x))
                 (b* ((msb (std::prod-car (cdr x)))
