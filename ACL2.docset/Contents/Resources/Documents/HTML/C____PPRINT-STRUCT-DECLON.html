Function: pprint-struct-declon

(defun pprint-struct-declon (member level)
       (declare (xargs :guard (and (struct-declonp member)
                                   (natp level))))
       (let ((__function__ 'pprint-struct-declon))
            (declare (ignorable __function__))
            (b* (((struct-declon member) member))
                (pprint-line (msg "~@0 ~@1;"
                                  (pprint-tyspecseq member.type)
                                  (pprint-declor member.declor))
                             (lnfix level)))))
