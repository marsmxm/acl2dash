Valid types for expressions.

Each expression should be either :vl-signed or :vl-unsigned. We may eventually expand this to include other types, such as real and string.

This is an ordinary defenum.

Function: vl-exprtype-p

(defun vl-exprtype-p (x)
       (declare (xargs :guard t))
       (or (eq x ':vl-signed)
           (eq x ':vl-unsigned)))

Theorem: type-when-vl-exprtype-p

(defthm type-when-vl-exprtype-p
        (implies (vl-exprtype-p x)
                 (if (symbolp x)
                     (if (not (equal x 't))
                         (not (equal x 'nil))
                         'nil)
                     'nil))
        :rule-classes :compound-recognizer)
