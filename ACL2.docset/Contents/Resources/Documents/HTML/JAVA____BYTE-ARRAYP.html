Function: byte-arrayp

(defun byte-arrayp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'byte-arrayp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :byte-array)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((components (std::da-nth 0 (cdr x))))
                     (and (byte-value-listp components)
                          (< (len components) (expt 2 31)))))))
