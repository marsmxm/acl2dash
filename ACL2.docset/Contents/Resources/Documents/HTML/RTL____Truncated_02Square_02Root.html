Function: rtz-sqrt

(defun rtz-sqrt (x n)
       (declare (xargs :guard (and (real/rationalp x) (natp n))))
       (if (zp n)
           0
           (let* ((lower (rtz-sqrt x (1- n)))
                  (upper (+ lower (expt 2 (- n)))))
                 (if (<= (* upper upper) x)
                     upper lower))))

Theorem: rtz-sqrt-bounds

(defthm rtz-sqrt-bounds
        (implies (and (rationalp x)
                      (<= 1/4 x)
                      (not (zp n)))
                 (and (<= 1/2 (rtz-sqrt x n))
                      (<= (rtz-sqrt x n)
                          (- 1 (expt 2 (- n))))))
        :rule-classes nil)

(defthm rtz-sqrt-unique
