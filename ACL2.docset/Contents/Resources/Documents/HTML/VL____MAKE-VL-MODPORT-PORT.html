Basic constructor macro for vl-modport-port structures.

Syntax
(make-vl-modport-port [:name <name>] 
                      [:dir <dir>] 
                      [:loc <loc>] 
                      [:expr <expr>] 
                      [:atts <atts>])

This is the usual way to construct vl-modport-port structures. It simply conses together a structure with the specified fields.

This macro generates a new vl-modport-port structure from scratch. See also change-vl-modport-port, which can "change" an existing structure, instead.

Definition

This is an ordinary make- macro introduced by defprod.

Macro: make-vl-modport-port

(defmacro make-vl-modport-port (&rest args)
          (std::make-aggregate 'vl-modport-port
                               args
                               '((:name)
                                 (:dir)
                                 (:loc . *vl-fakeloc*)
                                 (:expr)
                                 (:atts))
                               'make-vl-modport-port
                               nil))

(defun vl-modport-port (name dir loc expr atts)
       (declare (xargs :guard (and (stringp name)
                                   (vl-direction-p dir)
                                   (vl-location-p loc)
                                   (vl-maybe-expr-p expr)
                                   (vl-atts-p atts))))
       (declare (xargs :guard t))
       (let ((__function__ 'vl-modport-port))
            (declare (ignorable __function__))
            (b* ((name (mbe :logic (str-fix name) :exec name))
                 (dir (mbe :logic (vl-direction-fix dir)
                           :exec dir))
                 (loc (mbe :logic (vl-location-fix loc)
                           :exec loc))
                 (expr (mbe :logic (vl-maybe-expr-fix expr)
                            :exec expr))
                 (atts (mbe :logic (vl-atts-fix atts)
                            :exec atts)))
                (list (cons 'name name)
                      (cons 'dir dir)
                      (cons 'loc loc)
                      (cons 'expr expr)
                      (cons 'atts atts)))))
