Function: svtv->vars

(defun svtv->vars (svtv)
       (declare (xargs :guard (svtv-p svtv)))
       (let ((__function__ 'svtv->vars))
            (declare (ignorable __function__))
            (append (svtv->ins svtv)
                    (svtv->outs svtv))))
