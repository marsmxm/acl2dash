(defun
 isodata-gen-oldp-of-rec-call-args-under-contexts
 (rec-calls arg-isomaps)
 (declare
    (xargs :guard (and (pseudo-tests-and-call-listp rec-calls)
                       (isodata-symbol-isomap-alistp arg-isomaps))))
 (let
  ((__function__ 'isodata-gen-oldp-of-rec-call-args-under-contexts))
  (declare (ignorable __function__))
  (b*
   (((when (endp rec-calls)) *t*)
    (tests-and-call (car rec-calls))
    (tests (access tests-and-call tests-and-call :tests))
    (rec-call (access tests-and-call tests-and-call :call))
    (context (conjoin tests)))
   (conjoin2
     (implicate context
                (conjoin (isodata-gen-oldp-of-terms (fargs rec-call)
                                                    arg-isomaps)))
     (isodata-gen-oldp-of-rec-call-args-under-contexts
          (cdr rec-calls)
          arg-isomaps)))))
