Function: q-compose

(defun q-compose (x l)
       (declare (xargs :guard t))
       (cond ((atom x) x)
             ((atom l) (q-compose (cdr x) nil))
             ((hons-equal (car x) (cdr x))
              (q-compose (car x) (cdr l)))
             (t (q-ite (car l)
                       (q-compose (car x) (cdr l))
                       (q-compose (cdr x) (cdr l))))))
