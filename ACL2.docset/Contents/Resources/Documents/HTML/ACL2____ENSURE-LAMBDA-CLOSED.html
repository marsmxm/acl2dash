Function: ensure-lambda-closed

(defun ensure-lambda-closed
       (lambd description
              error-erp error-val ctx state)
       (declare (xargs :stobjs (state)))
       (declare (xargs :guard (and (pseudo-lambdap lambd)
                                   (msgp description))))
       (b* (((unless (lambda-closedp lambd))
             (er-soft+ ctx error-erp error-val
                       "~@0 must be closed." description)))
           (value nil)))
