       (let ((__function__ 'hidden-segment-registerbits-debug))
            (declare (ignorable __function__))
            (b* (((hidden-segment-registerbits x)))
                (cons (cons 'base-addr x.base-addr)
                      (cons (cons 'limit x.limit)
                            (cons (cons 'attr x.attr) nil))))))
