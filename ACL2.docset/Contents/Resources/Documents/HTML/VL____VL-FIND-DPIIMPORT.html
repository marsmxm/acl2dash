Function: vl-find-dpiimport

(defun vl-find-dpiimport (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-dpiimportlist-p x))))
       (let ((__function__ 'vl-find-dpiimport))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-dpiimport->name (car x)))
                   (vl-dpiimport-fix (car x)))
                  (t (vl-find-dpiimport name (cdr x))))))
