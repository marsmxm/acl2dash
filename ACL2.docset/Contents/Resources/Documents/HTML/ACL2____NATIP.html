Function: natip

(defun natip (x)
       (declare (xargs :guard t))
       (let ((__function__ 'natip))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x) (eq (car x) :finite))
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((get (std::da-nth 0 (cdr x))))
                                 (natp get))))
                       (t (and (eq (car x) :infinity)
                               (and (true-listp (cdr x))
                                    (eql (len (cdr x)) 0))
                               (b* nil t)))))))
