Function: vl-match-any$inline

(defun vl-match-any$inline (function tokstream)
       (declare (xargs :stobjs (tokstream)))
       (declare (xargs :guard (symbolp function)))
       (let ((__function__ 'vl-match-any))
            (declare (ignorable __function__))
            (b* ((tokens (vl-tokstream->tokens))
                 ((when (atom tokens))
                  (vl-parse-error "Unexpected EOF."
                                  :function function))
                 (tokstream (vl-tokstream-pop)))
                (mv nil (car tokens) tokstream))))
