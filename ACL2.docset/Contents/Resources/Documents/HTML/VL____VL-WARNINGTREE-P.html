(defun vl-warningtree-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-warningtree-p))
            (declare (ignorable __function__))
            (cond ((atom x) (and (eq x nil) (b* nil t)))
                  ((eq (car x) :vl-warning)
                   (b* ((warning x))
                       (vl-warning-p warning)))
                  ((eq (car x) :vl-warningtree-context)
                   (and (consp (cdr x))
                        (b* ((?context (cadr x)) (subtree (cddr x)))
                            (vl-warningtree-p subtree))))
                  (t (b* ((left (car x)) (right (cdr x)))
                         (and (vl-warningtree-p left)
                              (vl-warningtree-p right)))))))
