Specialization of def-primitive-unary to the case in which input and output types are boolean.

Macro: def-boolean-unary

(defmacro
 def-boolean-unary
 (name &key operation
       (parents 'nil parents-suppliedp)
       (short 'nil short-suppliedp)
       (long 'nil long-suppliedp))
 (cons
  'def-primitive-unary
  (cons
   name
   (cons
    ':in-type
    (cons
     '(primitive-type-boolean)
     (cons
      ':out-type
      (cons
       '(primitive-type-boolean)
       (cons
        ':operation
        (cons
           operation
           (append (and parents-suppliedp
                        (list :parents parents))
                   (append (and short-suppliedp (list :short short))
                           (and long-suppliedp
                                (list :long long)))))))))))))
