Recognizes well-formed index expressions into hierarchical identifier pieces, e.g., the bar[3][4][5] part of foo.bar[3][4][5].baz.

Signature
(vl-hidindex-p x)  bool
Arguments
x Guard (vl-expr-p x).

Definitions and Theorems

Function: vl-hidindex-p

(defun vl-hidindex-p (x)
       (declare (xargs :guard (vl-expr-p x)))
       (let ((__function__ 'vl-hidindex-p))
            (declare (ignorable __function__))
            (b* (((when (vl-atom-p x)) (vl-hidname-p x))
                 ((vl-nonatom x) x))
                (and (vl-op-equiv x.op :vl-index)
                     (vl-hidindex-p (first x.args))))))

Theorem: vl-hidname-p-when-vl-hidindex-p

(defthm vl-hidname-p-when-vl-hidindex-p
        (implies (and (vl-hidindex-p x) (vl-atom-p x))
                 (vl-hidname-p x)))

Theorem: vl-nonatom->op-when-vl-hidindex-p

(defthm vl-nonatom->op-when-vl-hidindex-p
        (implies (and (vl-hidindex-p x)
                      (force (not (vl-atom-p x))))
                 (equal (vl-nonatom->op x) :vl-index))
        :rule-classes ((:rewrite) (:forward-chaining)))

        (implies (and (vl-hidindex-p x)

Theorem: len-of-vl-nonatom->args-when-vl-hidindex-p

(defthm len-of-vl-nonatom->args-when-vl-hidindex-p
