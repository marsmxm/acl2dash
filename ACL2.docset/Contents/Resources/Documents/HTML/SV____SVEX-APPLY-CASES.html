(defun svex-apply-collect-args (n max argsvar)
       (let* ((n (nfix n)) (max (nfix max)))
             (if (zp (- max n))
                 nil
                 (cons (cons '4veclist-nth-safe
                             (cons n (cons argsvar 'nil)))
                       (svex-apply-collect-args (+ 1 n)
                                                max argsvar)))))
