Function: print-jclass-list

(defun print-jclass-list (classes indent-level)
       (declare (xargs :guard (and (jclass-listp classes)
                                   (natp indent-level))))
       (let ((__function__ 'print-jclass-list))
            (declare (ignorable __function__))
            (cond ((endp classes) nil)
                  (t (append (list (print-jline-blank))
                             (print-jclass (car classes)
                                           indent-level)
                             (print-jclass-list (cdr classes)
                                                indent-level))))))
