Theorem: bin-digitp-of-char-code

(defthm bin-digitp-of-char-code
        (equal (bin-digitp (char-code char))
               (str::bin-digit-char-p char)))

Theorem: bin-digit-listp-of-chars=>nats

(defthm bin-digit-listp-of-chars=>nats
        (equal (bin-digit-listp (chars=>nats chars))
               (str::bin-digit-char-listp chars)))
