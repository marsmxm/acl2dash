Function: vl-hidsolo-mark

(defun
 vl-hidsolo-mark
 (mtype force-bogusp x ss st ctx)
 (declare (xargs :guard (and (member mtype '(:used :set))
                             (booleanp force-bogusp)
                             (vl-scopeexpr-p x)
                             (vl-scopestack-p ss)
                             (vl-lucidstate-p st)
                             (vl-lucidctx-p ctx))))
 (let ((__function__ 'vl-hidsolo-mark))
      (declare (ignorable __function__))
      (b* (((mv err trace ?scopectx tail)
            (vl-follow-scopeexpr x ss))
           ((when err)
            (b* (((vl-lucidstate st)))
                (change-vl-lucidstate st
                                      :warnings st.warnings)))
           ((cons (vl-hidstep step) rest) trace)
           (key (make-vl-lucidkey
                     :item step.item
                     :scopestack (vl-normalize-scopestack step.ss)))
           (occ (if (and (not force-bogusp)
                         (vl-hidexpr-case tail :end))
                    (make-vl-lucidocc-solo :ctx ctx)
                    (make-vl-lucidocc-tail :ctx ctx)))
           (st (vl-hidtrace-mark-interfaces mtype rest st ctx))
           (st (vl-lucidstate-mark mtype key occ st ctx)))
          st)))
