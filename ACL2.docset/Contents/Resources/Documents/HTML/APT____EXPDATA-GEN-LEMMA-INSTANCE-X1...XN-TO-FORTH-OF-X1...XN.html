(defun
 expdata-gen-lemma-instance-x1...xn-to-forth-of-x1...xn
 (lemma old$ arg-surjmaps wrld)
 (declare
     (xargs :guard (and (or (symbolp lemma)
                            (symbol-listp lemma))
                        (symbolp old$)
                        (expdata-symbol-surjmap-alistp arg-surjmaps)
                        (plist-worldp wrld))))
 (declare (xargs :guard (= (len (formals old$ wrld))
                           (len arg-surjmaps))))
 (let
  ((__function__
        'expdata-gen-lemma-instance-x1...xn-to-forth-of-x1...xn))
  (declare (ignorable __function__))
  (b*
    ((x1...xn (formals old$ wrld))
     (forth-of-x1...xn
          (expdata-gen-forth-of-terms x1...xn arg-surjmaps))
     (inst (alist-to-doublets (pairlis$ x1...xn forth-of-x1...xn))))
    (cons ':instance
          (cons lemma
                (cons ':extra-bindings-ok inst))))))
