Function: sintp

(defun sintp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'sintp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :sint)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (sint-integerp get)))))
