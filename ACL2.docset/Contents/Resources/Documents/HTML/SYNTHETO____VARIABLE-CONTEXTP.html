Function: variable-contextp

(defun variable-contextp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (consp (car x))
                (identifierp (caar x))
                (typep (cdar x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (consp (cadr x))
                         (acl2::fast-<< (caar x) (caadr x))
                         (variable-contextp (cdr x)))))))
