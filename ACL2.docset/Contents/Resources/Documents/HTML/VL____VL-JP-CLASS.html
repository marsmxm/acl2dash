Function: vl-jp-class-fn

(defun vl-jp-class-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-class-p x)))
       (let ((__function__ 'vl-jp-class))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-class)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-class->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"warnings\": ")
                       (vl-jp-warninglist (vl-class->warnings x))
                       (vl-println? ", ")
                       (vl-print-str "\"minloc\": ")
                       (vl-jp-location (vl-class->minloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"maxloc\": ")
                       (vl-jp-location (vl-class->maxloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-class->atts x))
                       (vl-println? ", ")
                       (vl-print-str "\"comments\": ")
                       (vl-jp-commentmap (vl-class->comments x))
                       (vl-println? ", ")
                       (vl-print-str "\"virtualp\": ")
                       (jp-bool (vl-class->virtualp x))
                       (vl-println? ", ")
                       (vl-print-str "\"lifetime\": ")
                       (vl-jp-lifetime (vl-class->lifetime x))
                       (vl-println? "}"))))
