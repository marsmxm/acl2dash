Function: funtype-for-fundef

(defun funtype-for-fundef (fundef)
       (declare (xargs :guard (fundefp fundef)))
       (let ((__function__ 'funtype-for-fundef))
            (declare (ignorable __function__))
            (make-funtype :in (len (fundef->inputs fundef))
                          :out (len (fundef->outputs fundef)))))
