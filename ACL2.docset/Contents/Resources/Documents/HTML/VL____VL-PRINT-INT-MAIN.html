Function: vl-print-int-main-fn

(defun
 vl-print-int-main-fn (n ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (integerp n)))
 (declare (type signed-byte n))
 (declare (xargs :split-types t))
 (let
  ((__function__ 'vl-print-int-main))
  (declare (ignorable __function__))
  (b*
   ((n (lifix n))
    (ps
     (if
      (< n 0)
      (vl-ps-seq
          (vl-ps-update-rchars (cons #\- (vl-ps->rchars)))
          (vl-ps-update-col (the unsigned-byte (+ 1 (vl-ps->col)))))
      ps))
    (n (abs n))
    ((when (eql n 0))
     (vl-ps-seq
         (vl-ps-update-rchars (cons #\0 (vl-ps->rchars)))
         (vl-ps-update-col (the unsigned-byte (+ 1 (vl-ps->col))))))
    ((mv rchars col)
     (vl-print-natchars-aux n (vl-ps->rchars)
                            (vl-ps->col))))
   (vl-ps-seq (vl-ps-update-rchars rchars)
              (vl-ps-update-col col)))))
