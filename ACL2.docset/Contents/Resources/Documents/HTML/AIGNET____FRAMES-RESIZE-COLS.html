Function: framesl-resize-cols

(defun
 framesl-resize-cols
 (stobjs::ncols framesl-arr2)
 (declare (type (integer 0 *) stobjs::ncols)
          (xargs :guard (and (framesl-arr2-wfp framesl-arr2)
                             (<= stobjs::ncols (1- (expt 2 60)))
                             (<= (* stobjs::ncols
                                    (framesl-nrows framesl-arr2))
                                 (1- (expt 2 60))))))
 (let
  ((stobjs::nrows (framesl-nrows framesl-arr2)))
  (stobjs::2darr
      (nfix stobjs::ncols)
      (make-list stobjs::nrows
                 :initial-element (make-list (nfix stobjs::ncols)
                                             :initial-element 0)))))
