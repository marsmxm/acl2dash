Function: vl-jp-portlist-aux-fn

(defun vl-jp-portlist-aux-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-portlist-p x)))
       (let ((__function__ 'vl-jp-portlist-aux))
            (declare (ignorable __function__))
            (if (atom x)
                ps
                (vl-ps-seq (vl-jp-port (car x))
                           (if (atom (cdr x))
                               ps (vl-println? ", "))
                           (vl-jp-portlist-aux (cdr x))))))
