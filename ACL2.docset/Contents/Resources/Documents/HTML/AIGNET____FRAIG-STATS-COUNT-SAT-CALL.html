Function: fraig-stats-count-sat-call

(defun
 fraig-stats-count-sat-call
 (status fraig-stats)
 (declare (xargs :stobjs (fraig-stats)))
 (declare (xargs :guard (symbolp status)))
 (let
  ((__function__ 'fraig-stats-count-sat-call))
  (declare (ignorable __function__))
  (case
   status
   (:sat
       (update-fraig-sat-checks (+ 1 (fraig-sat-checks fraig-stats))
                                fraig-stats))
   (:unsat (update-fraig-unsat-checks
                (+ 1 (fraig-unsat-checks fraig-stats))
                fraig-stats))
   (otherwise (update-fraig-failed-checks
                   (+ 1 (fraig-failed-checks fraig-stats))
                   fraig-stats)))))
