Function: vl-jp-property-fn

(defun vl-jp-property-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-property-p x)))
       (let ((__function__ 'vl-jp-property))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-property)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-property->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"ports\": ")
                       (vl-jp-propportlist (vl-property->ports x))
                       (vl-println? ", ")
                       (vl-print-str "\"decls\": ")
                       (vl-jp-vardecllist (vl-property->decls x))
                       (vl-println? ", ")
                       (vl-print-str "\"spec\": ")
                       (vl-jp-propspec (vl-property->spec x))
                       (vl-println? ", ")
                       (vl-print-str "\"loc\": ")
                       (vl-jp-location (vl-property->loc x))
                       (vl-println? "}"))))
