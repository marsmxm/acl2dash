(defun vl-jp-initial-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-initial-p x)))
       (let ((__function__ 'vl-jp-initial))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-initial)
                       (vl-print ", ")
                       (vl-print-str "\"stmt\": ")
                       (vl-jp-stmt (vl-initial->stmt x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-initial->atts x))
                       (vl-println? ", ")
                       (vl-print-str "\"loc\": ")
                       (vl-jp-location (vl-initial->loc x))
                       (vl-println? "}"))))
