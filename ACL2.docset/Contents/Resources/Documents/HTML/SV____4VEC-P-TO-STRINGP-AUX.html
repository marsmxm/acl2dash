(defun
     4vec-p-to-stringp-aux
     (x.upper x.lower i ac)
     (declare (xargs :guard (and (integerp x.upper)
                                 (integerp x.lower)
                                 (natp i)
                                 (character-listp ac))))
     (let ((__function__ '4vec-p-to-stringp-aux))
          (declare (ignorable __function__))
          (if (zp i)
              (coerce (reverse ac) 'string)
              (4vec-p-to-stringp-aux
                   x.upper x.lower (1- i)
