Theorem: type-when-operand-type-code-p

(defthm type-when-operand-type-code-p
        (implies (operand-type-code-p x)
                 (if (symbolp x)
                     (if (not (equal x 't))
                         (not (equal x 'nil))
                         'nil)
                     'nil))
        :rule-classes :compound-recognizer)
