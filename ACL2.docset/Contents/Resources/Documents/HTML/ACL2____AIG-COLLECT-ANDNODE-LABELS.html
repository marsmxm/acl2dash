Function: aig-collect-andnode-labels

(defun aig-collect-andnode-labels (x map)
       (declare (xargs :guard t))
       (let ((__function__ 'aig-collect-andnode-labels))
            (declare (ignorable __function__))
            (b* (((when (atom x)) nil)
                 ((unless (cdr x))
                  (aig-collect-andnode-labels (car x)
                                              map))
                 (x-label (cdr (hons-get x map)))
                 (car-labels (aig-collect-andnode-labels (car x)
                                                         map))
                 (cdr-labels (aig-collect-andnode-labels (cdr x)
                                                         map)))
                (set::insert x-label
                             (set::union car-labels cdr-labels)))))
