Function: vl-jp-program-fn

(defun vl-jp-program-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-program-p x)))
       (let ((__function__ 'vl-jp-program))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-program)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-program->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"warnings\": ")
                       (vl-jp-warninglist (vl-program->warnings x))
                       (vl-println? ", ")
                       (vl-print-str "\"minloc\": ")
                       (vl-jp-location (vl-program->minloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"maxloc\": ")
                       (vl-jp-location (vl-program->maxloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-program->atts x))
                       (vl-println? ", ")
                       (vl-print-str "\"comments\": ")
                       (vl-jp-commentmap (vl-program->comments x))
                       (vl-println? "}"))))
