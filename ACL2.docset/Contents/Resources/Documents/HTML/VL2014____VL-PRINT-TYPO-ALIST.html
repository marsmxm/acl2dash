(defun
 vl-print-typo-alist-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (and (alistp x)
                             (vl-string-keys-p x)
                             (vl-string-list-values-p x))))
 (let
  ((__function__ 'vl-print-typo-alist))
  (declare (ignorable __function__))
  (cond
      ((atom x) ps)
      ((vl-ps->htmlp)
       (vl-ps-seq (vl-print-markup "<li>Should <tt class=\"typo_w\">")
                  (vl-print (caar x))
                  (vl-print-markup "</tt> be ")
                  (vl-print-typo-possibilities (cdar x))
                  (vl-println-markup "</li>")
                  (vl-print-typo-alist (cdr x))))
      (t (vl-ps-seq (vl-print "  ")
                    (vl-print (caar x))
                    (vl-print ": ")
                    (vl-indent 30)
                    (vl-print-typo-possibilities (cdar x))
                    (vl-println "")
                    (vl-print-typo-alist (cdr x)))))))
