Function: svar-lookup

(defun svar-lookup (k x)
       (declare (xargs :guard (and (svar-p k) (svar-alist-p x))))
       (let ((__function__ 'svar-lookup))
            (declare (ignorable __function__))
            (mbe :logic (if (atom x)
                            nil
                            (if (and (consp (car x))
                                     (svar-p (caar x))
                                     (equal (svar-fix k) (caar x)))
                                (car x)
                                (svar-lookup k (cdr x))))
                 :exec (hons-get k x))))

Theorem: svar-lookup-of-cons

(defthm svar-lookup-of-cons
        (equal (svar-lookup k (cons a b))
               (if (and (consp a)
                        (svar-p (car a))
                        (svar-equiv (car a) k))
                   a (svar-lookup k b))))

Theorem: svar-lookup-of-append

(defthm svar-lookup-of-append

               (or (svar-lookup k a)
                   (svar-lookup k b))))
