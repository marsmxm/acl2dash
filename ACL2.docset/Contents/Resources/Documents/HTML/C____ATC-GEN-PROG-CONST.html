(defun
 atc-gen-prog-const
 (prog-const tunit print)
 (declare (xargs :guard (and (symbolp prog-const)
                             (transunitp tunit)
                             (evmac-input-print-p print))))
 (let
  ((__function__ 'atc-gen-prog-const))
  (declare (ignorable __function__))
  (b*
   ((progress-start?
        (and (evmac-input-print->= print :info)
             (cons (cons 'cw-event
                         (cons '"~%Generating the named constant..."
                               'nil))
                   'nil)))
    (progress-end?
         (and (evmac-input-print->= print :info)
              (cons (cons 'cw-event (cons '" done.~%" 'nil))
                    'nil)))
    (defconst-event
         (cons 'defconst
               (cons prog-const
                     (cons (cons 'quote (cons tunit 'nil))
                           'nil))))
    (local-event
        (cons 'progn
              (append progress-start?
                      (cons (cons 'local (cons defconst-event 'nil))
                            progress-end?)))))
   (mv local-event defconst-event))))
