Function: vl-find-genelement

(defun vl-find-genelement (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-genelementlist-p x))))
       (let ((__function__ 'vl-find-genelement))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-genelement->blockname (car x)))
                   (vl-genelement-fix (car x)))
                  (t (vl-find-genelement name (cdr x))))))
