Function: vl-pp-paramargs-fn

(defun
 vl-pp-paramargs-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (vl-paramargs-p x)))
 (let
    ((__function__ 'vl-pp-paramargs))
    (declare (ignorable __function__))
    (without-mimicking-linebreaks
         (vl-paramargs-case
              x
