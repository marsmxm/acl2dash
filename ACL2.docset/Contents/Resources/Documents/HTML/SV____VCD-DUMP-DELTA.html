(defun
  vcd-dump-delta
  (time changes vcd-vals vcd-wiremap p)
  (declare (xargs :stobjs (vcd-vals vcd-wiremap)))
  (declare (xargs :guard (and (natp time)
                              (nat-listp changes)
                              (vl-printedlist-p p))))
  (declare
       (xargs :guard (and (<= (vcdwires-length vcd-wiremap)
                              (4vecs-length vcd-vals))

       (declare (ignorable __function__))
       (b* ((p (rlist* p #\# (natstr time) #\Newline))
