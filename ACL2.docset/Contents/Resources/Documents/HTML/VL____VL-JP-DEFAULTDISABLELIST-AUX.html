Function: vl-jp-defaultdisablelist-aux-fn

(defun
     vl-jp-defaultdisablelist-aux-fn (x ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard (vl-defaultdisablelist-p x)))
     (let ((__function__ 'vl-jp-defaultdisablelist-aux))
          (declare (ignorable __function__))
          (if (atom x)
              ps
              (vl-ps-seq (vl-jp-defaultdisable (car x))
                         (if (atom (cdr x))
                             ps (vl-println? ", "))
                         (vl-jp-defaultdisablelist-aux (cdr x))))))
