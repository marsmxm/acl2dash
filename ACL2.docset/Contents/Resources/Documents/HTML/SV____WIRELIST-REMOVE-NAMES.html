Function: wirelist-remove-names

(defun
   wirelist-remove-names (names x)
   (declare (xargs :guard (and (namelist-p names)
                               (wirelist-p x))))
   (let ((__function__ 'wirelist-remove-names))
        (declare (ignorable __function__))
        (if (atom names)
            (wirelist-fix x)
            (wirelist-remove-name (car names)
                                  (wirelist-remove-names (cdr names)
                                                         x)))))
