Function: vl-class-apply-reportcard

(defun
  vl-class-apply-reportcard (x reportcard)
  (declare (xargs :guard (and (vl-class-p x)
                              (vl-reportcard-p reportcard))))
  (let ((__function__ 'vl-class-apply-reportcard))
       (declare (ignorable __function__))
       (b* (((vl-class x) x)
            (entry (hons-get x.name (vl-reportcard-fix reportcard)))
            ((unless entry) (vl-class-fix x)))
           (change-vl-class
                x
                :warnings (revappend-without-guard (cdr entry)
                                                   x.warnings)))))
