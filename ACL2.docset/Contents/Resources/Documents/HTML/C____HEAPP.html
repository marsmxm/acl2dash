(defun heapp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (consp (car x))
                (addressp (caar x))
                (arrayp (cdar x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (consp (cadr x))
                         (acl2::fast-<< (caar x) (caadr x))
                         (heapp (cdr x)))))))
