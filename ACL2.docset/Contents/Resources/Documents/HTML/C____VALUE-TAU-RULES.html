Definitions and Theorems

Theorem: signed-integer-value-kinds

(defthm signed-integer-value-kinds
        (implies (or (scharp x)
                     (sshortp x)
                     (sintp x)
                     (slongp x)
                     (sllongp x))
                 (and (value-scalarp x)
                      (value-arithmeticp x)
                      (value-realp x)
                      (value-integerp x)
                      (value-signed-integerp x)))
        :rule-classes :tau-system)
