Function: lex-hex-number

(defun
 lex-hex-number (abnf::input)
 (declare (xargs :guard (nat-listp abnf::input)))
 (let
  ((__function__ 'lex-hex-number))
  (declare (ignorable __function__))
  (b*
   (((mv abnf::treess abnf::input)
     (b*
      (((mv abnf::treess1 abnf::input1)
        (b* (((mv abnf::tree abnf::input)
              (abnf::parse-schars "0x" abnf::input))
             ((when (resulterrp abnf::tree))
              (mv (fty::err-push abnf::tree)
                  abnf::input))
             (abnf::trees1 (list abnf::tree))
             ((mv abnf::trees2 abnf::input)
              (lex-repetition-*-hex-digit abnf::input))
             ((when (resulterrp abnf::trees2))
              (mv (fty::err-push abnf::trees2)
                  abnf::input))
             (abnf::treess (list abnf::trees1 abnf::trees2)))
            (mv abnf::treess abnf::input)))
       ((when (not (resulterrp abnf::treess1)))
        (mv abnf::treess1 abnf::input1)))
      (mv
       (err
        (list
            :found (list abnf::treess1)
            :required
            '(((:repetition (:repeat 1 (:finite 1))
                            (:char-val (:sensitive "0x")))
               (:repetition (:repeat 0 (:infinity))
                            (:rulename (:rulename "hex-digit")))))))
       abnf::input)))
    ((when (resulterrp abnf::treess))
     (mv (fty::err-push abnf::treess)
         (acl2::nat-list-fix abnf::input))))
   (mv (abnf::make-tree-nonleaf
            :rulename? (abnf::rulename "hex-number")
            :branches abnf::treess)
       abnf::input))))
