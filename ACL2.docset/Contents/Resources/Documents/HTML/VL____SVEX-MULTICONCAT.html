Function: svex-multiconcat

(defun svex-multiconcat (n x widths)
       (declare (xargs :guard (and (natp n)
                                   (sv::svexlist-p x)
                                   (nat-listp widths))))
       (declare (xargs :guard (eql (len x) (len widths))))
       (let ((__function__ 'svex-multiconcat))
            (declare (ignorable __function__))
            (svex-repeat n (sum-nats widths)
                         (svex-concat-list widths x))))
