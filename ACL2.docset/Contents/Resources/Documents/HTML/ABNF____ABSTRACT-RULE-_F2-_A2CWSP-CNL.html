A ( rule / (*c-wsp c-nl) ) parse tree is abstracted to either its rule (for the first alternative) or nothing (for the second alternative).

(defun abstract-rule-/-*cwsp-cnl (tree)
       (declare (xargs :guard (treep tree)))
       (b* (((fun (fail)) (abstract-fail))
            ((unless (tree-case tree :nonleaf))
             (fail))
            (treess (tree-nonleaf->branches tree))
            ((unless (consp treess)) (fail))
            (trees (car treess))
            ((unless (consp trees)) (fail))
            (subtree (car trees))

             (fail)))

                      *rule*)
