Macro: dl

(defmacro
   dl (&rest args)
   (mv-let
        (trees attributes)
        (partition-macro-args args 'dl)
        (let ((attributes (keyword-macro-args-to-terms attributes)))
             (list 'dl-fn
                   (cons 'list attributes)
                   (cons 'list trees)))))

(defun dl-fn (attributes trees)
       (declare (xargs :guard (and (keyword-tree-alistp attributes)
                                   (tree-listp trees))))
       (make-tree-tag :dl attributes trees))

Theorem: stringp-of-dl

(defthm stringp-of-dl
        (equal (treep (dl-fn attributes trees))
               (and (keyword-tree-alistp attributes)
                    (tree-listp trees))))
