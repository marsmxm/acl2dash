Definitions and Theorems

Theorem: oct-digitp-of-char-code

(defthm oct-digitp-of-char-code
        (equal (oct-digitp (char-code char))
               (str::oct-digit-char-p char)))

Theorem: oct-digit-listp-of-chars=>nats

(defthm oct-digit-listp-of-chars=>nats
        (equal (oct-digit-listp (chars=>nats chars))
               (str::oct-digit-char-listp chars)))
