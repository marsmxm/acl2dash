Function: vl-modulelist-propagate

(defun
 vl-modulelist-propagate (x limits)
 (declare (xargs :guard (and (vl-modulelist-p x)
                             (propagate-limits-p limits))))
 (let
  ((__function__ 'vl-modulelist-propagate))
  (declare (ignorable __function__))
  (mbe
    :logic (if (consp x)
               (cons (vl-module-propagate (car x) limits)
                     (vl-modulelist-propagate (cdr x)
                                              limits))
               nil)
    :exec (if (atom x)
              nil
              (with-local-nrev
                   (vl-modulelist-propagate-nrev x limits nrev))))))
