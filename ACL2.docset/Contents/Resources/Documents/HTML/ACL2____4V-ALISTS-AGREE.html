Function: 4v-alists-agree

(defun 4v-alists-agree (keys al1 al2)
       (declare (xargs :guard t))
       (or (atom keys)
           (and (equal (4v-lookup (car keys) al1)
                       (4v-lookup (car keys) al2))
                (4v-alists-agree (cdr keys) al1 al2))))
