Function: svtv-mod-alias-guard

(defun
 svtv-mod-alias-guard
 (modidx moddb aliases)
 (declare (xargs :stobjs (moddb aliases)))
 (declare (xargs :guard (and (natp modidx) (moddb-ok moddb))))
 (let
  ((__function__ 'svtv-mod-alias-guard))
  (declare (ignorable __function__))
  (flet
   ((fn (modidx moddb aliases)
        (and (< (lnfix modidx)
                (lnfix (moddb->nmods moddb)))
             (<= (stobj-let ((elab-mod (moddb->modsi modidx moddb)))
                            (nwires)
                            (elab-mod->totalwires elab-mod)
                            nwires)
                 (aliass-length aliases)))))
   (mbe :logic (non-exec (let ((moddb (moddb-fix moddb)))
                              (fn modidx moddb aliases)))
        :exec (fn modidx moddb aliases)))))
