Function: vl-import-p

(defun
     vl-import-p (x)
     (declare (xargs :guard t))
     (let ((__function__ 'vl-import-p))
          (declare (ignorable __function__))
          (and (consp x)
               (eq (car x) :vl-import)
               (std::prod-consp (cdr x))
               (std::prod-consp (std::prod-car (cdr x)))
               (std::prod-consp (std::prod-cdr (cdr x)))
               (b* ((pkg (std::prod-car (std::prod-car (cdr x))))
                    (part (std::prod-cdr (std::prod-car (cdr x))))
                    (loc (std::prod-car (std::prod-cdr (cdr x))))
                    (atts (std::prod-cdr (std::prod-cdr (cdr x)))))
