Function: exec-le

(defun exec-le (arg1 arg2)
       (declare (xargs :guard (and (valuep arg1) (valuep arg2))))
       (let ((__function__ 'exec-le))
            (declare (ignorable __function__))
            (b* ((arg1 (value-fix arg1))
                 (arg2 (value-fix arg2))
                 ((unless (value-realp arg1))
                  (error (list :mistype-le
                               :required :arithmetic
                               :supplied arg1)))
                 ((unless (value-realp arg2))
                  (error (list :mistype-le
                               :required :arithmetic
                               :supplied arg2)))
                 ((mv val1 val2)
                  (uaconvert-values arg1 arg2)))
                (cond ((uintp val1) (le-uint-uint val1 val2))
                      ((sintp val1) (le-sint-sint val1 val2))
                      ((ulongp val1)
                       (le-ulong-ulong val1 val2))
                      ((slongp val1)
                       (le-slong-slong val1 val2))
                      ((ullongp val1)
                       (le-ullong-ullong val1 val2))
                      ((sllongp val1)
                       (le-sllong-sllong val1 val2))
                      (t (error (impossible)))))))
