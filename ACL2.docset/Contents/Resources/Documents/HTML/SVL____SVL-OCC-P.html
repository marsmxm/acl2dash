(defun
     svl-occ-p (x)
     (declare (xargs :guard t))
     (let ((acl2::__function__ 'svl-occ-p))
          (declare (ignorable acl2::__function__))
          (and (consp x)
               (cond ((or (atom x) (eq (car x) :assign))
                      (and (true-listp (cdr x))
                           (eql (len (cdr x)) 2)
                           (b* ((output (std::da-nth 0 (cdr x)))
                                (svex (std::da-nth 1 (cdr x))))
                               (and (sv::svar-p output)
                                    (svex-p svex)))))

                             (and (true-listp (cdr x))
                                  (eql (len (cdr x)) 3))
