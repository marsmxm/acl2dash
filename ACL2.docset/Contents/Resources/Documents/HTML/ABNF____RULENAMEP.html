Function: rulenamep

(defun rulenamep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'rulenamep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :rulename)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (common-lisp::stringp get)))))

(defthm consp-when-rulenamep
