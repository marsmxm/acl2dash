Theorem: check-safe-statement-of-statement-dead

(defthm
 check-safe-statement-of-statement-dead
 (implies
    (and (statement-nofunp stmt)
         (statement-noloopinitp stmt))
    (b* ((varsmodes (check-safe-statement stmt varset funtab))
         (varsmodes-dead (check-safe-statement (statement-dead stmt)
                                               varset funtab)))
        (implies (not (resulterrp varsmodes))
                 (and (not (resulterrp varsmodes-dead))
                      (equal (vars+modes->vars varsmodes-dead)
                             (vars+modes->vars varsmodes))
                      (subset (vars+modes->modes varsmodes-dead)
                              (vars+modes->modes varsmodes)))))))
