Definitions and Theorems

Function: vl-string-values-p

(defun vl-string-values-p (x)
       (declare (xargs :guard t))
       (if (consp x)
           (and (consp (car x))
                (stringp (cdar x))
                (vl-string-values-p (cdr x)))
           (not x)))

Theorem: vl-string-values-p-when-not-consp

(defthm vl-string-values-p-when-not-consp
        (implies (not (consp x))
                 (equal (vl-string-values-p x) (not x))))

Theorem: vl-string-values-p-of-cons

(defthm vl-string-values-p-of-cons
        (equal (vl-string-values-p (cons a x))
               (and (consp a)
                    (stringp (cdr a))
                    (vl-string-values-p x))))
