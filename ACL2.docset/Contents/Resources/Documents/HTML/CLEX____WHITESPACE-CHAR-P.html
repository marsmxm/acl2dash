Function: whitespace-char-p

(defun whitespace-char-p (x)
       (declare (xargs :guard t))
       (and (characterp x)
            (or (eql x #\Space)
                (eql x #\Newline)
                (eql x #\Tab))))

Theorem: characterp-when-whitespace-char-p

(defthm characterp-when-whitespace-char-p
