Function: vl-module-deltemps

(defun
 vl-module-deltemps (x)
 (declare (xargs :guard (vl-module-p x)))
 (let
   ((__function__ 'vl-module-deltemps))
   (declare (ignorable __function__))
   (b* (((vl-module x) (vl-module-fix x)))
       (b* ((alwayses (vl-alwayslist-deltemps x.alwayses))
            (generates (vl-genelementlist-deltemps x.generates))
            (fundecls (vl-fundecllist-deltemps x.fundecls))
            (taskdecls (vl-taskdecllist-deltemps x.taskdecls))
            (initials (vl-initiallist-deltemps x.initials))
            (finals (vl-finallist-deltemps x.finals))
            (assertions (vl-assertionlist-deltemps x.assertions))
            (cassertions (vl-cassertionlist-deltemps x.cassertions))
            (elabtasks (vl-elabtasklist-deltemps x.elabtasks))
            (parse-temps ((lambda (x) (declare (ignore x)) nil)
                          x.parse-temps)))
           (change-vl-module x
                             :alwayses alwayses
                             :generates generates
                             :fundecls fundecls
                             :taskdecls taskdecls
                             :initials initials
                             :finals finals
                             :assertions assertions
                             :cassertions cassertions
                             :elabtasks elabtasks
                             :parse-temps parse-temps)))))
