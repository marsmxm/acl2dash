Function: rewrite-equiv$inline

(defun rewrite-equiv$inline (x y)
       (declare (xargs :guard (and (rewrite-p x) (rewrite-p y))))
       (equal (rewrite-fix x) (rewrite-fix y)))

Theorem: rewrite-equiv-is-an-equivalence

(defthm rewrite-equiv-is-an-equivalence
        (and (booleanp (rewrite-equiv x y))
             (rewrite-equiv x x)
             (implies (rewrite-equiv x y)
                      (rewrite-equiv y x))
             (implies (and (rewrite-equiv x y)
                           (rewrite-equiv y z))
                      (rewrite-equiv x z)))
        :rule-classes (:equivalence))
