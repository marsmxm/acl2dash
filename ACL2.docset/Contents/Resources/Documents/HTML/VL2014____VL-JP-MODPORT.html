Function: vl-jp-modport-fn

(defun
     vl-jp-modport-fn (x ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard (vl-modport-p x)))
     (let ((__function__ 'vl-jp-modport))
          (declare (ignorable __function__))
          (vl-ps-seq (vl-print "{\"tag\": ")
                     (jp-sym ':vl-modport)
                     (vl-print ", ")
                     (vl-print-str "\"name\": ")
                     (jp-str (vl-modport->name x))
                     (vl-println? ", ")
                     (vl-print-str "\"ports\": ")
                     (vl-jp-modport-portlist (vl-modport->ports x))
                     (vl-println? ", ")
                     (vl-print-str "\"atts\": ")
                     (vl-jp-atts (vl-modport->atts x))
                     (vl-println? ", ")
                     (vl-print-str "\"loc\": ")
                     (vl-jp-location (vl-modport->loc x))
                     (vl-println? "}"))))
