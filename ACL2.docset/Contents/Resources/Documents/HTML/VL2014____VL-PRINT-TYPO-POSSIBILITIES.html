Function: vl-print-typo-possibilities-fn

(defun
 vl-print-typo-possibilities-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (string-listp x)))
 (let
     ((__function__ 'vl-print-typo-possibilities))
     (declare (ignorable __function__))
     (cond ((atom x) ps)
           ((vl-ps->htmlp)
            (vl-ps-seq (vl-print-markup "<tt>")
                       (vl-print (car x))
                       (vl-print-markup "</tt>")
                       (cond ((and (consp (cdr x)) (consp (cddr x)))
                              (vl-print ", "))
                             ((consp (cdr x)) (vl-print ", or "))
                             (t (vl-print "?")))
                       (vl-print-typo-possibilities (cdr x))))
           (t (vl-ps-seq (vl-print (car x))
                         (if (consp (cdr x)) (vl-print ", ") ps)
                         (vl-print-typo-possibilities (cdr x)))))))
