Function: ucletter-char-p

(defun ucletter-char-p (x)
       (declare (xargs :guard t))
       (and (characterp x)
            (b* ((code (char-code x)))
                (and (<= (char-code #\A) code)
                     (<= code (char-code #\Z))))))
