Function: integer-type-minbits

(defun integer-type-minbits (type)
       (declare (xargs :guard (typep type)))
       (declare (xargs :guard (type-integerp type)))
       (let ((__function__ 'integer-type-minbits))
            (declare (ignorable __function__))
            (case (type-kind type)
                  ((:char :schar :uchar) 8)
                  ((:sshort :ushort) 16)
                  ((:sint :uint) 16)
                  ((:slong :ulong) 32)
                  ((:sllong :ullong) 64)
                  (t (prog2$ (impossible) 1)))))
