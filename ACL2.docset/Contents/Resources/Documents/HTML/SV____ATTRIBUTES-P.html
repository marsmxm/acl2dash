Function: attributes-p

(defun attributes-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'attributes-p))
            (declare (ignorable __function__))
            (if (atom x)
                (eq x nil)
                (and (consp (car x))
                     (stringp (caar x))
                     (maybe-svex-p (cdar x))
                     (attributes-p (cdr x))))))
