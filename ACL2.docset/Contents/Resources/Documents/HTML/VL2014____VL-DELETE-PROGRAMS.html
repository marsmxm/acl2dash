Function: vl-delete-programs

(defun
 vl-delete-programs (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-programlist-p x))))
 (let
  ((__function__ 'vl-delete-programs))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-program->name (car x))
                               (string-list-fix names))
                 (vl-delete-programs names (cdr x)))
                (t (cons (vl-program-fix (car x))
                         (vl-delete-programs names (cdr x)))))
   :exec
   (b*
    (((when (atom names)) (list-fix x))
     ((when (atom x)) nil)
     ((unless (longer-than-p 10 names))
      (vl-slow-delete-programs names x))
     (fal (make-lookup-alist names))
     (ans
       (with-local-nrev (vl-fast-delete-programs names fal x nrev)))
     (- (fast-alist-free fal)))
    ans))))
