Function: fundefp

(defun
 fundefp (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'fundefp))
  (declare (ignorable __function__))
  (and
      (consp x)
      (eq (car x) :fundef)
      (mbe :logic (and (alistp (cdr x))
                       (equal (strip-cars (cdr x))
                              '(result name params body)))
           :exec (fty::alist-with-carsp (cdr x)
                                        '(result name params body)))
      (b* ((result (cdr (std::da-nth 0 (cdr x))))
           (name (cdr (std::da-nth 1 (cdr x))))
           (params (cdr (std::da-nth 2 (cdr x))))
           (body (cdr (std::da-nth 3 (cdr x)))))
          (and (tyspecseqp result)
               (identp name)
               (param-declon-listp params)
               (block-item-listp body))))))
