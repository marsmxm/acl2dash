Theorem: expr-equiv-is-an-equivalence

(defthm expr-equiv-is-an-equivalence
        (and (booleanp (expr-equiv x y))
             (expr-equiv x x)
             (implies (expr-equiv x y)
                      (expr-equiv y x))
             (implies (and (expr-equiv x y) (expr-equiv y z))
                      (expr-equiv x z)))
        :rule-classes (:equivalence))
