Function: abstract-case-sensitive/insensitive-string

(defun abstract-case-sensitive/insensitive-string
       (tree)
       (declare (xargs :guard (treep tree)))
       (b* (((fun (fail))
             (prog2$ (abstract-fail) ""))
            ((unless (tree-case tree :nonleaf))
             (fail))
            (treess (tree-nonleaf->branches tree))
            ((unless (and (consp treess)
                          (consp (cdr treess))))
             (fail))
            (trees (cadr treess))
            ((unless (consp trees)) (fail))
            (subtree (car trees)))
           (abstract-quoted-string subtree)))
