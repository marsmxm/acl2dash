Function: svstate-fork

(defun
 svstate-fork (x)
 (declare (xargs :guard (svstate-p x)))
 (let
  ((__function__ 'svstate-fork))
  (declare (ignorable __function__))
  (b* (((svstate x)))
      (change-svstate x
                      :blkst (svstack-fork x.blkst)
                      :nonblkst (fast-alist-fork x.nonblkst nil)))))
