Theorem: logior*

(defthm logior*
        (implies (and (force (integerp i))
                      (force (integerp j)))
                 (equal (logior i j)
                        (logcons (b-ior (logcar i) (logcar j))
                                 (logior (logcdr i) (logcdr j)))))

        ((:definition :clique (binary-logior)
