Function: atj-gen-deep-main-cunit

(defun
 atj-gen-deep-main-cunit
 (java-package$ java-class$ verbose$)
 (declare (xargs :guard (and (maybe-stringp java-package$)
                             (stringp java-class$)
                             (booleanp verbose$))))
 (let
  ((__function__ 'atj-gen-deep-main-cunit))
  (declare (ignorable __function__))
  (b*
    ((class (atj-gen-deep-main-class java-class$ verbose$))
     ((run-when verbose$)
      (cw "~%Generate the main Java compilation unit.~%")))
    (make-jcunit
         :package? java-package$
         :imports (list (jimport nil (str::cat *aij-package* ".*")))
         :types (list class)))))
