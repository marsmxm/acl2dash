Function: ulongp

(defun ulongp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'ulongp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :ulong)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (ulong-integerp get)))))
