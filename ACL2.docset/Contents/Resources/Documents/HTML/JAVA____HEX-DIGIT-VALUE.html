Function: hex-digit-value

(defun hex-digit-value (x)
       (declare (xargs :guard (hex-digitp x)))
       (let ((__function__ 'hex-digit-value))
            (declare (ignorable __function__))
            (b* ((x (mbe :logic (hex-digit-fix x) :exec x)))
                (cond ((and (<= (char-code #\0) x)
                            (<= x (char-code #\9)))
                       (- x (char-code #\0)))
                      ((and (<= (char-code #\A) x)
                            (<= x (char-code #\F)))
                       (+ 10 (- x (char-code #\A))))
                      ((and (<= (char-code #\a) x)
                            (<= x (char-code #\f)))
                       (+ 10 (- x (char-code #\a))))
                      (t (impossible))))))
