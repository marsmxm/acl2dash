Function: vl-jp-typedef-fn

(defun vl-jp-typedef-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-typedef-p x)))
       (let ((__function__ 'vl-jp-typedef))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-typedef)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-typedef->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"type\": ")
                       (vl-jp-datatype (vl-typedef->type x))
                       (vl-println? ", ")
                       (vl-print-str "\"minloc\": ")
                       (vl-jp-location (vl-typedef->minloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"maxloc\": ")
                       (vl-jp-location (vl-typedef->maxloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-typedef->atts x))
                       (vl-println? ", ")
                       (vl-print-str "\"warnings\": ")
                       (vl-jp-warninglist (vl-typedef->warnings x))
                       (vl-println? ", ")
                       (vl-print-str "\"comments\": ")
                       (vl-jp-commentmap (vl-typedef->comments x))
                       (vl-println? "}"))))
