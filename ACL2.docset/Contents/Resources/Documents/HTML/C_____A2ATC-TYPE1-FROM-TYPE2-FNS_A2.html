Function: atc-type1-from-type2-fns-gen-aux

(defun
 atc-type1-from-type2-fns-gen-aux
 (type1 type2s)
 (cond
  ((endp type2s) nil)
  (t
   (append (and (not (equal type1 (car type2s)))
                (list (pack (type-kind type1)
                            '-from-
                            (type-kind (car type2s)))))
           (atc-type1-from-type2-fns-gen-aux type1 (cdr type2s))))))

(defun
