Function: field-list->type-list

(defun
 field-list->type-list (x)
 (declare (xargs :guard (field-listp x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'field-list->type-list))
  (declare (ignorable __function__))
  (mbe :logic (if (consp x)
                  (cons (field->type (car x))
                        (field-list->type-list (cdr x)))
                  nil)
       :exec (if (atom x)
                 nil
                 (acl2::with-local-nrev
                      (field-list->type-list-nrev x acl2::nrev))))))
