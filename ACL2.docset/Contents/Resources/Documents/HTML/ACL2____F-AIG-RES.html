Function: f-aig-res

(defun f-aig-res (x y)
       (declare (xargs :guard t))
       (let ((__function__ 'f-aig-res))
            (declare (ignorable __function__))
            (b* (((faig x1 x0) x) ((faig y1 y0) y))
