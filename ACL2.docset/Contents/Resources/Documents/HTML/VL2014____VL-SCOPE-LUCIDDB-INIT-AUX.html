Function: vl-scope-luciddb-init-aux

(defun vl-scope-luciddb-init-aux (locals ss db)
       (declare (xargs :guard (and (vl-scopeitem-alist-p locals)
                                   (vl-scopestack-p ss)
                                   (vl-luciddb-p db))))
       (let ((__function__ 'vl-scope-luciddb-init-aux))
            (declare (ignorable __function__))
            (b* ((locals (vl-scopeitem-alist-fix locals))
                 ((when (atom locals))
                  (vl-luciddb-fix db))
                 ((cons ?name item) (car locals))
                 (key (make-vl-lucidkey :item item
                                        :scopestack ss))
                 (db (hons-acons key *vl-empty-lucidval* db)))
                (vl-scope-luciddb-init-aux (cdr locals)
                                           ss db))))
