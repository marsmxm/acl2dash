Function: bfr-logapp-russ

(defun
 bfr-logapp-russ (n x y)
 (declare (xargs :guard (and (true-listp n)
                             (true-listp x)
                             (true-listp y))))
 (let
  ((__function__ 'bfr-logapp-russ))
  (declare (ignorable __function__))
  (if
   (atom n)
   (llist-fix y)
   (if
      (b* (((mv x1 & xend) (first/rest/end x))
           ((mv y1 & yend) (first/rest/end y)))
          (and xend yend (equal x1 y1)))
      (llist-fix x)
      (bfr-ite-bss
           (car n)
           (b* ((w (ash 1 (len (cdr n)))))
               (bfr-logapp-nus w (s-take w x)
                               (bfr-logapp-russ (cdr n)
                                                (bfr-logtail-ns w x)
                                                y)))
           (bfr-logapp-russ (cdr n) x y))))))
