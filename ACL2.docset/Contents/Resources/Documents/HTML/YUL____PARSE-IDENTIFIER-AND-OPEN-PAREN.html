 (declare (xargs :guard (abnf::tree-listp tokens)))
 (let
     ((__function__ 'parse-identifier-and-open-paren))
     (declare (ignorable __function__))
     (b* (((when (endp tokens))
           (mv (err "no id here") nil))
          ((mv id-ast?
               tokens-after-identifier-or-error)

               nil)))
