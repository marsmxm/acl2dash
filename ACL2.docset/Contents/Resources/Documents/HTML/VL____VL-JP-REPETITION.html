Function: vl-jp-repetition-fn

(defun
     vl-jp-repetition-fn (x ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard (vl-repetition-p x)))
     (let ((__function__ 'vl-jp-repetition))
          (declare (ignorable __function__))
          (vl-ps-seq (vl-print "{\"tag\": ")
                     (jp-sym ':vl-repetition)
                     (vl-print ", ")
                     (vl-print-str "\"type\": ")
                     (vl-jp-repetitiontype (vl-repetition->type x))
                     (vl-println? ", ")
                     (vl-print-str "\"left\": ")
                     (vl-jp-expr (vl-repetition->left x))
                     (vl-println? ", ")
                     (vl-print-str "\"right\": ")
                     (vl-jp-maybe-expr (vl-repetition->right x))
                     (vl-println? "}"))))
