Function: bfr-scons

(defun bfr-scons (b v)
       (declare (xargs :guard (true-listp v)))
       (let ((__function__ 'bfr-scons))
            (declare (ignorable __function__))
            (if (atom v)
                (if b (list b nil) '(nil))
                (if (and (atom (cdr v))
                         (hons-equal (car v) b))
                    (llist-fix v)
                    (cons b (llist-fix v))))))
