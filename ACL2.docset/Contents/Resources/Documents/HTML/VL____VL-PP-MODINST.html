Function: vl-pp-modinst-fn

(defun
 vl-pp-modinst-fn (x ss ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (and (vl-modinst-p x)
                             (vl-scopestack-p ss))))
 (let
  ((__function__ 'vl-pp-modinst))
  (declare (ignorable __function__))
  (b*
   (((vl-modinst x) x)
    (have-params-p
         (vl-paramargs-case
              x.paramargs
              :vl-paramargs-named (consp x.paramargs.args)
              :vl-paramargs-plain (consp x.paramargs.args))))
   (if
    (or x.str x.delay)
    (prog2$
     (cw
      "; Note: in vl-pp-modinst, dropping str/delay from ~x0 instance.~%"
      x.modname)
     ps)
    (vl-ps-seq
       (if x.atts (vl-pp-modinst-atts-begin x.atts)
           ps)
       (vl-progindent)
       (if (vl-ps->htmlp)
           (vl-pp-modulename-link x.modname ss)
           (vl-print-modname x.modname))
       (if (not have-params-p)
           ps
           (vl-ps-seq (vl-print " #(")
                      (vl-pp-paramargs x.paramargs)
                      (vl-println? ")")))
       (vl-print " ")
       (if x.instname
           (vl-print-wirename x.instname)
           (prog2$ (cw "Warning: instance of ~x0 has no instname.~%"
                       x.modname)
                   ps))
       (if (not x.range)
           ps
           (vl-ps-seq (vl-print " ")
                      (vl-pp-range x.range)))
       (vl-print " (")
       (vl-pp-arguments x.portargs)
       (vl-println ") ;"))))))
