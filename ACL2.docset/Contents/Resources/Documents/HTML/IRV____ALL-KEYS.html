Function: all-keys

(defun
  all-keys (val alst)
  (declare (xargs :guard (and (natp val) (count-alistp alst))))
  (let ((__function__ 'all-keys))
       (declare (ignorable __function__))
       (if (endp alst)
           nil
           (b* ((pair (rassoc-equal val alst)))
               (if (equal pair nil)
                   nil
                   (cons (car pair)
                         (all-keys val
                                   (remove1-assoc-equal (car pair)
                                                        alst))))))))
