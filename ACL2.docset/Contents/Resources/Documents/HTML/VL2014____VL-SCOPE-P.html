Function: vl-scope-p

(defun
     vl-scope-p (x)
     (declare (xargs :guard t))
     (let ((__function__ 'vl-scope-p))
          (declare (ignorable __function__))
          (common-lisp::case (tag x)
                             ((:vl-interface) (vl-interface-p x))
                             ((:vl-module) (vl-module-p x))
                             ((:vl-genblob) (vl-genblob-p x))
                             ((:vl-blockscope) (vl-blockscope-p x))
                             ((:vl-design) (vl-design-p x))
                             ((:vl-package) (vl-package-p x))
                             (otherwise (vl-scopeinfo-p x)))))
