(defun
 vl-plainarglist-update-ifports
 (x ports ss scopes)
 (declare (xargs :guard (and (vl-plainarglist-p x)
                             (vl-portlist-p ports)
                             (vl-scopestack-p ss)
                             (vl-elabscopes-p scopes))))
 (declare (xargs :guard (eql (len x) (len ports))))
 (let
    ((__function__ 'vl-plainarglist-update-ifports))
    (declare (ignorable __function__))
    (b* (((when (atom x))
          (mv nil (vl-portlist-fix ports) nil))
         ((mv err1 port1 ifportalist1)
          (vl-plainarg-update-ifports (car x)
                                      (car ports)
                                      ss scopes))
         ((mv err2 rest-ports rest-ifportalist)
          (vl-plainarglist-update-ifports (cdr x)
                                          (cdr ports)
                                          ss scopes)))
        (mv (vmsg-concat err1 err2)
            (cons-with-hint port1 rest-ports ports)
            (append-without-guard ifportalist1 rest-ifportalist)))))
