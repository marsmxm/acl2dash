Function: vl-binary->original-operator

(defun
    vl-binary->original-operator (x)
    (declare (xargs :guard (vl-expr-p x)))
    (declare (xargs :guard (vl-expr-case x :vl-binary)))
    (let ((__function__ 'vl-binary->original-operator))
         (declare (ignorable __function__))
         (b* (((vl-binary x))
              (orig (cdr (hons-assoc-equal "VL_ORIG_EXPR" x.atts))))
             (if orig
                 (vl-expr-case orig
                               :vl-binary orig.op
                               :otherwise x.op)
                 x.op))))
