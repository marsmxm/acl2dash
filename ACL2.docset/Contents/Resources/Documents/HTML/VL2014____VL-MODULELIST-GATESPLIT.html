 (declare (xargs :guard (vl-modulelist-p x)))
 (let
  ((__function__ 'vl-modulelist-gatesplit))
  (declare (ignorable __function__))
  (mbe
     :logic (if (consp x)
                (cons (vl-module-gatesplit (car x))
                      (vl-modulelist-gatesplit (cdr x)))
