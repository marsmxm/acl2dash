 check-ext-declon-list
 (exts funtab tagenv)

                             (fun-tablep funtab)
                             (tag-envp tagenv))))
 (let
  ((__function__ 'check-ext-declon-list))
  (declare (ignorable __function__))
  (b* (((when (endp exts))
        (make-funtab+tagenv :funs (fun-table-fix funtab)
