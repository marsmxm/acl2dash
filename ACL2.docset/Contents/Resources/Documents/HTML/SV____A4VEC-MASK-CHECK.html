Function: a4vec-mask-check

(defun
 a4vec-mask-check (mask idx vec upperp)
 (declare (xargs :guard (and (4vmask-p mask)
                             (natp idx)
                             (true-listp vec)
                             (booleanp upperp))))
 (let
     ((__function__ 'a4vec-mask-check))
     (declare (ignorable __function__))
     (b* ((idx (lnfix idx))
          (mask (4vmask-fix mask))
          ((mv first rest endp)
           (gl::first/rest/end vec))
          ((when endp)
           (or (mbe :logic (equal (logtail idx (sparseint-val mask))
                                  -1)
                    :exec (and (<= (sparseint-length mask) idx)
                               (eql 1 (sparseint-bit idx mask))))
               (equal first
                      (mbe :logic (bool-fix upperp)
                           :exec upperp)))))
         (and (or (eql 1 (sparseint-bit idx (4vmask-fix mask)))
                  (equal first
                         (mbe :logic (bool-fix upperp)
                              :exec upperp)))
              (a4vec-mask-check mask (1+ (lnfix idx))
                                rest upperp)))))

Theorem: a4vec-mask-check-correct

(defthm
 a4vec-mask-check-correct
 (and
  (implies (a4vec-mask-check mask 0 vec t)
           (equal (logior (lognot (sparseint-val (4vmask-fix mask)))
                          (aig-list->s vec env))
                  (aig-list->s vec env)))
  (implies (a4vec-mask-check mask 0 vec nil)
           (equal (logand (sparseint-val (4vmask-fix mask))
                          (aig-list->s vec env))
                  (aig-list->s vec env)))))
