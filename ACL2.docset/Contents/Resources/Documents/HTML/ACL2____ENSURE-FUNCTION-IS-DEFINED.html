Function: ensure-function-is-defined

(defun
  ensure-function-is-defined
  (fn description
      error-erp error-val ctx state)
  (declare (xargs :stobjs (state)))
  (declare (xargs :guard (and (msgp description)
                              (logic-function-namep fn (w state)))))
  (b* (((unless (definedp fn (w state)))
        (er-soft+ ctx error-erp error-val
                  "~@0 exists but has no definition."
                  description)))
      (value nil)))
