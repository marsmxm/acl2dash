Check that a function scope has no function definitions in the functions' bodies.

Signature
(funscope-nofunp funscope) → yes/no
Arguments
funscope — Guard (funscopep funscope).
Returns
yes/no — Type (booleanp yes/no).

Definitions and Theorems

Function: funscope-nofunp

(defun funscope-nofunp (funscope)
       (declare (xargs :guard (funscopep funscope)))
       (let ((__function__ 'funscope-nofunp))
            (declare (ignorable __function__))
            (or (not (mbt (funscopep funscope)))
                (omap::empty funscope)
                (and (b* (((mv & funinfo) (omap::head funscope)))
                         (funinfo-nofunp funinfo))
                     (funscope-nofunp (omap::tail funscope))))))

Theorem: booleanp-of-funscope-nofunp

(defthm booleanp-of-funscope-nofunp
        (b* ((yes/no (funscope-nofunp funscope)))
            (booleanp yes/no))
        :rule-classes :rewrite)

Theorem: funinfo-nofunp-of-cdr-of-in-when-funscope-nofunp

(defthm funinfo-nofunp-of-cdr-of-in-when-funscope-nofunp
        (implies (and (funscopep funscope)
                      (funscope-nofunp funscope)
                      (consp (omap::in fun funscope)))
                 (funinfo-nofunp (cdr (omap::in fun funscope)))))

Theorem: funscope-nofunp-of-update

(defthm
    funscope-nofunp-of-update
    (implies (and (funscopep funscope)
                  (funscope-nofunp funscope)
                  (funinfop funinfo)
                  (funinfo-nofunp funinfo))
             (funscope-nofunp (omap::update fun funinfo funscope))))
