Function: vl-paramargs-wildelim

(defun
 vl-paramargs-wildelim (x ctx warnings)
 (declare (xargs :guard (and (vl-paramargs-p x)
                             (vl-context-p ctx)
                             (vl-warninglist-p warnings))))
 (let
  ((__function__ 'vl-paramargs-wildelim))
  (declare (ignorable __function__))
  (b*
   ((x (vl-paramargs-fix x))
    (warnings (vl-warninglist-fix warnings)))
   (vl-paramargs-case
       x
       :vl-paramargs-named
       (b* (((mv warnings args)
             (vl-namedparamvaluelist-wildelim x.args ctx warnings)))
           (mv warnings
               (change-vl-paramargs-named x
                                          :args args)))
       :vl-paramargs-plain
       (b* (((mv warnings args)
             (vl-paramvaluelist-wildelim x.args ctx warnings)))
           (mv warnings
               (change-vl-paramargs-plain x
                                          :args args)))))))
