Function: vl-pp-bind-fn

(defun
 vl-pp-bind-fn (x ss ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (and (vl-bind-p x)
                             (vl-scopestack-p ss))))
 (let
  ((__function__ 'vl-pp-bind))
  (declare (ignorable __function__))
  (b*
   (((vl-bind x)))
   (vl-ps-seq
    (vl-progindent)
    (vl-ps-span "vl_key" (vl-print "bind "))
    (if
      x.scope
      (vl-ps-seq
           (vl-ps-span
                "vl_id"
                (vl-print-str (vl-maybe-escape-identifier x.scope)))
           (if (atom x.addto)
               ps
               (vl-ps-seq (vl-print " : ")
                          (vl-pp-exprlist x.addto))))
      (vl-pp-exprlist x.addto))
    (if
      (and (consp x.modinsts)
           (atom (cdr x.modinsts)))
      (vl-pp-modinst (car x.modinsts) ss)
      (vl-println "// BOZO print multiple modinsts in a BIND?"))))))
