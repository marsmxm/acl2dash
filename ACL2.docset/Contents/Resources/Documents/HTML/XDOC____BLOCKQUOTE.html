Macro: blockquote

(defmacro
   blockquote (&rest args)
   (mv-let
        (trees attributes)
        (partition-macro-args args 'blockquote)
        (let ((attributes (keyword-macro-args-to-terms attributes)))
             (list 'blockquote-fn
                   (cons 'list attributes)
                   (cons 'list trees)))))

(defthm stringp-of-blockquote
        (equal (treep (blockquote-fn attributes trees))
               (and (keyword-tree-alistp attributes)
