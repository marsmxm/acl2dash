Theorem: vl-typedef-equiv-is-an-equivalence

(defthm vl-typedef-equiv-is-an-equivalence
        (and (booleanp (vl-typedef-equiv x y))
             (vl-typedef-equiv x x)
             (implies (vl-typedef-equiv x y)
                      (vl-typedef-equiv y x))
             (implies (and (vl-typedef-equiv x y)
                           (vl-typedef-equiv y z))
                      (vl-typedef-equiv x z)))
        :rule-classes (:equivalence))
