(defun svtv-outputs->outalist (x phase)
       (declare (xargs :guard (and (svtv-lines-p x) (natp phase))))
       (let ((__function__ 'svtv-outputs->outalist))
            (declare (ignorable __function__))
            (b* (((when (atom x)) nil)
                 ((svtv-line xf) (car x))
                 (ent (or (nth phase xf.entries) '_))

                  (svtv-outputs->outalist (cdr x) phase))
