Function: count-alistp

(defun count-alistp (l)
       (declare (xargs :guard t))
       (let ((__function__ 'count-alistp))
            (declare (ignorable __function__))
            (cond ((atom l) (eq l nil))
                  (t (and (consp (car l))
                          (natp (caar l))
                          (natp (cdar l))
                          (count-alistp (cdr l)))))))
