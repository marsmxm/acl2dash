Function: cons-pos-alistp

(defun cons-pos-alistp (x)
       (declare (xargs :guard t))
       (if (consp x)
           (and (consp (car x))
                (consp (caar x))
                (posp (cdar x))
                (cons-pos-alistp (cdr x)))
           (null x)))
