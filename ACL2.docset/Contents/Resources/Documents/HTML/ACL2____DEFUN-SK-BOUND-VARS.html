(defun defun-sk-bound-vars (fn wrld)
       (declare (xargs :guard (and (plist-worldp wrld)
                                   (defun-sk-namep fn wrld))))
       (let ((__function__ 'defun-sk-bound-vars))
            (declare (ignorable __function__))
            (b* ((body (defun-sk-body fn wrld))
                 (var/vars (second body)))
                (if (atom var/vars)
                    (list var/vars)
                    var/vars))))
