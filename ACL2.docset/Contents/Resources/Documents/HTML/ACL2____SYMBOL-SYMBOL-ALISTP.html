Function: symbol-symbol-alistp

(defun symbol-symbol-alistp (x)
       (declare (xargs :guard t))
       (if (consp x)
           (and (consp (car x))
                (symbolp (caar x))
                (symbolp (cdar x))
                (symbol-symbol-alistp (cdr x)))
           (null x)))
