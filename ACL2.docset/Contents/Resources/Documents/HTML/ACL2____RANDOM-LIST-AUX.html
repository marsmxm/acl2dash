Function: random-list-aux

(defun random-list-aux (n limit acc state)
       (declare (xargs :guard (and (natp n)
                                   (posp limit)
                                   (true-listp acc))))
       (if (zp n)
           (mv (reverse acc) state)
           (b* (((mv x1 state) (random$ limit state)))
               (random-list-aux (- n 1)
                                limit (cons x1 acc)
                                state))))
