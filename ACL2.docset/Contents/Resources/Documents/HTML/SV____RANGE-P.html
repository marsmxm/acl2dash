Function: range-p

(defun range-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'range-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (b* ((lsb (std::prod-car x))
                      (width (std::prod-cdr x)))
                     (and (natp lsb) (maybe-posp width))))))

Theorem: consp-when-range-p

(defthm consp-when-range-p
        (implies (range-p x) (consp x))
