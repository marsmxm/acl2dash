Function: pseudo-event-landmarkp

(defun
  pseudo-event-landmarkp (val)
  (declare (xargs :guard t))
  (or (equal val '(-1 ((nil) 0)))
      (and (consp val)
           (or (natp (car val))
               (and (consp (car val))
                    (natp (car (car val)))
                    (natp (cdr (car val)))))
           (consp (cdr val))
           (if (symbolp (cadr val))
               (pseudo-event-formp (cdr val))
               (and (consp (cadr val))
                    (consp (car (cadr val)))
                    (symbolp (car (car (cadr val))))
                    (booleanp (cdr (car (cadr val))))
                    (consp (cdr (cadr val)))
                    (or (symbolp (cadr (cadr val)))
                        (stringp (cadr (cadr val)))
                        (equal 0 (cadr (cadr val)))
                        (string-or-symbol-listp (cadr (cadr val))))
                    (member-eq (cddr (cadr val))
                               '(nil :program
                                     :ideal :common-lisp-compliant))
                    (pseudo-event-formp (cddr val)))))))
