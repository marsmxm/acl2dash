Function: check-function-header-list

(defun
   check-function-header-list
   (headers ctxt)
   (declare (xargs :guard (and (function-header-listp headers)
                               (contextp ctxt))))
   (declare
        (xargs :guard (and (null (context->types ctxt))
                           (omap::empty (context->variables ctxt))
                           (null (context->obligation-vars ctxt))
                           (null (context->obligation-hyps ctxt)))))
   (let ((__function__ 'check-function-header-list))
        (declare (ignorable __function__))
        (b* (((when (endp headers)) nil)
             (err? (check-function-header (car headers)
                                          ctxt))
             ((when err?) err?)
             (err? (check-function-header-list (cdr headers)
                                               ctxt))
             ((when err?) err?))
            nil)))
