     !interp-flags->intro-bvars
     (intro-bvars x)
     (declare (xargs :guard (and (booleanp intro-bvars)
                                 (interp-flags-p x))))
     (mbe :logic (b* ((intro-bvars (bool->bit intro-bvars))
                      (x (interp-flags-fix x)))
                     (acl2::part-install intro-bvars x
                                         :width 1
                                         :low 0))
