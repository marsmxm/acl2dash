Definitions and Theorems

Function: moddb-path->wireidx

(defun
  moddb-path->wireidx (path modidx moddb)
  (declare (xargs :stobjs (moddb)))
  (declare (xargs :guard (and (path-p path)
                              (natp modidx)
                              (moddb-ok moddb))))
  (declare (xargs :guard (< modidx (moddb->nmods moddb))))
  (let ((__function__ 'moddb-path->wireidx))
       (declare (ignorable __function__))
       (b* (((mv err & idx bitsel)
             (moddb-path->wireidx/decl path modidx moddb))
            ((when err)
             (b* (((stobj-get name)
                   ((elab-mod (moddb->modsi modidx moddb)))
                   (elab-mod->name elab-mod)))
                 (cw "Error looking up ~x1: ~@0~%from module: ~x2~%"
                     err path name)))
            ((when bitsel)
             (cw "Didn't expect a bit select: ~x0"
                 path)))
           idx)))

(defthm moddb-path->wireidx-type

Theorem: moddb-path->wireidx-bound

(defthm moddb-path->wireidx-bound
        (b* ((?idx (moddb-path->wireidx path modidx moddb)))
            (implies (and (moddb-ok moddb)
                          (< (nfix modidx)

        :rule-classes :linear)
