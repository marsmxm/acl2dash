Arguments

    Guard (vl-bindlist-p x).

    Guard (vl-scopestack-p ss).
ctx — Where x occurs, for error messages.
    Guard (vl-bindcontext-p ctx).
itbl — For identifying local binds that we can handle.
    Guard (vl-bindelim-insttable-p itbl).
delta — The changes we're building.
    Guard (vl-binddelta-p delta).
genp — Are we inside a generate?.
    Guard (booleanp genp).
warnings — For collecting warnings about unsupported binds.
    Guard (vl-warninglist-p warnings).
Returns
warnings — Type (vl-warninglist-p warnings).
new-x — A list of binds to replace X. Essentially this contains any unsupported binds.
    Type (vl-bindlist-p new-x).
delta — Extended delta to apply in Pass 2.
    Type (vl-binddelta-p delta).

Function: vl-bindelim-bindlist

(defun
 vl-bindelim-bindlist
 (x ss ctx itbl delta genp warnings)
 (declare (xargs :guard (and (vl-bindlist-p x)
                             (vl-scopestack-p ss)
                             (vl-bindcontext-p ctx)
                             (vl-bindelim-insttable-p itbl)
                             (vl-binddelta-p delta)
                             (booleanp genp)
                             (vl-warninglist-p warnings))))
 (let ((__function__ 'vl-bindelim-bindlist))
      (declare (ignorable __function__))
      (b* (((when (atom x))
            (mv (ok) nil (vl-binddelta-fix delta)))
           ((mv warnings x1 delta)
            (vl-bindelim-main (car x)
                              ss ctx itbl delta genp warnings))
           ((mv warnings xrest delta)
            (vl-bindelim-bindlist (cdr x)
                                  ss ctx itbl delta genp warnings)))
          (mv warnings (append-without-guard x1 xrest)
              delta))))
