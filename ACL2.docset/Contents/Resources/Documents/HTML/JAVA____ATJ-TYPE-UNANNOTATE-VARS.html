Function: atj-type-unannotate-vars

(defun atj-type-unannotate-vars (vars)
       (declare (xargs :guard (symbol-listp vars)))
       (let ((__function__ 'atj-type-unannotate-vars))
            (declare (ignorable __function__))
            (b* (((when (endp vars)) nil)
                 ((mv var &)
                  (atj-type-unannotate-var (car vars)))
