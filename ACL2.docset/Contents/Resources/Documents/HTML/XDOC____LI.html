Macro: li

(defmacro
   li (&rest args)
   (mv-let
        (trees attributes)
        (partition-macro-args args 'li)
        (let ((attributes (keyword-macro-args-to-terms attributes)))
             (list 'li-fn
                   (cons 'list attributes)
                   (cons 'list trees)))))

Function: li-fn

(defun li-fn (attributes trees)
       (declare (xargs :guard (and (keyword-tree-alistp attributes)
                                   (tree-listp trees))))
       (make-tree-tag :li attributes trees))

Theorem: stringp-of-li

(defthm stringp-of-li
        (equal (treep (li-fn attributes trees))
               (and (keyword-tree-alistp attributes)
                    (tree-listp trees))))
