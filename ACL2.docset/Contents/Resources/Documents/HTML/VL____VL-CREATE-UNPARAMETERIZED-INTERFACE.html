(defun
   vl-create-unparameterized-interface-fn
   (x name final-paramdecls
      final-ports elabindex ledger config)
   (declare (xargs :stobjs (elabindex)))
   (declare (xargs :guard (and (vl-interface-p x)
                               (stringp name)
                               (vl-paramdecllist-p final-paramdecls)
                               (vl-portlist-p final-ports)
                               (vl-unparam-ledger-p ledger)
                               (vl-simpconfig-p config))))
   (let ((__function__ 'vl-create-unparameterized-interface))
        (declare (ignorable __function__))
        (b* ((name (string-fix name))
             (x (change-vl-interface x
                                     :name name
                                     :paramdecls final-paramdecls
                                     :ports final-ports))
             ((vl-interface x))
             (warnings x.warnings)
             (blob (vl-interface->genblob x))
             ((wmv warnings
                   keylist new-blob elabindex ledger
                   :ctx name)
              (vl-genblob-resolve blob elabindex ledger nil))
             (mod (vl-genblob->interface new-blob x))
             (mod (change-vl-interface mod
                                       :warnings warnings)))
            (mv mod keylist elabindex ledger))))
