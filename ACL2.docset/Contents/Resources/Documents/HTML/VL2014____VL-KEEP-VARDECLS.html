Function: vl-keep-vardecls

(defun
 vl-keep-vardecls (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-vardecllist-p x))))
 (let
  ((__function__ 'vl-keep-vardecls))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-vardecl->name (car x))
                               (string-list-fix names))
                 (cons (vl-vardecl-fix (car x))
                       (vl-keep-vardecls names (cdr x))))
                (t (vl-keep-vardecls names (cdr x))))
   :exec
   (b*
    (((when (or (atom names) (atom x))) nil)
     ((unless (longer-than-p 10 names))
      (vl-slow-keep-vardecls names x))
     (fal (make-lookup-alist names))
     (ans
         (with-local-nrev (vl-fast-keep-vardecls names fal x nrev)))
     (- (fast-alist-free fal)))
    ans))))
