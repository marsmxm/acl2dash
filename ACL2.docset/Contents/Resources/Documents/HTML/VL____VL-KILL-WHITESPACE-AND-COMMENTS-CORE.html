Function: vl-kill-whitespace-and-comments-core

(defun
 vl-kill-whitespace-and-comments-core
 (tokens nrev nrev2)
 (declare (xargs :stobjs (nrev nrev2)))
 (declare (xargs :guard (vl-tokenlist-p tokens)))
 (let
  ((__function__ 'vl-kill-whitespace-and-comments-core))
  (declare (ignorable __function__))
  (b*
   ((nrev (nrev-fix nrev))
    (nrev2 (nrev-fix nrev2))
    ((when (atom tokens)) (mv nrev nrev2))
    (type (vl-token->type (car tokens)))
    (nrev (if (or (eq type :vl-ws)
                  (eq type :vl-comment))
              nrev (nrev-push (car tokens) nrev)))
    (nrev2
     (if (eq type :vl-comment)
         (nrev-push
              (cons (change-vl-location (vl-token->loc (car tokens))
                                        :col 0)
                    (vl-token->string (car tokens)))
              nrev2)
         nrev2)))
   (vl-kill-whitespace-and-comments-core (cdr tokens)
                                         nrev nrev2))))
