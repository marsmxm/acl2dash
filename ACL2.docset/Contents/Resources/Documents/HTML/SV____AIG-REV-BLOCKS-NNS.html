(defun
 aig-rev-blocks-nns (nbits blocksz x)
 (declare (xargs :guard (and (natp nbits)
                             (posp blocksz)
                             (true-listp x))))
 (let
  ((__function__ 'aig-rev-blocks-nns))
  (declare (ignorable __function__))
  (b*
    ((nbits (lnfix nbits))
     (blocksz (lposfix blocksz))
     ((when (< nbits blocksz))
      (aig-loghead-ns nbits x))
     (next-nbits (- nbits blocksz))
     (rest (aig-rev-blocks-nns next-nbits

                    rest (aig-loghead-ns blocksz x)))))
