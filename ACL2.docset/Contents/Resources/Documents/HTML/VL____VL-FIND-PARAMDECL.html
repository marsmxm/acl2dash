Function: vl-find-paramdecl

(defun vl-find-paramdecl (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-paramdecllist-p x))))
       (let ((__function__ 'vl-find-paramdecl))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-paramdecl->name (car x)))
                   (vl-paramdecl-fix (car x)))
                  (t (vl-find-paramdecl name (cdr x))))))
