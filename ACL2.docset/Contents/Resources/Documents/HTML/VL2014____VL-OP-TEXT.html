Function: vl-op-text

(defun vl-op-text (x)
       (declare (xargs :guard (vl-op-p x)))
       (let ((__function__ 'vl-op-text))
            (declare (ignorable __function__))
            (b* ((op (vl-op-fix x))
                 (info (cdr (assoc op (vl-ops-table)))))
                (vl-opinfo->text info))))
