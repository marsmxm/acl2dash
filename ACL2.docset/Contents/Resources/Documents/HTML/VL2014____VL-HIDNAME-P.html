Function: vl-hidname-p

(defun vl-hidname-p (x)
       (declare (xargs :guard (vl-expr-p x)))
       (let ((__function__ 'vl-hidname-p))
            (declare (ignorable __function__))
            (and (vl-atom-p x)
                 (b* (((vl-atom x) x))
                     (or (vl-fast-hidpiece-p x.guts)
                         (vl-fast-id-p x.guts))))))
