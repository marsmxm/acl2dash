Definitions and Theorems

Function: primitivep

(defun
   primitivep (fn)
   (declare (xargs :guard (symbolp fn)))
   (let ((__function__ 'primitivep))
        (declare (ignorable __function__))
        (and (member-eq fn
                        (strip-cars *primitive-formals-and-guards*))
             t)))

Theorem: booleanp-of-primitivep

(defthm booleanp-of-primitivep
