Function: valuep

(defun
  valuep (x)
  (declare (xargs :guard t))
  (let ((__function__ 'valuep))
       (declare (ignorable __function__))
       (and (consp x)
            (cond ((or (atom x) (eq (car x) :number))
                   (and (true-listp (cdr x))
                        (eql (len (cdr x)) 1)
                        (b* ((get (std::da-nth 0 (cdr x))))
                            (acl2-numberp get))))
                  ((eq (car x) :character)
                   (and (true-listp (cdr x))
                        (eql (len (cdr x)) 1)
                        (b* ((get (std::da-nth 0 (cdr x))))
                            (characterp get))))
                  ((eq (car x) :string)
                   (and (true-listp (cdr x))
                        (eql (len (cdr x)) 1)
                        (b* ((get (std::da-nth 0 (cdr x))))
                            (stringp get))))

                   (and (true-listp (cdr x))
                        (eql (len (cdr x)) 1)
                        (b* ((get (std::da-nth 0 (cdr x))))
                            (symbol-valuep get))))
