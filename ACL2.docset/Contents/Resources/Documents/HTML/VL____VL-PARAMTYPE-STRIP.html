Function: vl-paramtype-strip

(defun
 vl-paramtype-strip (x)
 (declare (xargs :guard (vl-paramtype-p x)))
 (let
  ((__function__ 'vl-paramtype-strip))
  (declare (ignorable __function__))
  (vl-paramtype-case
   x
   :vl-implicitvalueparam
   (b* ((range (vl-maybe-range-strip x.range))
        (default (vl-maybe-expr-strip x.default)))
       (change-vl-implicitvalueparam x
                                     :range range
                                     :default default))
   :vl-explicitvalueparam
   (b* ((type (vl-datatype-strip x.type))
        (default (vl-maybe-expr-strip x.default)))
       (change-vl-explicitvalueparam x
                                     :type type
                                     :default default))
   :vl-typeparam (b* ((default (vl-maybe-datatype-strip x.default)))
                     (change-vl-typeparam x
                                          :default default)))))
