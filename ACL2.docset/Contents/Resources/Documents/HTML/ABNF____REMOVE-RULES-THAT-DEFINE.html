Definitions and Theorems

Function: remove-rules-that-define

(defun
 remove-rules-that-define
 (rulenames rules)
 (declare (xargs :guard (and (rulename-setp rulenames)
                             (rulelistp rules))))
 (cond
  ((endp rules) nil)
  (t (if (in (rule->name (car rules)) rulenames)
         (remove-rules-that-define rulenames (cdr rules))
         (cons (rule-fix (car rules))
               (remove-rules-that-define rulenames (cdr rules)))))))
