Function: vl-jp-fwdtypedef-fn

(defun
     vl-jp-fwdtypedef-fn (x ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard (vl-fwdtypedef-p x)))
     (let ((__function__ 'vl-jp-fwdtypedef))
          (declare (ignorable __function__))
          (vl-ps-seq (vl-print "{\"tag\": ")
                     (jp-sym ':vl-fwdtypedef)
                     (vl-print ", ")
                     (vl-print-str "\"atts\": ")
                     (vl-jp-atts (vl-fwdtypedef->atts x))
                     (vl-println? ", ")
                     (vl-print-str "\"kind\": ")
                     (vl-jp-fwdtypedefkind (vl-fwdtypedef->kind x))
                     (vl-println? ", ")
                     (vl-print-str "\"name\": ")
                     (jp-str (vl-fwdtypedef->name x))
                     (vl-println? ", ")
                     (vl-print-str "\"loc\": ")
                     (vl-jp-location (vl-fwdtypedef->loc x))
                     (vl-println? "}"))))
