(defun b-xor$inline (i j)
       (declare (xargs :guard (and (bitp i) (bitp j))))
       (let ((__function__ 'b-xor))

            (mbe :logic (if (zbp i)
                            (if (zbp j) 0 1)
                            (if (zbp j) 1 0))
