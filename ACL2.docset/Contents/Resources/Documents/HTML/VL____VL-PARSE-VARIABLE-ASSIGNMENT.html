Function: vl-parse-variable-assignment-fn

(defun vl-parse-variable-assignment-fn
       (tokstream config)
       (declare (xargs :stobjs (tokstream)))
       (declare (xargs :guard (vl-loadconfig-p config)))
       (declare (ignorable config))
       (let ((__function__ 'vl-parse-variable-assignment))
            (declare (ignorable __function__))
            (seq tokstream
                 (lvalue := (vl-parse-variable-lvalue))
                 (:= (vl-match-token :vl-equalsign))
                 (expr := (vl-parse-expression))
                 (return (cons lvalue expr)))))
