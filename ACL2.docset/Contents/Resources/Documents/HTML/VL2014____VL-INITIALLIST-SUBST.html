Function: vl-initiallist-subst

(defun
 vl-initiallist-subst (x sigma)
 (declare (xargs :guard (and (vl-initiallist-p x)
                             (vl-sigma-p sigma))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-initiallist-subst))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-initial-subst (car x) sigma)
                    (vl-initiallist-subst (cdr x) sigma))
              nil)
   :exec
   (if
      (atom x)
      nil
      (with-local-nrev (vl-initiallist-subst-nrev x sigma nrev))))))
