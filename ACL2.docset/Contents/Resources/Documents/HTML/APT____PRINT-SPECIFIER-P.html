Recognize a print specifier.

This is an ordinary std::defenum.

Function: print-specifier-p

(defun print-specifier-p (x)
       (declare (xargs :guard t))
       (or (eq x 'nil)
           (eq x ':error)
           (eq x ':result)
           (eq x ':info)
           (eq x ':all)))

Theorem: type-when-print-specifier-p

(defthm type-when-print-specifier-p
        (implies (print-specifier-p x)
                 (if (if (symbolp x)
                         (if (not (equal x 't))
                             (not (equal x 'nil))
                             'nil)
                         'nil)
                     't
                     (equal x 'nil)))
        :rule-classes :compound-recognizer)
