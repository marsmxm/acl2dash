 (declare (xargs :guard (vl-taskdecl-p x)))
 (let
  ((__function__ 'vl-taskdecl-strip))
  (declare (ignorable __function__))
  (b*
   (((vl-taskdecl x) (vl-taskdecl-fix x)))
   (b*
     ((body (vl-stmt-strip x.body))
