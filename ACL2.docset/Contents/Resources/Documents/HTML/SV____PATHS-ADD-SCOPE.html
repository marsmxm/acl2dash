Function: paths-add-scope

(defun paths-add-scope (scope paths)
       (declare (xargs :guard (and (name-p scope)
                                   (pathlist-p paths))))
       (let ((__function__ 'paths-add-scope))
            (declare (ignorable __function__))
            (if (atom paths)
                nil
                (cons (make-path-scope :namespace scope
                                       :subpath (car paths))
                      (paths-add-scope scope (cdr paths))))))
