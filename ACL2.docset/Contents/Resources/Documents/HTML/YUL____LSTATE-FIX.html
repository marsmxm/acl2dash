Theorem: lstate-equiv-is-an-equivalence

(defthm lstate-equiv-is-an-equivalence
        (and (booleanp (lstate-equiv x y))
             (lstate-equiv x x)
             (implies (lstate-equiv x y)
                      (lstate-equiv y x))
             (implies (and (lstate-equiv x y)
                           (lstate-equiv y z))
                      (lstate-equiv x z)))
        :rule-classes (:equivalence))
