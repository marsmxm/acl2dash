Function: vl-initial-p

(defun vl-initial-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-initial-p))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :vl-initial)
                 (std::prod-consp (cdr x))
                 (std::prod-consp (std::prod-cdr (cdr x)))
                 (b* ((stmt (std::prod-car (cdr x)))
                      (atts (std::prod-car (std::prod-cdr (cdr x))))
                      (loc (std::prod-cdr (std::prod-cdr (cdr x)))))
                     (and (vl-stmt-p stmt)
                          (vl-atts-p atts)
                          (vl-location-p loc))))))
