Signature

Arguments

Definitions and Theorems

Function: funinfo-dead

(defun
     funinfo-dead (info)
     (declare (xargs :guard (funinfop info)))
     (let ((__function__ 'funinfo-dead))
          (declare (ignorable __function__))
          (change-funinfo info
                          :body (block-dead (funinfo->body info)))))
