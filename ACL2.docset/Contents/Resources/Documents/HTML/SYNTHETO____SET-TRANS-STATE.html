Macro: set-trans-state

(defmacro set-trans-state (tstate)
          (declare (xargs :guard (trans-statep tstate)))
          (cons 'table
                (cons 'trans-state-table
                      (cons ':state
                            (cons (cons 'quote (cons tstate 'nil))
                                  'nil)))))
