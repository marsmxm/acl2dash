Function: vl-find-define

(defun
     vl-find-define (name x)
     (declare (xargs :guard (and (stringp name) (vl-defines-p x))))
     (let ((__function__ 'vl-find-define))
          (declare (ignorable __function__))
          (cond ((atom x) nil)
                ((equal (string-fix name)
                        (vl-define->name (car x)))
                 (vl-define-fix (car x)))
                (t (vl-find-define name (cdr x))))))

(defthm return-type-of-vl-find-define
        (b* ((guts (vl-find-define name x)))

        :rule-classes :rewrite)
