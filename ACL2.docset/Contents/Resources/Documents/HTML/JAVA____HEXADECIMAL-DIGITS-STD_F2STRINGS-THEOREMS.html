Theorem: hex-digitp-of-char-code

(defthm hex-digitp-of-char-code
        (equal (hex-digitp (char-code char))
               (str::hex-digit-char-p char)))

Theorem: hex-digit-listp-of-chars=>nats

(defthm hex-digit-listp-of-chars=>nats
        (equal (hex-digit-listp (chars=>nats chars))
               (str::hex-digit-char-listp chars)))
