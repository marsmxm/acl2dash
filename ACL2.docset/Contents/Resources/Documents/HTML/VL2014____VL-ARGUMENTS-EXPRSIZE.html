Function: vl-arguments-exprsize

(defun
 vl-arguments-exprsize
 (x ss ctx inst warnings)
 (declare (xargs :guard (and (vl-arguments-p x)
                             (vl-scopestack-p ss)
                             (vl-context-p ctx)
                             (vl-modinst-p inst)
                             (vl-warninglist-p warnings))))
 (let
  ((__function__ 'vl-arguments-exprsize))
  (declare (ignorable __function__))
  (b*
   ((x (vl-arguments-fix x))
    (warnings (vl-warninglist-fix warnings)))
   (vl-arguments-case
     x
     :vl-arguments-named
     (b* (((mv successp warnings args-prime)
           (vl-namedarglist-exprsize x.args ss ctx warnings))
          (x-prime (change-vl-arguments-named x
                                              :args args-prime)))
         (mv successp warnings x-prime))
     :vl-arguments-plain
     (b* (((vl-modinst inst))
          ((mv submod submod-ss)
           (vl-scopestack-find-definition/ss inst.modname ss))
          ((mv ports submod-ss)
           (b* (((unless (and submod
                              (or (eq (tag submod) :vl-module)
                                  (eq (tag submod) :vl-interface))))
                 (mv nil submod-ss))
                (submod-ss (vl-scopestack-push submod submod-ss))
                (ports (if (eq (tag submod) :vl-module)
                           (vl-module->ports submod)
                           (vl-interface->ports submod))))
               (mv ports submod-ss)))
          ((mv successp warnings args-prime)
           (vl-plainarglist-exprsize
                x.args ss ctx ports submod-ss warnings))
          (x-prime (change-vl-arguments-plain x
                                              :args args-prime)))
         (mv successp warnings x-prime))))))
