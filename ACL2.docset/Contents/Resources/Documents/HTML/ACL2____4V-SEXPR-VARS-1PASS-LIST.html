(defun 4v-sexpr-vars-1pass-list (x)
       (declare (xargs :guard t))
       (mbe :logic (4v-sexpr-vars-list x)
            :exec (b* (((mv seen vars)
                        (4v-sexpr-vars-1pass-list-exec x nil nil)))
