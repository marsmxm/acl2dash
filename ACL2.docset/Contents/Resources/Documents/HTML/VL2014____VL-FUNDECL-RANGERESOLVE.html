(defun
 vl-fundecl-rangeresolve (x ss warnings)
 (declare (xargs :guard (and (vl-fundecl-p x)
                             (vl-scopestack-p ss)
                             (vl-warninglist-p warnings))))
 (declare (ignorable ss))
 (let
  ((__function__ 'vl-fundecl-rangeresolve))
  (declare (ignorable __function__))
  (b*
     ((x (vl-fundecl-fix x))
      (warnings (vl-warninglist-fix warnings)))
     (b* (((vl-fundecl x) x)
