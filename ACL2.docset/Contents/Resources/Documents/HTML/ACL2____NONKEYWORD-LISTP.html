(defun nonkeyword-listp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'nonkeyword-listp))
            (declare (ignorable __function__))
            (if (consp x)
                (and (not (keywordp (car x)))
                     (nonkeyword-listp (cdr x)))
                (null x))))
