Function: expdata-gen-back-guard-instances-to-mv-nth

(defun
 expdata-gen-back-guard-instances-to-mv-nth
 (term res-surjmaps wrld)
 (declare
      (xargs :guard (and (pseudo-termp term)
                         (expdata-pos-surjmap-alistp res-surjmaps)
                         (plist-worldp wrld))))
 (let
  ((__function__ 'expdata-gen-back-guard-instances-to-mv-nth))
  (declare (ignorable __function__))
  (b*
   (((when (endp res-surjmaps)) nil)
    (j (caar res-surjmaps))
    (surjmap (cdar res-surjmaps))
    (back-guard (expdata-surjmap->back-guard surjmap))
    (var (expdata-formal-of-back surjmap wrld))
    (mv-nth-of-term (fcons-term* 'mv-nth (1- j) term))
    (instance
       (cons ':instance
             (cons back-guard
                   (cons ':extra-bindings-ok
                         (cons (cons var (cons mv-nth-of-term 'nil))
                               'nil)))))
    (instances (expdata-gen-back-guard-instances-to-mv-nth
                    term (cdr res-surjmaps)
                    wrld)))
   (cons instance instances))))
