(defun vl-jp-regularport-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-regularport-p x)))
       (let ((__function__ 'vl-jp-regularport))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-regularport)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-maybe-string (vl-regularport->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"expr\": ")
                       (vl-jp-maybe-expr (vl-regularport->expr x))
                       (vl-println? ", ")
                       (vl-print-str "\"loc\": ")
                       (vl-jp-location (vl-regularport->loc x))
                       (vl-println? "}"))))
