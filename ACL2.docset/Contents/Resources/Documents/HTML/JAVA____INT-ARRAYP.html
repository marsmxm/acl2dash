Function: int-arrayp

(defun int-arrayp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'int-arrayp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :int-array)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((components (std::da-nth 0 (cdr x))))
                     (and (int-value-listp components)
                          (< (len components) (expt 2 31)))))))
