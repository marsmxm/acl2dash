Raw constructor for 4vecs, using cons.

(make-4vec &key upper lower) builds a 4vec with the given upper and lower.

Macro: make-4vec

(defmacro make-4vec (&key upper lower)
          (cons '4vec
                (cons upper (cons lower 'nil))))

Function: 4vec

(defun 4vec (upper lower)
       (declare (xargs :guard (and (integerp upper)
                                   (integerp lower))))
       (b* (((the integer upper) (lifix upper))
            ((the integer lower) (lifix lower)))
           (if (eql upper lower)
               upper (cons upper lower))))

Theorem: 4vec-equal

(defthm 4vec-equal
        (equal (equal (4vec upper lower) x)
               (and (4vec-p x)
                    (equal (4vec->upper x) (ifix upper))
                    (equal (4vec->lower x) (ifix lower)))))
