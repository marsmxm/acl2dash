(defun
   vl-jp-interfaceport-fn (x ps)
   (declare (xargs :stobjs (ps)))
   (declare (xargs :guard (vl-interfaceport-p x)))
   (let ((__function__ 'vl-jp-interfaceport))
        (declare (ignorable __function__))
        (vl-ps-seq (vl-print "{\"tag\": ")
                   (jp-sym ':vl-interfaceport)
                   (vl-print ", ")
                   (vl-print-str "\"name\": ")

                   (vl-println? ", ")

                   (jp-str (vl-interfaceport->ifname x))
                   (vl-println? ", ")

                   (vl-jp-location (vl-interfaceport->loc x))
                   (vl-println? ", ")
                   (vl-print-str "\"modport\": ")
                   (jp-maybe-string (vl-interfaceport->modport x))
                   (vl-println? ", ")
                   (vl-print-str "\"udims\": ")
                   (vl-jp-dimensionlist (vl-interfaceport->udims x))
                   (vl-println? "}"))))
