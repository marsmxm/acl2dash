Function: rule-rename-rule

(defun
     rule-rename-rule (rule oldname newname)
     (declare (xargs :guard (and (rulep rule)
                                 (rulenamep oldname)
                                 (rulenamep newname))))
     (make-rule
          :name (if (equal (rule->name rule) oldname)
                    newname (rule->name rule))
          :incremental (rule->incremental rule)
          :definiens (alternation-rename-rule (rule->definiens rule)
                                              oldname newname)))
