Function: vl-find-file-aux

(defun vl-find-file-aux
       (filename searchpath state)
       (declare (xargs :guard (and (stringp filename)
                                   (string-listp searchpath)
                                   (state-p state))))
       (b* (((when (atom searchpath))
             (mv nil state))
            (attempt (vl-extend-pathname (car searchpath)
                                         filename))
            ((mv channel state)
             (open-input-channel attempt
                                 :character state))
            ((unless channel)
             (vl-find-file-aux filename (cdr searchpath)
                               state))
            (state (close-input-channel channel state)))
           (mv attempt state)))

 state-p1-of-vl-find-file-aux
 (implies
  (force (state-p1 state))
  (state-p1 (mv-nth 1
