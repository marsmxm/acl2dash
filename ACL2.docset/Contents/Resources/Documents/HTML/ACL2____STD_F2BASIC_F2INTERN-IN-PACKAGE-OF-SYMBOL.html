Theorem: equal-of-intern-in-package-of-symbols

(defthm
     equal-of-intern-in-package-of-symbols
     (implies (and (stringp a)
                   (stringp b)
                   (symbolp in-pkg))
              (equal (equal (intern-in-package-of-symbol a in-pkg)
