Function: funtypep

(defun
  funtypep (x)
  (declare (xargs :guard t))
  (let ((__function__ 'funtypep))
       (declare (ignorable __function__))
       (and (consp x)
            (eq (car x) :funtype)
            (mbe :logic (and (alistp (cdr x))
                             (equal (strip-cars (cdr x)) '(in out)))
                 :exec (fty::alist-with-carsp (cdr x)
                                              '(in out)))
            (b* ((in (cdr (std::da-nth 0 (cdr x))))
                 (out (cdr (std::da-nth 1 (cdr x)))))
                (and (natp in) (natp out))))))
