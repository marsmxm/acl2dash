Extend an alist by binding the names of vl-programs to their definitions.

Signature
(vl-programlist-alist x acc) → alist
Arguments
x — Guard (vl-programlist-p x).
Returns
alist — Type (equal (vl-program-alist-p alist) (vl-program-alist-p acc)).

This can be used as an alternative to vl-find-program when you need to perform a lot of lookups.

Definitions and Theorems

Function: vl-programlist-alist

(defun vl-programlist-alist (x acc)
       (declare (xargs :guard (vl-programlist-p x)))
       (let ((__function__ 'vl-programlist-alist))
            (declare (ignorable __function__))
            (if (atom x)
                acc
                (cons (cons (vl-program->name (car x))
                            (vl-program-fix (car x)))
                      (vl-programlist-alist (cdr x) acc)))))

Theorem: return-type-of-vl-programlist-alist

(defthm return-type-of-vl-programlist-alist
        (b* ((alist (vl-programlist-alist x acc)))
            (equal (vl-program-alist-p alist)
                   (vl-program-alist-p acc)))
        :rule-classes :rewrite)

Theorem: lookup-in-vl-programlist-alist-acc-elim

(defthm
 lookup-in-vl-programlist-alist-acc-elim
 (implies
     (syntaxp (not (equal acc ''nil)))
     (equal (hons-assoc-equal name (vl-programlist-alist x acc))
            (or (hons-assoc-equal name (vl-programlist-alist x nil))
                (hons-assoc-equal name acc)))))

Theorem: lookup-in-vl-programlist-alist-fast

(defthm
    lookup-in-vl-programlist-alist-fast
    (implies
         (stringp name)
         (equal (hons-assoc-equal name (vl-programlist-alist x nil))
                (let ((val (vl-find-program name x)))
                     (and val (cons name val))))))

Theorem: vl-programlist-alist-of-vl-programlist-fix-x

(defthm vl-programlist-alist-of-vl-programlist-fix-x
        (equal (vl-programlist-alist (vl-programlist-fix x)
                                     acc)
               (vl-programlist-alist x acc)))
