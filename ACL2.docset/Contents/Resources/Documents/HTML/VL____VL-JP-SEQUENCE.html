Function: vl-jp-sequence-fn

(defun vl-jp-sequence-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-sequence-p x)))
       (let ((__function__ 'vl-jp-sequence))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-sequence)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-sequence->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"ports\": ")
                       (vl-jp-propportlist (vl-sequence->ports x))
                       (vl-println? ", ")
                       (vl-print-str "\"decls\": ")
                       (vl-jp-vardecllist (vl-sequence->decls x))
                       (vl-println? ", ")
                       (vl-print-str "\"expr\": ")
                       (vl-jp-propexpr (vl-sequence->expr x))
                       (vl-println? ", ")
                       (vl-print-str "\"loc\": ")
                       (vl-jp-location (vl-sequence->loc x))
                       (vl-println? "}"))))
