(defun
 vl-property-elaborate-fn
 (x elabindex reclimit config)
 (declare (xargs :stobjs (elabindex)))
 (declare (xargs :guard (and (vl-property-p x)
                             (natp reclimit)
                             (vl-simpconfig-p config))))
 (let
  ((__function__ 'vl-property-elaborate))
  (declare (ignorable __function__))
  (b*
   (((vl-property x) (vl-property-fix x)))
   (b*
      (((mv ok warnings ports elabindex)
        (vl-propportlist-elaborate-fn
             x.ports elabindex reclimit config))
       ((mv ok1 warnings1 decls elabindex)
        (vl-vardecllist-elaborate-fn
             x.decls elabindex reclimit config))
       (ok (and ok1 ok))
       (warnings (append-without-guard warnings1 warnings))
       ((mv ok1 warnings1 spec elabindex)
        (vl-propspec-elaborate-fn x.spec elabindex reclimit config))
       (ok (and ok1 ok))
       (warnings (append-without-guard warnings1 warnings)))
      (mv ok warnings
          (change-vl-property x
                              :ports ports
                              :decls decls
                              :spec spec)
          elabindex)))))
