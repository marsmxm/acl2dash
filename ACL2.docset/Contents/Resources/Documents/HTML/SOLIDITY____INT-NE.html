Function: int-ne

(defun
     int-ne (left right)
     (declare (xargs :guard (and (intp left) (intp right))))
     (declare (xargs :guard (= (int->size left) (int->size right))))
     (b* ((x (int->value left))
          (y (int->value right)))
         (bool (not (equal x y)))))

(defthm boolp-of-int-ne
        (b* ((result (int-ne left right)))
            (boolp result))
        :rule-classes :rewrite)
