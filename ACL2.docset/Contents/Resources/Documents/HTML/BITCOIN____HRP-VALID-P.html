Function: hrp-valid-p

(defun
    hrp-valid-p (s)
    (declare (xargs :guard (stringp s)))
    (let ((__function__ 'hrp-valid-p))
         (declare (ignorable __function__))
         (and (hrp-valid-string-length-p s)
              (loop$ for x in (explode s)
                     always
                     (and (characterp x)
                          (hrp-valid-char-code-p (char-code x)))))))
