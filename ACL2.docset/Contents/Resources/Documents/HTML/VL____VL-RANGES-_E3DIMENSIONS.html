Function: vl-ranges->dimensions

(defun
 vl-ranges->dimensions (x)
 (declare (xargs :guard (vl-rangelist-p x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-ranges->dimensions))
  (declare (ignorable __function__))
  (mbe :logic (if (consp x)
                  (cons (vl-range->dimension (car x))
                        (vl-ranges->dimensions (cdr x)))
                  nil)
       :exec
       (if (atom x)
           nil
           (with-local-nrev (vl-ranges->dimensions-nrev x nrev))))))
