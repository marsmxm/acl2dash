Theorem: nth-4v-sexpr-eval-list

(defthm nth-4v-sexpr-eval-list
        (equal (nth n (4v-sexpr-eval-list x env))
               (if (< (nfix n) (len x))
                   (4v-sexpr-eval (nth n x) env)
                   nil)))
