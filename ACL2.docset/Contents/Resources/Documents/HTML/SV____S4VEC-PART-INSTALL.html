(defun
 s4vec-part-install (lsb width in val)
 (declare (xargs :guard (and (s4vec-p lsb)
                             (s4vec-p width)
                             (s4vec-p in)
                             (s4vec-p val))))
 (let
  ((__function__ 's4vec-part-install))
  (declare (ignorable __function__))
  (b*
     (((unless (and (s4vec-2vec-p lsb)
                    (s4vec-2vec-p width)))
       (s4vec-x))
      (widthint (s4vec->upper width))
      ((when (sparseint-< widthint 0))
       (s4vec-x))
      (lsbint (s4vec->upper lsb))
      ((unless (sparseint-< lsbint 0))
       (s4vec-concat
            lsb in
            (s4vec-concat
                 width val
                 (s4vec-rsh (s2vec (sparseint-plus lsbint widthint))
                            in))))
      (neg-lsb (sparseint-unary-minus lsbint))
      ((unless (sparseint-< neg-lsb widthint))
       (s4vec-fix in))
      (width+lsb (sparseint-plus lsbint widthint)))
     (s4vec-concat (s2vec width+lsb)
                   (s4vec-rsh (s2vec neg-lsb) val)
                   (s4vec-rsh (s2vec width+lsb) in)))))
