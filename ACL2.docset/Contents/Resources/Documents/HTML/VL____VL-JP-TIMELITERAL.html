Function: vl-jp-timeliteral-fn

(defun vl-jp-timeliteral-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-timeliteral-p x)))
       (let ((__function__ 'vl-jp-timeliteral))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-timeliteral)
                       (vl-print ", ")
                       (vl-print-str "\"quantity\": ")
                       (jp-str (vl-timeliteral->quantity x))
                       (vl-println? ", ")
                       (vl-print-str "\"units\": ")
                       (vl-jp-timeunit (vl-timeliteral->units x))
                       (vl-println? "}"))))
