Function: vl-collect-unsized-ints

(defun vl-collect-unsized-ints (x ss scopes)
       (declare (xargs :guard (and (vl-exprlist-p x)
                                   (vl-scopestack-p ss)
                                   (vl-elabscopes-p scopes))))
       (let ((__function__ 'vl-collect-unsized-ints))
            (declare (ignorable __function__))
            (b* (((when (atom x)) nil)
                 (x1 (vl-expr-fix (car x)))
                 (keep (vl-is-unsized-int x1 ss scopes)))
                (if keep
                    (cons x1
                          (vl-collect-unsized-ints (cdr x)
                                                   ss scopes))
                    (vl-collect-unsized-ints (cdr x)
                                             ss scopes)))))
