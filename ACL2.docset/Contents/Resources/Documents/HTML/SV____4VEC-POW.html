Function: 4vec-pow

(defun 4vec-pow (base exp)
       (declare (xargs :guard (and (4vec-p base) (4vec-p exp))))
       (let ((__function__ '4vec-pow))
            (declare (ignorable __function__))
            (if (and (2vec-p base) (2vec-p exp))
                (b* ((base (2vec->val base))
                     (exp (2vec->val exp))
                     ((when (or (natp exp)
                                (eql base 1)
                                (eql base -1)))
                      (2vec (expt base exp)))
                     ((when (eql base 0)) (4vec-x)))
                    (2vec 0))
                (4vec-x))))
