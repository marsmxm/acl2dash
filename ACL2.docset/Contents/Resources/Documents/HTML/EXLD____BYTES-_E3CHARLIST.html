Function: bytes->charlist

(defun bytes->charlist (bytes)
       (declare (xargs :guard (byte-listp bytes)))
       (let ((__function__ 'bytes->charlist))
            (declare (ignorable __function__))
            (if (endp bytes)
                nil
                (cons (code-char (car bytes))
                      (bytes->charlist (cdr bytes))))))
