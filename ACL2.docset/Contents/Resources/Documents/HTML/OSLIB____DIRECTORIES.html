Collect paths that are directories.

Signature
(directories paths &key (state 'state)) 
  → 
(mv err ans new-state)
Arguments
paths — Guard (string-listp paths).
Returns
err — NIL on success, or an error msg on failure.
ans — Meaningful only when there is no error.
    Type (string-listp ans), given (string-listp paths).
new-state — Type (state-p1 new-state), given (force (state-p1 state)).

Definitions and Theorems

Function: directories-exec-fn

(defun directories-exec-fn (paths acc state)
       (declare (xargs :stobjs (state)))
       (declare (xargs :guard (string-listp paths)))
       "Tail recursive version for execution."
       (let ((__function__ 'directories-exec))
            (declare (ignorable __function__))
            (b* (((when (atom paths)) (mv nil acc state))
                 ((mv err directory-p state)
                  (directory-p (car paths)))
                 ((when err) (mv err acc state))
                 (acc (if directory-p (cons (car paths) acc)
                          acc)))
                (directories-exec (cdr paths) acc))))

Function: directories-fn

(defun
  directories-fn (paths state)
  (declare (xargs :stobjs (state)))
  (declare (xargs :guard (string-listp paths)))
  (let ((__function__ 'directories))
       (declare (ignorable __function__))
       (mbe :logic (b* (((when (atom paths)) (mv nil nil state))
                        ((mv err directory-p state)
                         (directory-p (car paths)))
                        ((when err) (mv err nil state))
                        ((mv err rest state)
                         (directories (cdr paths)))
                        (ans (if directory-p (cons (car paths) rest)
                                 rest)))
                       (mv err ans state))
            :exec (b* (((mv err acc state)
                        (directories-exec paths nil)))
                      (mv err (reverse acc) state)))))

Theorem: directories-exec-removal

(defthm directories-exec-removal
        (equal (directories-exec paths acc)
               (b* (((mv err ans state)
                     (directories paths)))
                   (mv err (revappend ans acc) state))))

Theorem: true-listp-of-directories

(defthm true-listp-of-directories

        :rule-classes :type-prescription)

Theorem: w-state-of-directories

(defthm w-state-of-directories
        (b* (((mv ?err ?ans ?new-state)
              (directories-fn paths state)))
            (equal (w new-state) (w state))))
