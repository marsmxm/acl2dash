Function: vl-keep-modules

(defun
 vl-keep-modules (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-modulelist-p x))))
 (let
  ((__function__ 'vl-keep-modules))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-module->name (car x))
                               (string-list-fix names))
                 (cons (vl-module-fix (car x))
                       (vl-keep-modules names (cdr x))))
                (t (vl-keep-modules names (cdr x))))
   :exec
   (b*
    (((when (or (atom names) (atom x))) nil)
     ((unless (longer-than-p 10 names))
      (vl-slow-keep-modules names x))
     (fal (make-lookup-alist names))
     (ans (with-local-nrev (vl-fast-keep-modules names fal x nrev)))
     (- (fast-alist-free fal)))
    ans))))
