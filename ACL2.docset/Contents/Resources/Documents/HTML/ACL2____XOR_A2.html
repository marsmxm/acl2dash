Theorem: xor*-with-nil

(defthm xor*-with-nil
        (and (iff (xor* x nil) x)
             (iff (xor* nil x) x)))

Theorem: xor*-when-other

(defthm xor*-when-other
        (implies y
                 (and (equal (xor* x y) (not x))
                      (equal (xor* y x) (not x)))))
