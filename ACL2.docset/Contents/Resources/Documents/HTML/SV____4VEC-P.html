Function: 4vec-p

(defun 4vec-p (x)
       (declare (xargs :guard t))
       (let ((__function__ '4vec-p))
            (declare (ignorable __function__))
            (or (integerp x)
                (and (consp x)
                     (integerp (car x))
                     (integerp (cdr x))
                     (not (equal (car x) (cdr x)))))))
