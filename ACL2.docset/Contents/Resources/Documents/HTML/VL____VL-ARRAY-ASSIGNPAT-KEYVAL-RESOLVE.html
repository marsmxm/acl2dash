(defun
 vl-array-assignpat-keyval-resolve
 (lsb msb x)
 (declare (xargs :guard (and (integerp lsb)
                             (integerp msb)
                             (vl-svex-keyvallist-p x))))
 (let
  ((__function__ 'vl-array-assignpat-keyval-resolve))
  (declare (ignorable __function__))
  (b*
   ((first (or (vl-svex-keyval-index-lookup msb x)
               (vl-svex-keyval-default-lookup x)))
    (err1
        (and (not first)
             (vmsg "No assignment pattern entry for array index ~x0"
                   (lifix msb))))
    ((when (eql (lifix lsb) (lifix msb)))
     (mv err1 (list (or first (svex-x)))))
    (next (if (< (lifix lsb) (lifix msb))
              (- (lifix msb) 1)
              (+ (lifix msb) 1)))
    ((mv err rest)
     (vl-array-assignpat-keyval-resolve lsb next x)))
   (mv (or err1 err)
       (cons (or first (svex-x)) rest)))))
