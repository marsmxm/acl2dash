Function: cutinfo->truth

(defun cutinfo->truth (x)
       (declare (xargs :guard (cutinfo-p x)))
       (mbe :logic (let ((x (cutinfo-fix x)))
                        (part-select x :low 0 :width 16))
            :exec (the (unsigned-byte 16)
                       (logand (the (unsigned-byte 16) 65535)
                               (the (unsigned-byte 32) x)))))
