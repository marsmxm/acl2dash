Function: vl-namedarglist-wildelim

(defun vl-namedarglist-wildelim
       (x ctx warnings)
       (declare (xargs :guard (and (vl-namedarglist-p x)
                                   (vl-context-p ctx)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-namedarglist-wildelim))
            (declare (ignorable __function__))
            (b* (((when (atom x)) (mv (ok) x))
                 ((mv warnings car-prime)
                  (vl-namedarg-wildelim (car x)
                                        ctx warnings))
                 ((mv warnings cdr-prime)
                  (vl-namedarglist-wildelim (cdr x)
                                            ctx warnings)))
                (mv warnings (cons car-prime cdr-prime)))))
