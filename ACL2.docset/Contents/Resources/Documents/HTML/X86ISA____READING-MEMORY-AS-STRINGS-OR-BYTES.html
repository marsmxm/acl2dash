 read-memory-zero-terminated
 (ptr x86 acc)
 (declare (xargs :stobjs (x86)))
 (declare (type (signed-byte 49) ptr))
 (declare (xargs :guard (and (integerp ptr)
                             (<= (- *2^47*) ptr)
                             (<= ptr *2^47*)
                             (nat-listp acc))))
 (let
  ((__function__ 'read-memory-zero-terminated))
  (declare (ignorable __function__))
  (if
   (signed-byte-p 48 ptr)
   (b*
    (((mv flg (the (unsigned-byte 8) mem-val)

   (mv t (reverse acc) x86))))

Function: read-bytes-from-memory

(defun
 read-bytes-from-memory
 (ptr nbytes x86 acc)
 (declare (xargs :stobjs (x86)))
 (declare (type (signed-byte 49) ptr)
          (type (unsigned-byte 48) nbytes))
 (declare (xargs :split-types t
                 :guard (and (integerp nbytes)
                             (<= 0 nbytes)
                             (integerp ptr)
                             (<= (- *2^47*) ptr)
                             (< (+ ptr nbytes) *2^47*)
                             (nat-listp acc))))
 (let
  ((__function__ 'read-bytes-from-memory))
  (declare (ignorable __function__))
  (if
   (mbt (and (integerp nbytes)
             (integerp ptr)
             (<= (- *2^47*) ptr)
             (< (+ ptr nbytes) *2^47*)))
   (if
    (zp nbytes)
    (mv nil (reverse acc) x86)
    (b* (((mv flg (the (unsigned-byte 8) byte)
              x86)
          (rml08 ptr :r x86))
         ((when flg) (mv flg nil x86)))
        (read-bytes-from-memory (the (signed-byte 49) (1+ ptr))
                                (the (unsigned-byte 48) (1- nbytes))
                                x86 (cons byte acc))))
   (mv t (reverse acc) x86))))
