Function: get-defined-type-names

(defun
 get-defined-type-names (tops)
 (declare (xargs :guard (toplevel-listp tops)))
 (let
  ((__function__ 'get-defined-type-names))
  (declare (ignorable __function__))
  (b*
     (((when (endp tops)) nil)
      (top (car tops))
      (names (toplevel-case
                  top
                  :type (list (type-definition->name top.get))
                  :types (get-defined-type-names-in-type-definitions
                              (type-recursion->definitions top.get))
                  :function nil
                  :functions nil
                  :specification nil
                  :theorem nil
                  :transform nil))
      (more-names (get-defined-type-names (cdr tops))))
     (append names more-names))))

Theorem: defined-type-name-has-definition

(defthm
     defined-type-name-has-definition
     (implies (member-equal name (get-defined-type-names tops))
              (type-definitionp (get-type-definition name tops))))
