Function: index-listp

(defun index-listp (x numvars)
       (declare (xargs :guard (natp numvars)))
       (let ((__function__ 'index-listp))
            (declare (ignorable __function__))
            (if (atom x)
                (eq x nil)
                (and (natp (car x))
                     (< (car x) (lnfix numvars))
                     (index-listp (cdr x) numvars)))))
