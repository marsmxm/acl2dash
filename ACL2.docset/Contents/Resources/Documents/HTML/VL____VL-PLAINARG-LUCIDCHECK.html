Function: vl-plainarg-lucidcheck

(defun
 vl-plainarg-lucidcheck (x ss st ctx)
 (declare (xargs :guard (and (vl-plainarg-p x)
                             (vl-scopestack-p ss)
                             (vl-lucidstate-p st)
                             (vl-lucidctx-p ctx))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-plainarg-lucidcheck))
  (declare (ignorable __function__))
  (b*
   ((x (vl-plainarg-fix x))
    (ss (vl-scopestack-fix ss))
    (st (vl-lucidstate-fix st)))
   (b*
    (((vl-plainarg x))
     (ctx (change-vl-lucidctx ctx
                              :portname x.portname))
     ((unless x.expr) st)
     (mpname
      (let
       ((look
          (cdr (assoc-equal "VL_REMOVED_EXPLICIT_MODPORT" x.atts))))
       (and look (vl-string-expr->value look))))
     (ifname
      (and
        mpname
        (let ((look (cdr (assoc-equal "VL_INTERFACE_NAME" x.atts))))
             (and look (vl-string-expr->value look)))))
     (st (if ifname
             (vl-lucidst-mark-modport ifname mpname ss st ctx)
             st)))
    (case
      x.dir
      (:vl-input (vl-rhsexpr-lucidcheck x.expr ss st ctx))
      (:vl-output (vl-lhsexpr-lucidcheck x.expr ss st ctx))
      (otherwise (b* ((st (vl-rhsexpr-lucidcheck x.expr ss st ctx))
                      (st (vl-lhsexpr-lucidcheck x.expr ss st ctx)))
                     st)))))))
