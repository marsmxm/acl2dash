(defun
 vl-ports-resolve-interfaces
 (x ifport-alist)

      (xargs :guard (and (vl-portlist-p x)
                         (vl-interfaceport-alist-p ifport-alist))))
 (let
    ((__function__ 'vl-ports-resolve-interfaces))
    (declare (ignorable __function__))
    (b* (((when (atom x)) nil)
         (x1 (vl-port-fix (car x)))
         (rest (vl-ports-resolve-interfaces (cdr x)
                                            ifport-alist))

          (cons x1 rest))
         ((vl-regularport x1))
