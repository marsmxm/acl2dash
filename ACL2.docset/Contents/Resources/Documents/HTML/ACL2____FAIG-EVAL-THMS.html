Theorem: aig-env-equiv-implies-equal-faig-eval-2

(defthm aig-env-equiv-implies-equal-faig-eval-2
        (implies (aig-env-equiv env env-equiv)

Theorem: faig-eval-append-when-not-intersecting-alist-keys

(defthm faig-eval-append-when-not-intersecting-alist-keys
        (implies (not (intersectp-equal (alist-keys env)
                                        (faig-vars x)))
                 (equal (faig-eval x (append env rest))
                        (faig-eval x rest))))
