Function: pointerp

(defun
   pointerp (x)
   (declare (xargs :guard t))
   (let ((__function__ 'pointerp))
        (declare (ignorable __function__))
        (and (consp x)
             (eq (car x) :pointer)
             (mbe :logic (and (alistp (cdr x))
                              (equal (strip-cars (cdr x))
                                     '(address? reftype)))
                  :exec (fty::alist-with-carsp (cdr x)
                                               '(address? reftype)))
             (b* ((address? (cdr (std::da-nth 0 (cdr x))))
                  (reftype (cdr (std::da-nth 1 (cdr x)))))
                 (and (address-optionp address?)
                      (typep reftype))))))

(defthm consp-when-pointerp
        (implies (pointerp x) (consp x))
