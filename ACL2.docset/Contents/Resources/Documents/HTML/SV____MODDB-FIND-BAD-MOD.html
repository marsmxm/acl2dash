Function: moddb-find-bad-mod

(defun
     moddb-find-bad-mod (n moddb)
     (declare (xargs :stobjs (moddb)))
     (declare (xargs :guard (natp n)))
     (declare (xargs :guard (and (<= n (moddb->nmods moddb))
                                 (<= (moddb->nmods moddb)
                                     (moddb->mods-length moddb)))))
     (let ((__function__ 'moddb-find-bad-mod))
          (declare (ignorable __function__))
          (if (zp n)
              nil
              (if (moddb-mod-ok (1- n) moddb)
                  (moddb-find-bad-mod (1- n) moddb)
                  (1- n)))))
