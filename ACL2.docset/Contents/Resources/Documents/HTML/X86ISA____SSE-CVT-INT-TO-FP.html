 (let
  ((__function__ 'sse-cvt-int-to-fp))
  (declare (ignorable __function__))
  (b*
   ((mxcsr (mbe :logic (loghead 32 mxcsr)
                :exec mxcsr))

                          nil nil nil rc exp-width frac-width)))
