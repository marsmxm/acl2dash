Function: def-parse-order-permutationp

(defun def-parse-order-permutationp (order)
       (declare (xargs :guard (pos-listp order)))
       (let ((__function__ 'def-parse-order-permutationp))
            (declare (ignorable __function__))
            (def-parse-order-permutationp-aux
                 order
                 (integers-from-to 1 (len order)))))
