Theorem: chars=>nats-of-nats=>chars

   chars=>nats-of-nats=>chars
   (implies (unsigned-byte-listp 8 (true-list-fix nats))
            (equal (chars=>nats (nats=>chars nats))
                   (true-list-fix nats)))
