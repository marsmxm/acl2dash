Function: vl-relocate-assigns

(defun
 vl-relocate-assigns (loc x)
 (declare (xargs :guard (and (vl-location-p loc)
                             (vl-assignlist-p x))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-relocate-assigns))
  (declare (ignorable __function__))
  (mbe
     :logic (if (consp x)
                (cons (change-vl-assign (car x) :loc loc)
                      (vl-relocate-assigns loc (cdr x)))
                nil)
     :exec
     (if (atom x)
         nil
         (with-local-nrev (vl-relocate-assigns-nrev loc x nrev))))))
