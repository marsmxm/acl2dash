Function: pprint-param-declon

(defun pprint-param-declon (param)
       (declare (xargs :guard (param-declonp param)))
       (let ((__function__ 'pprint-param-declon))
            (declare (ignorable __function__))
            (b* (((param-declon param) param))
                (msg "~@0 ~@1" (pprint-tyspecseq param.type)
                     (pprint-declor param.declor)))))
