Theorem: bebytes=>nat-injectivity*

(defthm bebytes=>nat-injectivity*
        (implies (and (equal (trim-bendian* (byte-list-fix digits1))
                             digits1)
                      (equal (trim-bendian* (byte-list-fix digits2))
                             digits2))
                 (equal (equal (bebytes=>nat digits1)
                               (bebytes=>nat digits2))
                        (equal digits1 digits2))))
