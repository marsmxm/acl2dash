Function: lhs-decomp

(defun
 lhs-decomp (x)
 (declare (xargs :guard (lhs-p x)))
 (let
  ((__function__ 'lhs-decomp))
  (declare (ignorable __function__))
  (mbe
     :logic (mv (lhs-first x) (lhs-rest x))
     :exec (if (consp x)
               (if (consp (cdr x))
                   (if (lhrange-combinable (car x)
                                           (lhrange->atom (cadr x)))

                                          (lhrange->w (cadr x)))

                                       (cddr x))
                       (mv (car x) (cdr x)))
