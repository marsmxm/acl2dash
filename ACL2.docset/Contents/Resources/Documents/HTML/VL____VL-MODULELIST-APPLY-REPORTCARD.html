Function: vl-modulelist-apply-reportcard

(defun
 vl-modulelist-apply-reportcard
 (x reportcard)
 (declare (xargs :guard (and (vl-modulelist-p x)
                             (vl-reportcard-p reportcard))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-modulelist-apply-reportcard))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-module-apply-reportcard (car x)
                                                reportcard)
                    (vl-modulelist-apply-reportcard (cdr x)
                                                    reportcard))
              nil)
   :exec
   (if
    (atom x)
    nil
    (with-local-nrev
        (vl-modulelist-apply-reportcard-nrev x reportcard nrev))))))
