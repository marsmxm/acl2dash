Function: observability-size-check

(defun
     observability-size-check
     (lit-size full-size config)
     (declare (xargs :guard (and (natp lit-size)
                                 (natp full-size)
                                 (observability-config-p config))))
     (let ((__function__ 'observability-size-check))
          (declare (ignorable __function__))
          (b* (((observability-config config)))
              (and (or (not config.hyp-max-size)
                       (<= (lnfix lit-size)
                           config.hyp-max-size))
                   (or (not config.concl-min-size)
                       (>= (lnfix full-size)
                           config.concl-min-size))
                   (<= (* (numerator config.min-ratio)
                          (lnfix lit-size))
                       (* (denominator config.min-ratio)
                          (lnfix full-size)))))))
