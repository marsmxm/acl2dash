Function: vls-get-origsrc

(defun
 vls-get-origsrc (origname data)
 (declare (xargs :guard (and (stringp origname)
                             (vls-data-p data))))
 (let
   ((__function__ 'vls-get-origsrc))
   (declare (ignorable __function__))
   (b* (((vls-data data))
        (desc (cdr (hons-assoc-equal origname data.orig-descalist)))
        ((unless desc)
         (cat "Error: " origname " not found."))
        (ss (vl-scopestack-init data.orig))
        (ans (with-local-ps (vl-ps-update-htmlp t)
                            (vl-ppc-description desc ss))))
       ans)))
