Function: vl-port->name

(defun vl-port->name (x)
       (declare (xargs :guard (vl-port-p x)))
       (let ((__function__ 'vl-port->name))

            (b* ((x (vl-port-fix x)))
                (case (tag x)
                      (:vl-regularport (vl-regularport->name x))
