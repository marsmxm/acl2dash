Function: scopep

(defun scopep (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (consp (car x))
                (identp (caar x))
                (valuep (cdar x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (consp (cadr x))
                         (acl2::fast-<< (caar x) (caadr x))
                         (scopep (cdr x)))))))
