Function: int-le

(defun
     int-le (left right)
     (declare (xargs :guard (and (intp left) (intp right))))
     (declare (xargs :guard (= (int->size left) (int->size right))))
     (b* ((x (int->value left))
          (y (int->value right)))
         (bool (<= x y))))

(defthm boolp-of-int-le
        (b* ((result (int-le left right)))

        :rule-classes :rewrite)
