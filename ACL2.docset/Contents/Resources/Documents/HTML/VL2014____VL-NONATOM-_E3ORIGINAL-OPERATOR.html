Function: vl-nonatom->original-operator

(defun vl-nonatom->original-operator (x)
       (declare (xargs :guard (vl-expr-p x)))
       (declare (xargs :guard (not (vl-atom-p x))))
       (let ((__function__ 'vl-nonatom->original-operator))
            (declare (ignorable __function__))
            (b* ((atts (vl-nonatom->atts x))
                 (orig (cdr (hons-assoc-equal "VL_ORIG_EXPR" atts)))
                 ((when (and orig (not (vl-fast-atom-p orig))))
                  (vl-nonatom->op orig)))
                (vl-nonatom->op x))))
