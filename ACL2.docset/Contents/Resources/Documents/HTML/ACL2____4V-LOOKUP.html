Function: 4v-lookup

(defun
    4v-lookup (k env)
    (declare (xargs :guard t))
    (mbe :logic (let ((look (hons-get k env)))
                     (if look (4v-fix (cdr look)) (4vx)))
         :exec (let ((look (hons-get k env)))
                    (prog2$ (and (not look) (4v-lookup-not-found k))
                            (4v-fix (cdr look))))))
