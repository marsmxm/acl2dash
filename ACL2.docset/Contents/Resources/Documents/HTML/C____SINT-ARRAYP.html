Function: sint-arrayp

(defun sint-arrayp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'sint-arrayp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :sint-array)
                 (mbe :logic (and (alistp (cdr x))
                                  (equal (strip-cars (cdr x))
                                         '(elements)))
                      :exec (fty::alist-with-carsp (cdr x)
                                                   '(elements)))
                 (b* ((elements (cdr (std::da-nth 0 (cdr x)))))
                     (and (sint-listp elements)
                          (consp elements))))))
