Function: framep

(defun framep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'framep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :frame)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 2)
                 (b* ((term (std::da-nth 0 (cdr x)))
                      (binding (std::da-nth 1 (cdr x))))
                     (and (ttermp term)
                          (bindingp binding))))))
