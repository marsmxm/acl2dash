Function: check-toplevel-list

(defun
   check-toplevel-list (tops ctxt)
   (declare (xargs :guard (and (toplevel-listp tops)
                               (contextp ctxt))))
   (declare
        (xargs :guard (and (null (context->types ctxt))
                           (null (context->functions ctxt))
                           (omap::empty (context->variables ctxt))
                           (null (context->obligation-vars ctxt))
                           (null (context->obligation-hyps ctxt)))))
   (let ((__function__ 'check-toplevel-list))
        (declare (ignorable __function__))
        (b* (((when (endp tops)) (mv nil nil))
             ((mv err? obligs ctxt)
              (check-toplevel (car tops) ctxt))
             ((when err?) (mv err? nil))
             ((mv err? more-obligs)
              (check-toplevel-list (cdr tops) ctxt))
             ((when err?) (mv err? nil)))
            (mv nil (append obligs more-obligs)))))
