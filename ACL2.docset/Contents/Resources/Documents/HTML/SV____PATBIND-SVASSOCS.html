     patbind-svassocs
     (args acl2::forms acl2::rest-expr)
     (b* (((mv acl2::pre-bindings acl2::name rest)
           (if (and (consp (car acl2::forms))
                    (not (eq (caar acl2::forms) 'quote)))
               (mv (cons (cons 'acl2::?tmp-for-assocs
                               (cons (car acl2::forms) 'nil))
                         'nil)

                   (cons 'check-vars-not-free
                         (cons (cons 'acl2::tmp-for-assocs 'nil)
                               (cons acl2::rest-expr 'nil))))
               (mv nil (car acl2::forms)
                   acl2::rest-expr))))
