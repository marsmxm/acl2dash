Function: ushortp

(defun ushortp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'ushortp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :ushort)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (ushort-integerp get)))))
