                             (vl-sigma-p sigma))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-enumitemlist-subst))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-enumitem-subst (car x) sigma)
