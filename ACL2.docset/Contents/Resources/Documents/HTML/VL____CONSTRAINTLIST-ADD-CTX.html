Function: constraintlist-add-ctx

(defun
 constraintlist-add-ctx (x ctx)
 (declare (xargs :guard (sv::constraintlist-p x)))
 (let
  ((__function__ 'constraintlist-add-ctx))
  (declare (ignorable __function__))
  (if
   (atom x)
   nil
   (cons
       (b* (((sv::constraint x1) (car x)))
           (sv::change-constraint x1
                                  :name (if (and ctx (atom x1.name))
                                            (cons ctx x1.name)
                                            x1.name)))
       (constraintlist-add-ctx (cdr x) ctx)))))
