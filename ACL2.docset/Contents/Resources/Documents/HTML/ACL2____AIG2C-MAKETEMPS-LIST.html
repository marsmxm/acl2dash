Function: aig2c-maketemps-list

(defun
     aig2c-maketemps-list
     (x config tempmap db)
     (declare (xargs :guard (and (aig2c-config-p config)
                                 (string-listp (alist-vals tempmap))
                                 (vl::vl-namedb-p db))))
     (let ((__function__ 'aig2c-maketemps-list))
          (declare (ignorable __function__))
          (b* (((when (atom x)) (mv tempmap db))
               ((mv tempmap db)
                (aig2c-maketemps (car x)
                                 config tempmap db)))
              (aig2c-maketemps-list (cdr x)
                                    config tempmap db))))
