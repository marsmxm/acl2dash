(defun vl-jp-tagname-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-tagname-p x)))
       (let ((__function__ 'vl-jp-tagname))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-tagname)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-tagname->name x))
                       (vl-println? "}"))))
