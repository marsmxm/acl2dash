Function: vl-find-class

(defun vl-find-class (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-classlist-p x))))
       (let ((__function__ 'vl-find-class))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-class->name (car x)))
                   (vl-class-fix (car x)))
                  (t (vl-find-class name (cdr x))))))

Theorem: return-type-of-vl-find-class

(defthm return-type-of-vl-find-class
        (b* ((class? (vl-find-class name x)))
            (iff (vl-class-p class?) class?))
        :rule-classes :rewrite)

Theorem: vl-find-class-under-iff

(defthm vl-find-class-under-iff
        (iff (vl-find-class name x)
             (member-equal (string-fix name)
                           (vl-classlist->names x))))
