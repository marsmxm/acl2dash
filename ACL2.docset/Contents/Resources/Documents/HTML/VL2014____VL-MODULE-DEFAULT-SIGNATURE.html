Function: vl-module-default-signature

(defun
 vl-module-default-signature
 (modname ss warnings)
 (declare (xargs :guard (and (stringp modname)
                             (vl-scopestack-p ss)
                             (vl-warninglist-p warnings))))
 (let
  ((__function__ 'vl-module-default-signature))
  (declare (ignorable __function__))
  (b*
   ((modname (string-fix modname))
    (x (vl-scopestack-find-definition modname ss))
    ((unless (and x (eq (tag x) :vl-module)))
     (mv
       nil (vl-scopestack-fix ss)
       (fatal
            :type :vl-unparam-fail
            :msg "Programming error: top-level module ~s0 not found"
            :args (list modname))))
    ((mv ok warnings new-ss final-paramdecls)
     (vl-scope-finalize-params x (vl-module->paramdecls x)
                               (make-vl-paramargs-named)
                               warnings ss (vl-context modname)))
    ((unless ok)
     (mv nil (vl-scopestack-fix ss)
         warnings)))
   (mv (make-vl-unparam-signature :modname modname
                                  :final-params final-paramdecls)
       new-ss warnings))))
