Function: a3vec-?*

(defun
 a3vec-?* (x y y3p z z3p)
 (declare (xargs :guard (and (a4vec-p x)
                             (a4vec-p y)
                             (a4vec-p z))))
 (let
  ((__function__ 'a3vec-?*))
  (declare (ignorable __function__))
  (b*
   (((a4vec a) x)
    ((a4vec b) y)
    ((a4vec c) z)
    (a=1 (aig-not (aig-iszero-s a.lower)))
    (a=0 (aig-iszero-s a.upper))
    (a=x (aig-nor a=1 a=0))
    (boolcase (aig-logior-ss (aig-ite-bss-fn a=1 b.upper nil)
                             (aig-ite-bss-fn a=0 c.upper nil)))
    (upper
      (aig-logior-ss
           boolcase
           (aig-ite-bss
                a=x
                (aig-logior-ss (aig-logior-ss b.upper c.upper)
                               (aig-logxor-ss (if y3p nil b.lower)
                                              (if z3p nil c.lower)))
                nil)))
    (boolcase (aig-logior-ss (aig-ite-bss-fn a=1 b.lower nil)
                             (aig-ite-bss-fn a=0 c.lower nil)))
    (lower
     (aig-logior-ss
      boolcase
      (aig-ite-bss
          a=x
          (aig-logand-ss
               (aig-logand-ss b.lower c.lower)
               (aig-lognot-s (aig-logxor-ss (if y3p '(t) b.upper)
                                            (if z3p '(t) c.upper))))
          nil))))
   (a4vec upper lower))))
