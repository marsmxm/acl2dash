An 32-bit unsigned bitstruct type.

This is a bitstruct type introduced by fty::defbitstruct, represented as a unsigned 32-bit integer.

Fields
truth — truth::truth4
size — cutsize
valid — boolean
score — cutscore

  ((__function__ 'cutinfo-fix))

   (loghead

   :exec x)))

Function: cutinfo

(defun cutinfo (truth size valid score)
       (declare (xargs :guard (and (truth::truth4-p truth)
                                   (cutsize-p size)
                                   (booleanp valid)
                                   (cutscore-p score))))
       (let ((__function__ 'cutinfo))
            (declare (ignorable __function__))
            (b* ((truth (mbe :logic (truth::truth4-fix$ truth)
                             :exec truth))
                 (size (mbe :logic (cutsize-fix size)
                            :exec size))
                 (valid (bool->bit valid))
                 (score (mbe :logic (cutscore-fix score)
                             :exec score)))
                (logapp 16 truth
                        (logapp 3 size (logapp 1 valid score))))))

        (equal (!cutinfo->score score x)
               (change-cutinfo x :score score)))

Theorem: cutinfo->score-of-!cutinfo->score

(defthm cutinfo->score-of-!cutinfo->score
        (b* ((?new-x (!cutinfo->score score x)))
            (equal (cutinfo->score new-x)
                   (cutscore-fix score))))

Theorem: !cutinfo->score-equiv-under-mask

(defthm !cutinfo->score-equiv-under-mask
        (b* ((?new-x (!cutinfo->score score x)))
            (cutinfo-equiv-under-mask new-x x 1048575)))

Function: cutinfo-debug

(defun
   cutinfo-debug (x)
   (declare (xargs :guard (cutinfo-p x)))
   (let ((__function__ 'cutinfo-debug))
        (declare (ignorable __function__))
        (b* (((cutinfo x)))
            (cons (cons 'truth x.truth)
                  (cons (cons 'size x.size)
                        (cons (cons 'valid x.valid)
                              (cons (cons 'score x.score) nil)))))))

Theorem: cutinfo-debug-of-cutinfo-fix-x

(defthm cutinfo-debug-of-cutinfo-fix-x
        (equal (cutinfo-debug (cutinfo-fix x))
               (cutinfo-debug x)))

Theorem: cutinfo-debug-cutinfo-equiv-congruence-on-x

(defthm cutinfo-debug-cutinfo-equiv-congruence-on-x
        (implies (cutinfo-equiv x x-equiv)
                 (equal (cutinfo-debug x)
                        (cutinfo-debug x-equiv)))
        :rule-classes :congruence)
