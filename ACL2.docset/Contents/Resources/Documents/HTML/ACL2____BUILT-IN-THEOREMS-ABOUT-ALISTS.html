Theorem: consp-assoc-equal

(defthm
    consp-assoc-equal
    (implies (alistp l)
             (or (consp (assoc-equal name l))
                 (equal (assoc-equal name l) nil)))
    :rule-classes
    (:type-prescription
         (:forward-chaining :trigger-terms ((assoc-equal name l)))))
