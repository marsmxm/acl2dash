Function: nat-setp

(defun nat-setp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (natp (car x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (fast-<< (car x) (cadr x))
                         (nat-setp (cdr x)))))))
