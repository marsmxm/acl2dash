Function: tailrec-variantp

(defun tailrec-variantp (x)
       (declare (xargs :guard t))
       (or (eq x ':assoc)
           (eq x ':assoc-alt)
           (eq x ':monoid)
           (eq x ':monoid-alt)))

(defthm type-when-tailrec-variantp
        (implies (tailrec-variantp x)
                 (if (symbolp x)
                     (if (not (equal x 't))
                         (not (equal x 'nil))
