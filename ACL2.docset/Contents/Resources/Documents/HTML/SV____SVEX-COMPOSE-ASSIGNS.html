Function: svex-compose-assigns

(defun
 svex-compose-assigns (assigns)
 (declare (xargs :guard (svex-alist-p assigns)))
 (let
     ((__function__ 'svex-compose-assigns))
     (declare (ignorable __function__))
     (b* (((mv updates memo)
           (with-fast-alist
