Function: print-bin-integer-literal

(defun
  print-bin-integer-literal (lit)
  (declare (xargs :guard (bin-integer-literalp lit)))
  (let ((__function__ 'print-bin-integer-literal))
       (declare (ignorable __function__))
       (b* (((bin-integer-literal lit) lit))
           (msg "~@0~@1~@2"
                (if lit.prefix-upcase-p "0B" "0b")
