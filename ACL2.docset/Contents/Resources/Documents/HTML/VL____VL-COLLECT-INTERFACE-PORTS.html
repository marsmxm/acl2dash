Function: vl-collect-interface-ports

 vl-collect-interface-ports (x)
 (declare (xargs :guard (vl-portlist-p x)))
 (let
  ((__function__ 'vl-collect-interface-ports))
  (declare (ignorable __function__))
  (mbe
   :logic (b* (((when (atom x)) nil)
               (x1 (vl-port-fix (car x)))
               ((when (eq (tag x1) :vl-interfaceport))
                (cons x1
                      (vl-collect-interface-ports (cdr x)))))
              (vl-collect-interface-ports (cdr x)))

   (if
      (atom x)

      (with-local-nrev (vl-collect-interface-ports-exec x nrev))))))
