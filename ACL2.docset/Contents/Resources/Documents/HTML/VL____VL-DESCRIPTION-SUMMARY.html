Function: vl-description-summary

(defun vl-description-summary (x)
       (declare (xargs :guard (vl-description-p x)))
       (let ((__function__ 'vl-description-summary))
            (declare (ignorable __function__))
            (b* ((type (tag x))
                 (name (vl-description->name x))
                 (minloc (vl-description->minloc x)))
                (list (cons :name name)
                      (cons :type type)
                      (cons :file (vl-location->filename minloc))
                      (cons :line (vl-location->line minloc))
                      (cons :col (vl-location->col minloc))))))
