(defun
 vl-fast-keep-modinsts-by-modname
 (names fal x nrev)
 (declare (xargs :stobjs (nrev)))
 (declare
      (xargs :guard (and (string-listp names)
                         (vl-modinstlist-p x)
                         (equal fal (make-lookup-alist names)))))
 (let
     ((__function__ 'vl-fast-keep-modinsts-by-modname))
     (declare (ignorable __function__))
     (if (atom x)
         (nrev-fix nrev)
         (let ((nrev (if (fast-memberp (vl-modinst->modname (car x))
                                       (string-list-fix names)
                                       fal)
                         (nrev-push (vl-modinst-fix (car x))
                                    nrev)
                         nrev)))
              (vl-fast-keep-modinsts-by-modname names fal (cdr x)
                                                nrev)))))
