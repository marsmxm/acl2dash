(defun modinst-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'modinst-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (b* ((instname (std::prod-car x))
                      (modname (std::prod-cdr x)))
                     (and (name-p instname)
                          (modname-p modname))))))
