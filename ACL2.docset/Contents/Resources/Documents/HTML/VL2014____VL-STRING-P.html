Function: vl-string-p

(defun vl-string-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-string-p))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :vl-string)
                 (b* ((value (cdr x)))
                     (stringp value)))))
