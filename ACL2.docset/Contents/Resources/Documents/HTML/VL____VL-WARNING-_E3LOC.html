Function: vl-warning->loc

(defun vl-warning->loc (x)
       (declare (xargs :guard (vl-warning-p x)))
       (let ((__function__ 'vl-warning->loc))
            (declare (ignorable __function__))
            (b* ((args (vl-warning->args x)))
                (and (consp args)
                     (vl-location-p (car args))
                     (car args)))))
