Function: vl-taskdecl-elaborate-fn

(defun
 vl-taskdecl-elaborate-fn
 (x elabindex reclimit config)
 (declare (xargs :stobjs (elabindex)))
 (declare (xargs :guard (and (vl-taskdecl-p x)
                             (natp reclimit)
                             (vl-simpconfig-p config))))
 (let
  ((__function__ 'vl-taskdecl-elaborate))
  (declare (ignorable __function__))
  (b* (((vl-taskdecl x) (vl-taskdecl-fix x)))
      (b* (((mv ok warnings body elabindex)
            (vl-stmt-elaborate-fn x.body elabindex reclimit config))
           ((mv ok1 warnings1 portdecls elabindex)
            (vl-portdecllist-elaborate-fn
                 x.portdecls elabindex reclimit config))
           (ok (and ok1 ok))
           (warnings (append-without-guard warnings1 warnings))
           ((mv ok1 warnings1 vardecls elabindex)
            (vl-vardecllist-elaborate-fn
                 x.vardecls elabindex reclimit config))
           (ok (and ok1 ok))
           (warnings (append-without-guard warnings1 warnings))
           ((mv ok1 warnings1 paramdecls elabindex)
            (vl-paramdecllist-elaborate-fn
                 x.paramdecls elabindex reclimit config))
           (ok (and ok1 ok))
           (warnings (append-without-guard warnings1 warnings))
           ((mv ok1 warnings1 typedefs elabindex)
            (vl-typedeflist-elaborate-fn
                 x.typedefs elabindex reclimit config))
           (ok (and ok1 ok))
           (warnings (append-without-guard warnings1 warnings))
           ((mv ok1 warnings1 loaditems elabindex)
            (vl-portdecl-or-blockitem-list-elaborate-fn
                 x.loaditems elabindex reclimit config))
           (ok (and ok1 ok))
           (warnings (append-without-guard warnings1 warnings)))
          (mv ok warnings
              (change-vl-taskdecl x
                                  :body body
                                  :portdecls portdecls
                                  :vardecls vardecls
                                  :paramdecls paramdecls
                                  :typedefs typedefs
                                  :loaditems loaditems)
              elabindex)))))
