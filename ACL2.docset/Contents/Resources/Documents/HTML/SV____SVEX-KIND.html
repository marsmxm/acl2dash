Function: svex-kind$inline

(defun svex-kind$inline (x)
       (declare (xargs :guard (svex-p x)))
       (let ((__function__ 'svex-kind))
            (declare (ignorable __function__))
            (cond ((if (atom x)
                       (or (stringp x) (and x (symbolp x)))
                       (eq (car x) :var))
                   :var)
                  ((or (atom x) (integerp (car x)))
                   :quote)
                  (t :call))))
