(defun
 schar-array-write (array index element)
 (declare (xargs :guard (and (schar-arrayp array)
                             (integerp index)
                             (scharp element))))
 (declare (xargs :guard (schar-array-index-okp array index)))
 (let
  ((__function__ 'schar-array-write))
  (declare (ignorable __function__))
  (b*
   ((array (schar-array-fix array))
    (index (ifix index))
    (element (schar-fix element)))
   (if
    (mbt (schar-array-index-okp array index))
    (schar-array (update-nth index
                             element (schar-array->elements array)))
    array))))
