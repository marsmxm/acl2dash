Function: vl-shadowcheck-declare-name-fn

(defun
 vl-shadowcheck-declare-name-fn
 (name decl st warnings genblockp)
 (declare (xargs :guard (and (stringp name)
                             (any-p decl)
                             (vl-shadowcheck-state-p st)
                             (vl-warninglist-p warnings)
                             (booleanp genblockp))))
 (let
    ((__function__ 'vl-shadowcheck-declare-name))
    (declare (ignorable __function__))
    (b* (((vl-shadowcheck-state st))
         (- (vl-shadowcheck-debug
                 "    vl-shadowcheck-declare-name: ~s0 for ~a1.~%"
                 name decl))
         ((mv lexscopes warnings)
          (vl-lexscopes-declare-name name decl st.lexscopes warnings
                                     :genblockp genblockp))
         (st (change-vl-shadowcheck-state st
                                          :lexscopes lexscopes)))
        (mv st warnings))))
