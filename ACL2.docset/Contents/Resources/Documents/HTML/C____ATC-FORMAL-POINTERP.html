Function: atc-formal-pointerp

(defun
 atc-formal-pointerp
 (formal typed-formals)
 (declare
      (xargs :guard (and (symbolp formal)
                         (atc-symbol-type-alistp typed-formals))))
 (let
   ((__function__ 'atc-formal-pointerp))
   (declare (ignorable __function__))
   (b* ((pair (assoc-eq (symbol-fix formal)
                        (atc-symbol-type-alist-fix typed-formals))))
       (and (consp pair)
            (type-case (cdr pair) :pointer)))))
