Function: svex-varmasks->a4env-rec

(defun
   svex-varmasks->a4env-rec
   (vars masks boolmasks nextvar acc)
   (declare (xargs :guard (and (svarlist-p vars)
                               (svex-mask-alist-p masks)
                               (svar-boolmasks-p boolmasks)
                               (natp nextvar)
                               (nat-bool-a4env-p! acc))))
   (let ((__function__ 'svex-varmasks->a4env-rec))
        (declare (ignorable __function__))
        (b* ((acc (svex-a4vec-env-fix acc))
             ((when (atom vars))
              (mv nil acc (lnfix nextvar)))
             (mask (svex-mask-lookup (svex-var (car vars))
                                     masks))
             ((when (sparseint-< mask 0))
              (mv (msg "Negative mask: ~x0~%"
                       (svar-fix (car vars)))
                  acc (lnfix nextvar)))
             (boolmask (svar-boolmasks-lookup (car vars)
                                              boolmasks))
             (a4vec (4vmask-to-a4vec mask boolmask nextvar))
             (nextvar (+ (lnfix nextvar)
                         (4vmask-to-a4vec-varcount mask boolmask))))
            (svex-varmasks->a4env-rec
                 (cdr vars)
                 masks boolmasks nextvar
                 (cons (cons (svar-fix (car vars)) a4vec)
                       acc)))))
