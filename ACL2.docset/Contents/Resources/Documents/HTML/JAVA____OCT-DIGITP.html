Function: oct-digitp

(defun oct-digitp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'oct-digitp))
            (declare (ignorable __function__))
            (and (integerp x)
                 (<= (char-code #\0) x)
                 (<= x (char-code #\7)))))
