This is a step towards generating more modular and controlled loop proofs. The hints are more than needed for now, as they include rules about statement execution, which does not apply here. We will make the hints more nuanced later.

(defun
 atc-gen-loop-test-correct-thm
 (fn typed-formals loop-test
     test-term fn-thms names-to-avoid wrld)
 (declare (xargs :guard (and (symbolp fn)
                             (atc-symbol-type-alistp typed-formals)
                             (exprp loop-test)
                             (pseudo-termp test-term)
                             (symbol-symbol-alistp fn-thms)
                             (symbol-listp names-to-avoid)
                             (plist-worldp wrld))))
 (let
  ((__function__ 'atc-gen-loop-test-correct-thm))
  (declare (ignorable __function__))
  (b*
   ((correct-thm (cdr (assoc-eq fn fn-thms)))
    (correct-test-thm (add-suffix correct-thm "-TEST"))
    ((mv correct-test-thm names-to-avoid)
     (fresh-logical-name-with-$s-suffix correct-test-thm
                                        nil names-to-avoid wrld))
    (formals (strip-cars typed-formals))
    (compst-var (genvar 'atc "COMPST" nil formals))
    ((mv formals-bindings
         pointer-hyps & instantiation)
     (atc-gen-outer-bindings-and-hyps typed-formals compst-var t))
    (hyps
     (cons
      'and
      (cons
       (cons 'compustatep
             (cons compst-var 'nil))
       (cons (cons 'not
                   (cons (cons 'equal
                               (cons (cons 'compustate-frames-number
                                           (cons compst-var 'nil))
                                     (cons '0 'nil)))
                         'nil))
             (append pointer-hyps
                     (cons (untranslate (uguard+ fn wrld) nil wrld)
                           'nil))))))
    (concl
     (cons
      'equal
      (cons
         (cons 'exec-test
               (cons (cons 'exec-expr-pure
                           (cons (cons 'quote (cons loop-test 'nil))
                                 (cons compst-var 'nil)))
                     'nil))
         (cons test-term 'nil))))
    (formula (cons 'b*
                   (cons formals-bindings
                         (cons (cons 'implies
                                     (cons hyps (cons concl 'nil)))
                               'nil))))
    (hints
     (cons
      (cons
       '"Goal"
       (cons
        ':do-not-induct
        (cons
         't
         (cons
          ':in-theory
          (cons
           (cons
            'union-theories
            (cons
                (cons 'theory
                      (cons (cons 'quote (cons 'atc-all-rules 'nil))
                            'nil))
                (cons (cons 'quote
                            (cons (cons 'not 'nil) 'nil))
                      'nil)))
           (cons
            ':use
            (cons
             (cons (cons ':instance
                         (cons (cons ':guard-theorem (cons fn 'nil))
                               (cons ':extra-bindings-ok
                                     instantiation)))
                   'nil)
             (cons ':expand
                   (cons ':lambdas 'nil)))))))))
      'nil))
    ((mv correct-test-thm-event &)
     (evmac-generate-defthm correct-test-thm
                            :formula formula
                            :hints hints
                            :enable nil)))
   (mv (list correct-test-thm-event)
       correct-test-thm names-to-avoid))))
