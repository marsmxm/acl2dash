Function: ipasir-add-clauses

(defun
 ipasir-add-clauses (ipasir clauses)
 (declare (xargs :stobjs (ipasir)))
 (declare (xargs :guard (lit-list-listp clauses)))
 (declare
     (xargs :guard (and (not (eq (ipasir-get-status ipasir) :undef))
                        (ipasir-empty-new-clause ipasir))))
 (let ((__function__ 'ipasir-add-clauses))
      (declare (ignorable __function__))
      (if (atom clauses)
          (b* ((ipasir (ipasir-cancel-new-clause ipasir)))
              (ipasir-input ipasir))
          (b* ((ipasir (ipasir-add-list ipasir (car clauses))))
              (ipasir-add-clauses ipasir (cdr clauses))))))
