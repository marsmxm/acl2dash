Function: vl-final-size-warnings

(defun
   vl-final-size-warnings (x ss scopes)
   (declare (xargs :guard (and (vl-final-p x)
                               (vl-scopestack-p ss)
                               (vl-elabscopes-p scopes))))
   (let ((__function__ 'vl-final-size-warnings))
        (declare (ignorable __function__))
        (b* (((vl-final x) (vl-final-fix x))
             ((mv ?ok warnings ?svstmts)
              (vl-stmt->svstmts x.stmt ss scopes
                                (make-svstmt-config :nonblockingp t)
                                nil)))
            warnings)))
