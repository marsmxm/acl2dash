(defun atc-gen-appconds (fns wrld)
       (declare (xargs :guard (and (symbol-listp fns)
                                   (plist-worldp wrld))))
       (let ((__function__ 'atc-gen-appconds))
            (declare (ignorable __function__))
            (b* (((when (endp fns)) (mv nil nil))
                 (fn (car fns))
                 ((when (not (irecursivep+ fn wrld)))
                  (atc-gen-appconds (cdr fns) wrld))
                 (meas (measure+ fn wrld))
                 (name (packn-pos (list 'natp-of-measure-of- fn)
                                  :keyword))
                 (formula (untranslate (cons 'natp (cons meas 'nil))
                                       nil wrld))
                 (appcond (make-evmac-appcond :name name
                                              :formula formula))
                 ((mv appconds fn-appconds)
                  (atc-gen-appconds (cdr fns) wrld)))
                (mv (cons appcond appconds)
                    (acons fn name fn-appconds)))))
