(defun esim-sexpr-probe-steps (mod ins st)
       (declare (xargs :guard t))
       (let ((__function__ 'esim-sexpr-probe-steps))
            (declare (ignorable __function__))
            (b* (((when (atom ins)) (mv nil nil nil))
                 ((mv nst out internal)
                  (b* ((in (car ins)) ((with-fast in st)))
                      (esim-sexpr-probe mod in st)))
                 ((mv nsts outs internals)
                  (esim-sexpr-probe-steps mod (cdr ins)
                                          nst)))
                (mv (cons nst nsts)
                    (cons out outs)
                    (cons internal internals)))))
