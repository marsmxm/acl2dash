Function: float-valuep

(defun float-valuep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'float-valuep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :float)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((float (std::da-nth 0 (cdr x))))
                     (float-value-abs-p float)))))
