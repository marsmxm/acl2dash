Function: lex-comment

(defun
 lex-comment (abnf::input)
 (declare (xargs :guard (nat-listp abnf::input)))
 (let
  ((__function__ 'lex-comment))
  (declare (ignorable __function__))
  (b*
   (((mv abnf::treess abnf::input)
     (b*
      (((mv abnf::treess1 abnf::input1)
        (b* (((mv abnf::tree abnf::input)
              (lex-block-comment abnf::input))
             ((when (resulterrp abnf::tree))
              (mv (fty::err-push abnf::tree)
                  abnf::input))
             (abnf::trees1 (list abnf::tree))
             (abnf::treess (list abnf::trees1)))
            (mv abnf::treess abnf::input)))
       ((when (not (resulterrp abnf::treess1)))
        (mv abnf::treess1 abnf::input1))
       ((mv abnf::treess2 abnf::input1)
        (b* (((mv abnf::tree abnf::input)
              (lex-end-of-line-comment abnf::input))
             ((when (resulterrp abnf::tree))
              (mv (fty::err-push abnf::tree)
                  abnf::input))
             (abnf::trees1 (list abnf::tree))
             (abnf::treess (list abnf::trees1)))
            (mv abnf::treess abnf::input)))
       ((when (not (resulterrp abnf::treess2)))
        (mv abnf::treess2 abnf::input1)))
      (mv
       (err
        (list
          :found (list abnf::treess1 abnf::treess2)
          :required
          '(((:repetition (:repeat 1 (:finite 1))
                          (:rulename (:rulename "block-comment"))))
            ((:repetition
                  (:repeat 1 (:finite 1))
                  (:rulename (:rulename "end-of-line-comment")))))))
       abnf::input)))
    ((when (resulterrp abnf::treess))
     (mv (fty::err-push abnf::treess)
         (acl2::nat-list-fix abnf::input))))
   (mv
      (abnf::make-tree-nonleaf :rulename? (abnf::rulename "comment")
                               :branches abnf::treess)
      abnf::input))))
