Function: bin-integer-literalp

(defun
     bin-integer-literalp (x)
     (declare (xargs :guard t))
     (let ((__function__ 'bin-integer-literalp))
          (declare (ignorable __function__))
          (and (consp x)
               (eq (car x) :bin-integer-lit)
               (true-listp (cdr x))
               (eql (len (cdr x)) 3)
               (b* ((digits/uscores (std::da-nth 0 (cdr x)))
                    (prefix-upcase-p (std::da-nth 1 (cdr x)))
                    (suffix? (std::da-nth 2 (cdr x))))
                   (and (bindig/uscore-listp digits/uscores)
                        (booleanp prefix-upcase-p)
                        (optional-integer-type-suffix-p suffix?)
                        (bindig/uscore-list-wfp digits/uscores))))))
