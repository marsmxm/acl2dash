Function: pprint-obj-declon

(defun
 pprint-obj-declon (declon level options)
 (declare (xargs :guard (and (obj-declonp declon)
                             (natp level)
                             (pprint-options-p options))))
 (let
    ((__function__ 'pprint-obj-declon))
    (declare (ignorable __function__))
    (b* (((obj-declon declon) declon))
        (list (pprint-line (msg "~@0 ~@1 = ~@2;"
                                (pprint-tyspecseq declon.tyspec)
                                (pprint-obj-declor declon.declor)
                                (pprint-initer declon.init options))
                           (lnfix level))))))
