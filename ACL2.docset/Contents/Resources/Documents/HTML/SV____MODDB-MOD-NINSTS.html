Function: moddb-mod-ninsts

(defun moddb-mod-ninsts (modidx moddb)
       (declare (xargs :stobjs (moddb)))
       (declare (xargs :guard (and (natp modidx)
                                   (moddb-basics-ok moddb))))
       (declare (xargs :guard (< modidx (moddb->nmods moddb))))
       (let ((__function__ 'moddb-mod-ninsts))
            (declare (ignorable __function__))
            (b* (((unless (mbt (< (lnfix modidx)
                                  (moddb->nmods moddb))))
                  0)

                  (elab-mod-ninsts elab-mod)))
                ninsts)))
