Function: vl-consteval-concat

(defun
 vl-consteval-concat
 (x val-acc width-acc)
 (declare (xargs :guard (and (vl-exprlist-p x)
                             (natp val-acc)
                             (natp width-acc))))
 (declare (xargs :guard (and (vl-exprlist-resolved-p x)
                             (consp x)
                             (unsigned-byte-p width-acc val-acc))))
 (let
  ((__function__ 'vl-consteval-concat))
  (declare (ignorable __function__))
  (b*
     (((vl-constint x1)
       (vl-atom->guts (car x)))
      (val-acc (acl2::logapp x1.origwidth x1.value (lnfix val-acc)))
      (width-acc (+ x1.origwidth (lnfix width-acc)))
      ((when (atom (cdr x)))
       (vl-consteval-ans :value val-acc
                         :width width-acc
                         :type :vl-unsigned)))
     (vl-consteval-concat (cdr x)
                          val-acc width-acc))))
