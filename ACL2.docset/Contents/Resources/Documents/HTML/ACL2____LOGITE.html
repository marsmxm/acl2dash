Function: logite

(defun logite (test then else)
       (declare (type integer test)
                (type integer then)
                (type integer else))
       (let ((__function__ 'logite))
            (declare (ignorable __function__))
            (logior (logand test then)
                    (logand (lognot test) else))))

(defthm logite-type
