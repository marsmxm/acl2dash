Function: iconstp

(defun
 iconstp (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'iconstp))
  (declare (ignorable __function__))
  (and
    (consp x)
    (eq (car x) :iconst)
    (mbe :logic (and (alistp (cdr x))
                     (equal (strip-cars (cdr x))
                            '(value base unsignedp type)))
         :exec (fty::alist-with-carsp (cdr x)
                                      '(value base unsignedp type)))
    (b* ((value (cdr (std::da-nth 0 (cdr x))))
         (base (cdr (std::da-nth 1 (cdr x))))
         (unsignedp (cdr (std::da-nth 2 (cdr x))))
         (type (cdr (std::da-nth 3 (cdr x)))))
        (and (natp value)
             (iconst-basep base)
             (booleanp unsignedp)
             (iconst-tysuffixp type))))))
