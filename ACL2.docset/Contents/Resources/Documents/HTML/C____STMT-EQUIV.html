Theorem: stmt-equiv-is-an-equivalence

(defthm stmt-equiv-is-an-equivalence
        (and (booleanp (stmt-equiv x y))
             (stmt-equiv x x)
             (implies (stmt-equiv x y)
                      (stmt-equiv y x))
             (implies (and (stmt-equiv x y) (stmt-equiv y z))
                      (stmt-equiv x z)))
        :rule-classes (:equivalence))
