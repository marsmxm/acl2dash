(honsed-defdigits-info base

This is identical to defdigits-info, except that we hons the structure we are creating.

Definition

This is an ordinary honsing constructor introduced by std::defaggregate.

(defun
 honsed-defdigits-info
 (base digits-pred digits-fix bendian-to-nat
       lendian-to-nat nat-to-bendian
       nat-to-lendian digits-description
       digits-pred-correct digits-fix-correct)

                             (symbolp digits-pred)
                             (symbolp digits-fix)
                             (symbolp bendian-to-nat)

                             (symbolp nat-to-bendian)

                             (symbolp digits-pred-correct)
                             (symbolp digits-fix-correct))))

  :logic (defdigits-info base

  :exec

   (hons

    (hons

       (hons 'bendian-to-nat bendian-to-nat)
       (hons (hons 'lendian-to-nat lendian-to-nat)
             (hons (hons 'nat-to-bendian nat-to-bendian)
                   (hons (hons 'nat-to-lendian nat-to-lendian)
                         (hons (hons 'digits-description
                                     digits-description)
                               (hons (hons 'digits-pred-correct

                                     (hons (hons 'digits-fix-correct

                                           nil))))))))))))
