Rules for executing unary operations

Definitions and Theorems

Theorem: exec-plus-when-scharp

(defthm
 exec-plus-when-scharp
 (implies
  (and
   (syntaxp
    (or
     (atom x)
     (not
        (member-eq x
                   '(exec-ident exec-const
                                exec-iconst exec-arrsub exec-memberp
                                exec-arrsub-of-memberp exec-unary
                                exec-cast exec-binary-strict-pure
                                exec-test exec-expr-pure)))))
   (equal op (unop-plus))
   (scharp x))
  (equal (exec-unary op x)
         (plus-schar x))))
