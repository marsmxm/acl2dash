Function: term-equivs-p

(defun term-equivs-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'term-equivs-p))
            (declare (ignorable __function__))
            (if (atom x)
                (eq x nil)
                (and (consp (car x))
                     (fgl-object-p (caar x))
                     (nat-listp (cdar x))
                     (term-equivs-p (cdr x))))))
