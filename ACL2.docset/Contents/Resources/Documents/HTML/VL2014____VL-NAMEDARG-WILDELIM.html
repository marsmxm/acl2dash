Function: vl-namedarg-wildelim

(defun
     vl-namedarg-wildelim (x ctx warnings)
     (declare (xargs :guard (and (vl-namedarg-p x)
                                 (vl-context-p ctx)
                                 (vl-warninglist-p warnings))))
     (let ((__function__ 'vl-namedarg-wildelim))
          (declare (ignorable __function__))
          (b* ((x (vl-namedarg-fix x))
               (warnings (vl-warninglist-fix warnings)))
              (b* (((vl-namedarg x) x)
                   ((unless x.expr) (mv warnings x))
                   ((mv warnings expr-prime)
                    (vl-expr-wildelim x.expr ctx warnings))
                   (x-prime (change-vl-namedarg x
                                                :expr expr-prime)))
                  (mv warnings x-prime)))))
