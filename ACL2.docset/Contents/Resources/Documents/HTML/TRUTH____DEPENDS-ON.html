Function: depends-on

(defun depends-on (n truth numvars)
       (declare (xargs :guard (and (natp n)
                                   (integerp truth)
                                   (natp numvars))))
       (declare (xargs :guard (< n numvars)))
       (let ((__function__ 'depends-on))
            (declare (ignorable __function__))
            (b* ((var (var n numvars))
