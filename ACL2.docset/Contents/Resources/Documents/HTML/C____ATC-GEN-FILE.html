(defun
   atc-gen-file
   (tunit output-file pretty-printing state)
   (declare (xargs :stobjs (state)))
   (declare (xargs :guard (and (transunitp tunit)
                               (stringp output-file)
                               (pprint-options-p pretty-printing))))
   (let ((__function__ 'atc-gen-file))
        (declare (ignorable __function__))
        (b* ((lines (pprint-transunit tunit pretty-printing)))
