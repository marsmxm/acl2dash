Function: vl-parse-system-task-enable-fn

(defun
 vl-parse-system-task-enable-fn
 (atts tokstream config)
 (declare (xargs :stobjs (tokstream)))
 (declare (xargs :guard (vl-loadconfig-p config)))
 (declare (ignorable config))
 (declare (xargs :guard (and (vl-atts-p atts)
                             (vl-is-token? :vl-sysidtoken))))
 (let
  ((__function__ 'vl-parse-system-task-enable))
  (declare (ignorable __function__))
  (seq
   tokstream (loc := (vl-current-loc))
   (id := (vl-match))
   (when (vl-is-token? :vl-lparen)
         (:= (vl-match))
         (unless (vl-is-token? :vl-rparen)
                 (args := (vl-parse-sysfuncall-args)))
         (:= (vl-match-token :vl-rparen)))
   (:= (vl-match-token :vl-semi))
   (return
    (make-vl-callstmt
     :id
     (make-vl-scopeexpr-end
          :hid (make-vl-hidexpr-end :name (vl-sysidtoken->name id)))
     :voidp nil
     :typearg nil
     :systemp t
     :args args
     :atts atts
     :loc loc)))))
