Cause an error if any element of a true list is not a member of another true list.

Arguments

    Guard (true-listp super).

Function: ensure-list-subset

(defun
 ensure-list-subset
 (list super description
       error-erp error-val ctx state)
 (declare (xargs :stobjs (state)))
 (declare (xargs :guard (and (true-listp list)
                             (true-listp super)
                             (msgp description))))
 (b*
  (((unless (subsetp-equal list super))
    (er-soft+
     ctx error-erp error-val
     "~@0 must have only elements in the list ~x1, but it includes the ~@2."
     description super
     (let ((extra (remove-duplicates-equal
                       (set-difference-equal list super))))
          (if (= (len extra) 1)
              (msg "element ~x0" (car extra))
              (msg "elements ~&0" extra))))))
  (value nil)))
