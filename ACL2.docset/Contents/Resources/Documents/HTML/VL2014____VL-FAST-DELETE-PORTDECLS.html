(defun
  vl-fast-delete-portdecls
  (names fal x nrev)
  (declare (xargs :stobjs (nrev)))
  (declare
       (xargs :guard (and (string-listp names)
                          (vl-portdecllist-p x)
                          (equal fal (make-lookup-alist names)))))
  (let ((__function__ 'vl-fast-delete-portdecls))
       (declare (ignorable __function__))
       (if (atom x)
           (nrev-fix nrev)
           (let ((nrev (if (fast-memberp (vl-portdecl->name (car x))

                                         fal)
                           nrev

                                      nrev))))

                                          nrev)))))
