Theorem: logapp*

(defthm
 logapp*
 (implies
      (logapp-guard size i j)
      (equal (logapp size i j)
             (cond ((equal size 0) j)
                   (t (logcons (logcar i)
                               (logapp (1- size) (logcdr i) j))))))
 :rule-classes ((:definition :clique (logapp)
                             :controller-alist ((logapp t t nil)))))
