 svl-module-p (x)
 (declare (xargs :guard t))
 (let
  ((acl2::__function__ 'svl-module-p))
  (declare (ignorable acl2::__function__))
  (and
   (mbe
     :logic (and (alistp x)
                 (equal (strip-cars x)
                        '(rank inputs delayed-inputs outputs occs)))
     :exec (fty::alist-with-carsp
                x
                '(rank inputs delayed-inputs outputs occs)))
   (b* ((rank (cdr (std::da-nth 0 x)))
        (inputs (cdr (std::da-nth 1 x)))
        (delayed-inputs (cdr (std::da-nth 2 x)))
        (outputs (cdr (std::da-nth 3 x)))
        (occs (cdr (std::da-nth 4 x))))

            (sv::svarlist-p delayed-inputs)
