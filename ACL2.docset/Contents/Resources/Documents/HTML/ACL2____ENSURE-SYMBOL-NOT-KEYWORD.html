Function: ensure-symbol-not-keyword

(defun ensure-symbol-not-keyword
       (symb description
             error-erp error-val ctx state)
       (declare (xargs :stobjs (state)))
       (declare (xargs :guard (and (symbolp symb)
                                   (msgp description))))
       (b* (((when (keywordp symb))
             (er-soft+ ctx error-erp
                       error-val "~@0 must not be a keyword."
                       description)))
           (value nil)))
