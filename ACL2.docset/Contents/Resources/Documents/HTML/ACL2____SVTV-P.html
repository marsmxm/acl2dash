Function: svtv-p

(defun
 svtv-p (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'svtv-p))
  (declare (ignorable __function__))
  (and
    (mbe :logic
         (and (alistp x)
              (equal (strip-cars x)
                     '(name outexprs nextstate states
                            inmasks outmasks orig-ins orig-overrides
                            orig-outs orig-internals expanded-ins
                            expanded-overrides nphases)))
         :exec (fty::alist-with-carsp
                    x
                    '(name outexprs nextstate states
                           inmasks outmasks orig-ins orig-overrides
                           orig-outs orig-internals expanded-ins
                           expanded-overrides nphases)))
    (b* ((name (cdr (std::da-nth 0 x)))
         (outexprs (cdr (std::da-nth 1 x)))
         (nextstate (cdr (std::da-nth 2 x)))
         (states (cdr (std::da-nth 3 x)))
         (inmasks (cdr (std::da-nth 4 x)))
         (outmasks (cdr (std::da-nth 5 x)))
         (orig-ins (cdr (std::da-nth 6 x)))
         (orig-overrides (cdr (std::da-nth 7 x)))
         (orig-outs (cdr (std::da-nth 8 x)))
         (orig-internals (cdr (std::da-nth 9 x)))
         (expanded-ins (cdr (std::da-nth 10 x)))
         (expanded-overrides (cdr (std::da-nth 11 x)))
         (nphases (cdr (std::da-nth 12 x))))
        (and (symbolp name)
             (svex-alist-p outexprs)
             (svex-alist-p nextstate)
             (svex-alistlist-p states)
             (svar-boolmasks-p inmasks)
             (svar-boolmasks-p outmasks)
             (true-list-listp orig-ins)
             (true-list-listp orig-overrides)
             (true-list-listp orig-outs)
             (true-list-listp orig-internals)
             (svtv-lines-p expanded-ins)
             (svtv-lines-p expanded-overrides)
             (natp nphases))))))
