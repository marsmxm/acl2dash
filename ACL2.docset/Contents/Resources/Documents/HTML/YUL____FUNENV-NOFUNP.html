Function: funenv-nofunp

(defun funenv-nofunp (x)
       (declare (xargs :guard (funenvp x)))
       (let ((__function__ 'funenv-nofunp))
            (declare (ignorable __function__))
            (if (consp x)
                (and (funscope-nofunp (car x))
