Function: vl-delete-taskdecls

(defun
 vl-delete-taskdecls (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-taskdecllist-p x))))
 (let
  ((__function__ 'vl-delete-taskdecls))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-taskdecl->name (car x))
                               (string-list-fix names))
                 (vl-delete-taskdecls names (cdr x)))
                (t (cons (vl-taskdecl-fix (car x))
                         (vl-delete-taskdecls names (cdr x)))))
   :exec
   (b*
    (((when (atom names)) (list-fix x))
     ((when (atom x)) nil)
     ((unless (longer-than-p 10 names))
      (vl-slow-delete-taskdecls names x))
     (fal (make-lookup-alist names))
     (ans
      (with-local-nrev (vl-fast-delete-taskdecls names fal x nrev)))
     (- (fast-alist-free fal)))
    ans))))
