(defun
     head-to-head-competition-loser-p
     (l c xs)
     (declare (xargs :guard (and (natp l)
                                 (natp c)
                                 (irv-ballot-p xs))))
     (declare (xargs :guard (and (member-equal l (candidate-ids xs))
                                 (member-equal c (candidate-ids xs))
