Theorem: mod-equiv-is-an-equivalence

(defthm mod-equiv-is-an-equivalence
        (and (booleanp (mod-equiv x y))
             (mod-equiv x x)
             (implies (mod-equiv x y)
                      (mod-equiv y x))
             (implies (and (mod-equiv x y) (mod-equiv y z))

        :rule-classes (:equivalence))
