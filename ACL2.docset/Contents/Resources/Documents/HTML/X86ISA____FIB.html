(defun fib (n)
       (declare (xargs :guard (natp n)))
       (declare (xargs :guard (natp n)))
       (let ((__function__ 'fib))
            (declare (ignorable __function__))
            (cond ((zp n) 0)
                  ((eql n 1) 1)
                  (t (+ (fib (- n 1)) (fib (- n 2)))))))
