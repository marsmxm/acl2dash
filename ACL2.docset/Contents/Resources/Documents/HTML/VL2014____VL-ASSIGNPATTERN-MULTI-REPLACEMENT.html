Function: vl-assignpattern-multi-replacement

(defun
 vl-assignpattern-multi-replacement
 (lhs-type args orig-x ctx warnings)
 (declare (xargs :guard (and (vl-datatype-p lhs-type)
                             (vl-exprlist-p args)
                             (vl-expr-p orig-x)
                             (vl-context-p ctx)
                             (vl-warninglist-p warnings))))
 (let
  ((__function__ 'vl-assignpattern-multi-replacement))
  (declare (ignorable __function__))
  (b*
   ((ctx (vl-context-fix ctx))
    (orig-x (vl-expr-fix orig-x))
    ((unless (and (eql (len args) 2)
                  (vl-expr-resolved-p (first args))
                  (not (eq (vl-expr-kind (second args)) :atom))
                  (eq (vl-nonatom->op (second args))
                      :vl-concat)))
     (mv
      nil nil
      (warn
          :type :vl-assignpattern-elim-fail
          :msg "~a0: Ill-formed replication assignment pattern: ~a1"
          :args (list ctx orig-x))))
    (replications (vl-resolved->val (first args)))
    (fields (append-n replications
                      (vl-nonatom->args (second args)))))
   (vl-assignpattern-positional-replacement
        lhs-type fields orig-x ctx warnings))))
