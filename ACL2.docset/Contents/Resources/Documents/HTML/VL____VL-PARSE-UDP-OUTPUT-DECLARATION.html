Function: vl-parse-udp-output-declaration-fn

(defun
 vl-parse-udp-output-declaration-fn
 (atts tokstream config)
 (declare (xargs :stobjs (tokstream)))
 (declare (xargs :guard (vl-loadconfig-p config)))
 (declare (ignorable config))
 (declare (xargs :guard (vl-atts-p atts)))
 (let
  ((__function__ 'vl-parse-udp-output-declaration))
  (declare (ignorable __function__))
  (seq
   tokstream
   (:= (vl-match-token :vl-kwd-output))
   (when (vl-is-token? :vl-kwd-reg)
         (reg := (vl-match)))
   (id := (vl-match-token :vl-idtoken))
   (when
    (vl-is-token? :vl-equalsign)
    (return-raw
     (vl-parse-error
      "Assignments to UDP output reg declarations are not currently supported.")))
   (return (make-vl-portdecl :loc (vl-token->loc id)
                             :name (vl-idtoken->name id)
                             :dir :vl-output
                             :type (if reg *vl-plain-old-reg-type*
                                       *vl-plain-old-logic-type*)
                             :atts atts)))))
