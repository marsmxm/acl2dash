Extract and attach in-bounds warnings to a description.

Signature
(vl-description-inject-warnings x warnings) 
  → 
(mv new-x warnings)
Arguments
x — Description to attach to.

Returns

Definitions and Theorems

Function: vl-description-inject-warnings

(defun
    vl-description-inject-warnings
    (x warnings)
    (declare (xargs :guard (and (vl-description-p x)
                                (vl-warninglist-p warnings))))
    (let ((__function__ 'vl-description-inject-warnings))
         (declare (ignorable __function__))
         (b* ((x (vl-description-fix x))
              (warnings (vl-warninglist-fix warnings))
              ((when (atom warnings)) (mv x warnings))
              ((unless (vl-description-has-comments-p x))
               (mv x warnings))
              (min (vl-description->minloc x))
              (max (vl-description->maxloc x))
              ((mv between beyond)
               (vl-filter-warnings-by-loc min max warnings nil nil))
              ((unless (consp between))
               (mv x warnings))
              (new-x (vl-description-add-warnings x between)))
             (mv new-x beyond))))
