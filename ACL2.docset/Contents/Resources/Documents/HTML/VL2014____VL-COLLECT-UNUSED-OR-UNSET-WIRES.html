                                   (string-listp unused)
                                   (string-listp unset))))
       (let ((__function__ 'vl-collect-unused-or-unset-wires))
            (declare (ignorable __function__))
            (b* (((when (atom x)) (mv unused unset))
                 (name (caar x))
                 (info (cdar x))
                 (usedp (vl-wireinfo->usedp info))
                 (setp (vl-wireinfo->setp info)))
