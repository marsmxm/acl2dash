Recognizes well-formed hierarchical scope expressions.

Signature
(vl-scopeexpr-p x)  bool
Arguments
x Guard (vl-expr-p x).

Example: foo::bar::a.b[1][2].c.

Function: vl-scopeexpr-p

(defun vl-scopeexpr-p (x)
       (declare (xargs :guard (vl-expr-p x)))
       (let ((__function__ 'vl-scopeexpr-p))
            (declare (ignorable __function__))
            (b* (((when (vl-atom-p x))
                  (mbe :logic (vl-hidexpr-p x)
                       :exec (vl-hidname-p x)))
                 ((vl-nonatom x) x)
                 ((when (vl-op-equiv x.op :vl-scope))
                  (and (vl-scopeatom-p (first x.args))
                       (vl-scopeexpr-p (second x.args)))))
                (vl-hidexpr-p x))))
