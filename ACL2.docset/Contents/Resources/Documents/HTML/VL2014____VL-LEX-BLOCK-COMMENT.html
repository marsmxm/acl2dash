Function: vl-lex-block-comment

(defun
 vl-lex-block-comment (echars)
 (declare (xargs :guard (vl-echarlist-p echars)))
 (let
  ((__function__ 'vl-lex-block-comment))
  (declare (ignorable __function__))
  (b*
     (((unless (vl-matches-string-p "/*" echars))
       (mv nil echars))
      ((mv successp prefix remainder)
       (vl-read-through-literal "*/" (cddr echars)))
      ((unless successp)
       (mv (cw "Lexer error (~s0): block comment is never closed.~%"
               (vl-location-string (vl-echar->loc (car echars))))
           echars))
      (etext (list* (first echars)
                    (second echars)
                    prefix))
      (token (make-vl-plaintoken :etext etext
                                 :type :vl-comment)))
     (mv token remainder))))
