Function: vl-pp-fwdtypedef-fn

(defun
 vl-pp-fwdtypedef-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (vl-fwdtypedef-p x)))
 (let
  ((__function__ 'vl-pp-fwdtypedef))
  (declare (ignorable __function__))
  (b*
   (((vl-fwdtypedef x) x))
   (vl-ps-seq
       (vl-progindent)
       (if x.atts (vl-pp-atts x.atts) ps)
