Function: vl-keep-imports-by-package

(defun
 vl-keep-imports-by-package (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-importlist-p x))))
 (let
  ((__function__ 'vl-keep-imports-by-package))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-import->pkg (car x))
                               (string-list-fix names))
                 (cons (vl-import-fix (car x))
                       (vl-keep-imports-by-package names (cdr x))))
                (t (vl-keep-imports-by-package names (cdr x))))
   :exec
   (b*
     (((when (or (atom names) (atom x))) nil)
      ((unless (longer-than-p 10 names))
       (vl-slow-keep-imports-by-package names x))
      (fal (make-lookup-alist names))
      (ans (with-local-nrev
                (vl-fast-keep-imports-by-package names fal x nrev)))
      (- (fast-alist-free fal)))
     ans))))
