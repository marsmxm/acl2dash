Function: uint-arrayp

(defun uint-arrayp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'uint-arrayp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :uint-array)
                 (mbe :logic (and (alistp (cdr x))
                                  (equal (strip-cars (cdr x))
                                         '(elements)))
                      :exec (fty::alist-with-carsp (cdr x)
                                                   '(elements)))
                 (b* ((elements (cdr (std::da-nth 0 (cdr x)))))
