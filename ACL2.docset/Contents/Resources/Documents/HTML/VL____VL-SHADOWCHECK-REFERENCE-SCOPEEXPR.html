(defun
 vl-shadowcheck-reference-scopeexpr
 (x ctx st warnings)
 (declare (xargs :guard (and (vl-scopeexpr-p x)
                             (any-p ctx)
                             (vl-shadowcheck-state-p st)
                             (vl-warninglist-p warnings))))
 (let
  ((__function__ 'vl-shadowcheck-reference-scopeexpr))
  (declare (ignorable __function__))
  (b*
   ((st (vl-shadowcheck-state-fix st))
    (warnings (vl-warninglist-fix warnings)))
   (vl-scopeexpr-case
    x
    :colon (mv st warnings)
    :end
    (vl-hidexpr-case
     x.hid
     :end (vl-shadowcheck-reference-name x.hid.name ctx st warnings)
     :dot
     (b* ((name (vl-hidindex->name x.hid.first))
          ((unless (stringp name))
           (mv st warnings))
          ((vl-shadowcheck-state st))
          ((when (vl-scopestack-find-definition name st.ss))
           (mv st warnings)))
         (vl-shadowcheck-reference-name name ctx st warnings)))))))
