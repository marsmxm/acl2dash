Change the number of rows in the frames, preserving data.

Function: framesl-resize-rows

(defun
 framesl-resize-rows
 (stobjs::nrows framesl-arr2)
 (declare (type (integer 0 *) stobjs::nrows)
          (xargs :guard (and (framesl-arr2-wfp framesl-arr2)
                             (<= stobjs::nrows (1- (expt 2 60)))
                             (<= (* stobjs::nrows
                                    (framesl-ncols framesl-arr2))
                                 (1- (expt 2 60))))))
 (stobjs::2darr (stobjs::2darr->ncols framesl-arr2)
                (resize-list (stobjs::2darr->rows framesl-arr2)
                             (nfix stobjs::nrows)
                             (make-list (framesl-ncols framesl-arr2)
                                        :initial-element 0))))
