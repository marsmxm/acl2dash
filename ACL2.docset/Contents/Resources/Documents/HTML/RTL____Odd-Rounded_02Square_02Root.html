Function: rto-sqrt

(defun rto-sqrt (x n)
       (declare (xargs :guard (and (real/rationalp x) (posp n))))
       (let ((trunc (rtz-sqrt x (1- n))))
            (if (< (* trunc trunc) x)
                (+ trunc (expt 2 (- n)))
                trunc)))

(defthm exactp-cmp-rto-sqrt
        (implies (and (rationalp x)
                      (>= x 1/4)
                      (< x 1)
                      (rationalp q)
                      (> q 0)
                      (integerp n)
                      (> n 1)
                      (exactp q (1- n)))
