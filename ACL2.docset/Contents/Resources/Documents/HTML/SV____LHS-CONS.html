Function: lhs-cons

(defun lhs-cons (x y)
       (declare (xargs :guard (and (lhrange-p x) (lhs-p y))))
       (let ((__function__ 'lhs-cons))
            (declare (ignorable __function__))
            (if (atom y)
                (list (lhrange-fix x))
                (let ((comb (lhrange-combine x (car y))))
                     (if comb (cons comb (lhs-fix (cdr y)))
                         (cons (lhrange-fix x) (lhs-fix y)))))))
