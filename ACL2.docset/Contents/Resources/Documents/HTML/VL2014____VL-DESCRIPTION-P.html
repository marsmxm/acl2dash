Function: vl-description-p

(defun
     vl-description-p (x)
     (declare (xargs :guard t))
     (let ((__function__ 'vl-description-p))
          (declare (ignorable __function__))
          (common-lisp::case (tag x)
                             ((:vl-module) (vl-module-p x))
                             ((:vl-udp) (vl-udp-p x))
                             ((:vl-interface) (vl-interface-p x))
                             ((:vl-package) (vl-package-p x))
                             ((:vl-program) (vl-program-p x))
                             ((:vl-config) (vl-config-p x))
                             ((:vl-taskdecl) (vl-taskdecl-p x))
                             ((:vl-fundecl) (vl-fundecl-p x))
                             ((:vl-paramdecl) (vl-paramdecl-p x))
                             ((:vl-import) (vl-import-p x))
                             ((:vl-fwdtypedef) (vl-fwdtypedef-p x))
                             (otherwise (vl-typedef-p x)))))
