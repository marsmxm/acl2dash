Theorem: hexchars=>ubyte8-of-ubyte8=>hexchars

(defthm hexchars=>ubyte8-of-ubyte8=>hexchars
        (b* (((mv hi-char lo-char)
              (ubyte8=>hexchars byte)))
            (equal (hexchars=>ubyte8 hi-char lo-char)
                   (unsigned-byte-fix 8 byte))))
