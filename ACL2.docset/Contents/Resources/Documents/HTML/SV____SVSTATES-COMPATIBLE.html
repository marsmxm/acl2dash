Theorem: svstates-compatible-is-an-equivalence

(defthm svstates-compatible-is-an-equivalence
        (and (booleanp (svstates-compatible x y))
             (svstates-compatible x x)
             (implies (svstates-compatible x y)
                      (svstates-compatible y x))
             (implies (and (svstates-compatible x y)
                           (svstates-compatible y z))
                      (svstates-compatible x z)))
        :rule-classes (:equivalence))
