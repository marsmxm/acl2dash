(defun vl-jp-namedarg-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-namedarg-p x)))
       (let ((__function__ 'vl-jp-namedarg))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-namedarg)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-namedarg->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"expr\": ")
                       (vl-jp-maybe-expr (vl-namedarg->expr x))
                       (vl-println? ", ")
                       (vl-print-str "\"nameonly-p\": ")
                       (jp-bool (vl-namedarg->nameonly-p x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-namedarg->atts x))
                       (vl-println? "}"))))
