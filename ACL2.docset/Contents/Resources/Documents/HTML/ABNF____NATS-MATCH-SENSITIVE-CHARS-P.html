Function: nats-match-sensitive-chars-p

(defun nats-match-sensitive-chars-p
       (nats chars)
       (declare (xargs :guard (and (nat-listp nats)
                                   (character-listp chars))))
       (cond ((endp nats) (endp chars))

                     (nat-match-sensitive-char-p (car nats)
