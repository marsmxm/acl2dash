(defun ex-from-rp (term)
       (declare (xargs :guard t))
       (let ((acl2::__function__ 'ex-from-rp))
            (declare (ignorable acl2::__function__))
            (if (is-rp term)
                (ex-from-rp (caddr term))
                term)))
