Function: vl-vardecllist-reset-atts

(defun
 vl-vardecllist-reset-atts (x new-atts)
 (declare (xargs :guard (and (vl-vardecllist-p x)
                             (vl-atts-p new-atts))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-vardecllist-reset-atts))
  (declare (ignorable __function__))
  (mbe
      :logic (if (consp x)
                 (cons (change-vl-vardecl (car x)
                                          :atts new-atts)
                       (vl-vardecllist-reset-atts (cdr x)
                                                  new-atts))
                 nil)
      :exec
      (if (atom x)
          nil
          (with-local-nrev
               (vl-vardecllist-reset-atts-nrev x new-atts nrev))))))
