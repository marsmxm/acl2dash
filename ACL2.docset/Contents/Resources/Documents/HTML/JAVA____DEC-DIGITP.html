Function: dec-digitp

(defun dec-digitp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'dec-digitp))
            (declare (ignorable __function__))
            (and (integerp x)
                 (<= (char-code #\0) x)
                 (<= x (char-code #\9)))))
