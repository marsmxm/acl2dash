Extend an alist by binding the names of vl-modinsts to their definitions.

Signature
(vl-modinstlist-alist x acc) → alist
Arguments
x — Guard (vl-modinstlist-p x).
Returns
alist — Type (equal (vl-modinst-alist-p alist) (vl-modinst-alist-p acc)).

This can be used as an alternative to vl-find-modinst when you need to perform a lot of lookups.

Definitions and Theorems

Function: vl-modinstlist-alist

(defun vl-modinstlist-alist (x acc)
       (declare (xargs :guard (vl-modinstlist-p x)))
       (let ((__function__ 'vl-modinstlist-alist))
            (declare (ignorable __function__))
            (b* (((when (atom x)) acc)
                 (name (hons-copy (vl-modinst->instname (car x))))
                 ((when (stringp name))
                  (cons (cons name (vl-modinst-fix (car x)))
                        (vl-modinstlist-alist (cdr x) acc))))
                (vl-modinstlist-alist (cdr x) acc))))

Theorem: return-type-of-vl-modinstlist-alist

(defthm return-type-of-vl-modinstlist-alist
        (b* ((alist (vl-modinstlist-alist x acc)))
            (equal (vl-modinst-alist-p alist)
                   (vl-modinst-alist-p acc)))
        :rule-classes :rewrite)

Theorem: lookup-in-vl-modinstlist-alist-acc-elim

(defthm
 lookup-in-vl-modinstlist-alist-acc-elim
 (implies
     (syntaxp (not (equal acc ''nil)))
     (equal (hons-assoc-equal name (vl-modinstlist-alist x acc))
            (or (hons-assoc-equal name (vl-modinstlist-alist x nil))
                (hons-assoc-equal name acc)))))

Theorem: lookup-in-vl-modinstlist-alist-fast

(defthm
    lookup-in-vl-modinstlist-alist-fast
    (implies
         (stringp name)
         (equal (hons-assoc-equal name (vl-modinstlist-alist x nil))
                (let ((val (vl-find-modinst name x)))
                     (and val (cons name val))))))

Theorem: vl-modinstlist-alist-of-vl-modinstlist-fix-x

(defthm vl-modinstlist-alist-of-vl-modinstlist-fix-x
        (equal (vl-modinstlist-alist (vl-modinstlist-fix x)
                                     acc)
               (vl-modinstlist-alist x acc)))

Theorem: vl-modinstlist-alist-vl-modinstlist-equiv-congruence-on-x

(defthm vl-modinstlist-alist-vl-modinstlist-equiv-congruence-on-x
