Function: declonp

(defun
 declonp (x)
 (declare (xargs :guard t))
 (let
   ((__function__ 'declonp))
   (declare (ignorable __function__))
   (and (consp x)
        (cond ((or (atom x) (eq (car x) :var))
               (and (true-listp (cdr x))
                    (eql (len (cdr x)) 3)
                    (b* ((type (std::da-nth 0 (cdr x)))
                         (declor (std::da-nth 1 (cdr x)))
                         (init (std::da-nth 2 (cdr x))))
                        (and (tyspecseqp type)
                             (declorp declor)
                             (exprp init)))))
              (t (and (eq (car x) :struct)
                      (and (true-listp (cdr x))
                           (eql (len (cdr x)) 2))
                      (b* ((tag (std::da-nth 0 (cdr x)))
                           (members (std::da-nth 1 (cdr x))))
                          (and (identp tag)
                               (struct-declon-listp members)))))))))
