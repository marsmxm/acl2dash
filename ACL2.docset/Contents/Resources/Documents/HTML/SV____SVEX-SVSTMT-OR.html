(defun svex-svstmt-or (a b)
       (declare (xargs :guard (and (svex-p a) (svex-p b))))
       (let ((__function__ 'svex-svstmt-or))
            (declare (ignorable __function__))
            (b* ((a (svex-fix a))
                 (b (svex-fix b))
                 ((when (svex-case a
                                   :quote (2vec-p a.val)
                                   :otherwise nil))
                  (b* ((a (2vec->val (svex-quote->val a))))
                      (if (eql a 0) b -1)))

                                   :quote (2vec-p b.val)
                                   :otherwise nil))
                  (b* ((b (2vec->val (svex-quote->val b))))
                      (if (eql b 0) a -1))))
