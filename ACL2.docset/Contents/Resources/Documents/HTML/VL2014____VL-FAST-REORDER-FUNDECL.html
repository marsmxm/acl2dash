(defun
 vl-fast-reorder-fundecl (names x fal)
 (declare
      (xargs :guard (and (string-listp names)
                         (vl-fundecllist-p x)
                         (equal fal (vl-fundecllist-alist x nil)))))
 (let ((__function__ 'vl-fast-reorder-fundecl))
      (declare (ignorable __function__))
      (b* (((when (atom names)) nil)
           (look (hons-get (car names) fal))
           ((when look)
            (cons (cdr look)
                  (vl-fast-reorder-fundecl (cdr names)
                                           x fal))))
          (vl-fast-reorder-fundecl (cdr names)
                                   x fal))))
