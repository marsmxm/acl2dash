(defun vl-plainarg-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-plainarg-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (std::prod-consp (std::prod-car x))
                 (std::prod-consp (std::prod-cdr x))
                 (b* ((expr (std::prod-car (std::prod-car x)))
                      (portname (std::prod-cdr (std::prod-car x)))
                      (dir (std::prod-car (std::prod-cdr x)))
                      (atts (std::prod-cdr (std::prod-cdr x))))
                     (and (vl-maybe-expr-p expr)
                          (maybe-stringp portname)
                          (vl-maybe-direction-p dir)
                          (vl-atts-p atts))))))
