Function: rulelist-called-rules

(defun rulelist-called-rules (rules)
       (declare (xargs :guard (rulelistp rules)))
       (cond ((endp rules) nil)
             (t (union (rule-called-rules (car rules))
                       (rulelist-called-rules (cdr rules))))))

(defthm rulename-setp-of-rulelist-called-rules
