(defun
   spaces1 (n col hard-right-margin acc)
   (declare (xargs :guard (and (natp n)
                               (natp col)
                               (posp hard-right-margin))))
   (let ((acl2::__function__ 'spaces1))
        (declare (ignorable acl2::__function__))
        (b* ((n (nfix n))
             (col (nfix col))
             (hard-right-margin (acl2::pos-fix hard-right-margin)))
            (cond ((zp n) acc)
