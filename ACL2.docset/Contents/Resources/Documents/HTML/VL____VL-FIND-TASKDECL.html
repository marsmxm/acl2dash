Function: vl-find-taskdecl

(defun vl-find-taskdecl (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-taskdecllist-p x))))
       (let ((__function__ 'vl-find-taskdecl))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-taskdecl->name (car x)))
                   (vl-taskdecl-fix (car x)))
                  (t (vl-find-taskdecl name (cdr x))))))
