Function: vl-interfaceref-to-svar

(defun
 vl-interfaceref-to-svar (x ss)
 (declare (xargs :guard (and (vl-expr-p x)
                             (vl-scopestack-p ss))))
 (let
  ((__function__ 'vl-interfaceref-to-svar))
  (declare (ignorable __function__))
  (b*
   ((x (vl-expr-fix x)))
   (vl-expr-case
    x
    :vl-index
    (b*
     (((when (or (not (vl-partselect-case x.part :none))
                 (consp x.indices)))
       (mv
         (vmsg "Don't yet support referencing interface arrays: ~a0"
               x)
         (svex-x)))
      ((mv err hidtrace context tail)
       (vl-follow-scopeexpr x.scope ss))
      ((when err)
       (mv (vmsg "Couldn't resolve interface reference to ~a0: ~@1"
                 x err)
           (svex-x)))
      (declname (vl-hidexpr-case tail
                                 :end tail.name
                                 :otherwise nil))
      ((unless declname)
       (mv
        (vmsg
         "Extra indexing on interface reference ~a0: ~a1. Modports
                       should have been removed?"
         x
         (make-vl-index :scope (make-vl-scopeexpr-end :hid tail)))
        (svex-x)))
      ((unless (vl-hidtrace-resolved-p hidtrace))
       (mv
          (vmsg "Unresolved hid indices on interface reference: ~a0"
                x)
          (svex-x)))
      (path (vl-hidtrace-to-path hidtrace nil))
      ((mv err addr)
       (vl-scopecontext-to-addr context ss path))
      ((when err)
       (mv
        (vmsg
         "Couldn't resolve interface reference to ~a0: context was
                       problematic? ~@1"
         x err)
        (svex-x))))
     (mv nil
         (sv::make-svex-var :name (sv::address->svar addr))))
    :otherwise
    (mv (vmsg "Bad expression for interface reference: ~a0"
              x)
        (svex-x))))))
