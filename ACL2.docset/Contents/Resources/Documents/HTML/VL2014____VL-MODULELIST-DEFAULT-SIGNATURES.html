Function: vl-modulelist-default-signatures

(defun
     vl-modulelist-default-signatures
     (names ss warnings)
     (declare (xargs :guard (and (string-listp names)
                                 (vl-scopestack-p ss)
                                 (vl-warninglist-p warnings))))
     (let ((__function__ 'vl-modulelist-default-signatures))
          (declare (ignorable __function__))
          (if (atom names)
              (mv nil (vl-warninglist-fix warnings))
              (b* (((mv sig sig-ss warnings)
                    (vl-module-default-signature (car names)
                                                 ss warnings))
                   ((mv sigs warnings)
                    (vl-modulelist-default-signatures (cdr names)
                                                      ss warnings)))
                  (mv (if sig (cons (cons sig sig-ss) sigs)
                          sigs)
                      warnings)))))
