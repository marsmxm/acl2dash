Function: function-setp

(defun function-setp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (functionp (car x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (acl2::fast-<< (car x) (cadr x))
                         (function-setp (cdr x)))))))
