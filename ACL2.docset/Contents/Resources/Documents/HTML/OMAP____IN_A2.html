Check if every key in a non-empty set is in an omap.

Signature
(in* keys map) → yes/no
Arguments
keys — Guard (setp keys).
map — Guard (mapp map).
Returns
yes/no — Type (booleanp yes/no).

This lifts in to sets of keys. However, this returns a boolean, while in returns a listp.

Definitions and Theorems

Function: in*

(defun in* (keys map)
       (declare (xargs :guard (and (setp keys) (mapp map))))
       (let ((__function__ 'in*))
            (declare (ignorable __function__))
            (cond ((set::empty keys) t)
                  (t (and (in (set::head keys) map)
                          (in* (set::tail keys) map))))))

Theorem: booleanp-of-in*

(defthm booleanp-of-in*
        (b* ((yes/no (in* keys map)))
            (booleanp yes/no))
        :rule-classes :rewrite)

Theorem: in*-when-left-empty

(defthm in*-when-left-empty
        (implies (set::empty keys)
                 (in* keys map)))

Theorem: in*-when-rigth-empty

(defthm in*-when-rigth-empty
        (implies (empty map)
                 (equal (in* keys map)
                        (set::empty keys))))

Theorem: in*-of-tail

(defthm in*-of-tail
        (implies (in* keys map)
                 (in* (set::tail keys) map)))
