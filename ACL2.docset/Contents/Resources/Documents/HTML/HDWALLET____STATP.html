Function: statp

(defun statp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'statp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :state)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 2)
                 (b* ((keys (std::da-nth 0 (cdr x)))
                      (addresses (std::da-nth 1 (cdr x))))
                     (and (bitcoin::bip32-key-treep keys)
                          (natp addresses))))))

Theorem: consp-when-statp

(defthm consp-when-statp
        (implies (statp x) (consp x))
        :rule-classes :compound-recognizer)
