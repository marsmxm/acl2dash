Function: vl-portdecl-exprsize

(defun
     vl-portdecl-exprsize (x ss warnings)
     (declare (xargs :guard (and (vl-portdecl-p x)
                                 (vl-scopestack-p ss)
                                 (vl-warninglist-p warnings))))
     (let ((__function__ 'vl-portdecl-exprsize))
          (declare (ignorable __function__))
          (b* ((x (vl-portdecl-fix x))
               (warnings (vl-warninglist-fix warnings)))
              (b* (((vl-portdecl x) x)
