Function: vl-modulelist-original-sources

(defun
 vl-modulelist-original-sources
 (x filemap)
 (declare (xargs :guard (and (vl-modulelist-p x)
                             (vl-filemap-p filemap))))
 (let
  ((__function__ 'vl-modulelist-original-sources))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-module-original-source (car x)
                                               filemap)
                    (vl-modulelist-original-sources (cdr x)
                                                    filemap))
              nil)
   :exec
   (if
      (atom x)
      nil
      (with-local-nrev
           (vl-modulelist-original-sources-nrev x filemap nrev))))))
