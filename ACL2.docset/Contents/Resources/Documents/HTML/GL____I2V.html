Function: i2v

(defun i2v (n)
       (declare (xargs :guard t))
       (let ((__function__ 'i2v))
            (declare (ignorable __function__))
            (cond ((eql 0 (ifix n)) '(nil))
                  ((eql n -1) '(t))
                  (t (bfr-scons (equal (logcar n) 1)
                                (i2v (logcdr n)))))))
