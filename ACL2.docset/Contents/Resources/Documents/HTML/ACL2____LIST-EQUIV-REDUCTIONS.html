Definitions and Theorems

Theorem: list-equiv-of-cons-and-cons

(defthm list-equiv-of-cons-and-cons
        (equal (list-equiv (cons a x) (cons a y))
               (list-equiv x y)))

Theorem: list-equiv-of-append-and-append

(defthm list-equiv-of-append-and-append
        (equal (list-equiv (append a x) (append a y))
               (list-equiv x y)))

Theorem: list-equiv-of-revappend-and-revappend

(defthm list-equiv-of-revappend-and-revappend
        (equal (list-equiv (revappend a x)
                           (revappend a y))
               (list-equiv x y)))
