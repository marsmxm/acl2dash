Theorem: return-type-of-vl-edgesynth-stmt-assigns.assigns

(defthm
 return-type-of-vl-edgesynth-stmt-assigns.assigns
 (implies
  (and (force (vl-stmt-p x))
       (force (vl-edgesynth-stmt-p x)))
  (b*
   ((?assigns (vl-edgesynth-stmt-assigns x)))
   (and
       (vl-stmtlist-p assigns)
