Function: vl-elabscopes-push-named-fn

(defun
 vl-elabscopes-push-named-fn
 (key scopes allow-empty)
 (declare (xargs :guard (and (vl-elabkey-p key)
                             (vl-elabscopes-p scopes))))
 (let
  ((__function__ 'vl-elabscopes-push-named))
  (declare (ignorable __function__))
  (b* ((scopes (vl-elabscopes-fix scopes))
       ((when (atom scopes))
        (and (not allow-empty)
             (raise "No outer scope -- can't push named scope!"))
        (cons (cons (vl-elabkey-fix key)
                    (make-vl-elabscope))
              scopes))
       (curr-elabscope (cdar scopes))
       (new-elabscope (or (vl-elabscope-subscope key curr-elabscope)
                          (make-vl-elabscope))))
      (cons (cons (vl-elabkey-fix key)
                  new-elabscope)
            scopes))))
