(defun vl-slow-keep-modules (names x)
       (declare (xargs :guard (and (string-listp names)
                                   (vl-modulelist-p x))))
       (let ((__function__ 'vl-slow-keep-modules))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((member-equal (vl-module->name (car x))
                                 (string-list-fix names))
                   (cons (vl-module-fix (car x))
                         (vl-slow-keep-modules names (cdr x))))
                  (t (vl-slow-keep-modules names (cdr x))))))
