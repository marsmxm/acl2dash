Function: uintp

(defun uintp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'uintp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :uint)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((get (std::da-nth 0 (cdr x))))
                     (uint-integerp get)))))
