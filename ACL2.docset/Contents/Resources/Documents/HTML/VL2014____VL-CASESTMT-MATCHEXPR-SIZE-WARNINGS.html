Function: vl-casestmt-matchexpr-size-warnings

(defun
 vl-casestmt-matchexpr-size-warnings
 (test match-exprs ctx)
 (declare (xargs :guard (and (vl-expr-p test)
                             (vl-exprlist-p match-exprs)
                             (vl-modelement-p ctx))))
 (let
  ((__function__ 'vl-casestmt-matchexpr-size-warnings))
  (declare (ignorable __function__))
  (b*
   (((when (atom match-exprs)) nil)
    (test (vl-expr-fix test))
    (expr1 (vl-expr-fix (car match-exprs)))
    (ctx (vl-modelement-fix ctx))
    (rest
         (vl-casestmt-matchexpr-size-warnings test (cdr match-exprs)
                                              ctx))
    ((unless (vl-expr->finaltype expr1))
     (warn
      :type :vl-case-stmt-type
      :msg
      "In ~a0: failed to determine signedness of case-statement ~
                    match expression: ~a1."
      :args (list ctx expr1)
      :acc rest))
    ((unless (eql (vl-expr->finalwidth test)
                  (vl-expr->finalwidth expr1)))
     (warn
      :type :vl-case-stmt-size
      :msg
      "In ~a0: case statement sizes are incompatible:~%     ~
                      - ~x1-bit test:  ~a3~%     ~
                      - ~x2-bit match: ~a4"
      :args (list ctx (vl-expr->finalwidth test)
                  (vl-expr->finalwidth expr1)
                  test expr1)
      :acc rest)))
   rest)))
