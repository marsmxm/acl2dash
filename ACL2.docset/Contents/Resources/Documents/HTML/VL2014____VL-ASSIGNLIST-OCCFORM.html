Function: vl-assignlist-occform

(defun
 vl-assignlist-occform
 (x nf mod ialist warnings)
 (declare
    (xargs :guard (and (vl-assignlist-p x)
                       (vl-namefactory-p nf)
                       (vl-module-p mod)
                       (vl-warninglist-p warnings)
                       (equal ialist (vl-make-moditem-alist mod)))))
 (let ((__function__ 'vl-assignlist-occform))
      (declare (ignorable __function__))
      (b* (((when (atom x))
            (mv (ok)
                nil nil nil (vl-namefactory-fix nf)))
           ((mv warnings mods1 modinsts1 assigns1 nf)
            (vl-assign-occform (car x)
                               nf mod ialist warnings))
           ((mv warnings mods2 modinsts2 assigns2 nf)
            (vl-assignlist-occform (cdr x)
                                   nf mod ialist warnings))
           (mods (append mods1 mods2))
           (modinsts (append modinsts1 modinsts2))
           (assigns (append assigns1 assigns2)))
          (mv warnings mods modinsts assigns nf))))
