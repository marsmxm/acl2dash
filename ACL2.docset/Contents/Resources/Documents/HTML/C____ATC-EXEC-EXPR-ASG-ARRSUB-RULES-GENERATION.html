(defun
 atc-exec-expr-asg-arrsub-rules-gen
 (atype itype)
 (declare (xargs :guard (and (typep atype) (typep itype))))
 (declare (xargs :guard (and (type-integerp atype)
                             (type-integerp itype))))
 (let
  ((__function__ 'atc-exec-expr-asg-arrsub-rules-gen))
  (declare (ignorable __function__))
  (b*
   ((afixtype (integer-type-to-fixtype atype))
    (ifixtype (integer-type-to-fixtype itype))
    (apred (pack afixtype '-arrayp))
    (epred (pack afixtype 'p))
    (ipred (pack ifixtype 'p))
    (atype-array-itype-index-okp (pack afixtype '-array-
                                       ifixtype '-index-okp))
    (atype-array-index-okp (pack afixtype '-array-index-okp))
    (atype-array-write-itype
         (pack afixtype '-array-write- ifixtype))
    (atype-array-write-alt-def
         (pack afixtype '-array-write-alt-def))
    (elemtype-when-apred (pack 'value-array->elemtype-when-
                               apred))
    (name (pack 'exec-expr-asg-arrsub-when-
                apred '-and-
                ipred))
    (formula
     (cons
      'implies
      (cons
       (cons
        'and
        (cons
         '(syntaxp (quotep e))
         (cons
          '(equal (expr-kind e) :binary)
          (cons
           '(equal (binop-kind (expr-binary->op e))
                   :asg)
           (cons
            '(equal left (expr-binary->arg1 e))
            (cons
             '(equal right (expr-binary->arg2 e))
             (cons
              '(equal (expr-kind left) :arrsub)
              (cons
               '(equal arr (expr-arrsub->arr left))
               (cons
                '(equal sub (expr-arrsub->sub left))
                (cons
                 '(equal (expr-kind arr) :ident)
                 (cons
                  '(not (zp limit))
                  (cons
                   '(equal ptr
                           (read-var (expr-ident->get arr) compst))
                   (cons
                    '(valuep ptr)
                    (cons
                     '(value-case ptr :pointer)
                     (cons
                      '(not (value-pointer-nullp ptr))
                      (cons
                       (cons
                           'equal
                           (cons '(value-pointer->reftype ptr)
                                 (cons (type-to-maker atype) 'nil)))
                       (cons
                        '(equal array
                                (read-object
                                     (value-pointer->designator ptr)
                                     compst))
                        (cons
                         (cons apred '(array))
                         (cons
                          '(equal index (exec-expr-pure sub compst))
                          (cons
                           (cons ipred '(index))
                           (cons
                            (cons atype-array-itype-index-okp
                                  '(array index))
                            (cons
                             '(equal
                                  val (exec-expr-pure right compst))
                             (cons (cons epred '(val))
                                   'nil)))))))))))))))))))))))
       (cons
        (cons
         'equal
         (cons '(exec-expr-asg e compst fenv limit)
               (cons (cons 'write-object
                           (cons '(value-pointer->designator ptr)
                                 (cons (cons atype-array-write-itype
                                             '(array index val))
                                       '(compst))))
                     'nil)))
        'nil))))
    (event
     (cons
      'defruled
      (cons
       name
       (cons
        formula
        (cons
         ':enable
         (cons
          (cons
           'exec-expr-asg
           (cons
              'exec-integer
              (cons atype-array-itype-index-okp
                    (cons atype-array-write-itype
                          (cons atype-array-write-alt-def
                                (cons elemtype-when-apred 'nil))))))
          (cons
           ':prep-lemmas
           (cons
            (cons
             (cons
              'defrule
              (cons
               'lemma1
               (cons
                (cons
                 'implies
                 (cons
                  (cons
                   'and
                   (cons (cons atype-array-index-okp '(array index))
                         '((integerp index))))
                  '((not (< index 0)))))
                (cons ':enable
                      (cons atype-array-index-okp 'nil)))))
             (cons
              (cons
               'defrule
               (cons
                'lemma2
                (cons
                 (cons
                  'implies
                  (cons
                   (cons
                    'and
                    (cons
                     (cons apred '(array))
                     (cons
                      '(integerp index)
                      (cons
                         (cons atype-array-index-okp '(array index))
                         (cons (cons epred '(val)) 'nil)))))
                   '((not
                       (errorp
                            (value-array-write index val array))))))
                 (cons ':use
                       (cons (cons ':instance
                                   (cons atype-array-write-alt-def
                                         '((elem val))))
                             'nil)))))
              'nil))
            'nil)))))))))
   (mv name event))))
