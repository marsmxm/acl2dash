Function: vl-build-taskports

(defun
 vl-build-taskports (atts dir type names)
 (declare (xargs :guard (and (vl-atts-p atts)
                             (vl-direction-p dir)
                             (vl-datatype-p type)
                             (vl-idtoken-list-p names))))
 (let
    ((__function__ 'vl-build-taskports))
    (declare (ignorable __function__))
    (if (atom names)
        nil
        (cons (make-vl-portdecl :name (vl-idtoken->name (car names))
                                :dir dir
                                :nettype nil
                                :type type
                                :atts atts
                                :loc (vl-token->loc (car names)))
              (vl-build-taskports atts dir type (cdr names))))))
