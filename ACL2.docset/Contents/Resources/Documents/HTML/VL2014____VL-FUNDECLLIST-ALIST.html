Extend an alist by binding the names of vl-fundecls to their definitions.

Signature
(vl-fundecllist-alist x acc) → alist
Arguments
x — Guard (vl-fundecllist-p x).
Returns
alist — Type (equal (vl-fundecl-alist-p alist) (vl-fundecl-alist-p acc)).

This can be used as an alternative to vl-find-fundecl when you need to perform a lot of lookups.

Definitions and Theorems

Function: vl-fundecllist-alist

(defun vl-fundecllist-alist (x acc)
       (declare (xargs :guard (vl-fundecllist-p x)))
       (let ((__function__ 'vl-fundecllist-alist))
            (declare (ignorable __function__))
            (if (atom x)
                acc
                (cons (cons (vl-fundecl->name (car x))
                            (vl-fundecl-fix (car x)))
                      (vl-fundecllist-alist (cdr x) acc)))))

Theorem: return-type-of-vl-fundecllist-alist

(defthm return-type-of-vl-fundecllist-alist
        (b* ((alist (vl-fundecllist-alist x acc)))
            (equal (vl-fundecl-alist-p alist)
                   (vl-fundecl-alist-p acc)))
        :rule-classes :rewrite)

Theorem: lookup-in-vl-fundecllist-alist-acc-elim

(defthm
 lookup-in-vl-fundecllist-alist-acc-elim
 (implies
     (syntaxp (not (equal acc ''nil)))
     (equal (hons-assoc-equal name (vl-fundecllist-alist x acc))
            (or (hons-assoc-equal name (vl-fundecllist-alist x nil))
                (hons-assoc-equal name acc)))))

Theorem: lookup-in-vl-fundecllist-alist-fast

(defthm
    lookup-in-vl-fundecllist-alist-fast
    (implies
         (stringp name)
         (equal (hons-assoc-equal name (vl-fundecllist-alist x nil))
                (let ((val (vl-find-fundecl name x)))
                     (and val (cons name val))))))
