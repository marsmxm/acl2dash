Function: vl-maybe-integer-listp

(defun vl-maybe-integer-listp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-maybe-integer-listp))
            (declare (ignorable __function__))
            (if (consp x)
                (and (maybe-integerp (car x))
                     (vl-maybe-integer-listp (cdr x)))
                (null x))))
