Function: occs

(defun occs (x bag)
       (declare (xargs :guard (bagp bag)))
       (let ((__function__ 'occs))
            (declare (ignorable __function__))
            (cond ((empty bag) 0)
                  ((equal x (head bag))
                   (1+ (occs x (tail bag))))
                  (t (occs x (tail bag))))))

Theorem: natp-of-occs

(defthm natp-of-occs
        (b* ((occs (occs x bag))) (natp occs))
        :rule-classes :rewrite)
