(defun vl-find-interfaceport (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-interfaceportlist-p x))))
       (let ((__function__ 'vl-find-interfaceport))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
