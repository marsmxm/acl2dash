Function: check-tyname

(defun check-tyname (tyname tagenv)
       (declare (xargs :guard (and (tynamep tyname)
                                   (tag-envp tagenv))))
       (let ((__function__ 'check-tyname))
            (declare (ignorable __function__))
            (b* ((wf? (check-tyspecseq (tyname->tyspec tyname)
                                       tagenv))
                 ((when (errorp wf?)) wf?)
                 (wf? (check-obj-adeclor (tyname->declor tyname)))
                 ((when (errorp wf?)) wf?))
                :wellformed)))
