Function: vl-modulelist-basicsanity

(defun
 vl-modulelist-basicsanity (x)
 (declare (xargs :guard (vl-modulelist-p x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-modulelist-basicsanity))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-module-basicsanity (car x))
                    (vl-modulelist-basicsanity (cdr x)))
              nil)
   :exec (if (atom x)
             nil
             (with-local-nrev
                  (let ((nrev (nrev::nrev-set-hint x nrev)))
                       (vl-modulelist-basicsanity-nrev x nrev)))))))
