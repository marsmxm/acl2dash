Function: vl-oddexpr-check

(defun
 vl-oddexpr-check (x ctx ss)
 (declare (xargs :guard (and (vl-expr-p x)
                             (vl-context-p ctx)
                             (vl-scopestack-p ss))))
 (let
  ((__function__ 'vl-oddexpr-check))
  (declare (ignorable __function__))
  (b*
   ((details (append (vl-warn-odd-binary-expression x ss)))
    ((unless details) nil))
   (list
    (make-vl-warning
     :type :vl-warn-oddexpr
     :msg
     (cat
      "~a0: found ~s1 that suggest precedence problems may be ~
                      present.  Maybe add explicit parens?  Details:~%"
      (with-local-ps (vl-pp-oddexpr-details details)))
     :args (list ctx
                 (if (vl-plural-p details)
                     "subexpressions"
                     "a subexpression")))))))
