Function: rulelist-defined-rules

(defun rulelist-defined-rules (rules)
       (declare (xargs :guard (rulelistp rules)))
       (cond ((endp rules) nil)
             (t (insert (rule->name (car rules))
                        (rulelist-defined-rules (cdr rules))))))
