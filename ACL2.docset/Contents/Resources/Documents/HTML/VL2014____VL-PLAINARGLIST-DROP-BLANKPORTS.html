(defun
   vl-plainarglist-drop-blankports
   (args ports)

   (declare (xargs :guard (same-lengthp args ports)))
   (let ((__function__ 'vl-plainarglist-drop-blankports))
        (declare (ignorable __function__))
        (b* (((when (atom args)) nil)
             (port1 (vl-port-fix (car ports)))

                        (vl-regularport->expr port1)))
              (cons (vl-plainarg-fix (car args))
                    (vl-plainarglist-drop-blankports (cdr args)
                                                     (cdr ports)))))
            (vl-plainarglist-drop-blankports (cdr args)
                                             (cdr ports)))))
