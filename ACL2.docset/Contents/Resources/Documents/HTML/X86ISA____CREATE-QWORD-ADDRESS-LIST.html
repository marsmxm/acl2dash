(defun
 create-qword-address-list (count addr)
 (declare (type (unsigned-byte 52) addr))
 (declare (xargs :guard (natp count)))
 (declare
      (xargs :guard (physical-address-p (+ -1 (ash count 3) addr))))
 (let ((__function__ 'create-qword-address-list))
      (declare (ignorable __function__))
      (if (or (zp count)
              (not (physical-address-p addr))
              (not (physical-address-p (+ 7 addr))))
          nil

                                               (+ 8 addr)))))))
