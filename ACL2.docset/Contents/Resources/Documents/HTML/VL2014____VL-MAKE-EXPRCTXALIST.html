Function: vl-make-exprctxalist

(defun
 vl-make-exprctxalist (exprs ctx)
 (declare (xargs :guard (and (vl-exprlist-p exprs)
                             (vl-context1-p ctx))))
 (let
  ((__function__ 'vl-make-exprctxalist))
  (declare (ignorable __function__))
  (mbe
     :logic (if (atom exprs)
