Function: vl-jp-propspec-fn

(defun
     vl-jp-propspec-fn (x ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard (vl-propspec-p x)))
     (let ((__function__ 'vl-jp-propspec))
          (declare (ignorable __function__))
          (vl-ps-seq (vl-print "{\"tag\": ")
                     (jp-sym ':vl-propspec)
                     (vl-print ", ")
                     (vl-print-str "\"evatoms\": ")
                     (vl-jp-evatomlist (vl-propspec->evatoms x))
                     (vl-println? ", ")
                     (vl-print-str "\"disable\": ")
                     (vl-jp-maybe-exprdist (vl-propspec->disable x))
                     (vl-println? ", ")
                     (vl-print-str "\"prop\": ")
                     (vl-jp-propexpr (vl-propspec->prop x))
                     (vl-println? ", ")
                     (vl-print-str "\"loc\": ")
                     (vl-jp-location (vl-propspec->loc x))
                     (vl-println? "}"))))
