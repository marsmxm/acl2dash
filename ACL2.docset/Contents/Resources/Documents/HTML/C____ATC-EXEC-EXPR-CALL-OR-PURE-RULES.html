Definitions and Theorems

Theorem: exec-expr-call-or-pure-when-pure

(defthm exec-expr-call-or-pure-when-pure
        (implies (and (syntaxp (quotep e))
                      (not (equal (expr-kind e) :call))
                      (not (zp limit))
                      (compustatep compst))
                 (equal (exec-expr-call-or-pure e compst fenv limit)
                        (mv (exec-expr-pure e compst) compst))))

Theorem: exec-expr-call-of-pure-when-call

(defthm exec-expr-call-of-pure-when-call
        (implies (and (syntaxp (quotep e))
                      (equal (expr-kind e) :call)
                      (not (zp limit)))
                 (equal (exec-expr-call-or-pure e compst fenv limit)
                        (exec-expr-call (expr-call->fun e)
                                        (expr-call->args e)
                                        compst fenv (1- limit)))))
