Function: pseudo-term-const->val

(defun pseudo-term-const->val (x)
       (declare (xargs :guard (pseudo-termp x)))
       (declare (xargs :guard (member-eq (pseudo-term-kind x)
                                         '(:quote :null))))
       (let ((__function__ 'pseudo-term-const->val))
            (declare (ignorable __function__))
            (mbe :logic (and (eq (pseudo-term-kind x) :quote)
                             (pseudo-term-quote->val x))
                 :exec (cadr x))))

(defthm pseudo-term-const->val-when-quote
