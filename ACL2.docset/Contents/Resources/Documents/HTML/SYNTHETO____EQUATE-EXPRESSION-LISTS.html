Function: equate-expression-lists

(defun
     equate-expression-lists (lefts rights)
     (declare (xargs :guard (and (expression-listp lefts)
                                 (expression-listp rights))))
     (declare (xargs :guard (= (len lefts) (len rights))))
     (let ((__function__ 'equate-expression-lists))
          (declare (ignorable __function__))
          (cond ((endp lefts) nil)
                (t (cons (equate-expressions (car lefts)
                                             (car rights))
                         (equate-expression-lists (cdr lefts)
                                                  (cdr rights)))))))
