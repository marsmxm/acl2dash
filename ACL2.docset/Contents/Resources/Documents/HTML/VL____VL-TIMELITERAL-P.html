Function: vl-timeliteral-p

(defun vl-timeliteral-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-timeliteral-p))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :vl-timeliteral)
                 (consp (cdr x))
                 (b* ((quantity (car (cdr x)))
                      (units (cdr (cdr x))))
                     (and (stringp quantity)
                          (vl-timeunit-p units))))))
