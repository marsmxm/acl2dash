Function: atc-make-mv-nth-terms

(defun
 atc-make-mv-nth-terms (indices term)
 (declare (xargs :guard (and (nat-listp indices)
                             (pseudo-termp term))))
 (let
  ((__function__ 'atc-make-mv-nth-terms))
  (declare (ignorable __function__))
  (cond ((endp indices) nil)
        (t (cons (cons 'mv-nth
                       (cons (cons 'quote (cons (car indices) 'nil))
                             (cons (pseudo-term-fix term) 'nil)))
                 (atc-make-mv-nth-terms (cdr indices)
                                        term))))))
