A list of lists of trees matches a concatenation iff each list of trees matches the corresponding repetition.

Function: tree-list-list-match-concatenation-p

(defun
 tree-list-list-match-concatenation-p
 (treess concatenation rules)
 (declare (xargs :guard (and (tree-list-listp treess)
                             (concatenationp concatenation)
                             (rulelistp rules))))
 (cond
   ((endp treess) (endp concatenation))
   (t (and (consp concatenation)
           (tree-list-match-repetition-p (car treess)
                                         (car concatenation)
                                         rules)
           (tree-list-list-match-concatenation-p (cdr treess)
                                                 (cdr concatenation)
                                                 rules)))))
