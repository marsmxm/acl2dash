Function: vl-module-suppress-file-warnings

(defun
    vl-module-suppress-file-warnings
    (x suppress-files)
    (declare (xargs :guard (and (vl-module-p x)
                                (string-listp suppress-files))))
    (let ((__function__ 'vl-module-suppress-file-warnings))
         (declare (ignorable __function__))
         (if (and (consp (vl-module->warnings x))
                  (vl-filename-to-suppress-p
                       (vl-location->filename (vl-module->minloc x))
                       suppress-files))
             (change-vl-module x :warnings nil)
             (vl-module-fix x))))
