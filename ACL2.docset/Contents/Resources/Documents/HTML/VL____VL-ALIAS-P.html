(defun vl-alias-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-alias-p))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :vl-alias)
                 (std::prod-consp (cdr x))
                 (std::prod-consp (std::prod-car (cdr x)))
                 (std::prod-consp (std::prod-cdr (cdr x)))
