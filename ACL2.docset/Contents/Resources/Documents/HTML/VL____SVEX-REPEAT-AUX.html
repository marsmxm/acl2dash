Function: svex-repeat-aux

(defun svex-repeat-aux (n width x acc)
       (declare (xargs :guard (and (natp n)
                                   (natp width)
                                   (sv::svex-p x)
                                   (sv::svex-p acc))))
       (let ((__function__ 'svex-repeat-aux))
            (declare (ignorable __function__))
            (if (zp n)
                (sv::svex-fix acc)
