Definitions and Theorems

Function: check-obj-adeclor

(defun check-obj-adeclor (declor)
       (declare (xargs :guard (obj-adeclorp declor)))
       (let ((__function__ 'check-obj-adeclor))
            (declare (ignorable __function__))
            (obj-adeclor-case
                 declor
                 :none :wellformed
                 :pointer (check-obj-adeclor declor.decl)
                 :array (b* ((wf? (check-obj-adeclor declor.decl))
                             ((when (errorp wf?)) wf?)
                             ((unless declor.size) :wellformed)
                             (type? (check-iconst declor.size))
                             ((when (errorp type?)) type?))
                            :wellformed))))

(defthm wellformed-resultp-of-check-obj-adeclor
