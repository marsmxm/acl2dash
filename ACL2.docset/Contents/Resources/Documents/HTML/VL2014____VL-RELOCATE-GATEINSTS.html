Function: vl-relocate-gateinsts

(defun
 vl-relocate-gateinsts (loc x)
 (declare (xargs :guard (and (vl-location-p loc)
                             (vl-gateinstlist-p x))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-relocate-gateinsts))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (change-vl-gateinst (car x) :loc loc)
                    (vl-relocate-gateinsts loc (cdr x)))
              nil)
   :exec
   (if (atom x)
       nil
       (with-local-nrev (vl-relocate-gateinsts-nrev loc x nrev))))))
