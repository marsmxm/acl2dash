Function: vl-jp-portdecl-fn

(defun
  vl-jp-portdecl-fn (x ps)
  (declare (xargs :stobjs (ps)))
  (declare (xargs :guard (vl-portdecl-p x)))
  (let ((__function__ 'vl-jp-portdecl))
       (declare (ignorable __function__))
       (vl-ps-seq (vl-print "{\"tag\": ")
                  (jp-sym ':vl-portdecl)
                  (vl-print ", ")
                  (vl-print-str "\"name\": ")
                  (jp-str (vl-portdecl->name x))
                  (vl-println? ", ")
                  (vl-print-str "\"loc\": ")
                  (vl-jp-location (vl-portdecl->loc x))
                  (vl-println? ", ")
                  (vl-print-str "\"dir\": ")
                  (vl-jp-direction (vl-portdecl->dir x))
                  (vl-println? ", ")
                  (vl-print-str "\"type\": ")
                  (vl-jp-datatype (vl-portdecl->type x))
                  (vl-println? ", ")
                  (vl-print-str "\"default\": ")
                  (vl-jp-maybe-expr (vl-portdecl->default x))
                  (vl-println? ", ")
                  (vl-print-str "\"nettype\": ")
                  (vl-jp-maybe-nettypename (vl-portdecl->nettype x))
                  (vl-println? ", ")
                  (vl-print-str "\"atts\": ")
                  (vl-jp-atts (vl-portdecl->atts x))
                  (vl-println? "}"))))
