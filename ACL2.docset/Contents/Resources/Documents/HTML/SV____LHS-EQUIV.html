(defthm lhs-equiv-is-an-equivalence
        (and (booleanp (lhs-equiv x y))
             (lhs-equiv x x)
             (implies (lhs-equiv x y)
                      (lhs-equiv y x))
             (implies (and (lhs-equiv x y) (lhs-equiv y z))
                      (lhs-equiv x z)))
        :rule-classes (:equivalence))
