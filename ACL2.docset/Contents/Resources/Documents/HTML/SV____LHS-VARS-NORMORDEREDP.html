(defun
 lhs-vars-normorderedp (bound offset x)
 (declare (xargs :guard (and (integerp bound)
                             (natp offset)
                             (lhs-p x))))
 (let
  ((__function__ 'lhs-vars-normorderedp))
  (declare (ignorable __function__))
  (if
   (atom x)
