Function: lhs-eval

(defun lhs-eval (x env)
       (declare (xargs :guard (and (lhs-p x) (svex-env-p env))))
       (let ((__function__ 'lhs-eval))
            (declare (ignorable __function__))
            (if (atom x)
                (4vec-z)
                (4vec-concat (2vec (lhrange->w (car x)))
                             (lhrange-eval (car x) env)
                             (lhs-eval (cdr x) env)))))
