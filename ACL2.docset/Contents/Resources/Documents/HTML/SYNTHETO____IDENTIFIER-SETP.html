Function: identifier-setp

(defun identifier-setp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (identifierp (car x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (acl2::fast-<< (car x) (cadr x))
                         (identifier-setp (cdr x)))))))
