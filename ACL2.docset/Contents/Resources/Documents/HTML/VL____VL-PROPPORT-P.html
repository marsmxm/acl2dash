Function: vl-propport-p

(defun
 vl-propport-p (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-propport-p))
  (declare (ignorable __function__))
  (and
   (consp x)
   (eq (car x) :vl-propport)
   (mbe
       :logic (and (alistp (cdr x))
                   (equal (strip-cars (cdr x))
                          '(name localp dir type arg atts loc)))
       :exec
       (fty::alist-with-carsp (cdr x)
                              '(name localp dir type arg atts loc)))
   (b* ((name (cdr (std::da-nth 0 (cdr x))))
        (localp (cdr (std::da-nth 1 (cdr x))))
        (dir (cdr (std::da-nth 2 (cdr x))))
        (type (cdr (std::da-nth 3 (cdr x))))
        (arg (cdr (std::da-nth 4 (cdr x))))
        (atts (cdr (std::da-nth 5 (cdr x))))
        (loc (cdr (std::da-nth 6 (cdr x)))))
       (and (stringp name)
            (booleanp localp)
            (vl-direction-p dir)
            (vl-datatype-p type)
            (vl-propactual-p arg)

            (vl-location-p loc))))))
