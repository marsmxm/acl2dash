Function: matchstate-p

(defun matchstate-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'matchstate-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (b* ((index (std::prod-car x))
                      (backrefs (std::prod-cdr x)))
                     (and (natp index)
                          (backref-alist-p backrefs))))))
