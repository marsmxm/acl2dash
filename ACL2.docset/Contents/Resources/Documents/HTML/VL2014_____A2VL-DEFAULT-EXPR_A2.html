Definition: *vl-default-expr*

(defconst
 *vl-default-expr*
 (let
     ((guts (make-vl-constint :origwidth 1
                              :value 0
                              :origtype :vl-unsigned))
      (finalwidth nil)
      (finaltype nil)
      (atts nil))
     (cons :atom (std::prod-cons (std::prod-cons guts finalwidth)
                                 (std::prod-cons finaltype atts)))))
