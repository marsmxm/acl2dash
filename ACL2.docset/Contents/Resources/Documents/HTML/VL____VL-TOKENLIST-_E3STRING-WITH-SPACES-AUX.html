Function: vl-tokenlist->string-with-spaces-aux

(defun vl-tokenlist->string-with-spaces-aux
       (x acc)
       (declare (xargs :guard (vl-tokenlist-p x)))
       (let ((__function__ 'vl-tokenlist->string-with-spaces-aux))
            (declare (ignorable __function__))
            (b* (((when (atom x)) acc)
                 (echars1 (vl-token->etext (car x)))
                 (acc (vl-echarlist->chars-exec echars1 acc))
                 ((when (atom (cdr x))) acc)
                 (acc (cons #\Space acc)))
                (vl-tokenlist->string-with-spaces-aux (cdr x)
                                                      acc))))
