(defun string-keyed-alist-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'string-keyed-alist-p))
            (declare (ignorable __function__))
            (cond ((atom x) (null x))
                  (t (and (consp (car x))
                          (stringp (caar x))
                          (string-keyed-alist-p (cdr x)))))))
