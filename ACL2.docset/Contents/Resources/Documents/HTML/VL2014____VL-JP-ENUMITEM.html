Function: vl-jp-enumitem-fn

(defun vl-jp-enumitem-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-enumitem-p x)))
       (let ((__function__ 'vl-jp-enumitem))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-enumitem)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-enumitem->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"range\": ")
                       (vl-jp-maybe-range (vl-enumitem->range x))
                       (vl-println? ", ")
                       (vl-print-str "\"value\": ")
                       (vl-jp-maybe-expr (vl-enumitem->value x))
                       (vl-println? "}"))))
