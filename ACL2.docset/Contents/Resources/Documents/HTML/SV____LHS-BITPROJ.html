Function: lhs-bitproj

(defun lhs-bitproj (idx x)
       (declare (xargs :guard (and (natp idx) (lhs-p x))))
       (let ((__function__ 'lhs-bitproj))
            (declare (ignorable __function__))
            (if (atom x)
                (lhbit-z)
                (b* (((lhrange xf) (car x))
                     (idx (lnfix idx)))
                    (if (< idx xf.w)
                        (lhatom-bitproj idx xf.atom)
                        (lhs-bitproj (- idx xf.w) (cdr x)))))))
