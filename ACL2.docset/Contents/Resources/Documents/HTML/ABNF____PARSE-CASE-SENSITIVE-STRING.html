Function: parse-case-sensitive-string

(defun
 parse-case-sensitive-string (input)
 (declare (xargs :guard (nat-listp input)))
 (seq
    input
    (tree-%s := (parse-ichar2 #\% #\s input))
    (tree-qstring := (parse-quoted-string input))
    (return
         (make-tree-nonleaf :rulename? *case-sensitive-string*
                            :branches (list (list tree-%s)
                                            (list tree-qstring))))))
