 vl-iframe-p (x)

  ((__function__ 'vl-iframe-p))

      (std::prod-consp (std::prod-cdr x))
      (std::prod-consp (std::prod-cdr (std::prod-cdr x)))

           (some-thing-satisfiedp (std::prod-cdr (std::prod-car x)))
