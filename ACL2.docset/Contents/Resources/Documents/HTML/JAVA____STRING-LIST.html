Fixtype of lists of ACL2 strings, i.e. values recognized by string-listp.

This is not specific to Java, and it should be moved to a more general library eventually.

Definitions and Theorems

Function: string-list-fix

(defun string-list-fix (x)
       (declare (xargs :guard (string-listp x)))
       (mbe :logic (if (string-listp x) x nil)
            :exec x))

Theorem: string-listp-of-string-list-fix

(defthm string-listp-of-string-list-fix
        (b* ((fixed-x (string-list-fix x)))
            (string-listp fixed-x))
        :rule-classes :rewrite)

(defthm string-list-fix-when-string-listp

Function: string-list-equiv$inline

(defun string-list-equiv$inline
       (acl2::x acl2::y)
       (declare (xargs :guard (and (string-listp acl2::x)
                                   (string-listp acl2::y))))
       (equal (string-list-fix acl2::x)
              (string-list-fix acl2::y)))

Theorem: string-list-equiv-is-an-equivalence

(defthm string-list-equiv-is-an-equivalence
        (and (booleanp (string-list-equiv x y))
             (string-list-equiv x x)
             (implies (string-list-equiv x y)
                      (string-list-equiv y x))
             (implies (and (string-list-equiv x y)
                           (string-list-equiv y z))
                      (string-list-equiv x z)))
        :rule-classes (:equivalence))
