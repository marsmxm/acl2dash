B* binder to automatically join returned warnings to a vttree

This is a b* binder introduced with ACL2::def-b*-binder.

Macro: patbind-wvmv

(defmacro
 patbind-wvmv
 (args acl2::forms acl2::rest-expr)
 (b*
  (((mv ctx args)
    (b* ((mem (member :ctx args)))
        (if mem
            (mv (cadr mem)
                (append (take (- (len args) (len mem)) args)
                        (cddr mem)))
            (mv nil args)))))
  (cons
   'b*
   (cons
    (cons
     (if (equal args '(vttree))
         (cons '__tmp__warnings acl2::forms)
         (cons (cons 'mv
                     (subst '__tmp__warnings 'vttree args))
               acl2::forms))
     (cons
       (cons 'vttree
             (cons (cons 'vttree-add-warnings
                         (cons '__tmp__warnings
                               (cons 'vttree
                                     (cons ':ctx (cons ctx 'nil)))))
                   'nil))
       'nil))
    (cons acl2::rest-expr 'nil)))))
