Helper function for creating lists of primitive port declarations.

Signature
(vl-primitive-mkports prefix i n &key dir (loc '*vl-fakeloc*)) 
  → 
(mv exprs ports portdecls vardecls)
Arguments
prefix — name prefix for each port.
    Guard (stringp prefix).
i — starting index, inclusive!.
    Guard (natp i).
n — ending index, non-inclusive!.
    Guard (natp n).
dir — direction of each port.
    Guard (vl-direction-p dir).
loc — Guard (vl-location-p loc).
Returns
exprs — Type (vl-exprlist-p exprs), given the guard.
ports — Type (vl-portlist-p ports), given the guard.
portdecls — Type (vl-portdecllist-p portdecls), given the guard.
vardecls — Type (vl-vardecllist-p vardecls), given the guard.

Function: vl-primitive-mkports-fn

(defun vl-primitive-mkports-fn
       (prefix i n dir loc)
       (declare (xargs :guard (and (stringp prefix)
                                   (natp i)
                                   (natp n)
                                   (vl-direction-p dir)
                                   (vl-location-p loc))))
       (declare (xargs :guard (<= i n)))
       (let ((__function__ 'vl-primitive-mkports))
            (declare (ignorable __function__))
            (b* (((when (mbe :logic (zp (- (lnfix n) (lnfix i)))
                             :exec (eql i n)))
                  (mv nil nil nil nil))
                 (name1 (hons-copy (cat prefix (natstr i))))
                 ((mv expr1 port1 portdecl1 vardecl1)
                  (vl-primitive-mkport name1 dir))
                 ((mv exprs2 ports2 portdecls2 vardecls2)
                  (vl-primitive-mkports prefix (+ 1 (lnfix i))
                                        n
                                        :dir dir
                                        :loc loc)))
                (mv (cons expr1 exprs2)
                    (cons port1 ports2)
                    (cons portdecl1 portdecls2)
                    (cons vardecl1 vardecls2)))))

Theorem: len-of-vl-primitive-mkports

(defthm len-of-vl-primitive-mkports
        (b* (((mv exprs ports portdecls vardecls)
              (vl-primitive-mkports prefix i n
                                    :dir dir
                                    :loc loc))
             (len (nfix (- (nfix n) (nfix i)))))
            (and (equal (len exprs) len)
                 (equal (len ports) len)
                 (equal (len portdecls) len)
                 (equal (len vardecls) len))))
