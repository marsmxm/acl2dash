Function: lhs-rest

(defun
 lhs-rest (x)
 (declare (xargs :guard (lhs-p x)))
 (let
  ((__function__ 'lhs-rest))
  (declare (ignorable __function__))
  (mbe
   :logic (if (or (atom x) (atom (cdr x)))
              nil
              (if (lhrange-combine (car x) (cadr x))
                  (lhs-rest (cdr x))
                  (lhs-fix (cdr x))))
   :exec (if (consp x)
             (if (and (consp (cdr x))
                      (lhrange-combinable (car x)
                                          (lhrange->atom (cadr x))))
                 (lhs-rest-aux (+ (lhrange->w (car x))
                                  (lhrange->w (cadr x)))
                               (lhrange->atom (car x))
                               (cddr x))
                 (cdr x))
             x))))
