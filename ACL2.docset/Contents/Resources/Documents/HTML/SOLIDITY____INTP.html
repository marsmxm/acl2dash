Function: intp

(defun intp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'intp))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :int)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 2)
                 (b* ((size (std::da-nth 0 (cdr x)))
                      (value (std::da-nth 1 (cdr x))))
                     (and (bit-size-p size)
                          (integerp value)
                          (signed-byte-p size value))))))
