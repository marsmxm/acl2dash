Function: byte-valuep

(defun byte-valuep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'byte-valuep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :byte)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((int (std::da-nth 0 (cdr x))))
                     (sbyte8p int)))))
