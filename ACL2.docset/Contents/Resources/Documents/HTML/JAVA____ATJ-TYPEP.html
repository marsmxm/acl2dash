Function: atj-typep

(defun atj-typep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'atj-typep))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x) (eq (car x) :acl2))
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((get (std::da-nth 0 (cdr x))))
                                 (atj-atypep get))))
                       ((eq (car x) :jprim)
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((get (std::da-nth 0 (cdr x))))
                                 (primitive-typep get))))
                       (t (and (eq (car x) :jprimarr)
                               (and (true-listp (cdr x))
                                    (eql (len (cdr x)) 1))
                               (b* ((comp (std::da-nth 0 (cdr x))))
                                   (primitive-typep comp))))))))
