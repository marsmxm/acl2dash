Function: vl-modinst-maybe-argresolve

(defun vl-modinst-maybe-argresolve
       (flag x ss warnings)
       (declare (xargs :guard (and (booleanp flag)
                                   (vl-modinst-p x)
                                   (vl-scopestack-p ss)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-modinst-maybe-argresolve))
            (declare (ignorable __function__))
            (if flag
                (vl-modinst-argresolve x ss warnings)
                (mv (vl-warninglist-fix warnings)
                    (vl-modinst-fix x)))))
