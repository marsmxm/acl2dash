Function: bindingp

(defun
  bindingp (x)
  (declare (xargs :guard t))
  (let ((__function__ 'bindingp))
       (declare (ignorable __function__))
       (and (mbe :logic (and (alistp x)
                             (equal (strip-cars x)
                                    '(variables value)))
                 :exec (fty::alist-with-carsp x '(variables value)))
            (b* ((variables (cdr (std::da-nth 0 x)))
                 (value (cdr (std::da-nth 1 x))))
                (and (typed-variable-listp variables)
                     (expressionp value))))))
