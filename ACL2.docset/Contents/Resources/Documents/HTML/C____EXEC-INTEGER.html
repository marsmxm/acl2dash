Execute a value to obtain an (ACL2) integer.

Signature
(exec-integer arg)  result
Arguments
arg Guard (valuep arg).
Returns
result Type (integerp result).

This is used for operands such that only their mathematical values affect the result of the operation, and not their C types. Examples are the second operand of shift operations and the index operand of array subscript operations.

Function: exec-integer

(defun exec-integer (arg)
       (declare (xargs :guard (valuep arg)))
       (declare (xargs :guard (value-integerp arg)))
       (let ((__function__ 'exec-integer))
            (declare (ignorable __function__))
            (b* ((arg (value-fix arg)))
                (cond ((ucharp arg) (uchar-integer-value arg))
                      ((scharp arg) (schar-integer-value arg))
                      ((ushortp arg)
                       (ushort-integer-value arg))
                      ((sshortp arg)
                       (sshort-integer-value arg))
                      ((uintp arg) (uint-integer-value arg))
                      ((sintp arg) (sint-integer-value arg))
                      ((ulongp arg) (ulong-integer-value arg))
                      ((slongp arg) (slong-integer-value arg))
                      ((ullongp arg)
                       (ullong-integer-value arg))
                      ((sllongp arg)
                       (sllong-integer-value arg))
                      (t (prog2$ (impossible) 0))))))

Theorem: integerp-of-exec-integer

(defthm integerp-of-exec-integer
        (b* ((result (exec-integer arg)))
            (integerp result))
        :rule-classes :rewrite)
