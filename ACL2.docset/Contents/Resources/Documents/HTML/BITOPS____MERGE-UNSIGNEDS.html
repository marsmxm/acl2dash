Theorem: width-of-merge-unsigneds

(defthm width-of-merge-unsigneds
        (b* ((?packed (merge-unsigneds width elts)))
            (implies (and (<= (* (nfix width) (len elts)) n)
                          (natp n))
                     (unsigned-byte-p n packed))))
