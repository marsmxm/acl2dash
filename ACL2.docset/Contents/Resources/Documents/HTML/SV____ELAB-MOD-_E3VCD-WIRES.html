Function: elab-mod->vcd-wires

(defun
 elab-mod->vcd-wires (n codeidx elab-mod)
 (declare (xargs :stobjs (elab-mod)))
 (declare (xargs :guard (and (natp n) (natp codeidx))))
 (declare (xargs :guard (<= n (elab-mod-nwires elab-mod))))
 (let
  ((__function__ 'elab-mod->vcd-wires))
  (declare (ignorable __function__))
  (b*
   (((when (mbe :logic (zp (- (elab-mod-nwires elab-mod) (nfix n)))
                :exec (eql (elab-mod-nwires elab-mod) n)))
     nil)
    (x (elab-mod-wiretablei n elab-mod))
    ((wire x) x)
    (vcdwire
         (make-vcd-wire :name (name->string x.name)
                        :msb (if x.revp
                                 x.low-idx (+ -1 x.width x.low-idx))
                        :lsb (if x.revp (+ -1 x.width x.low-idx)
                                 x.low-idx)
                        :code (vcd-index->codestr codeidx)))
    (- (and (equal (vcd-wire->code vcdwire) "")
            (cw "Created empty-named wire for wire ~s0~%"
                x.name))))
   (cons vcdwire
         (elab-mod->vcd-wires (1+ (lnfix n))
                              (1+ (lnfix codeidx))
                              elab-mod)))))
