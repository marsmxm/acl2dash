Function: expr-typep

(defun expr-typep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'expr-typep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :expr-type)
                 (mbe :logic (and (alistp (cdr x))
                                  (equal (strip-cars (cdr x))
                                         '(type lvalue)))

                                                   '(type lvalue)))
