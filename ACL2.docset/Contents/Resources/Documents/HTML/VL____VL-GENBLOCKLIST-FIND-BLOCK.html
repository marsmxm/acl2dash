Function: vl-genblocklist-find-block

(defun vl-genblocklist-find-block (idx x)
       (declare (xargs :guard (and (integerp idx)
                                   (vl-genblocklist-p x))))
       (let ((__function__ 'vl-genblocklist-find-block))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((eql (vl-genblock->name (car x))
                        (lifix idx))
                   (vl-genblock-fix (car x)))
                  (t (vl-genblocklist-find-block idx (cdr x))))))
