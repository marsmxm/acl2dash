Function: vl-delete-fundecls

(defun
 vl-delete-fundecls (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-fundecllist-p x))))
 (let
  ((__function__ 'vl-delete-fundecls))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-fundecl->name (car x))
                               (string-list-fix names))
                 (vl-delete-fundecls names (cdr x)))
                (t (cons (vl-fundecl-fix (car x))
                         (vl-delete-fundecls names (cdr x)))))
   :exec
   (b*
    (((when (atom names)) (list-fix x))
     ((when (atom x)) nil)
     ((unless (longer-than-p 10 names))
      (vl-slow-delete-fundecls names x))
     (fal (make-lookup-alist names))
     (ans
       (with-local-nrev (vl-fast-delete-fundecls names fal x nrev)))
     (- (fast-alist-free fal)))
    ans))))
