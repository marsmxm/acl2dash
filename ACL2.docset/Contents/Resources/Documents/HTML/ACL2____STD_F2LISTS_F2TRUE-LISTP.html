Theorem: true-listp-when-atom

(defthm true-listp-when-atom
        (implies (atom x)
                 (equal (true-listp x) (not x))))

Theorem: true-listp-of-cons

(defthm true-listp-of-cons
        (equal (true-listp (cons a x))
