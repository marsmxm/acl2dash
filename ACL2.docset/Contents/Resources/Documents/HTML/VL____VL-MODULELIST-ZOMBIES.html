Function: vl-modulelist-zombies

(defun
   vl-modulelist-zombies
   (x suppress-fatals)
   (declare (xargs :guard (and (vl-modulelist-p x)
                               (symbol-listp suppress-fatals))))
   (let ((__function__ 'vl-modulelist-zombies))
        (declare (ignorable __function__))
        (cond ((atom x) nil)
              ((vl-some-warning-fatalp (vl-module->warnings (car x))
                                       suppress-fatals)
               (cons (vl-module->name (car x))
                     (vl-modulelist-zombies (cdr x)
                                            suppress-fatals)))
              (t (vl-modulelist-zombies (cdr x)
                                        suppress-fatals)))))
