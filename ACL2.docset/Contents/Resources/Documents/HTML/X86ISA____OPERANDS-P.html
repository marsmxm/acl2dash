(defun operands-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'operands-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (std::prod-consp (std::prod-car x))
                 (std::prod-consp (std::prod-cdr x))
                 (b* ((op1 (std::prod-car (std::prod-car x)))
                      (op2 (std::prod-cdr (std::prod-car x)))
                      (op3 (std::prod-car (std::prod-cdr x)))
