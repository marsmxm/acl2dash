Function: vl-jp-funname-fn

(defun vl-jp-funname-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-funname-p x)))
       (let ((__function__ 'vl-jp-funname))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-funname)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-funname->name x))
                       (vl-println? "}"))))
