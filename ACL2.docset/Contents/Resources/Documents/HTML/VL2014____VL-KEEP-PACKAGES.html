Function: vl-keep-packages

(defun
 vl-keep-packages (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-packagelist-p x))))
 (let
  ((__function__ 'vl-keep-packages))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-package->name (car x))
                               (string-list-fix names))
                 (cons (vl-package-fix (car x))
                       (vl-keep-packages names (cdr x))))
                (t (vl-keep-packages names (cdr x))))
   :exec
   (b*
    (((when (or (atom names) (atom x))) nil)
     ((unless (longer-than-p 10 names))
      (vl-slow-keep-packages names x))
     (fal (make-lookup-alist names))
     (ans
         (with-local-nrev (vl-fast-keep-packages names fal x nrev)))
     (- (fast-alist-free fal)))
    ans))))
