Function: vl-rhs-increwrite

(defun
 vl-rhs-increwrite (x loc)
 (declare (xargs :guard (and (vl-rhs-p x) (vl-location-p loc))))
 (let
  ((__function__ 'vl-rhs-increwrite))
  (declare (ignorable __function__))
  (b*
   ((x (vl-rhs-fix x)))
   (vl-rhs-case
    x
    (:vl-rhsexpr (b* (((unless (vl-expr-has-incexprs-p x.guts))
                       (mv x nil nil))
                      ((mv new-guts pre-rev post-rev)
                       (vl-expr-increwrite-aux x.guts nil nil loc)))
                     (mv (change-vl-rhsexpr x :guts new-guts)
                         (rev pre-rev)
                         (rev post-rev))))
    (:vl-rhsnew (b* (((mv new-arrsize pre1 post1)
                      (vl-maybe-expr-increwrite x.arrsize loc))
                     ((mv new-args pre2 post2)
                      (vl-exprlist-increwrite x.args loc)))
                    (mv (change-vl-rhsnew x
                                          :arrsize new-arrsize
                                          :args new-args)
                        (append-without-guard pre1 pre2)
                        (append-without-guard post1 post2))))))))
