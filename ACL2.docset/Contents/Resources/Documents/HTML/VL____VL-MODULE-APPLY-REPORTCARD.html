Function: vl-module-apply-reportcard

(defun
  vl-module-apply-reportcard
  (x reportcard)
  (declare (xargs :guard (and (vl-module-p x)
                              (vl-reportcard-p reportcard))))
  (let ((__function__ 'vl-module-apply-reportcard))
       (declare (ignorable __function__))
       (b* (((vl-module x) x)
            (entry (hons-get x.name (vl-reportcard-fix reportcard)))
            ((unless entry) (vl-module-fix x)))
           (change-vl-module
                x
                :warnings (revappend-without-guard (cdr entry)
                                                   x.warnings)))))
