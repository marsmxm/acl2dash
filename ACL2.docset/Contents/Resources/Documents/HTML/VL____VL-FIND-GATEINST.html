Function: vl-find-gateinst

(defun vl-find-gateinst (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-gateinstlist-p x))))
       (let ((__function__ 'vl-find-gateinst))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-gateinst->name (car x)))
                   (vl-gateinst-fix (car x)))
                  (t (vl-find-gateinst name (cdr x))))))
