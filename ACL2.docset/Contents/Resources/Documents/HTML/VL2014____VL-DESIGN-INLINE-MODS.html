(defun
 vl-design-inline-mods
 (mods-to-inline design)
 (declare
    (xargs :guard (and (and (vl-modulelist-p mods-to-inline)
                            (vl-ok-to-inline-list-p mods-to-inline))
                       (vl-design-p design))))
 (let ((__function__ 'vl-design-inline-mods))
      (declare (ignorable __function__))
      (b* ((design (vl-design-fix design))
           ((vl-design design) design)
           (new-mods (vl-inline-mods mods-to-inline design.mods)))
          (change-vl-design design
                            :mods new-mods))))
