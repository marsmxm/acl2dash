Theorem: logand*

(defthm logand*
        (implies (and (force (integerp i))
                      (force (integerp j)))
                 (equal (logand i j)
                        (logcons (b-and (logcar i) (logcar j))
                                 (logand (logcdr i) (logcdr j)))))
        :rule-classes
        ((:definition :clique (binary-logand)
                      :controller-alist ((binary-logand t t)))))
