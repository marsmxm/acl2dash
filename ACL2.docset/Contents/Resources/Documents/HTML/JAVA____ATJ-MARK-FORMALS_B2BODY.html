(defun atj-mark-formals+body (formals body)
       (declare (xargs :guard (and (symbol-listp formals)
                                   (pseudo-termp body))))
       (let ((__function__ 'atj-mark-formals+body))
            (declare (ignorable __function__))
            (b* ((marked-formals (atj-mark-vars-new formals))
                 ((mv marked-body &)
                  (atj-mark-term body formals nil formals)))
                (mv marked-formals marked-body))))
