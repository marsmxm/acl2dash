Function: vl-delete-define

(defun
     vl-delete-define (name x)
     (declare (xargs :guard (and (stringp name) (vl-defines-p x))))
     (let ((__function__ 'vl-delete-define))
          (declare (ignorable __function__))
          (cond ((atom x) nil)
                ((equal (string-fix name)
                        (vl-define->name (car x)))
                 (vl-delete-define name (cdr x)))

                         (vl-delete-define name (cdr x)))))))
