Function: vl-base-to-radix

(defun vl-base-to-radix (base)
       (declare (xargs :guard (character-listp base)))
       (let ((__function__ 'vl-base-to-radix))
            (declare (ignorable __function__))
            (cond ((or (member #\h base) (member #\H base))
                   16)
                  ((or (member #\d base) (member #\D base))
                   10)
                  ((or (member #\o base) (member #\O base))
                   8)
                  ((or (member #\b base) (member #\B base))
                   2)
                  (t (progn$ (raise "programming error")
                             2)))))
