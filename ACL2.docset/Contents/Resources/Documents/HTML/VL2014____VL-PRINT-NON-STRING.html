Function: vl-print-non-string-fn

(defun
     vl-print-non-string-fn (x ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard (and (vl-printable-p x)
                                 (not (stringp x)))))
     (let ((__function__ 'vl-print-non-string))
          (declare (ignorable __function__))
          (cond ((consp x) (vl-print-charlist-main x))
                ((characterp x)
                 (vl-print-charlist-main (list x)))
                ((natp x) (vl-print-nat-main x))
                (t (vl-print-charlist-main (explode-atom x 10))))))
