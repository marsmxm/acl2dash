Function: defobject-process-init

(defun
 defobject-process-init
 (init type ctx state)
 (declare (xargs :stobjs (state)))
 (declare (xargs :guard (and (typep type) (ctxp ctx))))
 (let
  ((__function__ 'defobject-process-init))
  (declare (ignorable __function__))
  (b*
   (((er &)
     (acl2::ensure-value-is-true-list$
          init (msg "The :INIT input ~x0" init)
          t nil))
    ((unless (type-case type :array))
     (acl2::value (raise "Internal error: not array type ~x0."
                         type)))
    ((unless (equal (len init)
                    (type-array->size type)))
     (er-soft+
      ctx t nil
      "The number ~x0 of elements of the :INIT input ~
                   must match the size ~x1 of the array ~
                   specified by the :TYPE input."
      (len init)
      (type-array->size type))))
   (defobject-process-init-terms init (type-array->of type)
                                 ctx state))))
