Function: bytep

(defun bytep (x)
       (declare (xargs :guard t))
       (mbe :logic (unsigned-byte-p (char-bit) x)
            :exec (and (integerp x)
                       (<= 0 x)
                       (< x (expt 2 (char-bit))))))
