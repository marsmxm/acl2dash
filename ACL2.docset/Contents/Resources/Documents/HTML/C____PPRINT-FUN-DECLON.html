Function: pprint-fun-declon

(defun
 pprint-fun-declon (declon level)
 (declare (xargs :guard (and (fun-declonp declon)
                             (natp level))))
 (let ((__function__ 'pprint-fun-declon))
      (declare (ignorable __function__))
      (b* (((fun-declon declon) declon))
          (list (pprint-line (msg "~@0 ~@1;"

                             (lnfix level))))))
