Function: vl-port-resolve-indexing

(defun
   vl-port-resolve-indexing (x ss warnings)
   (declare (xargs :guard (and (vl-port-p x)
                               (vl-scopestack-p ss)
                               (vl-warninglist-p warnings))))
   (let ((__function__ 'vl-port-resolve-indexing))
        (declare (ignorable __function__))
        (b* ((x (vl-port-fix x))
             (warnings (vl-warninglist-fix warnings)))
            (b* ((x (vl-port-fix x))
                 ((when (eq (tag x) :vl-interfaceport))
                  (mv warnings x))
                 ((vl-regularport x) x)
                 ((unless x.expr) (mv warnings x))
                 ((mv warnings expr-prime)
                  (vl-expr-resolve-indexing x.expr ss warnings))
                 (x-prime (change-vl-regularport x
                                                 :expr expr-prime)))
                (mv warnings x-prime)))))
