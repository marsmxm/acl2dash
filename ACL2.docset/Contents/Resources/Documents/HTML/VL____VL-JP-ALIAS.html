(defun vl-jp-alias-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-alias-p x)))
       (let ((__function__ 'vl-jp-alias))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-alias)
                       (vl-print ", ")
                       (vl-print-str "\"lhs\": ")
                       (vl-jp-expr (vl-alias->lhs x))
                       (vl-println? ", ")
                       (vl-print-str "\"rhs\": ")
                       (vl-jp-expr (vl-alias->rhs x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-alias->atts x))
                       (vl-println? ", ")
                       (vl-print-str "\"loc\": ")
                       (vl-jp-location (vl-alias->loc x))
                       (vl-println? "}"))))
