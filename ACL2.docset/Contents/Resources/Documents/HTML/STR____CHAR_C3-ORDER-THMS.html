Theorem: char<-antisymmetric

(defthm char<-antisymmetric
        (implies (char< x y) (not (char< y x))))

Theorem: char<-transitive

(defthm char<-transitive
        (implies (and (char< x y) (char< y z))
                 (char< x z)))

Theorem: char<-trichotomy-weak

(defthm char<-trichotomy-weak
        (implies (and (not (char< x y))

Theorem: char<-trichotomy-strong

(defthm char<-trichotomy-strong
        (equal (char< x y)
