(defun lhs-concat (w x y)
       (declare (xargs :guard (and (natp w) (lhs-p x) (lhs-p y))))
       (let ((__function__ 'lhs-concat))
            (declare (ignorable __function__))
            (b* (((when (zp w)) (lhs-fix y))
                 ((when (atom x))
                  (lhs-cons (lhrange w (lhatom-z)) y))
