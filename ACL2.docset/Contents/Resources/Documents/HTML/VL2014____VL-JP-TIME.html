(defun vl-jp-time-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-time-p x)))
       (let ((__function__ 'vl-jp-time))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-time)
                       (vl-print ", ")
                       (vl-print-str "\"quantity\": ")
                       (jp-str (vl-time->quantity x))
                       (vl-println? ", ")
                       (vl-print-str "\"units\": ")
                       (jp-timeunit (vl-time->units x))
                       (vl-println? "}"))))
