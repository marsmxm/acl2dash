Function: cutinfo-p

(defun cutinfo-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'cutinfo-p))
            (declare (ignorable __function__))
            (and (mbe :logic (unsigned-byte-p 32 x)
                      :exec (and (natp x) (< x 4294967296)))
                 (cutsize-p (part-select x :low 16 :width 3)))))
