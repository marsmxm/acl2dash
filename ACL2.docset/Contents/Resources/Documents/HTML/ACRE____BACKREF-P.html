Function: backref-p

(defun backref-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'backref-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (b* ((loc (std::prod-car x))
                      (len (std::prod-cdr x)))
                     (and (natp loc) (natp len))))))

Theorem: consp-when-backref-p

(defthm consp-when-backref-p
        (implies (backref-p x) (consp x))
