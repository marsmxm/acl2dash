Check the current bfr-mode.

Just returns T if the given bfr-mode matches the keyword. Like bfr-mode-case, the bfr-mode argument is optional. Usage:

;; Use the bfr-mode variable. T if aig mode, NIL if aignet or bdd.
(bfr-mode-case :aig) 

;; Use the given bfr-mode object. T if bdd mode, NIL if aignet or aig.
(bfr-mode-case :bdd my-bfr-mode)

(defun bfr-mode-is-fn (key mode)
       (cond ((eq key :aignet)
              (cons 'eql
                    (cons (cons 'bfr-mode-fix (cons mode 'nil))
                          '(0))))
             ((eq key :bdd)
              (cons 'eql
                    (cons (cons 'bfr-mode-fix (cons mode 'nil))
                          '(1))))

              (cons 'eql
                    (cons (cons 'bfr-mode-fix (cons mode 'nil))
                          '(2))))
