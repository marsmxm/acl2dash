Function: vl-scope->name

(defun vl-scope->name (x)
       (declare (xargs :guard (vl-scope-p x)))
       (let ((__function__ 'vl-scope->name))
            (declare (ignorable __function__))
            (b* ((x (vl-scope-fix x)))
                (case (tag x)
                      (:vl-interface (vl-interface->name x))
                      (:vl-module (vl-module->name x))
                      (:vl-genblob (vl-genblob->name x))
                      (:vl-blockscope (vl-blockscope->name x))
                      (:vl-package (vl-package->name x))
                      (otherwise nil)))))

Theorem: maybe-stringp-of-vl-scope->name

(defthm maybe-stringp-of-vl-scope->name
        (b* ((name (vl-scope->name x)))
            (maybe-stringp name))
        :rule-classes :type-prescription)

(defthm vl-scope->name-of-vl-scope-fix-x
