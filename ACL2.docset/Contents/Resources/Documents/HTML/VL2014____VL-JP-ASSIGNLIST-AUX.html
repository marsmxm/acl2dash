Definitions and Theorems

Function: vl-jp-assignlist-aux-fn

(defun vl-jp-assignlist-aux-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-assignlist-p x)))
       (let ((__function__ 'vl-jp-assignlist-aux))
            (declare (ignorable __function__))
            (if (atom x)
                ps
                (vl-ps-seq (vl-jp-assign (car x))
                           (if (atom (cdr x))
                               ps
                               (vl-ps-seq (vl-println ",")
                                          (vl-indent 4)))
                           (vl-jp-assignlist-aux (cdr x))))))
