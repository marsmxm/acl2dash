Function: ensure-tuple

(defun
 ensure-tuple
 (x n description
    error-erp error-val ctx state)
 (declare (xargs :stobjs (state)))
 (declare (xargs :guard (and (natp n) (msgp description))))
 (b*
    (((unless (tuplep n x))
      (er-soft+ ctx error-erp error-val
                "~@0 must be a NIL-terminated list of ~x1 elements."
                description n)))
    (value nil)))
