Function: gatesimp->xor-mode

(defun
 gatesimp->xor-mode (x)
 (declare (xargs :guard (gatesimp-p x)))
 (mbe
     :logic (let ((x (gatesimp-fix x)))
                 (part-select x :low 4 :width 2))
     :exec (the (unsigned-byte 2)
                (logand (the (unsigned-byte 2) 3)
                        (the (unsigned-byte 2)
                             (ash (the (unsigned-byte 6) x) -4))))))
