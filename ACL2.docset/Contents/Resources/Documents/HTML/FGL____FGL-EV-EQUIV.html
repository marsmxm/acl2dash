Function: fgl-ev-equiv

(defun fgl-ev-equiv (x y)
       (declare (xargs :non-executable t))
       (declare (xargs :guard t))
       (prog2$ (acl2::throw-nonexec-error 'fgl-ev-equiv
                                          (list x y))
