(defun floating-point-typep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'floating-point-typep))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x) (eq (car x) :float))
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 0)
                             (b* nil t)))
