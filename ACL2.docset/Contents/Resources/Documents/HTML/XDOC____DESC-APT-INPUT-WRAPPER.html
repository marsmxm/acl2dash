Build a description of the :wrapper input for the user documentation of an APT transformation.

Macro: desc-apt-input-wrapper

(defmacro
 xdoc::desc-apt-input-wrapper
 (&rest additional)
 (cons
  'xdoc::desc
  (cons
   '"@(':wrapper') &mdash; default @('nil')"
   (cons
    (cons
      'xdoc::p
      (cons '"Determines whether the wrapper function is generated."
            'nil))
    (cons
     (cons 'xdoc::p
           (cons '"It must be one of the following:"
                 'nil))
     (cons
      (cons 'xdoc::ul
            (cons (cons 'xdoc::li
                        (cons '"@('t'), to generate it." 'nil))
                  (cons (cons 'xdoc::li
                              (cons '"@('nil'), to not generate it."
                                    'nil))
                        'nil)))
      additional))))))
