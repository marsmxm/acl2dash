Function: addr-range

(defun addr-range (count addr)
       (declare (xargs :guard (natp count)))
       (let ((__function__ 'addr-range))
            (declare (ignorable __function__))
            (if (zp count)
                nil
                (cons (ifix addr)
                      (addr-range (1- count)
                                  (1+ (ifix addr)))))))
