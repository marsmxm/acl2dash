Function: lcletter/digit-char-p

(defun lcletter/digit-char-p (x)
       (declare (xargs :guard t))
       (and (characterp x)
            (b* ((code (char-code x)))
                (or (and (<= (char-code #\a) code)
                         (<= code (char-code #\z)))
                    (and (<= (char-code #\0) code)
                         (<= code (char-code #\9)))))))
