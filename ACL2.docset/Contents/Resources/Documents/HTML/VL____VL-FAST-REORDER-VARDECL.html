(defun
 vl-fast-reorder-vardecl (names x fal)
 (declare
      (xargs :guard (and (string-listp names)
                         (vl-vardecllist-p x)
                         (equal fal (vl-vardecllist-alist x nil)))))
 (let ((__function__ 'vl-fast-reorder-vardecl))
      (declare (ignorable __function__))
      (b* (((when (atom names)) nil)
           (look (hons-get (car names) fal))
           ((when look)
            (cons (cdr look)
                  (vl-fast-reorder-vardecl (cdr names)
                                           x fal))))
          (vl-fast-reorder-vardecl (cdr names)
                                   x fal))))
