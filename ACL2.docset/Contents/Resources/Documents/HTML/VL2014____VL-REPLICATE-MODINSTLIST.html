Function: vl-replicate-modinstlist

(defun vl-replicate-modinstlist
       (x nf ss warnings)
       (declare (xargs :guard (and (vl-modinstlist-p x)
                                   (vl-namefactory-p nf)
                                   (vl-scopestack-p ss)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-replicate-modinstlist))
            (declare (ignorable __function__))
            (b* (((when (atom x)) (mv (ok) nil nf))
                 ((mv warnings car-insts nf)
                  (vl-replicate-modinst (car x)
                                        nf ss warnings))
                 ((mv warnings cdr-insts nf)
                  (vl-replicate-modinstlist (cdr x)
                                            nf ss warnings)))
                (mv warnings (append car-insts cdr-insts)
                    nf))))
