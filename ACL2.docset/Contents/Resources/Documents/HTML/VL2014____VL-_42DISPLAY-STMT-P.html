(defun vl-$display-stmt-p (x)
       (declare (xargs :guard (vl-stmt-p x)))
       (declare (xargs :guard (vl-stmt-p x)))
       (let ((__function__ 'vl-$display-stmt-p))
            (declare (ignorable __function__))
            (b* (((unless (vl-enablestmt-p x)) nil)
                 ((vl-enablestmt x) x)
                 ((unless (vl-fast-atom-p x.id)) nil)
                 ((vl-atom x.id) x.id)
                 ((unless (vl-fast-sysfunname-p x.id.guts))
                  nil)
                 ((vl-sysfunname x.id.guts) x.id.guts))
                (equal x.id.guts.name "$display"))))
