Function: vl-paramvaluelist-wildelim

(defun vl-paramvaluelist-wildelim
       (x ctx warnings)
       (declare (xargs :guard (and (vl-paramvaluelist-p x)
                                   (vl-context-p ctx)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-paramvaluelist-wildelim))
            (declare (ignorable __function__))
            (b* (((when (atom x)) (mv (ok) x))
                 ((mv warnings car-prime)
                  (vl-paramvalue-wildelim (car x)
                                          ctx warnings))
                 ((mv warnings cdr-prime)
                  (vl-paramvaluelist-wildelim (cdr x)
                                              ctx warnings)))
                (mv warnings (cons car-prime cdr-prime)))))
