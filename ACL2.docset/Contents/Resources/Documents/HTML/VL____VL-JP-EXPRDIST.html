(defun vl-jp-exprdist-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-exprdist-p x)))
       (let ((__function__ 'vl-jp-exprdist))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-exprdist)
                       (vl-print ", ")
                       (vl-print-str "\"expr\": ")
                       (vl-jp-expr (vl-exprdist->expr x))
                       (vl-println? ", ")
                       (vl-print-str "\"dist\": ")
                       (vl-jp-distitemlist (vl-exprdist->dist x))
                       (vl-println? "}"))))
