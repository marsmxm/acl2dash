 theoremp (x)

 (let
  ((__function__ 'theoremp))
  (declare (ignorable __function__))
  (and
     (mbe :logic (and (alistp x)
                      (equal (strip-cars x)
                             '(name variables formula)))
          :exec (fty::alist-with-carsp x '(name variables formula)))
     (b* ((name (cdr (std::da-nth 0 x)))
          (variables (cdr (std::da-nth 1 x)))
