Signature
(atj-type-annotate-args args var-types mv-typess guards$ wrld) 
  → 
(mv annotated-args resulting-types new-mv-typess)
Arguments
args — Guard (pseudo-term-listp args).
var-types — Guard (atj-symbol-type-alistp var-types).
mv-typess — Guard (atj-type-list-listp mv-typess).
guards$ — Guard (booleanp guards$).
wrld — Guard (plist-worldp wrld).
Returns
annotated-args — Type (and (pseudo-term-listp annotated-args) (equal (len annotated-args) (len args))) .
resulting-types — Type (and (atj-type-listp resulting-types) (equal (len resulting-types) (len args))) .
new-mv-typess — Type (and (atj-type-list-listp new-mv-typess) (cons-listp new-mv-typess)).
