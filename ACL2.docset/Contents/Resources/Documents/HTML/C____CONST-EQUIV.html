Theorem: const-equiv-is-an-equivalence

(defthm const-equiv-is-an-equivalence
        (and (booleanp (const-equiv x y))
             (const-equiv x x)
             (implies (const-equiv x y)
                      (const-equiv y x))
             (implies (and (const-equiv x y)
                           (const-equiv y z))
                      (const-equiv x z)))
        :rule-classes (:equivalence))
