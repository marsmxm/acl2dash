(defun vl-msg-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-msg-p))
            (declare (ignorable __function__))
            (and (or (atom x) (cdr x))
                 (b* ((msg (if (atom x) x (car x)))
                      (args (and (consp x) (cdr x))))
                     (and (stringp msg)
                          (true-listp args))))))
