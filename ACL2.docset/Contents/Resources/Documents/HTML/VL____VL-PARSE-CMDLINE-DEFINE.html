Function: vl-parse-cmdline-define

(defun vl-parse-cmdline-define (x loc stickyp)
       (declare (xargs :guard (and (stringp x)
                                   (vl-location-p loc)
                                   (booleanp stickyp))))
       (let ((__function__ 'vl-parse-cmdline-define))
            (declare (ignorable __function__))
            (b* ((x (string-fix x))
                 (pos (position #\= x))
                 ((unless pos)
                  (mv t x
                      (make-vl-define :formals nil
                                      :body ""
                                      :loc loc
                                      :stickyp stickyp)))
                 ((unless (< (+ 1 pos) (length x)))
                  (mv nil nil nil))
                 (name (subseq x 0 pos))
                 (body (subseq x (+ 1 pos) nil)))
                (mv t name
                    (make-vl-define :formals nil
                                    :body body
                                    :loc loc
                                    :stickyp stickyp)))))
