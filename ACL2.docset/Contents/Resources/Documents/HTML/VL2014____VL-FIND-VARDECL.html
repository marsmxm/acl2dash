Function: vl-find-vardecl

(defun vl-find-vardecl (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-vardecllist-p x))))
       (let ((__function__ 'vl-find-vardecl))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-vardecl->name (car x)))
                   (vl-vardecl-fix (car x)))
                  (t (vl-find-vardecl name (cdr x))))))
