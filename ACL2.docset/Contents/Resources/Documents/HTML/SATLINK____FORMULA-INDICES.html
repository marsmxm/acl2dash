Function: formula-indices

(defun formula-indices (formula)
       (declare (xargs :guard (lit-list-listp formula)))
       (let ((__function__ 'formula-indices))
            (declare (ignorable __function__))
            (if (atom formula)
                nil
                (append (clause-indices (car formula))
                        (formula-indices (cdr formula))))))

Theorem: formula-indices-when-atom

(defthm formula-indices-when-atom
        (implies (atom formula)
                 (equal (formula-indices formula) nil)))

Theorem: formula-indices-of-cons

(defthm formula-indices-of-cons
        (equal (formula-indices (cons clause formula))
               (append (clause-indices clause)
                       (formula-indices formula))))

Theorem: formula-indices-of-lit-list-list-fix-formula

(defthm formula-indices-of-lit-list-list-fix-formula
        (equal (formula-indices (lit-list-list-fix formula))
               (formula-indices formula)))
