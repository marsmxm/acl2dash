Function: candidate-ids

(defun
  candidate-ids (xs)
  (declare (xargs :guard (irv-ballot-p xs)))
  (let ((__function__ 'candidate-ids))
       (declare (ignorable __function__))
       (acl2::<-sort (remove-duplicates-equal (acl2::flatten xs)))))
