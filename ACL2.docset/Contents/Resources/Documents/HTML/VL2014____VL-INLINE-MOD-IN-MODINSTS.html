Function: vl-inline-mod-in-modinsts

(defun vl-inline-mod-in-modinsts
       (sub x nf warnings)
       (declare (xargs :guard (and (and (vl-module-p sub)
                                        (vl-ok-to-inline-p sub))
                                   (vl-modinstlist-p x)
                                   (vl-namefactory-p nf)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-inline-mod-in-modinsts))
            (declare (ignorable __function__))
            (b* (((when (atom x))
                  (mv nf nil nil nil nil (ok)))
                 ((mv nf modinsts1
                      gateinsts1 assigns1 vardecls1 warnings)
                  (vl-inline-mod-in-modinst sub (car x)
                                            nf warnings))
                 ((mv nf modinsts2
                      gateinsts2 assigns2 vardecls2 warnings)
                  (vl-inline-mod-in-modinsts sub (cdr x)
                                             nf warnings)))
                (mv nf (append modinsts1 modinsts2)
                    (append gateinsts1 gateinsts2)
                    (append assigns1 assigns2)
                    (append vardecls1 vardecls2)
                    warnings))))
