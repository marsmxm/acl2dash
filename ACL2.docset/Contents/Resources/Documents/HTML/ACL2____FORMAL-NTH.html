Function: formal-nth

(defun
  formal-nth (n lst)
  (declare (xargs :guard (and (integerp n)
                              (<= 0 n)
                              (pseudo-termp lst)
                              (equal (formal-true-listp lst) *t*))))
  (case-match lst
              (('quote x)
               (cons 'quote (cons (nth n x) 'nil)))
              (& (cond ((zp n) (fargn lst 1))
                       (t (formal-nth (- n 1) (fargn lst 2)))))))
