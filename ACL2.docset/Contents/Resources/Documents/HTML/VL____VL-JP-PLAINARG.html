(defun vl-jp-plainarg-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-plainarg-p x)))
       (let ((__function__ 'vl-jp-plainarg))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-plainarg)
                       (vl-print ", ")
                       (vl-print-str "\"expr\": ")
                       (vl-jp-maybe-expr (vl-plainarg->expr x))
                       (vl-println? ", ")
                       (vl-print-str "\"portname\": ")
                       (jp-maybe-string (vl-plainarg->portname x))
                       (vl-println? ", ")
                       (vl-print-str "\"dir\": ")
                       (vl-jp-maybe-direction (vl-plainarg->dir x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-plainarg->atts x))
                       (vl-println? "}"))))
