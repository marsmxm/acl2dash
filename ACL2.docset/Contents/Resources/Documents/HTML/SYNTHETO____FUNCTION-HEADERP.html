 function-headerp (x)
 (declare (xargs :guard t))
 (let
   ((__function__ 'function-headerp))
   (declare (ignorable __function__))
   (and (mbe :logic (and (alistp x)
                         (equal (strip-cars x)
                                '(name inputs outputs)))
             :exec (fty::alist-with-carsp x '(name inputs outputs)))
