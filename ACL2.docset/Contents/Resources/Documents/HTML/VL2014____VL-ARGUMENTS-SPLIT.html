Function: vl-arguments-split

(defun
 vl-arguments-split (x elem delta)
 (declare (xargs :guard (and (vl-arguments-p x)
                             (vl-modelement-p elem)
                             (vl-delta-p delta))))
 (let
  ((__function__ 'vl-arguments-split))
  (declare (ignorable __function__))
  (b*
   (((when (eq (vl-arguments-kind x)
               :vl-arguments-named))
     (mv
      x
      (dwarn
       :type :vl-bad-arguments
       :msg
       "~a0: expected to only encounter plain arguments, ~
                           but found a named argument list.  Not actually ~
                           splitting anything."
       :args (list elem)
       :fatalp t
       :fn 'vl-arguments-split)))
    ((mv new-args delta)
     (vl-plainarglist-split (vl-arguments-plain->args x)
                            elem delta))
    (x-prime (change-vl-arguments-plain x
                                        :args new-args)))
   (mv x-prime delta))))
