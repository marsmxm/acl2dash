Function: subset-p

(defun subset-p (x y)
       (declare (xargs :guard (and (true-listp x) (true-listp y))))
       (let ((__function__ 'subset-p))
            (declare (ignorable __function__))
            (cond ((atom x) t)
                  ((member-p (car x) y)
                   (subset-p (cdr x) y))
                  (t nil))))
