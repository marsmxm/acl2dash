Function: vl-parse-global-clocking-declaration-fn

(defun
 vl-parse-global-clocking-declaration-fn
 (atts tokstream config)
 (declare (xargs :stobjs (tokstream)))
 (declare (xargs :guard (vl-loadconfig-p config)))
 (declare (ignorable config))
 (declare (xargs :guard (vl-atts-p atts)))
 (let
  ((__function__ 'vl-parse-global-clocking-declaration))
  (declare (ignorable __function__))
  (seq
   tokstream
   (:= (vl-match-token :vl-kwd-global))
   (clk := (vl-match-token :vl-kwd-clocking))
   (when (vl-is-token? :vl-idtoken)
         (name := (vl-match)))
   (event := (vl-parse-clocking-event))
   (:= (vl-match-token :vl-kwd-endclocking))
   (when
      name
      (:= (vl-parse-endblock-name (and name (vl-idtoken->name name))
                                  "clocking")))
   (return
        (make-vl-gclkdecl :name (and name (vl-idtoken->name name))
                          :event event
                          :loc (vl-token->loc clk)
                          :atts atts)))))
