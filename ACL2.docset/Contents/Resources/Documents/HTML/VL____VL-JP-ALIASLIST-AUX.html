Function: vl-jp-aliaslist-aux-fn

(defun vl-jp-aliaslist-aux-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-aliaslist-p x)))
       (let ((__function__ 'vl-jp-aliaslist-aux))
            (declare (ignorable __function__))
            (if (atom x)
                ps
                (vl-ps-seq (vl-jp-alias (car x))
                           (if (atom (cdr x))
                               ps (vl-println? ", "))
                           (vl-jp-aliaslist-aux (cdr x))))))
