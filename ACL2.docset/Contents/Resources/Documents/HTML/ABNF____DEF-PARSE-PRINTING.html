Pretty-print ABNF elements, alternations, concatenations, and repetitions to ACL2 strings.

This is used to generate portions of documentation strings in the generated parsing functions. Not all the ABNF elements are supported.

Definitions and Theorems

Function: def-parse-print-element

(defun
 def-parse-print-element (elem)
 (declare (xargs :guard (elementp elem)))
 (let
  ((__function__ 'def-parse-print-element))
  (declare (ignorable __function__))
  (element-case
   elem
   :rulename (rulename->get elem.get)
   :group (str::cat "( "
                    (def-parse-print-alternation elem.get)
                    " )")
   :option (str::cat "[ "
                     (def-parse-print-alternation elem.get)
                     " ]")
   :char-val
   (char-val-case
     elem.get
     :insensitive (str::cat "\"" (char-val-insensitive->get elem.get)
                            "\"")
     :sensitive (str::cat "%s\"" (char-val-sensitive->get elem.get)
                          "\""))
   :num-val (prog2$ (raise "Printing of ~x0 not supported."
                           elem.get)
                    "")
   :prose-val (prog2$ (raise "Printing of ~x0 not supported."
                             elem.get)
                      ""))))

Function: def-parse-print-alternation

(defun
 def-parse-print-alternation (alt)
 (declare (xargs :guard (alternationp alt)))
 (let
     ((__function__ 'def-parse-print-alternation))
     (declare (ignorable __function__))
     (cond ((endp alt) "")
           ((endp (cdr alt))
            (def-parse-print-concatenation (car alt)))
           (t (str::cat (def-parse-print-concatenation (car alt))
                        " / "
                        (def-parse-print-alternation (cdr alt)))))))

Function: def-parse-print-concatenation

(defun
 def-parse-print-concatenation (conc)
 (declare (xargs :guard (concatenationp conc)))
 (let
  ((__function__ 'def-parse-print-concatenation))
  (declare (ignorable __function__))
  (cond ((endp conc) "")
        ((endp (cdr conc))
         (def-parse-print-repetition (car conc)))
        (t (str::cat (def-parse-print-repetition (car conc))
                     " "
                     (def-parse-print-concatenation (cdr conc)))))))

Function: def-parse-print-repetition

(defun
 def-parse-print-repetition (rep)
 (declare (xargs :guard (repetitionp rep)))
 (let
  ((__function__ 'def-parse-print-repetition))
  (declare (ignorable __function__))
  (b*
     (((repetition rep) rep)
      ((repeat-range range) rep.range)
      ((when (and (equal range.min 1)
                  (equal range.max (nati-infinity))))
       (def-parse-print-element rep.element))
      ((when (equal range.max (nati-finite range.min)))
       (str::cat (str::nat-to-dec-string range.min)
                 (def-parse-print-element rep.element))))
     (str::cat
          (if (equal range.min 0)
              "" (str::nat-to-dec-string range.min))
          "*"
          (if (nati-case range.max :infinity)
              ""
              (str::nat-to-dec-string (nati-finite->get range.max)))
          (def-parse-print-element rep.element)))))
