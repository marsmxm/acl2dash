Function: isodata-gen-back-of-forth-instances-to-mv-nth

(defun
 isodata-gen-back-of-forth-instances-to-mv-nth
 (terms res-isomaps wrld)
 (declare (xargs :guard (and (pseudo-term-listp terms)
                             (isodata-pos-isomap-alistp res-isomaps)
                             (plist-worldp wrld))))
 (declare (xargs :guard (= (len terms) (len res-isomaps))))
 (let
  ((__function__ 'isodata-gen-back-of-forth-instances-to-mv-nth))
  (declare (ignorable __function__))
  (b*
   (((when (endp terms)) nil)
    (term (car terms))
    (j (caar res-isomaps))
    (isomap (cdar res-isomaps))
    (back-of-forth (isodata-isomap->back-of-forth isomap))
    (var (isodata-formal-of-forth isomap wrld))
    (mv-nth-of-term (fcons-term* 'mv-nth (1- j) term))
    (instance
       (cons ':instance
             (cons back-of-forth
                   (cons ':extra-bindings-ok
                         (cons (cons var (cons mv-nth-of-term 'nil))
                               'nil)))))
    (instances (isodata-gen-back-of-forth-instances-to-mv-nth
                    (cdr terms)
                    (cdr res-isomaps)
                    wrld)))
   (cons instance instances))))
