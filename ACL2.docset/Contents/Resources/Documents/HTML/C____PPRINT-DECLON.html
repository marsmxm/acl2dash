Function: pprint-declon

(defun
 pprint-declon (declon level)
 (declare (xargs :guard (and (declonp declon) (natp level))))
 (let
  ((__function__ 'pprint-declon))
  (declare (ignorable __function__))
  (declon-case
   declon
   :var
   (list
       (pprint-line (msg "~@0 ~@1 = ~@2;"
                         (pprint-tyspecseq declon.type)
                         (pprint-declor declon.declor)
                         (pprint-expr declon.init (expr-grade-top)))
                    (lnfix level)))
   :struct
   (append
       (list (pprint-line (msg "struct ~@0 {"
                               (pprint-ident declon.tag))
                          (lnfix level)))
       (pprint-struct-declon-list declon.members (1+ (lnfix level)))
       (list (pprint-line "}" (lnfix level)))))))
