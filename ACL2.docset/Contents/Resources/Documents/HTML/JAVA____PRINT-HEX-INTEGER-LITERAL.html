Function: print-hex-integer-literal

(defun
  print-hex-integer-literal (lit)
  (declare (xargs :guard (hex-integer-literalp lit)))
  (let ((__function__ 'print-hex-integer-literal))
       (declare (ignorable __function__))
       (b* (((hex-integer-literal lit) lit))
           (msg "~@0~@1~@2"
                (if lit.prefix-upcase-p "0X" "0x")
                (print-hexdig/uscore-list lit.digits/uscores)
                (print-optional-integer-type-suffix lit.suffix?)))))
