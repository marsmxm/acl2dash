Extend an alist by binding the names of vl-gateinsts to their definitions.

Signature
(vl-gateinstlist-alist x acc) → alist
Arguments
x — Guard (vl-gateinstlist-p x).
Returns
alist — Type (equal (vl-gateinst-alist-p alist) (vl-gateinst-alist-p acc)).

This can be used as an alternative to vl-find-gateinst when you need to perform a lot of lookups.

Definitions and Theorems

Function: vl-gateinstlist-alist

(defun vl-gateinstlist-alist (x acc)
       (declare (xargs :guard (vl-gateinstlist-p x)))
       (let ((__function__ 'vl-gateinstlist-alist))
            (declare (ignorable __function__))
            (b* (((when (atom x)) acc)
                 (name (hons-copy (vl-gateinst->name (car x))))
                 ((when (stringp name))
                  (cons (cons name (vl-gateinst-fix (car x)))
                        (vl-gateinstlist-alist (cdr x) acc))))
                (vl-gateinstlist-alist (cdr x) acc))))

Theorem: return-type-of-vl-gateinstlist-alist

(defthm return-type-of-vl-gateinstlist-alist
        (b* ((alist (vl-gateinstlist-alist x acc)))
            (equal (vl-gateinst-alist-p alist)
                   (vl-gateinst-alist-p acc)))
        :rule-classes :rewrite)

Theorem: lookup-in-vl-gateinstlist-alist-acc-elim

(defthm
 lookup-in-vl-gateinstlist-alist-acc-elim
 (implies
    (syntaxp (not (equal acc ''nil)))
    (equal (hons-assoc-equal name (vl-gateinstlist-alist x acc))
           (or (hons-assoc-equal name (vl-gateinstlist-alist x nil))
               (hons-assoc-equal name acc)))))

Theorem: lookup-in-vl-gateinstlist-alist-fast

(defthm
   lookup-in-vl-gateinstlist-alist-fast
   (implies
        (stringp name)
        (equal (hons-assoc-equal name (vl-gateinstlist-alist x nil))
               (let ((val (vl-find-gateinst name x)))
                    (and val (cons name val))))))
