Theorem: append-preserves-rational-listp

(defthm append-preserves-rational-listp
        (implies (true-listp x)
                 (equal (rational-listp (append x y))
                        (and (rational-listp x)
                             (rational-listp y)))))

Theorem: rationalp-car-rational-listp

(defthm rationalp-car-rational-listp
        (implies (and (rational-listp x) x)
                 (rationalp (car x)))
        :rule-classes :forward-chaining)
