Function: vl-expr-has-patterns

(defun
 vl-expr-has-patterns (x)
 (declare (xargs :guard (vl-expr-p x)))
 (let
     ((__function__ 'vl-expr-has-patterns))
     (declare (ignorable __function__))
     (b* (((when (vl-atom-p x)) nil)
          ((vl-nonatom x))
          ((when (member x.op
                         '(:vl-pattern-positional
                               :vl-pattern-keyvalue
