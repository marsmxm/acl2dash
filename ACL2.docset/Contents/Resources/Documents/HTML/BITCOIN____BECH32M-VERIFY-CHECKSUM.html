Function: bech32m-verify-checksum

(defun
 bech32m-verify-checksum (hrp data)
 (declare (xargs :guard (and (stringp hrp)
                             (unsigned-byte-listp 8 data))))
 (let
  ((__function__ 'bech32m-verify-checksum))
  (declare (ignorable __function__))
  (and (hrp-valid-p hrp)
       (<= 6 (len data))
       (warrant bvshr)
       (warrant bvand)
       (equal (bech32-polymod (append (bech32-hrp-expand hrp) data))
              *bech32m-const*))))
