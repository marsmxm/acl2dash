Function: vl-pp-typedef-fn

(defun
 vl-pp-typedef-fn (x ps)
 (declare (xargs :stobjs (ps)))
 (declare (xargs :guard (vl-typedef-p x)))
 (let
     ((__function__ 'vl-pp-typedef))
     (declare (ignorable __function__))
     (b* (((vl-typedef x) x))
         (vl-ps-seq (vl-progindent)
                    (if x.atts (vl-pp-atts x.atts) ps)
                    (vl-ps-span "vl_key" (vl-print "typedef "))
                    (vl-pp-datatype x.type)
                    (vl-print " ")
                    (vl-print-wirename x.name)
                    (let ((udims (vl-datatype->udims x.type)))
                         (if (consp udims)
                             (vl-ps-seq (vl-print " ")
                                        (vl-pp-dimensionlist udims))
                             ps))
                    (vl-println " ;")))))

Theorem: vl-pp-typedef-fn-of-vl-typedef-fix-x

(defthm vl-pp-typedef-fn-of-vl-typedef-fix-x
        (equal (vl-pp-typedef-fn (vl-typedef-fix x) ps)
               (vl-pp-typedef-fn x ps)))

Theorem: vl-pp-typedef-fn-vl-typedef-equiv-congruence-on-x

(defthm vl-pp-typedef-fn-vl-typedef-equiv-congruence-on-x
        (implies (vl-typedef-equiv x x-equiv)
                 (equal (vl-pp-typedef-fn x ps)
