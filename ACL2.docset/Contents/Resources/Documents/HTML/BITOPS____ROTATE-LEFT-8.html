(defun
 rotate-left-8$inline (x places)
 (declare (type (unsigned-byte 8) x)
          (type (unsigned-byte 8) places))
 (let
  ((__function__ 'rotate-left-8))
  (declare (ignorable __function__))
  (mbe
   :logic (rotate-left x 8 places)
   :exec
   (let*
       ((x (mbe :logic (loghead 8 x) :exec x))
        (places (mbe :logic (lnfix places)
                     :exec places))
        (places (mbe :logic (mod places 8)
                     :exec (if (< places 8)
                               (the (integer 0 7) places)
                               (the (integer 0 7) (rem places 8)))))
        (low-num (the (integer 0 8) (- 8 places)))
        (mask (the (unsigned-byte 8)
                   (1- (the (unsigned-byte 9)
                            (ash 1 low-num)))))
        (xl (the (unsigned-byte 8) (logand x mask)))
        (xh (the (unsigned-byte 8)
                 (logand x (the (signed-byte 9) (lognot mask)))))
        (xh-shift (the (unsigned-byte 8)
                       (ash xh (the (integer -8 0) (- low-num)))))
        (xl-shift (the (unsigned-byte 8)
                       (ash (the (unsigned-byte 8) xl)
                            (the (integer 0 8) places))))
        (ans (the (unsigned-byte 8)
                  (logior xl-shift xh-shift))))
       ans))))
