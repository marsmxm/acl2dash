Function: list-elements-equal

(defun list-elements-equal (e x)
       (declare (xargs :guard (true-listp x)))
       (let ((__function__ 'list-elements-equal))
            (declare (ignorable __function__))
            (if (endp x)
                t
                (and (equal (car x) e)
                     (list-elements-equal e (cdr x))))))
