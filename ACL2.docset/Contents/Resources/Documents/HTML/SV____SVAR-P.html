(defun
 svar-p (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'svar-p))
  (declare (ignorable __function__))
  (and
   (if (atom x)
       (or (stringp x)
           (and (symbolp x) (not (booleanp x))))
       (and (eq (car x) :var)
            (consp (cdr x))
            (let* ((name (cadr x)) (bits (cddr x)))
                  (and (integerp bits)
                       (implies (< bits 0)
                                (not (eql (loghead 3 bits) 0)))
                       (not (and (or (stringp name)
                                     (and (symbolp name)
                                          (not (booleanp name))))
                                 (eql bits 0)))))))
   (b* ((?name (if (atom x) x (cadr x)))
        (delay (if (atom x)
                   0
                   (if (< (cddr x) 0)
                       (logtail 3 (lognot (cddr x)))
                       (cddr x))))
        (nonblocking (if (atom x)
                         nil
                         (let ((bits (cddr x)))
                              (and (< bits 0) (logbitp 2 bits)))))
        (override-test (if (atom x)
                           nil
                           (let ((bits (cddr x)))
                                (and (< bits 0) (logbitp 0 bits)))))
        (override-val (if (atom x)
                          nil
                          (let ((bits (cddr x)))
                               (and (< bits 0) (logbitp 1 bits))))))
       (and (natp delay)
            (booleanp nonblocking)
            (booleanp override-test)
            (booleanp override-val))))))
