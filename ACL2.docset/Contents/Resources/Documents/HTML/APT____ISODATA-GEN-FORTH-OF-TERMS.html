Apply the forthk or forth_rk function to the corresponding term in a list of terms of length equal to the number of formals or results of old.

Function: isodata-gen-forth-of-terms

(defun
 isodata-gen-forth-of-terms
 (terms arg/res-isomaps)
 (declare
  (xargs
       :guard (and (pseudo-term-listp terms)
                   (isodata-symbol-isomap-alistp arg/res-isomaps))))
 (declare (xargs :guard (= (len terms) (len arg/res-isomaps))))
 (let
  ((__function__ 'isodata-gen-forth-of-terms))
  (declare (ignorable __function__))
  (b*
    (((when (endp terms)) nil)
     (term (car terms))
     (isomap (cdar arg/res-isomaps))
     (forth (isodata-isomap->forth isomap))
     (new-term (apply-term* forth term))
     (new-terms (isodata-gen-forth-of-terms (cdr terms)
                                            (cdr arg/res-isomaps))))
    (cons new-term new-terms))))
