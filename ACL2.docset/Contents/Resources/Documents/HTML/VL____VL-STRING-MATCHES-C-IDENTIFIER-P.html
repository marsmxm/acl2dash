(defun vl-string-matches-c-identifier-p (x)
       (declare (xargs :guard (stringp x)))
       (let ((__function__ 'vl-string-matches-c-identifier-p))
            (declare (ignorable __function__))
            (let ((chars (explode x)))
                 (and (consp chars)
                      (vl-simple-id-head-p (car chars))
                      (vl-simple-id-tail-list-p (cdr chars))))))
