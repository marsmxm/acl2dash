Function: vl-scopestack->path-aux

(defun
 vl-scopestack->path-aux (x rchars)
 (declare (xargs :guard (vl-scopestack-p x)))
 (let
  ((__function__ 'vl-scopestack->path-aux))
  (declare (ignorable __function__))
  (vl-scopestack-case
   x
   :null (str::revappend-chars ":null" rchars)
   :global (str::revappend-chars "$root" rchars)
   :local
   (b* ((rchars (vl-scopestack->path-aux x.super rchars))
        (id (vl-scope->id x.top)))
       (cond ((stringp id)
              (b* ((rchars (cons #\. rchars)))
                  (str::revappend-chars id rchars)))
             ((integerp id)
              (b* ((rchars (cons #\[ rchars))
                   (rchars (if (< id 0) (cons #\- rchars) rchars))
                   (rchars (str::revappend-nat-to-dec-chars (abs id)
                                                            rchars))
                   (rchars (cons #\] rchars)))
                  rchars))
             (t (str::revappend-chars
                     (cat "<unnamed-"
                          (symbol-name (vl-scope->scopetype x.top))
                          ">")
                     rchars)))))))
