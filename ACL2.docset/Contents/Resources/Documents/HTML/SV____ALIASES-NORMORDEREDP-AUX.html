Function: aliases-normorderedp-aux

(defun
  aliases-normorderedp-aux (n aliases)
  (declare (xargs :stobjs (aliases)))
  (declare (xargs :guard (natp n)))
  (declare (xargs :guard (<= n (aliass-length aliases))))
  (let ((__function__ 'aliases-normorderedp-aux))
       (declare (ignorable __function__))
       (if (zp n)
           t
           (and (lhs-vars-normorderedp (1- n)
                                       0 (get-alias (1- n) aliases))
                (aliases-normorderedp-aux (1- n)
                                          aliases)))))
