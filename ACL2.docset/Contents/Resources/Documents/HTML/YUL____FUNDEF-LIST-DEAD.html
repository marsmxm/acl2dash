Eliminate dead code in lists of function definitions.

Signature
(fundef-list-dead x) → new-fundefs
Arguments
x — Guard (fundef-listp x).
Returns
new-fundefs — Type (fundef-listp new-fundefs).

This is an ordinary std::defprojection.

Function: fundef-list-dead-exec

(defun
   fundef-list-dead-exec (x acc)
   (declare (xargs :guard (fundef-listp x)))
   (declare (xargs :guard t))
   (let ((__function__ 'fundef-list-dead-exec))
        (declare (ignorable __function__))
        (if (consp x)
            (fundef-list-dead-exec (cdr x)
                                   (cons (fundef-dead (car x)) acc))
            acc)))

Function: fundef-list-dead

(defun
 fundef-list-dead (x)
 (declare (xargs :guard (fundef-listp x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'fundef-list-dead))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (fundef-dead (car x))
                    (fundef-list-dead (cdr x)))
              nil)
   :exec
   (if
    (atom x)
    nil
    (acl2::with-local-nrev (fundef-list-dead-nrev x acl2::nrev))))))
