Definitions and Theorems

Function: vl-find-typedef

(defun vl-find-typedef (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-typedeflist-p x))))
       (let ((__function__ 'vl-find-typedef))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-typedef->name (car x)))
                   (vl-typedef-fix (car x)))
                  (t (vl-find-typedef name (cdr x))))))

Theorem: return-type-of-vl-find-typedef

(defthm return-type-of-vl-find-typedef
        (b* ((typedef? (vl-find-typedef name x)))
            (iff (vl-typedef-p typedef?) typedef?))
        :rule-classes :rewrite)

Theorem: vl-find-typedef-under-iff

(defthm vl-find-typedef-under-iff
        (iff (vl-find-typedef name x)
             (member-equal (string-fix name)
                           (vl-typedeflist->names x))))

Theorem: vl-typedef->name-of-vl-find-typedef

(defthm vl-typedef->name-of-vl-find-typedef
        (implies (vl-find-typedef name x)

(defthm tag-of-vl-find-typedef

Theorem: member-equal-of-vl-find-typedef

(defthm member-equal-of-vl-find-typedef
        (implies (force (vl-typedeflist-p x))
                 (iff (member-equal (vl-find-typedef name x)
                                    x)
                      (vl-find-typedef name x))))

Theorem: consp-of-vl-find-typedef-when-member-equal

(defthm consp-of-vl-find-typedef-when-member-equal
