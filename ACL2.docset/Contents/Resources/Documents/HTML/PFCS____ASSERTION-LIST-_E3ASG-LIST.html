 (declare (xargs :guard t))
 (let
  ((__function__ 'assertion-list->asg-list))
  (declare (ignorable __function__))
  (mbe
    :logic (if (consp x)
               (cons (assertion->asg (car x))
                     (assertion-list->asg-list (cdr x)))
