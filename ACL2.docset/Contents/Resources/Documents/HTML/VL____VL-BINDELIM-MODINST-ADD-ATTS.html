(defun
 vl-bindelim-modinst-add-atts (x intent)
 (declare (xargs :guard (and (vl-modinst-p x)
                             (vl-bindintent-p intent))))
 (let
  ((__function__ 'vl-bindelim-modinst-add-atts))
  (declare (ignorable __function__))
  (b*
   (((vl-modinst x))
    ((vl-bindintent intent))
    (val (cat "Added by bind statement from "
              (vl-bindcontext->shortdescription intent.ctx)
              " at "
              (vl-location-string (vl-bind->loc intent.source))))
    (atts
     (cons (cons "VL_FROM_BIND"
                 (make-vl-literal :val (make-vl-string :value val)))
           x.atts)))
   (change-vl-modinst x :atts atts))))
