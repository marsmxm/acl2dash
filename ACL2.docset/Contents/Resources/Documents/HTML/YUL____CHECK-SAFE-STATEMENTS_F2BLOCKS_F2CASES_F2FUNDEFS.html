Check if statements, blocks, cases, and function definitions are safe.

These are checked in the context of a variable table for visible and accessible variable vartab, and a function table funtab.

A successful check returns a set of possible modes in which execution may complete. The set of modes is used to check the conditions on the occurrence of leave, break, and continue statements described in [Yul: Specification of Yul: Restrictions on the Grammar]. The conditions there are phrased in terms of occurrences of constructs, but it is more convenient, in this formal static semantics, to check these conditions by putting restrictions on the possible modes statically calculated for statements and related constructs. This formulation is also more easily related to the dynamic semantics. At some point we may also formalize the conditions on the occurrence of leave, break, and continue statements described in [Yul: Specification of Yul: Restrictions on the Grammar], and show that they are equivalent to our formulation here.

Definitions and Theorems

Function: check-safe-statement

(defun
 check-safe-statement
 (stmt vartab funtab)
 (declare (xargs :guard (and (statementp stmt)
                             (vartablep vartab)
                             (funtablep funtab))))
 (let
  ((__function__ 'check-safe-statement))
  (declare (ignorable __function__))
  (statement-case
   stmt
   :block (b* (((ok modes)
                (check-safe-block stmt.get vartab funtab)))
              (make-vartable-modes :variables (vartable-fix vartab)
                                   :modes modes))
   :variable-single
   (b* (((ok vartab)
         (check-safe-variable-single
              stmt.name stmt.init vartab funtab)))
       (make-vartable-modes :variables vartab
                            :modes (insert (mode-regular) nil)))
   :variable-multi
   (b* (((ok vartab)
         (check-safe-variable-multi
              stmt.names stmt.init vartab funtab)))
       (make-vartable-modes :variables vartab
                            :modes (insert (mode-regular) nil)))
   :assign-single
   (b* (((ok &)
         (check-safe-assign-single
              stmt.target stmt.value vartab funtab)))
       (make-vartable-modes :variables (vartable-fix vartab)
                            :modes (insert (mode-regular) nil)))
   :assign-multi
   (b* (((ok &)
         (check-safe-assign-multi
              stmt.targets stmt.value vartab funtab)))
       (make-vartable-modes :variables (vartable-fix vartab)
                            :modes (insert (mode-regular) nil)))
   :funcall
   (b* (((ok results)
         (check-safe-funcall stmt.get vartab funtab))
        ((unless (= results 0))
         (err (list :discarded-values stmt.get))))
       (make-vartable-modes :variables (vartable-fix vartab)
                            :modes (insert (mode-regular) nil)))
   :if
   (b* (((ok results)
         (check-safe-expression stmt.test vartab funtab))
        ((unless (= results 1))
         (err (list :multi-valued-if-test stmt.test)))
        ((ok modes)
         (check-safe-block stmt.body vartab funtab)))
       (make-vartable-modes :variables (vartable-fix vartab)
                            :modes (insert (mode-regular) modes)))
   :for (b* ((stmts (block->statements stmt.init))
             ((ok funtab)
              (add-funtypes (statements-to-fundefs stmts)
                            funtab))
             ((ok vartab-modes)
              (check-safe-statement-list stmts vartab funtab))
             (vartab1 (vartable-modes->variables vartab-modes))
             (init-modes (vartable-modes->modes vartab-modes))
             ((when (in (mode-break) init-modes))
              (err (list :break-in-loop-init stmt.init)))
             ((when (in (mode-continue) init-modes))
              (err (list :continue-in-loop-init stmt.init)))
             ((ok results)
              (check-safe-expression stmt.test vartab1 funtab))
             ((unless (= results 1))
              (err (list :multi-valued-for-test stmt.test)))
             ((ok update-modes)
              (check-safe-block stmt.update vartab1 funtab))
             ((when (in (mode-break) update-modes))
              (err (list :break-in-loop-update stmt.update)))
             ((when (in (mode-continue) update-modes))
              (err (list :continue-in-loop-update stmt.update)))
             ((ok body-modes)
              (check-safe-block stmt.body vartab1 funtab))
             (modes (if (or (in (mode-leave) init-modes)
                            (in (mode-leave) update-modes)
                            (in (mode-leave) body-modes))
                        (insert (mode-leave)
                                (insert (mode-regular) nil))
                        (insert (mode-regular) nil))))
            (make-vartable-modes :variables (vartable-fix vartab)
                                 :modes modes))
   :switch
   (b*
    (((ok results)
      (check-safe-expression stmt.target vartab funtab))
     ((unless (= results 1))
      (err (list :multi-valued-switch-target stmt.target)))
     ((unless (or (consp stmt.cases) stmt.default))
      (err (list :no-cases-in-switch (statement-fix stmt))))
     ((unless
        (no-duplicatesp-equal (swcase-list->value-list stmt.cases)))
      (err (list :duplicate-switch-cases (statement-fix stmt))))
     ((ok cases-modes)
      (check-safe-swcase-list stmt.cases vartab funtab))
     ((ok default-modes)
      (check-safe-block-option stmt.default vartab funtab)))
    (make-vartable-modes :variables (vartable-fix vartab)
                         :modes (union cases-modes default-modes)))
   :leave (make-vartable-modes :variables (vartable-fix vartab)
                               :modes (insert (mode-leave) nil))
   :break (make-vartable-modes :variables (vartable-fix vartab)
                               :modes (insert (mode-break) nil))
   :continue
   (make-vartable-modes :variables (vartable-fix vartab)
                        :modes (insert (mode-continue) nil))
   :fundef
   (b* (((ok &)
         (check-safe-fundef stmt.get funtab)))
       (make-vartable-modes :variables (vartable-fix vartab)
                            :modes (insert (mode-regular) nil))))))

Function: check-safe-statement-list

(defun
 check-safe-statement-list
 (stmts vartab funtab)
 (declare (xargs :guard (and (statement-listp stmts)
                             (vartablep vartab)
                             (funtablep funtab))))
 (let
     ((__function__ 'check-safe-statement-list))
     (declare (ignorable __function__))
     (b* (((when (endp stmts))
           (make-vartable-modes :variables (vartable-fix vartab)
                                :modes (insert (mode-regular) nil)))
          ((ok vartab-modes)
           (check-safe-statement (car stmts)
                                 vartab funtab))
          (vartab (vartable-modes->variables vartab-modes))
          (first-modes (vartable-modes->modes vartab-modes))
          ((ok vartab-modes)
           (check-safe-statement-list (cdr stmts)
                                      vartab funtab))
          (vartab (vartable-modes->variables vartab-modes))
          (rest-modes (vartable-modes->modes vartab-modes))
          (modes (if (in (mode-regular) first-modes)
                     (union first-modes rest-modes)
                     first-modes)))
         (make-vartable-modes :variables vartab
                              :modes modes))))

Function: check-safe-block

(defun check-safe-block (block vartab funtab)
       (declare (xargs :guard (and (blockp block)
                                   (vartablep vartab)
                                   (funtablep funtab))))
       (let ((__function__ 'check-safe-block))
            (declare (ignorable __function__))
            (b* ((stmts (block->statements block))
                 ((ok funtab)
                  (add-funtypes (statements-to-fundefs stmts)
                                funtab))
                 ((ok vartab-modes)
                  (check-safe-statement-list stmts vartab funtab)))
                (vartable-modes->modes vartab-modes))))

Function: check-safe-block-option

(defun
   check-safe-block-option
   (block? vartab funtab)
   (declare (xargs :guard (and (block-optionp block?)
                               (vartablep vartab)
                               (funtablep funtab))))
   (let ((__function__ 'check-safe-block-option))
        (declare (ignorable __function__))
        (block-option-case
             block?
             :none (insert (mode-regular) nil)
             :some (check-safe-block (block-option-some->val block?)
                                     vartab funtab))))

Function: check-safe-swcase

(defun check-safe-swcase (case vartab funtab)
       (declare (xargs :guard (and (swcasep case)
                                   (vartablep vartab)
                                   (funtablep funtab))))
       (let ((__function__ 'check-safe-swcase))
            (declare (ignorable __function__))
            (b* (((swcase case) case)
                 ((ok &)
                  (check-safe-literal case.value)))
                (check-safe-block case.body vartab funtab))))

Function: check-safe-swcase-list

(defun check-safe-swcase-list
       (cases vartab funtab)
       (declare (xargs :guard (and (swcase-listp cases)
                                   (vartablep vartab)
                                   (funtablep funtab))))
       (let ((__function__ 'check-safe-swcase-list))
            (declare (ignorable __function__))
            (b* (((when (endp cases)) nil)
                 ((ok first-modes)
                  (check-safe-swcase (car cases)
                                     vartab funtab))
                 ((ok rest-modes)
                  (check-safe-swcase-list (cdr cases)
                                          vartab funtab)))
                (union first-modes rest-modes))))

Function: check-safe-fundef

(defun
 check-safe-fundef (fundef funtab)
 (declare (xargs :guard (and (fundefp fundef)
                             (funtablep funtab))))
 (let
    ((__function__ 'check-safe-fundef))
    (declare (ignorable __function__))
    (b* (((fundef fundef) fundef)
         ((ok vartab)
          (add-vars (append fundef.inputs fundef.outputs)
                    nil))
         ((ok modes)
          (check-safe-block fundef.body vartab funtab))
         ((when (in (mode-break) modes))
          (err (list :break-from-function (fundef-fix fundef))))
         ((when (in (mode-continue) modes))
          (err (list :continue-from-function (fundef-fix fundef)))))
        nil)))
