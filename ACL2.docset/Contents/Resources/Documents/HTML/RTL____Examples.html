Function: rcp24

(defun rcp24 (b)
       (let* ((i (fl (* (expt 2 7) (1- b))))
              (x (- b (1+ (* (expt 2 -7) i))))
              (c0 (* (expt 2 -27) (nth i (rcp0))))
              (c1 (- (* (expt 2 -17) (nth i (rcp1)))))
              (c2 (* (expt 2 -12) (nth i (rcp2)))))
             (rne (+ c0 (* c1 x) (* c2 x x)) 24)))

Theorem: rcp24-spec

(defthm rcp24-spec
        (implies (and (rationalp b)
                      (exactp b 24)
                      (<= 1 b)
                      (< b 2))
                 (and (exactp (rcp24 b) 24)
                      (<= 1/2 (rcp24 b))
                      (<= (rcp24 b) 1)
                      (< (abs (- 1 (* b (rcp24 b))))
                         (expt 2 -23))))
        :rule-classes nil)

Function: divsp

(defun divsp (a b mode)
       (let* ((y0 (rcp24 b))
              (q0 (rne (* a y0) 24))
              (e0 (rne (- 1 (* b y0)) 24))
              (r0 (rne (- a (* b q0)) 24))
              (y1 (rne (+ y0 (* e0 y0)) 24))
              (q1 (rne (+ q0 (* r0 y1)) 24))
              (r1 (rne (- a (* b q1)) 24)))
             (rnd (+ q1 (* r1 y1)) mode 24)))

Theorem: divsp-correct

(defthm divsp-correct
        (implies (and (rationalp a)
                      (rationalp b)
                      (<= 1 a)
                      (< a 2)
                      (<= 1 b)
                      (< b 2)
                      (exactp a 24)
                      (exactp b 24)
                      (ieee-rounding-mode-p mode))
                 (= (divsp a b mode)
                    (rnd (/ a b) mode 24)))
        :rule-classes nil)
