Function: svstate-p

(defun svstate-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'svstate-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (b* ((blkst (std::prod-car x))
                      (nonblkst (std::prod-cdr x)))
                     (and (svstack-p blkst)
                          (svex-alist-p nonblkst)
                          (consp blkst))))))
