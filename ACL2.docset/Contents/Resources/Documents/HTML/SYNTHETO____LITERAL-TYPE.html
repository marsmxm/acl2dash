Function: literal-type

(defun literal-type (lit)
       (declare (xargs :guard (literalp lit)))
       (let ((__function__ 'literal-type))
            (declare (ignorable __function__))
            (literal-case lit
                          :boolean (type-boolean)
                          :character (type-character)
                          :string (type-string)
                          :integer (type-integer))))

Theorem: typep-of-literal-type

(defthm typep-of-literal-type
        (b* ((type (literal-type lit)))
            (typep type))
        :rule-classes :rewrite)

Theorem: literal-type-of-literal-fix-lit

(defthm literal-type-of-literal-fix-lit
