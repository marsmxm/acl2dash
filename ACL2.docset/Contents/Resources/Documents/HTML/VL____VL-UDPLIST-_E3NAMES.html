Function: vl-udplist->names

(defun
 vl-udplist->names (x)
 (declare (xargs :guard (vl-udplist-p x)))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-udplist->names))
  (declare (ignorable __function__))
  (mbe
     :logic (if (consp x)
                (cons (vl-udp->name (car x))
                      (vl-udplist->names (cdr x)))
                nil)
     :exec (if (atom x)
               nil
               (with-local-nrev (vl-udplist->names-nrev x nrev))))))
