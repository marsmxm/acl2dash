Function: svar-named->indexed

(defun
 svar-named->indexed (x modidx moddb)
 (declare (xargs :stobjs (moddb)))
 (declare (xargs :guard (and (svar-p x)
                             (natp modidx)
                             (moddb-ok moddb))))
 (declare (xargs :guard (and (< modidx (moddb->nmods moddb))
                             (svar-addr-p x))))
 (let
  ((__function__ 'svar-named->indexed))
  (declare (ignorable __function__))
  (b* ((addr (svar->address x))
       ((address addr))
       ((unless (eql 0 addr.scope))
        (mv nil
            (change-svar x
                         :name (change-address addr :index nil)
                         :override-test nil
                         :override-val nil)))
       (idx (moddb-path->wireidx addr.path modidx moddb))
       ((unless idx)
        (mv (msg "Did not find wire: ~x0 in module ~s1~%"
                 addr.path
                 (stobj-let ((elab-mod (moddb->modsi modidx moddb)))
                            (name)
                            (elab-mod->name elab-mod)
                            name))
            (change-svar x
                         :name (change-address addr :index nil)
                         :override-test nil
                         :override-val nil))))
      (mv nil
          (change-svar x
                       :name (change-address addr :index idx)
                       :override-test nil
                       :override-val nil)))))
