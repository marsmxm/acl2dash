Function: vl-module-stmttemps

(defun vl-module-stmttemps (x)
       (declare (xargs :guard (vl-module-p x)))
       (let ((__function__ 'vl-module-stmttemps))
            (declare (ignorable __function__))
            (b* (((vl-module x) x)
                 ((when (vl-module->hands-offp x)) x)
                 ((unless x.alwayses) x)
                 (delta (vl-starting-delta x))
                 (delta (change-vl-delta delta
                                         :vardecls x.vardecls
                                         :assigns x.assigns))
                 ((mv alwayses delta)
                  (vl-alwayslist-stmttemps x.alwayses delta))
                 ((vl-delta delta)
                  (vl-free-delta delta)))
                (change-vl-module x
                                  :alwayses alwayses
                                  :assigns delta.assigns
                                  :vardecls delta.vardecls
                                  :warnings delta.warnings))))
