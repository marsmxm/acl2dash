Function: vl-configlist->names

(defun
 vl-configlist->names (x)
 (declare (xargs :guard (vl-configlist-p x)))
 (declare (xargs :guard t))
 (let
   ((__function__ 'vl-configlist->names))
   (declare (ignorable __function__))
   (mbe :logic (if (consp x)
                   (cons (vl-config->name (car x))
                         (vl-configlist->names (cdr x)))
                   nil)
        :exec
        (if (atom x)
            nil
            (with-local-nrev (vl-configlist->names-nrev x nrev))))))
