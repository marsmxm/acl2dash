Function: binary-and*

(defun binary-and* (a b)
       (declare (xargs :guard t))
       (and a b))

Function: and*-macro

(defun and*-macro (lst)
       (if (atom lst)
           t
           (if (atom (cdr lst))
               (car lst)
               (list 'binary-and*
                     (car lst)
                     (and*-macro (cdr lst))))))

Theorem: iff-implies-equal-and*-1

(defthm iff-implies-equal-and*-1
        (implies (iff a a-equiv)
                 (equal (and* a b) (and* a-equiv b)))
        :rule-classes (:congruence))

Theorem: iff-implies-iff-and*-2

(defthm iff-implies-iff-and*-2
        (implies (iff b b-equiv)
                 (iff (and* a b) (and* a b-equiv)))
        :rule-classes (:congruence))
