Function: vl-importlist->star-packages

(defun vl-importlist->star-packages (x)
       (declare (xargs :guard (vl-importlist-p x)))
       (let ((__function__ 'vl-importlist->star-packages))
            (declare (ignorable __function__))
            (b* (((when (atom x)) nil)
                 ((vl-import x1) (car x)))
                (if (eq x1.part :vl-import*)
