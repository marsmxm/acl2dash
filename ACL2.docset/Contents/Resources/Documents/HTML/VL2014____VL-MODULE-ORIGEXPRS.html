Function: vl-module-origexprs

(defun
  vl-module-origexprs (x)
  (declare (xargs :guard (vl-module-p x)))
  (let ((__function__ 'vl-module-origexprs))
       (declare (ignorable __function__))
       (let ((x (vl-module-fix x)))
            (b* (((when (vl-module->hands-offp x)) x)
                 ((vl-module x) x)
                 (assigns (vl-assignlist-origexprs x.assigns))
                 (gateinsts (vl-gateinstlist-origexprs x.gateinsts))
                 (modinsts (vl-modinstlist-origexprs x.modinsts))
                 (alwayses (vl-alwayslist-origexprs x.alwayses))
                 (initials (vl-initiallist-origexprs x.initials)))
                (change-vl-module x
                                  :assigns assigns
                                  :gateinsts gateinsts
                                  :modinsts modinsts
                                  :alwayses alwayses
                                  :initials initials)))))
