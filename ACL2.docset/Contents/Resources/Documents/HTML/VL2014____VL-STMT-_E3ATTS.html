Theorem: vl-atts-p-of-vl-stmt->atts

(defthm vl-atts-p-of-vl-stmt->atts
        (b* ((atts (vl-stmt->atts x)))
            (vl-atts-p atts))
        :rule-classes :rewrite)

Theorem: vl-stmt->atts-of-vl-stmt-fix-x

(defthm vl-stmt->atts-of-vl-stmt-fix-x
        (equal (vl-stmt->atts (vl-stmt-fix x))
               (vl-stmt->atts x)))

Theorem: vl-stmt->atts-vl-stmt-equiv-congruence-on-x

(defthm vl-stmt->atts-vl-stmt-equiv-congruence-on-x
        (implies (vl-stmt-equiv x x-equiv)
                 (equal (vl-stmt->atts x)

        :rule-classes :congruence)
