b* binder form LOCAL-IPASIR (placeholder).

This is a b* binder introduced with ACL2::def-b*-binder.

Macro: patbind-local-ipasir

(defmacro
 patbind-local-ipasir
 (args acl2::forms acl2::rest-expr)
 (b*
  ((forms acl2::forms)
   (rest-expr acl2::rest-expr)
   ((unless (or (not args)
                (and (consp args)
                     (symbolp (car args))
                     (not (cdr args)))))
    (er hard? 'local-ipasir
        "Bad local-ipasir args: ~x0~%" args))
   ((unless (and (eql (len forms) 1)
                 (symbol-listp (car forms))
                 (eq (caar forms) 'mv)))
    (er
     hard? 'local-ipasir
     "Bound form must be an mv of some symbols, giving the return values.")))
  (cons
   'with-local-ipasir
   (append
    args
    (cons
     (cons
      'mv-let
      (cons (cdar forms)
            (cons rest-expr
                  (b* ((returns (remove (if args (car args) 'ipasir)
                                        (cdar forms))))
                      (if (consp (cdr returns))
                          (cons (cons 'mv returns) 'nil)
                          returns)))))
     'nil)))))
