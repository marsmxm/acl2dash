Definition: completion-of-<

(defaxiom
     completion-of-<
     (equal (< x y)
            (if (and (real/rationalp x)
                     (real/rationalp y))
                (< x y)
                (let ((x1 (if (acl2-numberp x) x 0))
                      (y1 (if (acl2-numberp y) y 0)))
                     (or (< (realpart x1) (realpart y1))
                         (and (equal (realpart x1) (realpart y1))
                              (< (imagpart x1) (imagpart y1)))))))
     :rule-classes nil)
