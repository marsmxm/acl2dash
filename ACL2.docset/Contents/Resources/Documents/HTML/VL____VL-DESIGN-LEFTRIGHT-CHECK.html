Function: vl-design-leftright-check

(defun
  vl-design-leftright-check (x)
  (declare (xargs :guard (vl-design-p x)))
  (let ((__function__ 'vl-design-leftright-check))
       (declare (ignorable __function__))
       (b* (((vl-design x))
            (ss (vl-scopestack-init (vl-design-fix x)))
            (mods (vl-modulelist-leftright-check x.mods ss))
            (ifs (vl-interfacelist-leftright-check x.interfaces ss))
            (pkgs (vl-packagelist-leftright-check x.packages ss))
            (warnings (append (vl-ctxexprlist-leftright-check
                                   (vl-design-toplevel-ctxexprs x))
                              x.warnings)))
           (change-vl-design x
                             :mods mods

                             :packages pkgs
                             :warnings warnings))))
