Function: sd-keylist->indicies

(defun

 (declare (xargs :guard (sd-keylist-p x)))
 (let
   ((__function__ 'sd-keylist->indicies))
   (declare (ignorable __function__))
   (mbe :logic (if (consp x)
                   (cons (sd-key->index (car x))
                         (sd-keylist->indicies (cdr x)))
