Helper function for creating lists of port declarations.

Signature
(vl-occform-mkports prefix i n &key dir width (loc '*vl-fakeloc*)) 
  → 
(mv exprs ports portdecls vardecls)
Arguments
prefix — name prefix for each port.
    Guard (stringp prefix).
i — starting index, inclusive!.
    Guard (natp i).
n — ending index, non-inclusive!.
    Guard (natp n).
dir — direction of each port.
    Guard (vl-direction-p dir).
width — width of each port.
    Guard (posp width).
loc — Guard (vl-location-p loc).
Returns
exprs — Type (vl-exprlist-p exprs).
ports — Type (vl-portlist-p ports).
portdecls — Type (vl-portdecllist-p portdecls).
vardecls — Type (vl-vardecllist-p vardecls).

Function: vl-occform-mkports-fn

(defun vl-occform-mkports-fn
       (prefix i n dir width loc)
       (declare (xargs :guard (and (stringp prefix)
                                   (natp i)
                                   (natp n)
                                   (vl-direction-p dir)
                                   (posp width)
                                   (vl-location-p loc))))
       (declare (xargs :guard (<= i n)))
       (let ((__function__ 'vl-occform-mkports))
            (declare (ignorable __function__))
            (b* (((when (mbe :logic (zp (- (lnfix n) (lnfix i)))
                             :exec (eql i n)))
                  (mv nil nil nil nil))
                 (name1 (hons-copy (cat prefix (natstr i))))
                 ((mv expr1 port1 portdecl1 vardecl1)
                  (vl-occform-mkport name1 dir width))
                 ((mv exprs2 ports2 portdecls2 vardecls2)
                  (vl-occform-mkports prefix (+ 1 (lnfix i))
                                      n
                                      :dir dir
                                      :width width
                                      :loc loc)))
                (mv (cons expr1 exprs2)
                    (cons port1 ports2)
                    (cons portdecl1 portdecls2)
                    (cons vardecl1 vardecls2)))))

Theorem: len-of-vl-occform-mkports

(defthm len-of-vl-occform-mkports
        (b* (((mv exprs ports portdecls vardecls)
              (vl-occform-mkports prefix i n
                                  :dir dir
                                  :width width
                                  :loc loc))
             (len (nfix (- (nfix n) (nfix i)))))
            (and (equal (len exprs) len)
                 (equal (len ports) len)
                 (equal (len portdecls) len)
                 (equal (len vardecls) len))))
