(defun vl-jp-hidpiece-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-hidpiece-p x)))
       (let ((__function__ 'vl-jp-hidpiece))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-hidpiece)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-hidpiece->name x))
                       (vl-println? "}"))))
