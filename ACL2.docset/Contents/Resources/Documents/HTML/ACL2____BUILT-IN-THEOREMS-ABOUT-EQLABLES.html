Built-in theorems about eqlables.

Theorem: eqlablep-recog

(defthm eqlablep-recog
        (equal (eqlablep x)
               (or (acl2-numberp x)
                   (symbolp x)
                   (characterp x)))
        :rule-classes :compound-recognizer)

Theorem: eqlablep-nth

(defthm eqlablep-nth
        (implies (eqlable-listp x)
                 (eqlablep (nth n x))))
