 proof-obligationp (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'proof-obligationp))
  (declare (ignorable __function__))
  (and
   (mbe
      :logic (and (alistp x)
                  (equal (strip-cars x)
                         '(variables hypotheses
                                     conclusion source-expression)))

                 x
                 '(variables hypotheses
                             conclusion source-expression)))
   (b* ((variables (cdr (std::da-nth 0 x)))
        (hypotheses (cdr (std::da-nth 1 x)))
