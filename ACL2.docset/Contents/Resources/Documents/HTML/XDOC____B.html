Macro: b

(defmacro
   b (&rest args)
   (mv-let
        (trees attributes)
        (partition-macro-args args 'b)
        (let ((attributes (keyword-macro-args-to-terms attributes)))
             (list 'b-fn
                   (cons 'list attributes)
                   (cons 'list trees)))))

(defun b-fn (attributes trees)
       (declare (xargs :guard (and (keyword-tree-alistp attributes)
                                   (tree-listp trees))))
       (make-tree-tag :b attributes trees))

Theorem: stringp-of-b

(defthm stringp-of-b
        (equal (treep (b-fn attributes trees))
               (and (keyword-tree-alistp attributes)
                    (tree-listp trees))))
