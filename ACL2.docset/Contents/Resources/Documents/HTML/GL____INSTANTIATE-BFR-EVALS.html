(defun
 instantiate-bfr-evals (a b vals)
 (declare (xargs :guard t))
 (let
  ((__function__ 'instantiate-bfr-evals))
  (declare (ignorable __function__))
  (if
   (atom vals)
   nil
   (cons
    (cons
      'not
      (cons (cons 'equal
                  (cons (cons 'bfr-eval
