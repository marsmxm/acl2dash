Function: atj-type-annotate-vars

(defun
     atj-type-annotate-vars (vars types)
     (declare (xargs :guard (and (symbol-listp vars)
                                 (atj-type-listp types))))
     (declare (xargs :guard (int= (len vars) (len types))))
     (let ((__function__ 'atj-type-annotate-vars))
          (declare (ignorable __function__))
          (cond ((endp vars) nil)
                (t (cons (atj-type-annotate-var (car vars)
                                                (list (car types)))
                         (atj-type-annotate-vars (cdr vars)
                                                 (cdr types)))))))
