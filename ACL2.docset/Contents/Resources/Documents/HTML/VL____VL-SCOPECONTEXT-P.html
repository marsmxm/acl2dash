Function: vl-scopecontext-p

(defun
     vl-scopecontext-p (x)
     (declare (xargs :guard t))
     (let ((__function__ 'vl-scopecontext-p))
          (declare (ignorable __function__))
          (and (consp x)
               (cond ((or (atom x) (eq (car x) :local))
                      (and (b* ((levels (cdr x))) (natp levels))))
                     ((eq (car x) :root)
                      (and (eq (cdr x) nil) (b* nil t)))
                     ((eq (car x) :package)
                      (and (b* ((pkg (cdr x)))
                               (vl-package-p pkg))))
                     ((eq (car x) :module)
                      (and (b* ((mod (cdr x))) (vl-module-p mod))))
                     (t (and (eq (car x) :interface)
                             (and)
                             (b* ((iface (cdr x)))
                                 (vl-interface-p iface))))))))
