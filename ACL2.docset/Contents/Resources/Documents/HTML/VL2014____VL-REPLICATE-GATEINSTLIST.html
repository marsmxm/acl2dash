(defun vl-replicate-gateinstlist
       (x nf ss warnings)
       (declare (xargs :guard (and (vl-gateinstlist-p x)
                                   (vl-namefactory-p nf)
                                   (vl-scopestack-p ss)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-replicate-gateinstlist))
            (declare (ignorable __function__))
            (b* (((when (atom x))
                  (mv (ok) nil (vl-namefactory-fix nf)))
                 ((mv warnings car-prime nf)
                  (vl-replicate-gateinst (car x)

                                             nf ss warnings))

                (mv warnings new-gateinsts nf))))
