(defun vl-udp-suppress-file-warnings
       (x suppress-files)
       (declare (xargs :guard (and (vl-udp-p x)
                                   (string-listp suppress-files))))
       (let ((__function__ 'vl-udp-suppress-file-warnings))
            (declare (ignorable __function__))
            (if (and (consp (vl-udp->warnings x))
                     (vl-filename-to-suppress-p
