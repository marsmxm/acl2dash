Function: max-supertypes

(defun max-supertypes (types tops)
       (declare (xargs :guard (and (type-listp types)
                                   (toplevel-listp tops))))
       (let ((__function__ 'max-supertypes))
            (declare (ignorable __function__))
            (if (endp types)
                nil
                (cons (or (max-supertype (car types) tops)
                          (type-fix (car types)))
                      (max-supertypes (cdr types) tops)))))
