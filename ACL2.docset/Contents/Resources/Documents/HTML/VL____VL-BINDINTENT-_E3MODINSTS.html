Function: vl-bindintent->modinsts

(defun
  vl-bindintent->modinsts (x)
  (declare (xargs :guard (vl-bindintent-p x)))
  (let ((__function__ 'vl-bindintent->modinsts))
       (declare (ignorable __function__))
       (b* (((vl-bindintent x))
            ((vl-bind x.source)))
           (vl-bindelim-modinstlist-add-atts x.source.modinsts x))))
