Function: wordp

(defun wordp (atom)
       (declare (xargs :guard t))
       (let ((acl2::__function__ 'wordp))
            (declare (ignorable acl2::__function__))
            (if (or (acl2-numberp atom)
                    (symbolp atom)
                    (characterp atom)
                    (stringp atom))
                t nil)))

Function: word-fix

(defun word-fix (atom)
       (declare (xargs :guard (wordp atom)))
       (let ((acl2::__function__ 'word-fix))

            (mbe :logic (if (wordp atom) atom nil)
                 :exec atom)))
