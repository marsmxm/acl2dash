(defun
     vl-atomify-assignpairs (x)
     (declare (xargs :guard (and (alistp x)
                                 (vl-idtoken-list-p (strip-cars x))
                                 (vl-exprlist-p (strip-cdrs x)))))
     (let ((__function__ 'vl-atomify-assignpairs))
          (declare (ignorable __function__))

              nil
