Function: vl-arguments-p

(defun vl-arguments-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-arguments-p))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x)
                            (eq (car x) :vl-arguments-plain))
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((args (std::da-nth 0 (cdr x))))
                                 (vl-plainarglist-p args))))
                       (t (and (eq (car x) :vl-arguments-named)
                               (and (true-listp (cdr x))
                                    (eql (len (cdr x)) 2))
                               (b* ((args (std::da-nth 0 (cdr x)))
                                    (starp (std::da-nth 1 (cdr x))))
                                   (and (vl-namedarglist-p args)
                                        (booleanp starp)))))))))
