(defun
 simple-sat-fn (formula config)
 (declare (xargs :guard (and (lit-list-listp formula)

 "Just returns STATUS, not env."
 (let
    ((__function__ 'simple-sat))
    (declare (ignorable __function__))
    (progn$
         (tshell-ensure)
         (with-local-stobj env$
                           (mv-let (status env$)
