(defthm nat-listp-when-not-consp

Theorem: nat-listp-of-cons

(defthm nat-listp-of-cons
        (equal (nat-listp (cons a x))

Theorem: car-nat-listp

(defthm car-nat-listp
        (implies (and (nat-listp x) x)

        :rule-classes :forward-chaining)
