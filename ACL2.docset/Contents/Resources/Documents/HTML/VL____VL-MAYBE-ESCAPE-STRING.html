(defun
    vl-maybe-escape-string (x i len acc)

         (xargs :guard (and (stringp x)
                            (natp i)
                            (character-listp acc)
                            (and (natp len) (eql len (length x))))))
    (declare (xargs :guard (<= i len)))
    (let ((__function__ 'vl-maybe-escape-string))
