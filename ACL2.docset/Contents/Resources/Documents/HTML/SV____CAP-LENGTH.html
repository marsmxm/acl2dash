Function: cap-length

(defun cap-length (n x)
       (declare (xargs :guard (natp n)))
       (let ((__function__ 'cap-length))
            (declare (ignorable __function__))
            (if (< (lnfix n) (len x))
                (take n (list-fix x))
                x)))

Theorem: cap-length-of-nfix-n

(defthm cap-length-of-nfix-n
        (equal (cap-length (nfix n) x)
               (cap-length n x)))
