Function: vl-find-modport

(defun vl-find-modport (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-modportlist-p x))))
       (let ((__function__ 'vl-find-modport))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-modport->name (car x)))
                   (vl-modport-fix (car x)))
                  (t (vl-find-modport name (cdr x))))))
