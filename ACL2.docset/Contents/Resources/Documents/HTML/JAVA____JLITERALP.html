(defun jliteralp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'jliteralp))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x) (eq (car x) :integer))
                        (and (true-listp (cdr x))
