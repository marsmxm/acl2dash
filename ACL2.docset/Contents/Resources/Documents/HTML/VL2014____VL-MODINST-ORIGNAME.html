Function: vl-modinst-origname

(defun
 vl-modinst-origname (x)
 (declare (xargs :guard (vl-modinst-p x)))
 (let
  ((__function__ 'vl-modinst-origname))
  (declare (ignorable __function__))
  (b* (((vl-modinst x) x)
       (x.instname (mbe :logic (and (stringp x.instname) x.instname)
                        :exec x.instname))
       (look (assoc-equal "VL_REPLICATE_ORIGNAME" x.atts))
       ((when (not (cdr look))) x.instname)
       ((when (not (and (vl-atom-p (cdr look))
                        (vl-string-p (vl-atom->guts (cdr look))))))
        (raise "Malformed VL_REPLICATE_ORIGNAME attribute: ~x0~%"
               (cdr look))
        x.instname))
      (vl-string->value (vl-atom->guts (cdr look))))))
