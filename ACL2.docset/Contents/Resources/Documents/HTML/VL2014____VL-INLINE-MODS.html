Function: vl-inline-mods

(defun
     vl-inline-mods (x all-mods)
     (declare (xargs :guard (and (and (vl-modulelist-p x)
                                      (vl-ok-to-inline-list-p x))
                                 (vl-modulelist-p all-mods))))
     (let ((__function__ 'vl-inline-mods))
          (declare (ignorable __function__))
          (if (atom x)
              all-mods
              (b* ((all-mods (vl-inline-mod-in-mods-aux (car x)
                                                        all-mods)))
