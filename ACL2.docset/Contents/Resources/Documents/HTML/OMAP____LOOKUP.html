Function: lookup

(defun lookup (key map)
       (declare (xargs :guard (mapp map)))
       (declare (xargs :guard (in key map)))
       (let ((__function__ 'lookup))
            (declare (ignorable __function__))
            (cdr (in key map))))

Theorem: lookup-when-empty

(defthm lookup-when-empty
        (implies (empty map)
                 (not (lookup key map)))
        :rule-classes (:rewrite :type-prescription))

Theorem: acl2-count-lookup-<-map

(defthm acl2-count-lookup-<-map
        (implies (not (empty map))
                 (< (acl2-count (lookup key map))
                    (acl2-count map))))
