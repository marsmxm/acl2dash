Function: vl-namemangle-gateinsts

(defun vl-namemangle-gateinsts
       (prefix gateinsts nf)
       (declare (xargs :guard (and (stringp prefix)
                                   (vl-gateinstlist-p gateinsts)
                                   (vl-namefactory-p nf))))
       (let ((__function__ 'vl-namemangle-gateinsts))
            (declare (ignorable __function__))
            (b* (((when (atom gateinsts))
                  (mv nil (vl-namefactory-fix nf)))
                 (instname1 (or (vl-gateinst->name (car gateinsts))
                                "g"))
                 (want1 (cat prefix "_" instname1))
                 ((mv fresh1 nf)
                  (vl-namefactory-plain-name want1 nf))
                 (inst1 (change-vl-gateinst (car gateinsts)
                                            :name fresh1))
                 ((mv insts2 nf)
                  (vl-namemangle-gateinsts prefix (cdr gateinsts)
                                           nf)))
                (mv (cons inst1 insts2) nf))))
