Function: svex-select-to-lhs

(defun
 svex-select-to-lhs (x)
 (declare (xargs :guard (svex-select-p x)))
 (declare (xargs :guard (svex-select-staticp x)))
 (let
  ((__function__ 'svex-select-to-lhs))
  (declare (ignorable __function__))
  (svex-select-case
   x
   :var
   (if
     (eql x.width 0)
     nil
     (list (make-lhrange :atom (make-lhatom-var :name x.name :rsh 0)
                         :w x.width)))
   :part (b* ((inner-lhs (svex-select-to-lhs x.subexp))
              (rsh (2vec->val (svex-quote->val x.lsb))))
             (lhs-rsh rsh
                      (lhs-concat (+ rsh x.width)
                                  inner-lhs nil))))))
