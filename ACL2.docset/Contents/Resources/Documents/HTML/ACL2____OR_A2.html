Function: binary-or*

(defun binary-or* (a b)
       (declare (xargs :guard t))
       (or a b))

Function: or*-macro

(defun or*-macro (lst)
       (if (atom lst)
           nil
           (if (atom (cdr lst))
               (car lst)
               (list 'binary-or*
                     (car lst)
                     (or*-macro (cdr lst))))))

Theorem: iff-implies-iff-or*-1

(defthm iff-implies-iff-or*-1
        (implies (iff a a-equiv)
                 (iff (or* a b) (or* a-equiv b)))
        :rule-classes (:congruence))
