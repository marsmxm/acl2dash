Theorem: return-type-of-all-lambdas.final-ans

(defthm return-type-of-all-lambdas.final-ans
        (implies (and (pseudo-termp term)
                      (pseudo-lambda-listp ans))
                 (b* ((?final-ans (all-lambdas term ans)))
                     (pseudo-lambda-listp final-ans)))
        :rule-classes :rewrite)
