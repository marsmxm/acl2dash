Prints out the elements of a vl-warninglist-p without the enclosing brackets.

Arguments
x — Guard (vl-warninglist-p x).

Definitions and Theorems

Function: vl-jp-warninglist-aux-fn

(defun
    vl-jp-warninglist-aux-fn (x no-html ps)
    (declare (xargs :stobjs (ps)))
    (declare (xargs :guard (vl-warninglist-p x)))
    (let ((__function__ 'vl-jp-warninglist-aux))
         (declare (ignorable __function__))
         (if (atom x)
             ps
             (vl-ps-seq (vl-jp-warning (car x) :no-html no-html)
                        (if (atom (cdr x))
                            ps
                            (vl-ps-seq (vl-println ",")
                                       (vl-indent 4)))
                        (vl-jp-warninglist-aux (cdr x)
                                               :no-html no-html)))))
