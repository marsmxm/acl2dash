Function: 4v-nsexpr-alist-p

(defun 4v-nsexpr-alist-p (x)
       "Alist whose values are sexprs with natp variables."
       (declare (xargs :guard t))
       (or (atom x)
           (and (consp (car x))
                (4v-nsexpr-p (cdar x))
                (4v-nsexpr-alist-p (cdr x)))))
