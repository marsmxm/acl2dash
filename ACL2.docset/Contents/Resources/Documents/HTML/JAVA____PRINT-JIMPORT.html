Function: print-jimport

(defun print-jimport (import indent-level)
       (declare (xargs :guard (and (jimportp import)
                                   (natp indent-level))))
       (let ((__function__ 'print-jimport))
            (declare (ignorable __function__))
            (print-jline (msg "import ~s0~@1;"
                              (if (jimport->static? import)
                                  "static " "")
                              (jimport->target import))
                         indent-level)))
