(defun
 satlink-handle-lines
 (lines saw-unsat-p saw-sat-p saw-zero-p env$)
 (declare (xargs :stobjs (env$)))
 (declare (xargs :guard (string-listp lines)))
 (let
  ((__function__ 'satlink-handle-lines))
  (declare (ignorable __function__))
  (b* (((when (atom lines))
        (mv nil
            saw-unsat-p saw-sat-p saw-zero-p env$))
       ((mv error-p
            saw-unsat-p saw-sat-p saw-zero-p env$)
        (satlink-handle-line (car lines)
                             saw-unsat-p saw-sat-p saw-zero-p env$))
       ((when error-p)
        (mv error-p
            saw-unsat-p saw-sat-p saw-zero-p env$)))
      (satlink-handle-lines (cdr lines)
                            saw-unsat-p
                            saw-sat-p saw-zero-p env$))))
