Macro: patbind-symlet

(defmacro
 patbind-symlet (args forms rest-expr)
 (cond
  ((and (eql (len args) 1)
        (eql (len forms) 1))
   (cons 'let-syms
         (cons (cons (cons (car args)
                           (cons (car forms) 'nil))
                     'nil)
               (cons rest-expr 'nil))))
  ((and (doublet-listp args) (not forms))
   (cons 'let-syms
         (cons args (cons rest-expr 'nil))))
  (t
   (er
    hard? 'symlet
    "Symlet takes either exactly 1 argument and 1 ~
                            form, or a doublet-list as arguments and no form"))))
