Definitions and Theorems

Function: delete*

(defun delete* (keys map)
       (declare (xargs :guard (and (setp keys) (mapp map))))
       (let ((__function__ 'delete*))
            (declare (ignorable __function__))
            (cond ((set::empty keys) (mfix map))
                  (t (delete (set::head keys)
                             (delete* (set::tail keys) map))))))

Theorem: mapp-of-delete*

(defthm mapp-of-delete*

Theorem: delete*-when-left-empty

(defthm delete*-when-left-empty
        (implies (set::empty keys)
                 (equal (delete* keys map) (mfix map))))

Theorem: delete*-when-right-empty

(defthm delete*-when-right-empty
        (implies (empty map)
                 (equal (delete* keys map) nil)))
