Function: fundef-disabledp

(defun fundef-disabledp (fn state)
       (declare (xargs :stobjs (state)))
       (declare (xargs :guard (symbolp fn)))
       (let ((__function__ 'fundef-disabledp))
            (declare (ignorable __function__))
            (if (member-equal (cons ':definition (cons fn 'nil))
                              (disabledp fn))
                t nil)))
