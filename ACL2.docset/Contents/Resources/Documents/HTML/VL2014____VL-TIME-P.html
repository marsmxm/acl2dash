Function: vl-time-p

(defun vl-time-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-time-p))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :vl-time)
                 (std::prod-consp (cdr x))
                 (b* ((quantity (std::prod-car (cdr x)))
                      (units (std::prod-cdr (cdr x))))
                     (and (stringp quantity)
                          (vl-timeunit-p units))))))
