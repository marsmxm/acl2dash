Arguments

Function: element-called-rules

(defun element-called-rules (element)
       (declare (xargs :guard (elementp element)))
       (element-case element
                     :rulename (insert element.get nil)
                     :group (alternation-called-rules element.get)
                     :option (alternation-called-rules element.get)
                     :char-val nil
                     :num-val nil
                     :prose-val nil))
