Function: vl-descriptionlist-inject-warnings

(defun vl-descriptionlist-inject-warnings
       (x warnings)
       (declare (xargs :guard (and (vl-descriptionlist-p x)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-descriptionlist-inject-warnings))
            (declare (ignorable __function__))
            (b* ((x (vl-descriptionlist-fix x))
                 (warnings (vl-warninglist-fix warnings))
                 ((when (atom x)) (mv nil warnings))
                 ((when (atom warnings)) (mv x nil))
                 ((mv first warnings)
                  (vl-description-inject-warnings (car x)
                                                  warnings))
                 ((mv rest warnings)
                  (vl-descriptionlist-inject-warnings (cdr x)
                                                      warnings)))
                (mv (cons first rest) warnings))))
