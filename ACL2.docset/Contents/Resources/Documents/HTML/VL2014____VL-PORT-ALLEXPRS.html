Function: vl-port-allexprs

(defun
 vl-port-allexprs (x)
 (declare (xargs :guard (vl-port-p x)))
 (let
  ((__function__ 'vl-port-allexprs))
  (declare (ignorable __function__))
  (mbe
   :logic
   (b*
    ((x (vl-port-fix x)))
    (b*
     ((x (vl-port-fix x)))
     (if
       (eq (tag x) :vl-interfaceport)
       (vl-packeddimensionlist-allexprs (vl-interfaceport->udims x))
       (vl-maybe-expr-allexprs (vl-regularport->expr x)))))
   :exec (with-local-nrev (vl-port-allexprs-nrev x nrev)))))
