Function: bfr-mod-ss

(defun
 bfr-mod-ss (a b)
 (declare (xargs :guard (and (true-listp a) (true-listp b))))
 (let ((__function__ 'bfr-mod-ss))
      (declare (ignorable __function__))
      (bfr-ite-bss
           (bfr-=-ss b nil)
           (llist-fix a)
           (bfr-logext-ns
                (integer-length-bound-s b)
