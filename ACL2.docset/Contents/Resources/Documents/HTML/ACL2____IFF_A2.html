Theorem: iff*-is-an-equivalence

(defthm iff*-is-an-equivalence
        (and (booleanp (iff* x y))
             (iff* x x)
             (implies (iff* x y) (iff* y x))
             (implies (and (iff* x y) (iff* y z))
                      (iff* x z)))
        :rule-classes (:equivalence))

Theorem: iff*-refines-iff

(defthm iff*-refines-iff

Theorem: iff*-with-nil

(defthm iff*-with-nil
        (and (equal (iff* x nil) (not x))
             (equal (iff* nil x) (not x))))

Theorem: iff*-when-other

(defthm iff*-when-other
        (implies y
                 (and (iff (iff* x y) x)
                      (iff (iff* y x) x))))
