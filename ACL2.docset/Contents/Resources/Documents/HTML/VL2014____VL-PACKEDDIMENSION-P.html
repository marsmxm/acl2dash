Function: vl-packeddimension-p

(defun vl-packeddimension-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-packeddimension-p))
            (declare (ignorable __function__))
            (or (eq x :vl-unsized-dimension)
                (vl-range-p x))))
