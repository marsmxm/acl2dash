Function: vl-bind-p

(defun
 vl-bind-p (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-bind-p))
  (declare (ignorable __function__))
  (and
   (consp x)
   (eq (car x) :vl-bind)
   (std::prod-consp (cdr x))
   (std::prod-consp (std::prod-car (cdr x)))
   (std::prod-consp (std::prod-cdr (cdr x)))
   (std::prod-consp (std::prod-cdr (std::prod-cdr (cdr x))))
   (b*
    ((scope (std::prod-car (std::prod-car (cdr x))))
     (addto (std::prod-cdr (std::prod-car (cdr x))))
     (modinsts (std::prod-car (std::prod-cdr (cdr x))))
     (loc (std::prod-car (std::prod-cdr (std::prod-cdr (cdr x)))))
     (atts (std::prod-cdr (std::prod-cdr (std::prod-cdr (cdr x))))))
    (and (maybe-stringp scope)
         (vl-exprlist-p addto)
         (vl-modinstlist-p modinsts)
         (vl-location-p loc)
         (vl-atts-p atts))))))
