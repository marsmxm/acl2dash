Function: bin-digitp

(defun bin-digitp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'bin-digitp))
            (declare (ignorable __function__))
            (or (eql x (char-code #\0))
                (eql x (char-code #\1)))))
