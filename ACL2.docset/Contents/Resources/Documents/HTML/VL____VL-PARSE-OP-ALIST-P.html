(defun vl-parse-op-alist-p (arity x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-parse-op-alist-p))
            (declare (ignorable __function__))
            (if (atom x)
                (not x)
                (and (consp (car x))
                     (vl-tokentype-p (caar x))
                     (or (and (eql arity 2)
                              (vl-binaryop-p (cdar x)))

                              (vl-unaryop-p (cdar x))))
