Function: lhs-rsh

(defun
 lhs-rsh (sh x)
 (declare (xargs :guard (and (natp sh) (lhs-p x))))
 (let
  ((__function__ 'lhs-rsh))
  (declare (ignorable __function__))
  (b*
     (((when (zp sh)) (lhs-fix x))
      ((when (atom x)) nil)
      ((lhrange xf) (car x))
      ((when (<= xf.w sh))
       (lhs-rsh (- sh xf.w) (cdr x)))
      (newatom
           (lhatom-case
                xf.atom
                :z (lhatom-z)
                :var (lhatom-var xf.atom.name (+ xf.atom.rsh sh)))))
     (lhs-cons (lhrange (- xf.w sh) newatom)
               (cdr x)))))
