(defun
 vl-interface-resolve-ansi-portdecls
 (x ss)
 (declare (xargs :guard (and (vl-interface-p x)
                             (vl-scopestack-p ss))))
 (let
  ((__function__ 'vl-interface-resolve-ansi-portdecls))
  (declare (ignorable __function__))
  (b*
   (((vl-interface x) (vl-interface-fix x))
    ((unless x.parse-temps) x)
    ((vl-parse-temps x.temps) x.parse-temps)
    ((unless x.temps.ansi-p) x)
    ((mv warnings ports portdecls ?vardecls)
     (vl-resolve-ansi-portdecls
          x.temps.ansi-ports
          nil nil nil (vl-scopestack-push x ss))))

     x

     :ports ports
     :portdecls portdecls
     :vardecls x.vardecls
     :parse-temps
     (change-vl-parse-temps
          x.temps
          :loaditems (append-without-guard
                          (vl-modelementlist->genelements portdecls)
                          x.temps.loaditems))))))
