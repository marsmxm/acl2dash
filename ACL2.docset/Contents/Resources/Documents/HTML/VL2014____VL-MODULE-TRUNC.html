Function: vl-module-trunc

(defun
 vl-module-trunc (x)
 (declare (xargs :guard (vl-module-p x)))
 (let
   ((__function__ 'vl-module-trunc))
   (declare (ignorable __function__))
   (b* ((x (vl-module-fix x))
        ((vl-module x) x)
        ((when (vl-module->hands-offp x)) x)
        ((when (vl-assignlist-can-skip-trunc-p x.assigns))
         x)
        (delta (vl-starting-delta x))
        (delta (change-vl-delta delta
                                :vardecls x.vardecls))
        ((mv assigns delta)
         (vl-assignlist-trunc x.assigns delta))
        ((vl-delta delta)
         (vl-free-delta delta)))
       (change-vl-module
            x
            :vardecls delta.vardecls
            :assigns (revappend-without-guard delta.assigns assigns)
            :warnings delta.warnings))))
