(defun
 atc-exec-binary-rules-gen-op-ltype-rtype
 (op ltype rtype)
 (declare (xargs :guard (and (binopp op)
                             (typep ltype)
                             (typep rtype))))
 (declare (xargs :guard (and (type-integerp ltype)
                             (type-integerp rtype))))
 (let
  ((__function__ 'atc-exec-binary-rules-gen-op-ltype-rtype))
  (declare (ignorable __function__))
  (b*
   ((lfixtype (atc-integer-type-fixtype ltype))
    (rfixtype (atc-integer-type-fixtype rtype))
    (rpred (pack rfixtype 'p))
    (op-kind (binop-kind op))
    (exec-op (pack 'exec- op-kind))
    (exec-binary-strict-pure-of-op-and-ltype
         (pack 'exec-binary-strict-pure-of-
               op-kind '-and-
               lfixtype))
    (type (uaconvert-types ltype rtype))
    (name (pack exec-binary-strict-pure-of-op-and-ltype
                '-when-
                rfixtype))
    (op-ltype-rtype (pack op-kind '- lfixtype '- rfixtype))
    (op-ltype-rtype-okp
         (and (or (member-eq op-kind '(:div :rem :shl :shr))
                  (and (member-eq op-kind '(:add :sub :mul))
                       (type-signed-integerp type)))
              (pack op-ltype-rtype '-okp)))
    (formula
     (cons
      'implies
      (cons
       (cons 'and
             (cons (atc-syntaxp-hyp-for-expr-pure 'y)
                   (cons (cons rpred (cons 'y 'nil))
                         (and op-ltype-rtype-okp
                              (cons (cons op-ltype-rtype-okp
                                          (cons 'x (cons 'y 'nil)))
                                    'nil)))))
       (cons
        (cons
          'equal
          (cons (cons exec-binary-strict-pure-of-op-and-ltype
                      (cons 'x (cons 'y 'nil)))
                (cons (cons op-ltype-rtype (cons 'x (cons 'y 'nil)))
                      'nil)))
        'nil))))
    (event
     (cons
      'defruled
      (cons
       name
       (cons
        formula
        (cons
         ':enable
         (cons
          (cons
           exec-binary-strict-pure-of-op-and-ltype
           (cons
            exec-op
            (append
             (and (or (not (equal type ltype))
                      (not (equal type rtype))
                      (member-eq op-kind '(:shl :shr)))
                  (list op-ltype-rtype))
             (append
               (and op-ltype-rtype-okp
                    (or (not (equal type ltype))
                        (not (equal type rtype))
                        (member-eq op-kind '(:shl :shr)))
                    (list op-ltype-rtype-okp))
               (append
                    (and (member-eq op-kind '(:shl :shr))
                         (not (equal ltype (promote-type ltype)))
                         (list (pack op-kind '- lfixtype)
                               (pack op-kind '- lfixtype '-okp)))
                    (append (and (member-eq op-kind '(:shl :shr))
                                 (cons 'exec-integer
                                       *atc-integer-value-rules*))
                            (append *atc-uaconvert-values-rules*
                                    *atc-promote-value-rules*)))))))
          'nil)))))))
   (mv name event))))
