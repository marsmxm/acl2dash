(defmacro patbind-captures!
          (args acl2::forms acl2::rest-expr)
          (b* ((args args)
               (forms acl2::forms)
               ((unless (atom (cdr forms)))
                (er hard? 'captures
                    "Too many forms: ~x0" forms))
               (bindings (captures-bindings args 1 (car forms)
                                            t)))
              (cons 'b*
                    (cons bindings (cons acl2::rest-expr 'nil)))))
