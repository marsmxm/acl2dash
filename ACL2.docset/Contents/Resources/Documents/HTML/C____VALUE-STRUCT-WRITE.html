Write a member of a structure.

Signature
(value-struct-write name val struct) → new-struct
Arguments
name — Guard (identp name).
val — Guard (valuep val).
struct — Guard (valuep struct).
Returns
new-struct — Type (value-resultp new-struct).

We look up the members in order; given that the members have distinct names (see value), the search order is immaterial. The new value must have the same type as the old value.

Function: value-struct-write-aux

(defun
     value-struct-write-aux
     (name val members)
     (declare (xargs :guard (and (identp name)
                                 (valuep val)
                                 (member-value-listp members))))
     (let ((__function__ 'value-struct-write-aux))
          (declare (ignorable __function__))
          (b* (((when (endp members))
                (error (list :member-not-found (ident-fix name))))
               ((member-value member) (car members))
               ((when (equal member.name (ident-fix name)))
                (if (equal (type-of-value member.value)
                           (type-of-value val))
                    (cons (make-member-value :name name
                                             :value val)
                          (member-value-list-fix (cdr members)))
                    (error (list :mistype-member (ident-fix name)
                                 :old-value member.value
                                 :new-value (value-fix val)))))
               (new-cdr-members
                    (value-struct-write-aux name val (cdr members)))
               ((when (errorp new-cdr-members))
                new-cdr-members))
              (cons (member-value-fix (car members))
                    new-cdr-members))))

Function: value-struct-write

(defun
 value-struct-write (name val struct)
 (declare (xargs :guard (and (identp name)
                             (valuep val)
                             (valuep struct))))
 (declare (xargs :guard (value-case struct :struct)))
 (let
  ((__function__ 'value-struct-write))
  (declare (ignorable __function__))
  (b*
   ((new-members
        (value-struct-write-aux name
                                val (value-struct->members struct)))
    ((when (errorp new-members))
     new-members))
   (change-value-struct struct
                        :members new-members))))
