    defmapping-gen-thms
    (doma$ domb$ alpha$ beta$ a1...an b1...bm
           beta-of-alpha-thm$ alpha-of-beta-thm$

                    alpha-of-beta-thm$ unconditional$
                    thm-names$ appcond-thm-names wrld)))
