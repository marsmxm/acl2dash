Function: vl-simplenet-p

(defun vl-simplenet-p (x)
       (declare (xargs :guard (vl-vardecl-p x)))
       (let ((__function__ 'vl-simplenet-p))
            (declare (ignorable __function__))
            (b* (((vl-vardecl x) x)
                 ((unless x.nettype) nil)
                 ((unless (eq (vl-datatype-kind x.type)
                              :vl-coretype))
                  nil)
                 ((vl-coretype x.type)))
                (and (eq x.type.name :vl-logic)
                     (atom x.type.udims)
                     (or (atom x.type.pdims)
                         (and (not (eq (car x.type.pdims)
                                       :vl-unsized-dimension))
                              (atom (cdr x.type.pdims))))))))
