Function: vl-typedef-strip

(defun vl-typedef-strip (x)
       (declare (xargs :guard (vl-typedef-p x)))
       (let ((__function__ 'vl-typedef-strip))
            (declare (ignorable __function__))
            (b* (((vl-typedef x) (vl-typedef-fix x)))
                (b* ((type (vl-datatype-strip x.type))
                     (atts ((lambda (x) (declare (ignore x)) nil)
                            x.atts)))
