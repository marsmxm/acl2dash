(defun
   defflatsum-theorems
   (summand-predicates sum-predicate)
   (declare (xargs :guard (and (symbol-listp summand-predicates)
                               (symbolp sum-predicate))))
   (let ((__function__ 'defflatsum-theorems))
        (declare (ignorable __function__))
        (cond ((endp summand-predicates) nil)
              (t (cons (defflatsum-theorem (car summand-predicates)
                                           sum-predicate)
                       (defflatsum-theorems (cdr summand-predicates)
                                            sum-predicate))))))
