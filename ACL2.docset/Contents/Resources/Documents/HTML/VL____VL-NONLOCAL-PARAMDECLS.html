Function: vl-nonlocal-paramdecls

(defun vl-nonlocal-paramdecls (x)
       (declare (xargs :guard (vl-paramdecllist-p x)))
       (let ((__function__ 'vl-nonlocal-paramdecls))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((vl-paramdecl->localp (car x))
                   (vl-nonlocal-paramdecls (cdr x)))
                  (t (cons (vl-paramdecl-fix (car x))
                           (vl-nonlocal-paramdecls (cdr x)))))))
