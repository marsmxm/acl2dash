Function: svtv-compile-lazy

(defun
 svtv-compile-lazy
 (nphases ins overrides outs prev-state updates
          state-updates in-vars state-machine)
 (declare (xargs :guard (and (posp nphases)
                             (svtv-lines-p ins)
                             (svtv-overridelines-p overrides)
                             (svtv-lines-p outs)
                             (svex-alist-p prev-state)
                             (svex-alist-p updates)
                             (svex-alist-p state-updates)
                             (svarlist-p in-vars))))
 (let
  ((__function__ 'svtv-compile-lazy))
  (declare (ignorable __function__))
  (b* (((with-fast prev-state updates state-updates))
       (in-alists
            (svtv-allphases-inputs 0 nphases ins overrides in-vars))
       (data (make-svtv-composedata :nextstates state-updates
                                    :input-substs in-alists
                                    :initst prev-state))
       ((mv outalist final-state)
        (svtv-compile-phases-lazy 0 nphases
                                  outs updates data state-machine)))
      (fast-alist-free-list in-alists)
      (clear-memoize-table 'svex-compose)
      (clear-memoize-table 'svex-compose-svtv-phases-call)
      (mv outalist final-state))))
