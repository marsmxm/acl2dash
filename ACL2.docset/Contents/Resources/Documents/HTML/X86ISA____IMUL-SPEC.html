Function: imul-spec$inline

(defun
   imul-spec$inline (size dst src)
   (declare (type (member 1 2 4 8) size))
   (declare (xargs :guard (case size (1 (and (n08p src) (n08p dst)))
                                (2 (and (n16p src) (n16p dst)))
                                (4 (and (n32p src) (n32p dst)))
                                (8 (and (n64p src) (n64p dst)))
                                (otherwise nil))))
   (case size (1 (imul-spec-8 dst src))
         (2 (imul-spec-16 dst src))
         (4 (imul-spec-32 dst src))
         (8 (imul-spec-64 dst src))
         (otherwise (mv 0 0 0 0))))

  (:type-prescription

     :hints
     (("Goal" :in-theory '(unsigned-byte-p integer-range-p natp))))
