Function: div-sint-sint-okp

(defun div-sint-sint-okp (x y)
       (declare (xargs :guard (and (sintp x) (sintp y))))
       (and (not (equal (sint->get y) 0))
            (sint-integerp (truncate (sint->get x)
                                     (sint->get y)))))
