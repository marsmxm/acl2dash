Function: vl-parse-bind-directive-scoped-fn

(defun
 vl-parse-bind-directive-scoped-fn
 (atts tokstream config)
 (declare (xargs :stobjs (tokstream)))
 (declare (xargs :guard (vl-loadconfig-p config)))
 (declare (ignorable config))
 (declare (xargs :guard (vl-atts-p atts)))
 (let
  ((__function__ 'vl-parse-bind-directive-scoped))
  (declare (ignorable __function__))
  (seq
     tokstream
     (:= (vl-match-token :vl-kwd-bind))
     (target := (vl-match-token :vl-idtoken))
     (when (vl-is-token? :vl-colon)
           (:= (vl-match))
           (addto := (vl-parse-1+-expressions-separated-by-commas)))
     (modinsts := (vl-parse-module-instantiation nil))
     (return (make-vl-bind :scope (vl-idtoken->name target)
                           :addto addto
                           :modinsts modinsts
                           :loc (vl-token->loc target)
                           :atts atts)))))
