Function: svtv-debug-eval-aliases

(defun
 svtv-debug-eval-aliases
 (n aliases wirevals vcd-vals)
 (declare (xargs :stobjs (aliases vcd-vals)))
 (declare (xargs :guard (and (natp n) (svex-env-p wirevals))))
 (declare (xargs :guard (and (<= n (aliass-length aliases))
                             (<= (aliass-length aliases)
                                 (4vecs-length vcd-vals)))))
 (let
   ((__function__ 'svtv-debug-eval-aliases))
   (declare (ignorable __function__))
   (b* (((when (mbe :logic (zp (- (aliass-length aliases) (nfix n)))
                    :exec (eql (aliass-length aliases) n)))
         vcd-vals)
        (lhs (get-alias n aliases))
        (val (svtv-debug-lhs-eval lhs n wirevals vcd-vals))
        (vcd-vals (set-4vec n val vcd-vals)))
       (svtv-debug-eval-aliases (1+ (lnfix n))
                                aliases wirevals vcd-vals))))
