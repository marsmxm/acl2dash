Theorem: bed-eval-of-bed-from-aig

(defthm bed-eval-of-bed-from-aig
        (b* (((mv bed ?order) (bed-from-aig aig)))
            (equal (bed-eval bed env)
                   (bool->bit (aig-eval aig env)))))
