Function: vl-rhsatom-lucidcheck

(defun
 vl-rhsatom-lucidcheck (x ss st ctx)
 (declare (xargs :guard (and (vl-expr-p x)
                             (vl-scopestack-p ss)
                             (vl-lucidstate-p st)
                             (vl-context1-p ctx))))
 (declare (xargs :guard (vl-atom-p x)))
 (let
  ((__function__ 'vl-rhsatom-lucidcheck))
  (declare (ignorable __function__))
  (b*
   ((x (vl-expr-fix x))
    (guts (vl-atom->guts x))
    (st (vl-lucidstate-fix st)))
   (case
    (tag guts)
    (:vl-funname (vl-lucid-mark-simple :used (vl-funname->name guts)
                                       ss st ctx))
    (:vl-typename
         (vl-lucid-mark-simple :used (vl-typename->name guts)
                               ss st ctx))
    (:vl-tagname st)
    ((:vl-id :vl-hidpiece)
     (progn$
      (raise
       "Should never get here because we should check for hidexpr-p first.")
      st))
    ((:vl-constint :vl-weirdint :vl-extint
                   :vl-string :vl-real
                   :vl-keyguts :vl-time
                   :vl-basictype :vl-sysfunname)
     st)
    (otherwise (progn$ (impossible) st))))))
