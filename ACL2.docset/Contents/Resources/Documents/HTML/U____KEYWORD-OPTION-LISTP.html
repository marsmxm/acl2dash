Function: keyword-option-listp

(defun keyword-option-listp (l)
       (declare (xargs :guard t))
       (cond ((atom l) (null l))
             (t (and ((lambda (x)
                              (if (atom x)
                                  (keywordp x)
                                  (and (true-listp x)
                                       (keywordp (car x)))))
                      (car l))
                     (keyword-option-listp (cdr l))))))
