Function: errorp

(defun
    errorp (x)
    (declare (xargs :guard t))
    (let ((__function__ 'errorp))
         (declare (ignorable __function__))
         (and (consp x)
              (eq (car x) :error)
              (mbe :logic (and (alistp (cdr x))
                               (equal (strip-cars (cdr x)) '(info)))
                   :exec (fty::alist-with-carsp (cdr x) '(info)))
              (b* ((info (cdr (std::da-nth 0 (cdr x)))))
                  (acl2::any-p info)))))
