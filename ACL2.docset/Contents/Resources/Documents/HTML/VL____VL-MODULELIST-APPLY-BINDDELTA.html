Function: vl-modulelist-apply-binddelta

(defun
 vl-modulelist-apply-binddelta (x delta)
 (declare (xargs :guard (and (vl-modulelist-p x)
                             (vl-binddelta-p delta))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-modulelist-apply-binddelta))
  (declare (ignorable __function__))
  (mbe
     :logic (if (consp x)
                (cons (vl-module-apply-binddelta (car x)
                                                 delta)
                      (vl-modulelist-apply-binddelta (cdr x)
                                                     delta))
                nil)
     :exec
     (if (atom x)
         nil
         (with-local-nrev
              (vl-modulelist-apply-binddelta-nrev x delta nrev))))))
