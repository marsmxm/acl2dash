(defun
 aliases-finish-canonicalize (n aliases)
 (declare (xargs :stobjs (aliases)))
 (declare (xargs :guard (and (natp n)
                             (aliases-normorderedp aliases))))
 (declare (xargs :guard (<= n (aliass-length aliases))))
 (let
   ((__function__ 'aliases-finish-canonicalize))
   (declare (ignorable __function__))
   (b* (((when (mbe :logic (zp (- (aliass-length aliases) (nfix n)))
