Function: vl-namedarg-p

(defun
 vl-namedarg-p (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-namedarg-p))
  (declare (ignorable __function__))
  (and
   (mbe
       :logic (and (alistp x)
                   (equal (strip-cars x)
                          '(name expr nameonly-p atts)))
       :exec (fty::alist-with-carsp x '(name expr nameonly-p atts)))
   (b* ((name (cdr (std::da-nth 0 x)))
        (expr (cdr (std::da-nth 1 x)))
        (nameonly-p (cdr (std::da-nth 2 x)))
        (atts (cdr (std::da-nth 3 x))))
       (and (stringp name)
            (vl-maybe-expr-p expr)
            (booleanp nameonly-p)
            (vl-atts-p atts))))))
