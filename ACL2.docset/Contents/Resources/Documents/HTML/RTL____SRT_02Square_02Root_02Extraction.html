Definitions and Theorems

Function: quot%

(defun quot% (j)
       (if (zp j)
           1
           (+ (quot% (1- j))
              (* (q% j) (expt (r%) (- j))))))

Function: rem%

(defun rem% (j)
       (* (expt (r%) j)
          (- (x%) (* (quot% j) (quot% j)))))

Theorem: int-quot-sqrt

(defthm int-quot-sqrt
        (implies (natp j)
                 (integerp (* (expt (r%) j) (quot% j)))))

Theorem: rem0-sqrt-rewrite

(defthm rem0-sqrt-rewrite
        (equal (rem% 0) (1- (x%))))

Theorem: rem-sqrt-recurrence

(defthm rem-sqrt-recurrence
        (implies (natp j)
                 (equal (rem% (+ 1 j))
                        (- (* (r%) (rem% j))
                           (* (q% (1+ j))
                              (+ (* 2 (quot% j))
                                 (* (expt (r%) (- (1+ j)))
                                    (q% (1+ j)))))))))

Function: blo%

(defun blo% (j)

          (* (rho%) (rho%) (expt (r%) (- j)))))

Function: bhi%

(defun bhi% (j)
       (+ (* 2 (rho%) (quot% j))
          (* (rho%) (rho%) (expt (r%) (- j)))))

Theorem: blohi

(defthm
     blohi
     (implies (natp j)
              (iff (and (<= (expt (- (quot% j)
                                     (* (rho%) (expt (r%) (- j))))
                                  2)
                            (x%))
                        (>= (expt (+ (quot% j)
                                     (* (rho%) (expt (r%) (- j))))
                                  2)
                            (x%)))
                   (and (<= (blo% j) (rem% j))
                        (>= (bhi% j) (rem% j)))))
     :rule-classes nil)

Theorem: rem0-sqrt-bounds

(defthm rem0-sqrt-bounds
        (and (<= (blo% 0) (rem% 0))
             (>= (bhi% 0) (rem% 0))))

Function: sel-upper-sqrt

(defun sel-upper-sqrt (k j)
       (+ (* 2 (+ k (rho%)) (quot% j))
          (* (+ k (rho%))
             (+ k (rho%))
             (expt (r%) (- (1+ j))))))

Function: sel-lower-sqrt

(defun sel-lower-sqrt (k j)
       (+ (* 2 (- k (rho%)) (quot% j))
          (* (- k (rho%))
             (- k (rho%))
             (expt (r%) (- (1+ j))))))

Theorem: rem-sqrt-bnds-next

(defthm rem-sqrt-bnds-next
        (implies (and (natp j)
                      (<= (sel-lower-sqrt (q% (1+ j)) j)
                          (* (r%) (rem% j)))
                      (>= (sel-upper-sqrt (q% (1+ j)) j)
                          (* (r%) (rem% j))))
                 (and (<= (blo% (1+ j)) (rem% (1+ j)))
                      (>= (bhi% (1+ j)) (rem% (1+ j)))))
        :rule-classes nil)

Theorem: sqrt-containment

(defthm sqrt-containment
        (implies (natp j)
                 (and (equal (sel-upper-sqrt (a%) j)
                             (* (r%) (bhi% j)))
                      (equal (sel-lower-sqrt (- (a%)) j)
                             (* (r%) (blo% j))))))

Function: ms4*8

(defun ms4*8 (i j k)
       (case i
             (0 (case k (2 12)

Function: ms4

(defun ms4 (i j k) (/ (ms4*8 i j k) 8))

Function: i%

(defun i% (j)
       (* 16 (- (quot% (min (nfix j) 2)) 1/2)))

Function: select-digit-s4

(defun select-digit-s4 (a i j)
       (cond ((<= (ms4 i j 2) a) 2)
             ((<= (ms4 i j 1) a) 1)
             ((<= (ms4 i j 0) a) 0)
             ((<= (ms4 i j -1) a) -1)
             (t -2)))

Function: quot%-bnds-inv

(defun quot%-bnds-inv (j)
       (and (<= 1/2 (quot% j))
            (>= 1 (quot% j))))

Function: rem%-bnds-inv

(defun rem%-bnds-inv (j)
       (and (<= (blo% j) (rem% j))
            (>= (bhi% j) (rem% j))))

Function: approx%-bounds

(defun approx%-bounds (j k)
       (and (implies (< (approx% j) (ms4 (i% j) j k))
                     (< (* 4 (rem% j)) (ms4 (i% j) j k)))
            (implies (>= (approx% j) (ms4 (i% j) j k))
                     (> (* 4 (rem% j))
                        (- (ms4 (i% j) j k) 1/32)))))

Function: approx%-inv

(defun approx%-inv (j)
       (and (= (q% (1+ j))
               (select-digit-s4 (approx% j) (i% j) j))
            (approx%-bounds j 2)
            (approx%-bounds j 1)
            (approx%-bounds j 0)
            (approx%-bounds j -1)))

Function: s4-inv

(defun s4-inv (j)
       (and (quot%-bnds-inv j)
            (rem%-bnds-inv j)
            (approx%-inv j)))

Function: s4-hyp

(defun s4-hyp (j)
       (if (zp j)
           (s4-inv 0)
           (and (s4-inv j) (s4-hyp (1- j)))))

Theorem: srt-sqrt-rad-4

(defthm srt-sqrt-rad-4
        (implies (and (= (r%) 4)
                      (= (a%) 2)
                      (natp j)
                      (s4-hyp j))
                 (and (quot%-bnds-inv (1+ j))
                      (rem%-bnds-inv (1+ j)))))
