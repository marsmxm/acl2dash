Function: get-option-entry

(defun get-option-entry (option l)
       (declare (xargs :guard (and (keywordp option)
                                   (keyword-option-listp l))))
       (cond ((atom l) 'nil)
             (((lambda (x)
                       (or (eq option x)
                           (and (consp x) (eq option (car x)))))
               (car l))
              ((lambda (x) x) (car l)))
             (t (get-option-entry option (cdr l)))))
