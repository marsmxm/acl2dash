Function: identifier-start-p

(defun identifier-start-p (char)
       (declare (xargs :guard (unicodep char)))
       (let ((__function__ 'identifier-start-p))
            (declare (ignorable __function__))
            (b* ((char (mbe :logic (unicode-fix char)
                            :exec char)))
                (cond ((asciip char)
                       (ascii-identifier-start-p char))
                      (t (nonascii-identifier-start-p char))))))
