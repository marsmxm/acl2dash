Function: vl-alwayslist-stmtrewrite

(defun
     vl-alwayslist-stmtrewrite
     (x unroll-limit warnings)
     (declare (xargs :guard (and (vl-alwayslist-p x)
                                 (natp unroll-limit)
                                 (vl-warninglist-p warnings))))
     (let ((__function__ 'vl-alwayslist-stmtrewrite))
          (declare (ignorable __function__))
          (b* (((when (atom x)) (mv warnings nil))
               ((mv warnings car-prime)
                (vl-always-stmtrewrite (car x)
                                       unroll-limit warnings))
               ((mv warnings cdr-prime)
                (vl-alwayslist-stmtrewrite (cdr x)
                                           unroll-limit warnings)))
              (mv warnings
                  (if (vl-nullstmt-p (vl-always->stmt car-prime))
                      cdr-prime
                      (cons car-prime cdr-prime))))))
