Function: dab-digit-listp

(defun dab-digit-listp (base x)
       (declare (xargs :guard (dab-basep base)))
       (let ((__function__ 'dab-digit-listp))
            (declare (ignorable __function__))
            (if (consp x)
                (and (dab-digitp base (car x))
                     (dab-digit-listp base (cdr x)))
                (null x))))
