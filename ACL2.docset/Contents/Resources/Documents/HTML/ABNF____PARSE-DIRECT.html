Function: parse-direct

(defun parse-direct (nats input)
       (declare (xargs :guard (and (nat-listp nats)
                                   (nat-listp input))))
       (let ((__function__ 'parse-direct))
            (declare (ignorable __function__))
            (b* (((mv nats input)
                  (parse-direct-aux nats input))
                 ((when (resulterrp nats))
                  (mv (err-push nats) input)))
                (mv (tree-leafterm nats) input))))
