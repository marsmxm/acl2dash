Function: vl-interface-size

(defun
 vl-interface-size (x ss)
 (declare (xargs :guard (and (vl-interface-p x)
                             (vl-scopestack-p ss))))
 (let
  ((__function__ 'vl-interface-size))
  (declare (ignorable __function__))
  (b*
   ((x (vl-interface-fix x))
    ((mv err1 type)
     (vl-interface-mocktype x ss))
    ((unless (vl-datatype-resolved-p type))
     (mv (vmsg "Mocktype of interface ~a0 not resolved: ~a1"
               (vl-interface->name x)
               type)
         nil))
    ((mv err2 size)
     (vl-datatype-size type)))
   (mv
    (vmsg-concat
     err1
     (or
       err2
       (and (not size)
            (vmsg "Mocktype of interface ~a0 couldn't be sized: ~a1"
                  (vl-interface->name x)
                  type))))
    size))))
