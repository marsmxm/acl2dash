Function: double-valuep

(defun double-valuep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'double-valuep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :double)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((double (std::da-nth 0 (cdr x))))
                     (double-value-abs-p double)))))
