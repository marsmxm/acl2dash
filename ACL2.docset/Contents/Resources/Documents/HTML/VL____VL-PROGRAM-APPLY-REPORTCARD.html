Function: vl-program-apply-reportcard

(defun
  vl-program-apply-reportcard
  (x reportcard)
  (declare (xargs :guard (and (vl-program-p x)
                              (vl-reportcard-p reportcard))))
  (let ((__function__ 'vl-program-apply-reportcard))
       (declare (ignorable __function__))
       (b* (((vl-program x) x)
            (entry (hons-get x.name (vl-reportcard-fix reportcard)))
            ((unless entry) (vl-program-fix x)))
           (change-vl-program
                x
                :warnings (revappend-without-guard (cdr entry)
                                                   x.warnings)))))
