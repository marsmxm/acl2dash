Function: vl-edgetable-p

(defun vl-edgetable-p (x)
       (declare (xargs :guard t))
       (if (consp x)
           (and (consp (car x))
                (stringp (caar x))
                (vl-edgesynth-edge-p (cdar x))
                (vl-edgetable-p (cdr x)))
           t))
