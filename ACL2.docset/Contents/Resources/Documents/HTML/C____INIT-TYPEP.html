Function: init-typep

(defun init-typep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'init-typep))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x) (eq (car x) :single))
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((get (std::da-nth 0 (cdr x))))
                                 (typep get))))
                       (t (and (eq (car x) :list)
                               (and (true-listp (cdr x))
                                    (eql (len (cdr x)) 1))
                               (b* ((get (std::da-nth 0 (cdr x))))
                                   (type-listp get))))))))
