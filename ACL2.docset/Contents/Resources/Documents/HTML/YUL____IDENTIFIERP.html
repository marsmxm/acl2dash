Function: identifierp

(defun
     identifierp (x)
     (declare (xargs :guard t))
     (let ((__function__ 'identifierp))
          (declare (ignorable __function__))
          (and (consp x)
               (eq (car x) :identifier)
               (mbe :logic (and (alistp (cdr x))
