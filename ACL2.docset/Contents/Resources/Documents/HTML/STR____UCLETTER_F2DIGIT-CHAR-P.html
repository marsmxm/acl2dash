Function: ucletter/digit-char-p

(defun ucletter/digit-char-p (x)
       (declare (xargs :guard t))
       (and (characterp x)
            (b* ((code (char-code x)))
                (or (and (<= (char-code #\A) code)
                         (<= code (char-code #\Z)))
                    (and (<= (char-code #\0) code)
                         (<= code (char-code #\9)))))))
