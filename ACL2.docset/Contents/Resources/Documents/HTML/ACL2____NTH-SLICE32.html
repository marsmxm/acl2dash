Function: nth-slice32$inline

(defun
  acl2::nth-slice32$inline (n x)
  (declare (xargs :guard (and (natp n) (integerp x))))
  (let ((__function__ 'nth-slice32))
       (declare (ignorable __function__))
       (mbe :logic (logand (ash (ifix x) (* (nfix n) -32))
                           (1- (expt 2 32)))

                       (logand (ash x (the (integer * 0) (* n -32)))
                               4294967295)))))
