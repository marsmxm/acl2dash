Function: permute-polarity

(defun permute-polarity (n mask truth numvars)
       (declare (xargs :guard (and (natp n)
                                   (integerp mask)
                                   (integerp truth)
                                   (natp numvars))))
       (declare (xargs :guard (<= n numvars)))
       (let ((__function__ 'permute-polarity))
            (declare (ignorable __function__))
            (b* (((when (mbe :logic (zp (- (nfix numvars) (nfix n)))
                             :exec (eql n numvars)))
                  (truth-norm truth numvars))
                 (truth (if (logbitp n mask)
                            (swap-polarity n truth numvars)
                            truth)))
                (permute-polarity (1+ (lnfix n))
                                  mask truth numvars))))
