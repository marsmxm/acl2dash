Function: vl-assign-wildelim

(defun
  vl-assign-wildelim (x warnings)
  (declare (xargs :guard (and (vl-assign-p x)
                              (vl-warninglist-p warnings))))
  (let ((__function__ 'vl-assign-wildelim))
       (declare (ignorable __function__))
       (b* ((x (vl-assign-fix x))
            (warnings (vl-warninglist-fix warnings)))
           (b* (((vl-assign x) x)
                (ctx x)
                ((mv warnings lhs-prime)
                 (vl-expr-wildelim x.lvalue ctx warnings))
                ((mv warnings rhs-prime)
                 (vl-expr-wildelim x.expr ctx warnings))
                ((mv warnings delay-prime)
                 (vl-maybe-gatedelay-wildelim x.delay ctx warnings))
                (x-prime (change-vl-assign x
                                           :lvalue lhs-prime
                                           :expr rhs-prime
                                           :delay delay-prime)))
               (mv warnings x-prime)))))
