(defun
 moddb-address->wiredecl
 (addr scope moddb)
 (declare (xargs :stobjs (moddb)))
 (declare (xargs :guard (and (address-p addr)
                             (modscope-p scope)
                             (moddb-ok moddb))))
 (declare (xargs :guard (modscope-okp scope moddb)))
 (let
   ((__function__ 'moddb-address->wiredecl))
   (declare (ignorable __function__))
