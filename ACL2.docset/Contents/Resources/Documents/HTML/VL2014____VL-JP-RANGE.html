(defun vl-jp-range-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-range-p x)))
       (let ((__function__ 'vl-jp-range))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-range)
                       (vl-print ", ")
                       (vl-print-str "\"msb\": ")
                       (vl-jp-expr (vl-range->msb x))
                       (vl-println? ", ")
                       (vl-print-str "\"lsb\": ")
                       (vl-jp-expr (vl-range->lsb x))
                       (vl-println? "}"))))
