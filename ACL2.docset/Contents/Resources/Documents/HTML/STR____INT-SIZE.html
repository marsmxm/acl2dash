Function: int-size

(defun int-size (x print-base)
       (declare (xargs :guard (and (integerp x)
                                   (print-base-p print-base))))
       (let ((acl2::__function__ 'int-size))
            (declare (ignorable acl2::__function__))

                (+ 1 (nat-size (- x) print-base))
