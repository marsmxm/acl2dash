(defun vl-make-msb-to-lsb-bitselects
       (expr msb lsb)
       (declare (xargs :guard (and (vl-expr-p expr)
                                   (natp msb)
                                   (natp lsb))))
       (let ((__function__ 'vl-make-msb-to-lsb-bitselects))
            (declare (ignorable __function__))
            (b* ((msb (lnfix msb))
                 (lsb (lnfix lsb))
                 (left-right-p (>= msb lsb))
                 (low (if left-right-p lsb msb))
                 (high (if left-right-p msb lsb))
                 (list (vl-make-list-of-bitselects expr low high)))
                (if left-right-p (reverse list) list))))
