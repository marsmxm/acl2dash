Safely add a dependency onto a package. If the package doesn't exist, add a warning instead of a dependency.

Definitions and Theorems

Function: vl-immdeps-add-pkgdep-fn

(defun vl-immdeps-add-pkgdep-fn
       (pkgname ans ss ctx)
       (declare (xargs :guard (and (stringp pkgname)
                                   (vl-immdeps-p ans)
                                   (vl-scopestack-p ss)
                                   (any-p ctx))))
       (let ((__function__ 'vl-immdeps-add-pkgdep))
            (declare (ignorable __function__))
            (b* ((pkgname (string-fix pkgname))
                 (pkg (vl-scopestack-find-package pkgname ss))
                 ((unless pkg)
                  (vl-immdeps-add-error
                       ans
                       :type :vl-missing-package
                       :msg "~a0: reference to unknown package ~a1."
                       :args (list ctx pkgname)
                       :fatalp t))
                 (ans (vl-immdeps-add-raw-dependency pkgname ans)))
                ans)))
