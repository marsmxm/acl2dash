Function: vl-jp-import-fn

(defun vl-jp-import-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-import-p x)))
       (let ((__function__ 'vl-jp-import))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-import)
                       (vl-print ", ")
                       (vl-print-str "\"pkg\": ")
                       (jp-str (vl-import->pkg x))
                       (vl-println? ", ")
                       (vl-print-str "\"part\": ")
                       (vl-jp-importpart (vl-import->part x))
                       (vl-println? ", ")
                       (vl-print-str "\"loc\": ")
                       (vl-jp-location (vl-import->loc x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-import->atts x))
                       (vl-println? "}"))))
