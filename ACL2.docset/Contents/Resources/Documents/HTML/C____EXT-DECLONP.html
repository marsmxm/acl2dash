Function: ext-declonp

(defun ext-declonp (x)
       (declare (xargs :guard t))
       (let ((__function__ 'ext-declonp))
            (declare (ignorable __function__))
            (and (consp x)
                 (cond ((or (atom x) (eq (car x) :fundef))
                        (and (true-listp (cdr x))
                             (eql (len (cdr x)) 1)
                             (b* ((get (std::da-nth 0 (cdr x))))
                                 (fundefp get))))
                       (t (and (eq (car x) :declon)
                               (and (true-listp (cdr x))
                                    (eql (len (cdr x)) 1))
                               (b* ((get (std::da-nth 0 (cdr x))))
                                   (declonp get))))))))
