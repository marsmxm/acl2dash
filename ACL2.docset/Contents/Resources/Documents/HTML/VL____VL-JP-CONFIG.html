Definitions and Theorems

Function: vl-jp-config-fn

(defun vl-jp-config-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-config-p x)))
       (let ((__function__ 'vl-jp-config))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-config)
                       (vl-print ", ")
                       (vl-print-str "\"name\": ")
                       (jp-str (vl-config->name x))
                       (vl-println? ", ")
                       (vl-print-str "\"warnings\": ")
                       (vl-jp-warninglist (vl-config->warnings x))
                       (vl-println? ", ")
                       (vl-print-str "\"minloc\": ")
                       (vl-jp-location (vl-config->minloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"maxloc\": ")
                       (vl-jp-location (vl-config->maxloc x))
                       (vl-println? ", ")
                       (vl-print-str "\"atts\": ")
                       (vl-jp-atts (vl-config->atts x))
                       (vl-println? ", ")
                       (vl-print-str "\"comments\": ")
                       (vl-jp-commentmap (vl-config->comments x))
                       (vl-println? "}"))))
