Theorem: vl-save-and-restore-ppst-correct

(defthm vl-save-and-restore-ppst-correct
        (implies (vl-ppst-p ppst)
                 (equal (vl-restore-ppst (vl-save-ppst)
                                         :ppst (create-ppst))
                        ppst)))
