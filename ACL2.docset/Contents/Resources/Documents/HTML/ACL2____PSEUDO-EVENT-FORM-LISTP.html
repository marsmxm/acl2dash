Function: pseudo-event-form-listp

(defun pseudo-event-form-listp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (equal x nil)
           (and (pseudo-event-formp (car x))
                (pseudo-event-form-listp (cdr x)))))
