(defun
  atj-gen-shallow-fndef-methods
  (fn fns-that-may-throw fn-types
      formals body method-name qconsts
      pkg-class-names fn-method-names curr-pkg
      mv-typess guards$ no-aij-types$ wrld)

                              (symbol-listp fns-that-may-throw)
                              (atj-function-type-listp fn-types)
                              (symbol-listp formals)
                              (pseudo-termp body)
                              (stringp method-name)
                              (atj-qconstants-p qconsts)
                              (string-string-alistp pkg-class-names)
                              (symbol-string-alistp fn-method-names)
                              (stringp curr-pkg)
                              (atj-type-list-listp mv-typess)
                              (booleanp guards$)
                              (booleanp no-aij-types$)
                              (plist-worldp wrld))))
  (declare (xargs :guard (and (not (aij-nativep fn))
                              (not (equal curr-pkg ""))
                              (cons-listp mv-typess))))
  (let ((__function__ 'atj-gen-shallow-fndef-methods))
       (declare (ignorable __function__))
       (b* (((when (endp fn-types))
             (mv nil qconsts mv-typess))
            ((mv first-methods qconsts mv-typess)
             (atj-gen-shallow-fndef-method
                  fn fns-that-may-throw (car fn-types)
                  formals body method-name qconsts
                  pkg-class-names fn-method-names curr-pkg
                  mv-typess guards$ no-aij-types$ wrld))
            ((mv rest-methods qconsts mv-typess)
             (atj-gen-shallow-fndef-methods
                  fn fns-that-may-throw (cdr fn-types)
                  formals body method-name qconsts
                  pkg-class-names fn-method-names curr-pkg
                  mv-typess guards$ no-aij-types$ wrld)))
           (mv (cons first-methods rest-methods)
               qconsts mv-typess))))
