Function: sig-p

(defun
  sig-p (x)
  (declare (xargs :guard t))
  (let ((acl2::__function__ 'sig-p))
       (declare (ignorable acl2::__function__))
       (and (mbe :logic (and (alistp x)
                             (equal (strip-cars x) '(module index)))
                 :exec (fty::alist-with-carsp x '(module index)))
            (b* ((module (cdr (std::da-nth 0 x)))
                 (index (cdr (std::da-nth 1 x))))
                (and (symbolp module)
                     (integerp index))))))
