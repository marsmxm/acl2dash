Function: ident-setp

(defun ident-setp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (identp (car x))
                (or (null (cdr x))
                    (and (consp (cdr x))
                         (acl2::fast-<< (car x) (cadr x))
                         (ident-setp (cdr x)))))))
