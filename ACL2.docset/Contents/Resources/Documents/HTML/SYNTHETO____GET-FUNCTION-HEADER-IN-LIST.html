Function: get-function-header-in-list

(defun
 get-function-header-in-list
 (name headers)
 (declare (xargs :guard (and (identifierp name)
                             (function-header-listp headers))))
 (let
  ((__function__ 'get-function-header-in-list))
  (declare (ignorable __function__))
  (cond
     ((endp headers) nil)
     (t (if (identifier-equiv name
                              (function-header->name (car headers)))
            (function-header-fix (car headers))
            (get-function-header-in-list name (cdr headers)))))))
