Function: vl-paramdecl-expand-function-calls

(defun
 vl-paramdecl-expand-function-calls
 (x ss nf vardecls assigns warnings)
 (declare (xargs :guard (and (vl-paramdecl-p x)
                             (vl-scopestack-p ss)
                             (vl-namefactory-p nf)
                             (vl-vardecllist-p vardecls)
                             (vl-assignlist-p assigns)
                             (vl-warninglist-p warnings))))
 (let
  ((__function__ 'vl-paramdecl-expand-function-calls))
  (declare (ignorable __function__))
  (b*
   ((nf (vl-namefactory-fix nf))
    (assigns (vl-assignlist-fix assigns))
    (vardecls (vl-vardecllist-fix vardecls)))
   (b* ((x (vl-paramdecl-fix x))
        ((mv okp warnings)
         (vl-check-bad-funcalls x (vl-paramdecl-allexprs x)
                                "parameter declarations" warnings)))
       (mv okp warnings nf x vardecls assigns)))))
