(defun
 vl-enumbasetype-wildelim
 (x ctx warnings)
 (declare (xargs :guard (and (vl-enumbasetype-p x)
                             (vl-context-p ctx)
                             (vl-warninglist-p warnings))))
 (let
  ((__function__ 'vl-enumbasetype-wildelim))
  (declare (ignorable __function__))
  (b* ((x (vl-enumbasetype-fix x))
       (warnings (vl-warninglist-fix warnings)))
