Function: get-option-keywords

(defun get-option-keywords (l)
       (declare (xargs :guard (keyword-option-listp l)))
       (cond ((atom l) nil)
             (t (cons ((lambda (x) (if (atom x) x (car x)))
                       (car l))
                      (get-option-keywords (cdr l))))))
