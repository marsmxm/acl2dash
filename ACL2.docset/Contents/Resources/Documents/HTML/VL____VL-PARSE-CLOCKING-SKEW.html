Function: vl-parse-clocking-skew-fn

(defun
 vl-parse-clocking-skew-fn
 (tokstream config)
 (declare (xargs :stobjs (tokstream)))
 (declare (xargs :guard (vl-loadconfig-p config)))
 (declare (ignorable config))
 (let
  ((__function__ 'vl-parse-clocking-skew))
  (declare (ignorable __function__))
  (seq
   tokstream
   (when (vl-is-some-token?
              '(:vl-kwd-posedge :vl-kwd-negedge :vl-kwd-edge))
         (edge := (vl-match)))
   (when (vl-is-token? :vl-pound)
         (delay := (vl-parse-delay-control)))
   (unless
       (or edge delay)
       (return-raw
            (vl-parse-error "Expected edge identifier or #delay.")))
   (return (make-vl-clkskew
                :delay (and delay (vl-delaycontrol->value delay))
                :edge (case (and edge (vl-token->type edge))
                            (:vl-kwd-posedge :vl-posedge)
                            (:vl-kwd-negedge :vl-negedge)
                            (:vl-kwd-edge :vl-edge)
                            (otherwise :vl-noedge)))))))
