Theorem: exec-expr-asg-ident

(defthm
   exec-expr-asg-ident
   (implies
        (and (syntaxp (quotep e))
             (equal (expr-kind e) :binary)
             (equal (binop-kind (expr-binary->op e))
                    :asg)
             (not (zp limit))
             (equal e1 (expr-binary->arg1 e))
             (equal (expr-kind e1) :ident)
             (equal val+compst1
                    (exec-expr-call-or-pure (expr-binary->arg2 e)
                                            compst fenv (1- limit)))
             (equal val (mv-nth 0 val+compst1))
             (equal compst1 (mv-nth 1 val+compst1))
             (valuep val))
        (equal (exec-expr-asg e compst fenv limit)
               (write-var (expr-ident->get e1)
                          val compst1))))
