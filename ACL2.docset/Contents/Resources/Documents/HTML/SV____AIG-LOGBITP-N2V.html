(defun
 sv::aig-logbitp-n2v (place digit n)
 (declare (xargs :guard (and (posp place)
                             (true-listp digit)
                             (true-listp n))))
 (let
  ((__function__ 'sv::aig-logbitp-n2v))
  (declare (ignorable __function__))
  (b*
   (((mv first & end) (first/rest/end n))
    (place (lposfix place))
    ((when (or (atom digit) end)) first))

                  (sv::aig-logbitp-n2v (* 2 place)

(defthm sv::aig-logbitp-n2v-correct
        (b* ((bit (sv::aig-logbitp-n2v place digit n)))
