Function: vl-range-wildelim

(defun vl-range-wildelim (x ctx warnings)
       (declare (xargs :guard (and (vl-range-p x)
                                   (vl-context-p ctx)
                                   (vl-warninglist-p warnings))))
       (let ((__function__ 'vl-range-wildelim))
            (declare (ignorable __function__))
            (b* ((x (vl-range-fix x))
                 (warnings (vl-warninglist-fix warnings)))
                (b* (((vl-range x) x)
                     ((mv warnings msb-prime)
                      (vl-expr-wildelim x.msb ctx warnings))
                     ((mv warnings lsb-prime)
                      (vl-expr-wildelim x.lsb ctx warnings))
                     (x-prime (change-vl-range x
                                               :msb msb-prime
                                               :lsb lsb-prime)))
                    (mv warnings x-prime)))))
