(defun
 vl-possible-typo-warnings (typo-alist)
 (declare (xargs :guard (alistp typo-alist)))
 (declare (xargs :guard (and (vl-string-keys-p typo-alist)
                             (vl-string-list-values-p typo-alist))))
 (let
  ((__function__ 'vl-possible-typo-warnings))
  (declare (ignorable __function__))
  (b*
    (((when (atom typo-alist)) nil)
     ((cons badwire good-alternatives)
      (car typo-alist))
     (w (make-vl-warning
             :type :vl-warn-possible-typo
             :msg "Possible typo: implicit wire ~s0 looks like ~&1."
             :args (list badwire good-alternatives)
             :fn __function__
             :fatalp nil)))
    (cons w
          (vl-possible-typo-warnings (cdr typo-alist))))))
