Theorem: normalize-<-/-to-*

(defthm normalize-<-/-to-*
        (implies (and (real/rationalp x)
                      (real/rationalp y)
                      (not (equal y 0)))
                 (and (equal (< x (/ y))
                             (if (< y 0)
                                 (< 1 (* x y))
