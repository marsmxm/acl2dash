Function: vl-pp-assign-fn

(defun
  vl-pp-assign-fn (x ps)
  (declare (xargs :stobjs (ps)))
  (declare (xargs :guard (vl-assign-p x)))
  (let ((__function__ 'vl-pp-assign))
       (declare (ignorable __function__))
       (b* (((vl-assign x) x))
           (vl-ps-seq (if x.atts
                          (vl-ps-seq (vl-println "")
                                     (vl-progindent)
                                     (vl-pp-atts x.atts)
                                     (vl-println ""))
                          ps)
                      (vl-progindent)
                      (vl-ps-span "vl_key" (vl-print "assign "))
                      (if (not x.strength)
                          ps
                          (vl-ps-seq (vl-pp-gatestrength x.strength)
                                     (vl-print " ")))
                      (if (not x.delay)
                          ps
                          (vl-ps-seq (vl-pp-gatedelay x.delay)
                                     (vl-print " ")))
                      (vl-pp-expr x.lvalue)
                      (vl-println? " = ")
                      (vl-pp-expr x.expr)
                      (vl-println " ;")))))
