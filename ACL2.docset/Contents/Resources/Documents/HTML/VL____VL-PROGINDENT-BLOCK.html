Function: vl-progindent-block-start-fn

(defun vl-progindent-block-start-fn (ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard t))
       (let ((__function__ 'vl-progindent-block-start))
            (declare (ignorable __function__))
            (let* ((ind (vl-ps->autowrap-ind))
                   (ps (vl-ps-update-autowrap-ind (+ 2 ind)))
                   (col (vl-ps->autowrap-col))
                   (ps (vl-ps-update-autowrap-col (+ 2 col))))
                  ps)))

Function: vl-progindent-block-end-fn

(defun
     vl-progindent-block-end-fn (ps)
     (declare (xargs :stobjs (ps)))
     (declare (xargs :guard t))
     (let ((__function__ 'vl-progindent-block-end))
          (declare (ignorable __function__))
          (let* ((ind (vl-ps->autowrap-ind))
                 (ps (vl-ps-update-autowrap-ind (nfix (+ -2 ind))))
                 (col (vl-ps->autowrap-col))
                 (ps (vl-ps-update-autowrap-col (nfix (+ -2 col)))))
                ps)))
