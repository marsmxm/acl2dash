Theorem: exec-cast-of-schar-when-scharp

(defthm
 exec-cast-of-schar-when-scharp
 (implies
  (and
   (syntaxp
    (or
     (atom x)
     (not (member-eq x
                     '(exec-ident exec-const
                                  exec-iconst exec-arrsub exec-unary
                                  exec-cast exec-binary-strict-pure
                                  exec-test exec-expr-pure)))))
   (scharp x))
  (equal (exec-cast '(:tyname (specs :schar) (pointerp))
                    x)
         x)))
