Function: print-comma-sep

(defun print-comma-sep (parts)
       (declare (xargs :guard (msg-listp parts)))
       (let ((__function__ 'print-comma-sep))
            (declare (ignorable __function__))
            (cond ((endp parts) "")
                  ((endp (cdr parts)) (car parts))
                  (t (msg "~@0, ~@1" (car parts)
                          (print-comma-sep (cdr parts)))))))
