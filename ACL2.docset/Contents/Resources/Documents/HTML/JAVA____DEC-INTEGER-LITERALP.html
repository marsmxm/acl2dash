Function: dec-integer-literalp

(defun
     dec-integer-literalp (x)
     (declare (xargs :guard t))
     (let ((__function__ 'dec-integer-literalp))
          (declare (ignorable __function__))
          (and (consp x)
               (eq (car x) :dec-integer-lit)
               (true-listp (cdr x))
               (eql (len (cdr x)) 2)
               (b* ((digits/uscores (std::da-nth 0 (cdr x)))
                    (suffix? (std::da-nth 1 (cdr x))))
                   (and (decdig/uscore-listp digits/uscores)
                        (optional-integer-type-suffix-p suffix?)
                        (decdig/uscore-list-wfp digits/uscores))))))
