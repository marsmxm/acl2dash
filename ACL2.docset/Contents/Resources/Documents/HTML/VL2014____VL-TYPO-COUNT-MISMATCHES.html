(defun vl-typo-count-mismatches (x y)
       (declare (xargs :guard (same-lengthp x y)))
       (let ((__function__ 'vl-typo-count-mismatches))
            (declare (ignorable __function__))
            (cond ((atom x) 0)
                  ((equal (car x) (car y))
                   (vl-typo-count-mismatches (cdr x)
                                             (cdr y)))
                  (t (+ 1
                        (vl-typo-count-mismatches (cdr x)
                                                  (cdr y)))))))
