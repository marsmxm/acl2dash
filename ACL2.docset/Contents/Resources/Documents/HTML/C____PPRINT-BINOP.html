Function: pprint-binop

(defun pprint-binop (op)
       (declare (xargs :guard (binopp op)))
       (let ((__function__ 'pprint-binop))
            (declare (ignorable __function__))
            (binop-case op
                        :mul "*"
                        :div "/"
                        :rem "%"
                        :add "+"
                        :sub "-"
                        :shl "<<"
                        :shr ">>"
                        :lt "<"
                        :gt ">"
                        :le "<="
                        :ge ">="
                        :eq "=="
                        :ne "!="
                        :bitand "&"
                        :bitxor "^"
                        :bitior "|"
                        :logand "&&"
                        :logor "||"
                        :asg "="
                        :asg-mul "*="

Theorem: msgp-of-pprint-binop

(defthm msgp-of-pprint-binop
        (b* ((part (pprint-binop op)))
            (msgp part))
        :rule-classes :rewrite)

Theorem: pprint-binop-of-binop-fix-op

(defthm pprint-binop-of-binop-fix-op
        (equal (pprint-binop (binop-fix op))
               (pprint-binop op)))

Theorem: pprint-binop-binop-equiv-congruence-on-op

(defthm pprint-binop-binop-equiv-congruence-on-op
        (implies (binop-equiv op op-equiv)
