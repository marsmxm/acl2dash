(defun vl-jp-string-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-string-p x)))
       (let ((__function__ 'vl-jp-string))
            (declare (ignorable __function__))
            (vl-ps-seq (vl-print "{\"tag\": ")
                       (jp-sym ':vl-string)
                       (vl-print ", ")
                       (vl-print-str "\"value\": ")
                       (jp-str (vl-string->value x))
                       (vl-println? "}"))))
