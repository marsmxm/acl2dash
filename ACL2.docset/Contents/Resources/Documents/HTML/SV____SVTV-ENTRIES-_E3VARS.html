(defun
 svtv-entries->vars (x)
 (declare (xargs :guard (svtv-entrylist-p x)))
 (let
   ((__function__ 'svtv-entries->vars))
   (declare (ignorable __function__))
   (if (atom x)
       nil
       (let ((xf (svtv-entry-fix (car x))))
            (cond ((and (symbolp xf)
                        (not (svtv-dontcare-p xf))
                        (not (equal (symbol-name xf) "~"))
                        (not (eq xf :ones)))
                   (cons xf (svtv-entries->vars (cdr x))))

                        (let ((xf (svtv-condoverride->value xf)))
