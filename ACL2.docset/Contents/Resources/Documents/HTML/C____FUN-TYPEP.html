Function: fun-typep

(defun
 fun-typep (x)
 (declare (xargs :guard t))
 (let ((__function__ 'fun-typep))
      (declare (ignorable __function__))
      (and (mbe :logic (and (alistp x)
                            (equal (strip-cars x) '(inputs output)))
                :exec (fty::alist-with-carsp x '(inputs output)))
           (b* ((inputs (cdr (std::da-nth 0 x)))
                (output (cdr (std::da-nth 1 x))))

                    (typep output))))))
