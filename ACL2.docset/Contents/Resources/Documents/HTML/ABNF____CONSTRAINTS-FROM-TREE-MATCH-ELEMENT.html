Theorem: constraints-from-tree-match-element

(defthm
   constraints-from-tree-match-element
   (implies
        (and (tree-match-element-p tree (element-rulename *element*)
                                   *all-concrete-syntax-rules*)
             (tree-terminatedp tree))
        (or (and (<= (char-code #\A)
                     (car (tree->string tree)))
                 (<= (car (tree->string tree))
                     (char-code #\Z)))
            (and (<= (char-code #\a)
                     (car (tree->string tree)))
                 (<= (car (tree->string tree))
                     (char-code #\z)))
            (member-equal (car (tree->string tree))
                          (chars=>nats '(#\" #\% #\( #\< #\[)))))
   :rule-classes nil)
