Function: vl-echar

(defun
 vl-echar (char loc)
 (declare (xargs :guard (and (characterp char)
                             (vl-location-p loc))))
 (let
     ((__function__ 'vl-echar))
     (declare (ignorable __function__))
     (b* (((vl-location loc) loc))
         (vl-echar-raw
              loc.filename
              (vl-echarpack (the (unsigned-byte 8) (char-code char))
                            loc.line loc.col)))))
