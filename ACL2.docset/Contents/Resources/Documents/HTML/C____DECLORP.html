Function: declorp

(defun
     declorp (x)
     (declare (xargs :guard t))
     (let ((__function__ 'declorp))
          (declare (ignorable __function__))
          (and (consp x)
               (eq (car x) :declor)
               (mbe :logic (and (alistp (cdr x))
                                (equal (strip-cars (cdr x))
                                       '(pointerp ident)))
                    :exec (fty::alist-with-carsp (cdr x)
                                                 '(pointerp ident)))
               (b* ((pointerp (cdr (std::da-nth 0 (cdr x))))
                    (ident (cdr (std::da-nth 1 (cdr x)))))
                   (and (booleanp pointerp)
                        (identp ident))))))
