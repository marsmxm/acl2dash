(defun
 defmapping-gen-appcond-thms
 (appconds appcond-thm-names thm-names$ wrld)
 (declare
      (xargs :guard (and (evmac-appcond-listp appconds)
                         (symbol-symbol-alistp appcond-thm-names)
                         (symbol-symbol-alistp thm-names$)
                         (plist-worldp wrld))))
 (let
  ((__function__ 'defmapping-gen-appcond-thms))
  (declare (ignorable __function__))
  (b*
   (((when (endp appconds)) (mv nil nil))
    ((mv local-event exported-event)
     (defmapping-gen-appcond-thm (car appconds)

       (cons exported-event exported-events)))))
