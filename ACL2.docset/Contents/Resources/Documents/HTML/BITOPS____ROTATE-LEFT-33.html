   :logic (rotate-left x 33 places)
   :exec

                   :exec (if (< places 33)

                     (ash xh (the (integer -33 0) (- low-num)))))
      (xl-shift (the (unsigned-byte 33)
                     (ash (the (unsigned-byte 33) xl)
                          (the (integer 0 33) places))))
      (ans (the (unsigned-byte 33)
                (logior xl-shift xh-shift))))
     ans))))
