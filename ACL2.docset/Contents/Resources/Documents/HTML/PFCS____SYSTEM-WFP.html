Check if a system is well-formed.

Signature
(system-wfp sys) → yes/no
Arguments
sys — Guard (systemp sys).
Returns
yes/no — Type (booleanp yes/no).

The empty system is well-formed, and every well-formed system can be extended by adding a definition that is well-formed in that system. To iterate through the list more easily, we use an auxiliary function that operates on the reversed system.

Definitions and Theorems

Function: system-wfp-aux

(defun system-wfp-aux (rev-sys)
       (declare (xargs :guard (systemp rev-sys)))
       (let ((__function__ 'system-wfp-aux))
            (declare (ignorable __function__))
            (or (endp rev-sys)
                (and (system-wfp-aux (cdr rev-sys))
                     (definition-wfp (car rev-sys)
                                     (cdr rev-sys))))))

Theorem: booleanp-of-system-wfp-aux

(defthm booleanp-of-system-wfp-aux
        (b* ((yes/no (system-wfp-aux rev-sys)))
            (booleanp yes/no))
        :rule-classes :rewrite)

Theorem: system-wfp-aux-of-system-fix-rev-sys

(defthm system-wfp-aux-of-system-fix-rev-sys
        (equal (system-wfp-aux (system-fix rev-sys))
               (system-wfp-aux rev-sys)))

Theorem: system-wfp-aux-system-equiv-congruence-on-rev-sys

(defthm system-wfp-aux-system-equiv-congruence-on-rev-sys
        (implies (system-equiv rev-sys rev-sys-equiv)
                 (equal (system-wfp-aux rev-sys)
                        (system-wfp-aux rev-sys-equiv)))
        :rule-classes :congruence)

Function: system-wfp

(defun system-wfp (sys)
       (declare (xargs :guard (systemp sys)))
       (let ((__function__ 'system-wfp))
            (declare (ignorable __function__))
            (system-wfp-aux (rev (system-fix sys)))))

Theorem: booleanp-of-system-wfp

(defthm booleanp-of-system-wfp
        (b* ((yes/no (system-wfp sys)))
            (booleanp yes/no))
        :rule-classes :rewrite)

(defthm system-wfp-of-system-fix-sys
