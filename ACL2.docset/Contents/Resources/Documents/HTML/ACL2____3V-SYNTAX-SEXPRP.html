Function: 3v-syntax-sexprp

(defun 3v-syntax-sexprp (x)
       (declare (xargs :guard t))
       (and (consp x)
            (let ((fn (car x)))
                 (not (or (eq fn (4vz))
                          (eq fn 'id)
                          (eq fn 'res)
                          (eq fn 'zif)
                          (eq fn 'tristate))))))
