Function: short-valuep

(defun short-valuep (x)
       (declare (xargs :guard t))
       (let ((__function__ 'short-valuep))
            (declare (ignorable __function__))
            (and (consp x)
                 (eq (car x) :short)
                 (true-listp (cdr x))
                 (eql (len (cdr x)) 1)
                 (b* ((int (std::da-nth 0 (cdr x))))
                     (sbyte16p int)))))
