Function: svex-select-replace-indices

(defun
 svex-select-replace-indices (x indices)
 (declare (xargs :guard (and (svex-select-p x)
                             (svexlist-p indices))))
 (declare (xargs :guard (equal (len indices)
                               (len (svex-select->indices x)))))
 (let
  ((__function__ 'svex-select-replace-indices))
  (declare (ignorable __function__))
  (b*
   ((x (svex-select-fix x)))
   (svex-select-case
      x
      :var x
      :part
      (change-svex-select-part
           x
           :lsb (car indices)
           :subexp
           (svex-select-replace-indices x.subexp (cdr indices)))))))
