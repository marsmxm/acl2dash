Function: vl-find-portdecl

(defun vl-find-portdecl (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-portdecllist-p x))))
       (let ((__function__ 'vl-find-portdecl))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-portdecl->name (car x)))
                   (vl-portdecl-fix (car x)))
                  (t (vl-find-portdecl name (cdr x))))))
