Function: print-to-jfile

(defun
     print-to-jfile (lines filename state)
     (declare (xargs :stobjs (state)))
     (declare (xargs :guard (and (msg-listp lines)
                                 (stringp filename))))
     (let ((__function__ 'print-to-jfile))
          (declare (ignorable __function__))
          (b* (((mv channel state)
                (open-output-channel! filename
                                      :character state))
               (state (print-jlines-to-channel lines channel state))
               (state (close-output-channel channel state)))
              state)))
