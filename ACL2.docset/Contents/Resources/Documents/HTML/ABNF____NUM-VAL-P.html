Function: num-val-p

(defun
     num-val-p (x)
     (declare (xargs :guard t))
     (let ((__function__ 'num-val-p))
          (declare (ignorable __function__))
          (and (consp x)
               (cond ((or (atom x) (eq (car x) :direct))
                      (and (true-listp (cdr x))
                           (eql (len (cdr x)) 1)
                           (b* ((get (std::da-nth 0 (cdr x))))
                               (nat-listp get))))
                     (t (and (eq (car x) :range)
                             (and (true-listp (cdr x))
                                  (eql (len (cdr x)) 2))
                             (b* ((min (std::da-nth 0 (cdr x)))
                                  (max (std::da-nth 1 (cdr x))))
