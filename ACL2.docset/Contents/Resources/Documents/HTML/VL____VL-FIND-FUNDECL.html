Function: vl-find-fundecl

(defun vl-find-fundecl (name x)
       (declare (xargs :guard (and (stringp name)
                                   (vl-fundecllist-p x))))
       (let ((__function__ 'vl-find-fundecl))
            (declare (ignorable __function__))
            (cond ((atom x) nil)
                  ((equal (string-fix name)
                          (vl-fundecl->name (car x)))
                   (vl-fundecl-fix (car x)))
                  (t (vl-find-fundecl name (cdr x))))))

(defthm vl-find-fundecl-under-iff
        (iff (vl-find-fundecl name x)
