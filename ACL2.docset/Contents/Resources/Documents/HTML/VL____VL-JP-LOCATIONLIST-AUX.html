Arguments
x — Guard (vl-locationlist-p x).

Definitions and Theorems

Function: vl-jp-locationlist-aux-fn

(defun vl-jp-locationlist-aux-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-locationlist-p x)))
       (let ((__function__ 'vl-jp-locationlist-aux))
            (declare (ignorable __function__))
            (if (atom x)
                ps
                (vl-ps-seq (vl-jp-location (car x))
                           (if (atom (cdr x))
                               ps
                               (vl-ps-seq (vl-println ",")
                                          (vl-indent 8)))
                           (vl-jp-locationlist-aux (cdr x))))))
