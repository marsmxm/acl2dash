Function: v2i

(defun v2i (v)
       (declare (xargs :guard (true-listp v)))
       (let ((__function__ 'v2i))
            (declare (ignorable __function__))
            (mbe :logic (if (s-endp v)
                            (bool->sign (car v))
                            (logcons (bool->bit (car v))
                                     (v2i (scdr v))))
                 :exec (if (atom v)
                           0
                           (if (atom (cdr v))
                               (if (car v) -1 0)
                               (logcons (bool->bit (car v))
                                        (v2i (cdr v))))))))
