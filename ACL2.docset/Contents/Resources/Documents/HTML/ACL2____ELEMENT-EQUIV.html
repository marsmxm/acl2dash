Theorem: element-equiv-is-an-equivalence

(defthm element-equiv-is-an-equivalence
        (and (booleanp (element-equiv x y))
             (element-equiv x x)
             (implies (element-equiv x y)
                      (element-equiv y x))
             (implies (and (element-equiv x y)
                           (element-equiv y z))
                      (element-equiv x z)))
        :rule-classes (:equivalence))
