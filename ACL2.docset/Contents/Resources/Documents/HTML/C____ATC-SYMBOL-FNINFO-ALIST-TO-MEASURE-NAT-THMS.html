(defun
 atc-symbol-fninfo-alist-to-measure-nat-thms
 (prec-fns among)
 (declare (xargs :guard (and (atc-symbol-fninfo-alistp prec-fns)
                             (symbol-listp among))))
 (let
  ((__function__ 'atc-symbol-fninfo-alist-to-measure-nat-thms))
  (declare (ignorable __function__))
  (cond
   ((endp prec-fns) nil)
   ((member-eq (caar prec-fns) among)
    (b*
     ((thm (atc-fn-info->measure-nat-thm (cdr (car prec-fns)))))
     (if
      thm
      (cons
         thm
         (atc-symbol-fninfo-alist-to-measure-nat-thms (cdr prec-fns)
                                                      among))
      (atc-symbol-fninfo-alist-to-measure-nat-thms (cdr prec-fns)
                                                   among))))
   (t (atc-symbol-fninfo-alist-to-measure-nat-thms (cdr prec-fns)
                                                   among)))))
