Function: set-all-integerp

(defun set-all-integerp (set)
       (declare (xargs :guard (setp set)))
       (let ((acl2::__function__ 'set-all-integerp))
            (declare (ignorable acl2::__function__))
            (or (empty set)
                (and (integerp (head set))
                     (set-all-integerp (tail set))))))
