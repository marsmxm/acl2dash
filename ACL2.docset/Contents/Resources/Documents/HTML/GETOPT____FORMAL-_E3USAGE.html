(defun formal->usage (x)
       (declare (xargs :guard (formal-p x)))
       (let ((__function__ 'formal->usage))
            (declare (ignorable __function__))
            (b* (((formal x) x))
                (or (cdr (assoc :usage x.opts))
                    x.doc))))
