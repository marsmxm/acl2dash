Function: string-or-symbol-listp

(defun string-or-symbol-listp (x)
       (declare (xargs :guard t))
       (if (atom x)
           (null x)
           (and (or (stringp (car x)) (symbolp (car x)))
                (string-or-symbol-listp (cdr x)))))
