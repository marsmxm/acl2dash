Function: vl-programlist-zombies

(defun
 vl-programlist-zombies (x)
 (declare (xargs :guard (vl-programlist-p x)))
 (let ((__function__ 'vl-programlist-zombies))
      (declare (ignorable __function__))
      (cond ((atom x) nil)
            ((vl-some-warning-fatalp (vl-program->warnings (car x)))
             (cons (vl-program->name (car x))
                   (vl-programlist-zombies (cdr x))))
            (t (vl-programlist-zombies (cdr x))))))
