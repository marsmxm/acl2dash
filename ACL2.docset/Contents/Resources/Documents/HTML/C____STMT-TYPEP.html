Function: stmt-typep

(defun
 stmt-typep (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'stmt-typep))
  (declare (ignorable __function__))
  (and
     (mbe :logic (and (alistp x)
                      (equal (strip-cars x)
                             '(return-types variables)))
          :exec (fty::alist-with-carsp x '(return-types variables)))
     (b* ((return-types (cdr (std::da-nth 0 x)))
          (variables (cdr (std::da-nth 1 x))))
         (and (type-setp return-types)
              (var-tablep variables)
              (not (empty return-types)))))))
