Function: vl-jp-interfaceportlist-aux-fn

(defun vl-jp-interfaceportlist-aux-fn (x ps)
       (declare (xargs :stobjs (ps)))
       (declare (xargs :guard (vl-interfaceportlist-p x)))
       (let ((__function__ 'vl-jp-interfaceportlist-aux))
            (declare (ignorable __function__))
            (if (atom x)
                ps
                (vl-ps-seq (vl-jp-interfaceport (car x))
                           (if (atom (cdr x))
                               ps
                               (vl-ps-seq (vl-println ",")
                                          (vl-indent 4)))
                           (vl-jp-interfaceportlist-aux (cdr x))))))
