Function: vl-unparam-actualkeys

(defun
 vl-unparam-actualkeys (x inst-ss mod-ss)
 (declare (xargs :guard (and (vl-paramdecllist-p x)
                             (vl-scopestack-p inst-ss)
                             (vl-scopestack-p mod-ss))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-unparam-actualkeys))
  (declare (ignorable __function__))
  (mbe
    :logic (if (consp x)
               (cons (vl-unparam-actualkey (car x)
                                           inst-ss mod-ss)
                     (vl-unparam-actualkeys (cdr x)
                                            inst-ss mod-ss))
               nil)
    :exec
    (if (atom x)
        nil
        (with-local-nrev
             (vl-unparam-actualkeys-nrev x inst-ss mod-ss nrev))))))
