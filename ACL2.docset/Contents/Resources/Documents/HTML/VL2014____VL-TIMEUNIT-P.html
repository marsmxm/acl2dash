Function: vl-timeunit-p

(defun vl-timeunit-p (x)
       (declare (xargs :guard t))
       (or (eq x ':vl-s)
           (eq x ':vl-ms)
           (eq x ':vl-us)
           (eq x ':vl-ns)
           (eq x ':vl-ps)
           (eq x ':vl-fs)))
