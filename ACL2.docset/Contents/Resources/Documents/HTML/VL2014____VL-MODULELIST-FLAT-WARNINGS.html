Gather a flat list of all warnings from a vl-modulelist-p.

This is an ordinary defmapappend.

Function: vl-modulelist-flat-warnings

(defun
     vl-modulelist-flat-warnings (x)
     (declare (xargs :guard (vl-modulelist-p x)))
     (mbe :logic (if (consp x)
                     (append (vl-module->warnings (car x))
                             (vl-modulelist-flat-warnings (cdr x)))
                     nil)
          :exec (reverse (vl-modulelist-flat-warnings-exec x nil))))
