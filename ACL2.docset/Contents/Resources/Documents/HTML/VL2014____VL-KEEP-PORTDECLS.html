Function: vl-keep-portdecls

(defun
 vl-keep-portdecls (names x)
 (declare (xargs :guard (and (string-listp names)
                             (vl-portdecllist-p x))))
 (let
  ((__function__ 'vl-keep-portdecls))
  (declare (ignorable __function__))
  (mbe
   :logic (cond ((atom x) nil)
                ((member-equal (vl-portdecl->name (car x))
                               (string-list-fix names))
                 (cons (vl-portdecl-fix (car x))
                       (vl-keep-portdecls names (cdr x))))
                (t (vl-keep-portdecls names (cdr x))))
   :exec
   (b*
    (((when (or (atom names) (atom x))) nil)
     ((unless (longer-than-p 10 names))
      (vl-slow-keep-portdecls names x))
     (fal (make-lookup-alist names))
     (ans
        (with-local-nrev (vl-fast-keep-portdecls names fal x nrev)))
     (- (fast-alist-free fal)))
    ans))))
