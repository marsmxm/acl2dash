Function: vl-scope-find-item-fast

(defun
 vl-scope-find-item-fast
 (name scope design)
 (declare (xargs :guard (and (stringp name)
                             (vl-scope-p scope)
                             (vl-maybe-design-p design))))
 (let
  ((__function__ 'vl-scope-find-item-fast))
  (declare (ignorable __function__))
  (mbe
     :logic (vl-scope-find-item name scope design)
     :exec
     (vl-scopeinfo-find-item name (vl-scope->scopeinfo scope design)
                             design))))
