Theorem: funtype-equiv-is-an-equivalence

(defthm funtype-equiv-is-an-equivalence
        (and (booleanp (funtype-equiv x y))
             (funtype-equiv x x)
             (implies (funtype-equiv x y)
                      (funtype-equiv y x))
             (implies (and (funtype-equiv x y)
                           (funtype-equiv y z))
                      (funtype-equiv x z)))
