       (let ((__function__ 'vl-portlist-msb-bit-pattern))
            (declare (ignorable __function__))
            (b* (((when (atom x)) (mv t nil nil))
                 ((mv successp1 warnings1 wires1)
                  (vl-port-msb-bits (car x) walist))
