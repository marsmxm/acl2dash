(defun vl-merged-index-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-merged-index-p))
            (declare (ignorable __function__))
            (or (not x)
                (integerp x)
                (and (consp x)
                     (integerp (car x))
                     (integerp (cdr x))
                     (< (car x) (cdr x))))))
