(defun
 vl-paramdecl-p (x)
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-paramdecl-p))
  (declare (ignorable __function__))
  (and
   (consp x)
   (eq (car x) :vl-paramdecl)
   (std::prod-consp (cdr x))
   (std::prod-consp (std::prod-car (cdr x)))
   (std::prod-consp (std::prod-cdr (cdr x)))
   (std::prod-consp (std::prod-cdr (std::prod-cdr (cdr x))))
   (b*
     ((name (std::prod-car (std::prod-car (cdr x))))
      (localp (std::prod-cdr (std::prod-car (cdr x))))
      (type (std::prod-car (std::prod-cdr (cdr x))))
      (atts (std::prod-car (std::prod-cdr (std::prod-cdr (cdr x)))))
