(defun vl-def-context-p (x)
       (declare (xargs :guard t))
       (let ((__function__ 'vl-def-context-p))
            (declare (ignorable __function__))
            (and (std::prod-consp x)
                 (b* ((activep (std::prod-car x))
                      (loc (std::prod-cdr x)))
                     (and (booleanp activep)
                          (vl-location-p loc))))))
