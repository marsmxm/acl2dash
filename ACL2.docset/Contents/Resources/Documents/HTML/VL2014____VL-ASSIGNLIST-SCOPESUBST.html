Function: vl-assignlist-scopesubst

(defun
 vl-assignlist-scopesubst (x ss)
 (declare (xargs :guard (and (vl-assignlist-p x)
                             (vl-scopestack-p ss))))
 (declare (xargs :guard t))
 (let
  ((__function__ 'vl-assignlist-scopesubst))
  (declare (ignorable __function__))
  (mbe
   :logic (if (consp x)
              (cons (vl-assign-scopesubst (car x) ss)
                    (vl-assignlist-scopesubst (cdr x) ss))
              nil)
   :exec
   (if
     (atom x)
     nil
     (with-local-nrev (vl-assignlist-scopesubst-nrev x ss nrev))))))
