Function: vl-udp-level-symbol-token-p

(defun
 vl-udp-level-symbol-token-p (x)
 (declare (xargs :guard (vl-token-p x)))
 (let
    ((__function__ 'vl-udp-level-symbol-token-p))
    (declare (ignorable __function__))
    (case (vl-token->type x)
          (:vl-inttoken
               (let ((etext (vl-inttoken->etext x)))
                    (and (consp etext)
                         (not (consp (cdr etext)))
                         (consp (member (vl-echar->char (car etext))
                                        '(#\0 #\1))))))
          (:vl-idtoken
               (let ((name (vl-idtoken->name x)))
                    (consp (member-equal name '("x" "X" "b" "B")))))
          (:vl-qmark t)
          (otherwise nil))))
